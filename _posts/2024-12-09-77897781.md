    <pre><code class="language-java">import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Calendar;
import java.util.Date;
import java.util.List;

public class DayUtils {

	public static  List&lt;String&gt; getBetweenDays(String stime,String etime){
		SimpleDateFormat df = new SimpleDateFormat("yyyy-MM-dd");
		Date sdate = null;
		Date eDate = null;

		try {
			sdate = df.parse(stime);
			eDate = df.parse(etime);
		} catch (ParseException e) {
			
			e.printStackTrace();
		}
		long betweendays = (long) ((eDate.getTime() - sdate.getTime())
				/ (1000 * 60 * 60 * 24) + 0.5);// 天数间隔
		Calendar c = Calendar.getInstance();
		List&lt;String&gt; list = new ArrayList&lt;String&gt;();
		while (sdate.getTime() &lt;= eDate.getTime()) {
			list.add(df.format(sdate));
			c.setTime(sdate);
			c.add(Calendar.DATE, 1); // 日期加1天
			sdate = c.getTime();
		}
		return list;
  }
}
