      1、接口支持私有方法
     </strong>
     <br/>

    }

    <span class="hljs-keyword">private</span> String <span class="hljs-title">getStu</span>() {
        <span class="hljs-keyword">return</span> <span class="hljs-string">"hello world!"</span>;
    }

}
