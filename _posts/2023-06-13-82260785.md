     <span style="color:#ff0000;">

    <ol>
     <li>
      <span style="color:#000000;">
       下载安装包：
      </span>
     </li>
    </ol>
    <p style="margin-left:0pt;">
     <span style="color:#000000;">
      wget
     </span>
     <a href="https://www.python.org/ftp/python/3.7.0/Python-3.7.0.tgz" rel="nofollow">
      <u>
       <span style="color:#0000ff;">

       </span>

      <u>
       <span style="color:#0000ff;">
        <u>
         .python.org
        </u>
       </span>
      </u>
      <u>
       <span style="color:#0000ff;">
        <u>
         /ftp/python/
        </u>
       </span>
      </u>
      <u>
       <span style="color:#0000ff;">
        <u>
         3.7.0
        </u>
       </span>
      </u>
      <u>
       <span style="color:#0000ff;">
        <u>
         /Python-
        </u>
       </span>
      </u>
      <u>
       <span style="color:#0000ff;">
        <u>
         3.7.0.tgz
        </u>
       </span>
      </u>
     </a>
    </p>
    <ol>
     <li>
      <span style="color:#000000;">
       安装编译器：
      </span>
     </li>
    </ol>
    <p style="margin-left:0pt;">
     yum -y install gcc gcc-c++
    </p>
    <ol>
     <li>
      <span style="color:#000000;">
       安装python3依赖：
      </span>
     </li>
    </ol>
    <p style="margin-left:0pt;">
     yum -y install zlib-devel bzip2 bzip2-devel readline-devel sqlite sqlite-devel openssl-devel xz xz-devel
    </p>
    <p style="margin-left:0pt;">
    </p>
    <p style="margin-left:0pt;">
     <span style="color:#ff0000;">
      python3.7以上版本还需安装此依赖
     </span>
    </p>
    <p style="margin-left:0pt;">
     yum -y install libffi-devel
    </p>
    <ol>
     <li>
      <span style="color:#000000;">
       安装：
      </span>
     </li>
    </ol>
    <p style="margin-left:0pt;">
     tar -zxvf Python-3.7.0.tgz
    </p>
    <p style="margin-left:0pt;">
     cd Python-3.7.0/
    </p>
    <p style="margin-left:0pt;">
     ./configure --prefix=/usr/local/python3
    </p>
    <p style="margin-left:0pt;">
     <span style="color:#ff0000;">
      (将python3安装到/
     </span>
     <span style="color:#ff0000;">
      U
     </span>
     <span style="color:#ff0000;">
      sr/local/python3路径下)
     </span>
    </p>
    <p style="margin-left:0pt;">
     make &amp;&amp; make install
    </p>
    <p style="margin-left:0pt;">
     创建软连接：
    </p>
    <p style="margin-left:0pt;">
     <span style="color:#000000;">
      ln -s /usr/local/python3/bin/python3.
     </span>
     <span style="color:#000000;">
      7
     </span>
     <span style="color:#000000;">
      /usr/bin/python3
     </span>
     <br/>
     <span style="color:#000000;">
      ln -s /usr/local/python3/bin/pip3 /usr/bin/pip3
     </span>
    </p>
    <p style="margin-left:0pt;">
     <span style="color:#ff0000;">
      （以上两个ln命令是为了方便在终端中直接使用python3和pip3命令）
     </span>
    </p>
