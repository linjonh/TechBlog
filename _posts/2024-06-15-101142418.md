     <p>
      一． 加载方式
      <br/>
      //class 加载方式
      <br/>
      &lt;input id="ss" class="easyui-searchbox" style="width:300px"
      <br/>
      data-options="searcher:qq,prompt:'Please Input
      <br/>
      Value',menu:'#box'"&gt;&lt;/input&gt;
      <br/>
      &lt;div id="box" style="width:120px"&gt;
      <br/>
      &lt;div data-options="name:'all',iconCls:'icon-ok'"&gt;All News&lt;/div&gt;
      <br/>
      &lt;div data-options="name:'sports'"&gt;Sports News&lt;/div&gt;
      <br/>
      &lt;/div&gt;
     </p>
     <p>
      <br/>
      //JS 加载调用
      <br/>
      $('#ss').searchbox({
      <!-- -->
      <br/>
      searcher : function (value, name) {
      <!-- -->
      <br/>
      alert(value + ',' + name);
      <br/>
      },
      <br/>
      menu : '#box',
      <br/>
      prompt : '请输入内容',
      <br/>
      });
     </p>
     <p>
     </p>
     <p>
      二．属性列表
     </p>
     <p>
      <img alt="" src="https://i-blog.csdnimg.cn/blog_migrate/5b30e3ed505b121691a1643650c569c0.png"/>
     </p>
     <p>
      <img alt="" src="https://i-blog.csdnimg.cn/blog_migrate/df1128e0f7dee5e528cadd7846008c9e.png"/>
     </p>
     <p>
      //属性设置
      <br/>
      $('#ss').searchbox({
      <!-- -->
      <br/>
      width : 300,
      <br/>
      height : 30,
      <br/>
      value : '固定内容',
      <br/>
      searcher : function (value, name) {
      <!-- -->
      <br/>
      alert(value + ',' + name);
      <br/>
      },
      <br/>
      menu : '#box',
      <br/>
      prompt : '请输入内容',
      <br/>
      disabled : true,
      <br/>
      });
     </p>
     <p>
     </p>
     <p>
      三． 方法列表
     </p>
     <p>
      <img alt="" src="https://i-blog.csdnimg.cn/blog_migrate/b3dd0a927c6293267dac562c99ad46d7.png"/>
     </p>
     <p>
      //返回属性对象
      <br/>
      console.log($('#ss').searchbox('options'));
     </p>
     <p>
      <br/>
      //返回文本框对象
      <br/>
      console.log($('#ss').searchbox('textbox'));
     </p>
     <p>
      <br/>
      //返回当前索引值
      <br/>
      console.log($('#ss').searchbox('getValue'));
     </p>
     <p>
      <br/>
      //改变当前索引值
      <br/>
      $('#ss').searchbox('setValue','改变值')
     </p>
     <p>
      <br/>
      //选择指定的搜索类型
      <br/>
      $('#ss').searchbox('selectName', 'sports');
     </p>
     <p>
      <br/>
      //返回当前索引类型值
      <br/>
      console.log($('#ss').searchbox('getName'));
     </p>
     <p>
      <br/>
      //重置搜索框宽度
      <br/>
      $('#ss').searchbox('resize', 200);
     </p>
     <p>
      <br/>
      //销毁搜索框
      <br/>
      $('#ss').searchbox('destroy');
     </p>
     <p>
      <br/>
      //禁用和启用
      <br/>
      $('#ss').searchbox('disable');
      <br/>
      $('#ss').searchbox('enable');
     </p>
     <p>
      <br/>
      //清理搜索框内容
      <br/>
      $('#ss').searchbox('clear');
     </p>
     <p>
      <br/>
      //重置搜索框内容
      <br/>
      $(document).dblclick(function () {
      <!-- -->
      <br/>
      $('#ss').searchbox('reset');
      <br/>
      });
     </p>
     <p>
      <br/>
      //返回类型名对象
      <br/>
      var m = $('#ss').searchbox('menu');
      <br/>
      var item = m.menu('findItem', '体育');
      <br/>
      m.menu('setIcon', {
      <!-- -->
      <br/>
      target: item.target,
      <br/>
      iconCls: 'icon-save'
      <br/>
      });
     </p>
     <p>
     </p>
     <p>
      PS：我们可以使用$.fn.searchbox.defaults 重写默认值对象。
     </p>
