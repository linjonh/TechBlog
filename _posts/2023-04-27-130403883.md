       4.在方法列表中查找调用该方法的对象
      </p>
      <p>
       如果我们在方法列表中查找的对象是一个类，那么我们可以使用该对象的名称来访问该方法，也可以使用其接口来访问该对象。例如，我们可以在方法列表中查找“input”这个对象，并使用该对象的接口来访问它。 为了说明方法列表中的所有方法，我们假设这些方法都是通过接口调用的。例如，如果一个类声明了一个接口“address”，那么在类的实例中就可以找到“address”这个方法。

     <br/>
     1. 调用 Java 方法
     <br/>
     ```c++
     <br/>
     #include &lt;jni.h&gt;
     <br/>
     #include &lt;iostream&gt;
     <br/>
     int main() {
     <!-- -->
     <br/>
     JavaVM* jvm;
     <br/>
     JNIEnv* env;
     <br/>
     JavaVMInitArgs args;
     <br/>
     JavaVMOption options[1];
     <br/>
     jclass cls;
     <br/>
     jmethodID mid;
     <br/>
     jint square;
     <br/>
     jintArray arr;
     <br/>
     jint buf[10];
     <br/>
     options[0].optionString = "-Djava.class.path=./";
     <br/>
     args.version = JNI_VERSION_1_6;
     <br/>
     args.nOptions = 1;
     <br/>
     args.options = options;
     <br/>
     args.ignoreUnrecognized = JNI_FALSE;
     <br/>
     JNI_CreateJavaVM(&amp;jvm, (void**)&amp;env, &amp;args);
     <br/>
     cls = env-&gt;FindClass("Test");
     <br/>
     mid = env-&gt;GetStaticMethodID(cls, "square", "([I)I");
     <br/>
     arr = env-&gt;NewIntArray(10);
     <br/>
     for (int i = 0; i &lt; 10; i++) {
     <!-- -->
     <br/>
     buf[i] = i;
     <br/>
     }
     <br/>
     env-&gt;SetIntArrayRegion(arr, 0, 10, buf);
     <br/>
     square = env-&gt;CallStaticIntMethod(cls, mid, arr);
     <br/>
     std::cout &lt;&lt; "Square of 5 is " &lt;&lt; square &lt;&lt; std::endl;
     <br/>
     jvm-&gt;DestroyJavaVM();
     <br/>
     return 0;
     <br/>
     }
     <br/>
     ```
