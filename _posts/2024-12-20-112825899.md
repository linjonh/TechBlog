     <p>
      print '遍历列表方法1：'
     </p>
     <p>
      for i in list:

      print ("序号：%s 值：%s" % (list.index(i) + 1, i))

      print '\n遍历列表方法2：'
     </p>
     <p>

     </p>

      for i in range(len(list)):
     </p>

      print ("序号：%s 值：%s" % (i + 1, list[i]))

     </p>

      print ("序号：%s 值：%s" % (i + 1, val))
     </p>
     <p>
      # 方法3
     </p>

      print '\n遍历列表方法3 (设置遍历开始初始位置，只改变了起始序号)：'

     <p>
      for i, val in enumerate(list, 2):
     </p>
     <p>
