    <p>
     -- 查看数据库 a  的表名和数据量  table_name(表名),table_rows(数据量)
    </p>
    <pre class="has"><code> select table_name,table_rows from tables where TABLE_SCHEMA = 'a'  ;</code></pre>
    <p style="text-indent:0;">
     -- 查看当前数据库a 有记录的TABLE_NAME（表名）
    </p>
    <pre class="has"><code> select TABLE_NAME ,table_rows from information_schema.tables where TABLE_SCHEMA='a' and table_rows&gt;0;</code></pre>
    <p>
     <br/>
     -- 查看数据库记录为空的表名和数据量TABLE_NAME (表名),table_rows(数据量)
    </p>
    <pre class="has"><code>
 select TABLE_NAME ,table_rows from information_schema.tables where TABLE_SCHEMA='a' and table_rows = 0;</code></pre>
    <p>
    </p>
    <p>
     <br/>
     -- 查看当前数据库 有记录的表名和数据量 并分组
    </p>
    <pre class="has"><code> use information_schema; 
 select table_name,table_rows from tables  where TABLE_SCHEMA = 'ls_tx' and table_rows&gt;0 order by table_rows desc;

</code></pre>
    <p>
     <span style="color:#f33b45;">
      <strong>
       注: use information_schema;  一定要有
      </strong>
     </span>
