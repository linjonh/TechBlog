     <p style="margin-left:.0001pt;text-align:justify;">
      思路：利用现有的单链表LA和LB中的结点空间建立新表LC；

     </p>
     <p style="margin-left:.0001pt;text-align:justify;">
      将LA或LB的较小元件插入LC；
     </p>
     <p style="margin-left:.0001pt;text-align:justify;">
      移动到列表中插入该元素的下一个元素；
     </p>
     <p style="margin-left:.0001pt;text-align:justify;">
      继续这个过程，直到两个列表的所有元素都合并到LC里面为止；

    <pre><code class="language-cpp">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;

//  定义单链表的节点结构
struct Node {/_结点类型定义 node_/
    int data;//数据域（int）
    struct Node\* next;
};

//  向单链表尾部插入节点，把具有给定 new_data 值的新节点插入末尾的链表里面。
//它等下指向列表头的指针 head 和新数据 new_data 作为输入参数
void insertNode(struct Node\** head, int new_data) {
    struct Node* new_node=(struct Node*)malloc(sizeof(struct Node));//为新节点分配内存
    struct Node* last=\*head;//最后的指针设为列表头

new_node-&gt;data=new_data;//初始化新节点的 data 和 next 字段。
    new_node-&gt;next=NULL;

if (*head ==NULL) { 
        *head =new_node;
        return;           //检查列表空的话，头指针设为新节点
    }

while (last-&gt;next !=NULL) {
        last=last-&gt;next; //last 指针遍历列表，直到到达末尾，
    }
    last-&gt;next=new_node; //last-&gt;next 设为新结点
}

//  构建用于合并两个排序的链表 LA 和 LB 的函数
struct Node* mergeLists(struct Node* LA, struct Node* LB) {
    struct Node* LC=NULL;  //创建结果合并列表
    struct Node\* tail =NULL;

while (1) {          
        if (LA==NULL) {
            tail-&gt;next=LB;
            break;           //检查 LA 是空的话将尾部节点 tail 的 next 字段设置为 LB 并中断循环
        }
        if (LB==NULL) {
            tail-&gt;next=LA;
            break;
        }

if (LA-&gt;data&lt;=LB-&gt;data) {
            insertNode(&amp;LC, LA-&gt;data);
            LA=LA-&gt;next;
        } else {
            insertNode(&amp;LC, LB-&gt;data);
            LB =LB-&gt;next;  //比较当前节点 LA 和 LB 的数值。如果 LA-&gt;data 小于或等于 LB-&gt;data，用 insertNode（）把 LA-&gt;data 的数据插入 LC，然后，LA 针移动到下一个节点，如果 LA-&gt;data 大于 LB-&gt;data，LB-&gt;data 值插入 LC
        }   
        if (tail==NULL) {
            tail=LC;
        } else {
            tail = tail-&gt;next;
        }
    }//持续到 LALB 完全被处理     
   return LC;//返回  LC  作为合并后的新单链表的头节点
}

//  打印单链表
void printList(struct Node\* node) {
    while (node != NULL) {
        printf("%d ", node-&gt;data);
        node = node-&gt;next;
    }
}

int main() {
    struct Node* LA = NULL;
    struct Node* LB = NULL;    
    //  将元素插入 LA
    insertNode(&amp;LA, 1);
    insertNode(&amp;LA, 2);
    insertNode(&amp;LA, 5);

//  将元素插入 LB
    insertNode(&amp;LB, 2);
    insertNode(&amp;LB, 4);
    insertNode(&amp;LB, 6);
    insertNode(&amp;LB, 7);

// LALB 合并到 LC
    struct Node\* LC=mergeLists(LA, LB);

//   输出 LC
    printf("合并完成的列表: ");
    printList(LC);

return 0;
}

2*、*用邻接矩阵法创建有向图，编写算法，实现图的深度优先遍历，输出遍历序列、领接矩阵。
</p>
<blockquote>
<p style="margin-left:.0001pt;text-align:justify;">
知识点回顾：图是比树更一般、更复杂的非线性数据结构。图结构中结点之间的关系是任意的，每个元素都可以和其他任何元素相关，即元素之间是多对多的关系。
<img alt="" height="417" src="https://i-blog.csdnimg.cn/blog_migrate/55deaa5dc1c59c45d83df868c61e64ea.png" width="1200"/>
</p>
<p style="margin-left:.0001pt;text-align:justify;">
<span style="background-color:#ffffff;">
<span style="color:#4d4d4d;">
有向图：一个有向的、无环的图，其中顶点（节点）之间有边相连。有向图可以用邻接矩阵来表示，其中矩阵的行和列分别表示顶点。在邻接矩阵中，如果两个顶点之间有边相连，则在相应的矩阵元素中存储 1，否则为 0。
</span>
</span>
</p>
<p style="margin-left:.0001pt;text-align:justify;">
</p>
<p style="margin-left:.0001pt;text-align:justify;">
</p>
<p style="margin-left:.0001pt;text-align:justify;">
<span style="background-color:#ffffff;">
<span style="color:#4d4d4d;">
领接矩阵：采用两个数组表示图，一个用于存储顶点信息的一维数组；另一个用于存储图中顶点之间的关联关系的二维数组。
</span>
</span>
</p>
<p style="margin-left:.0001pt;text-align:justify;">
</p>

<pre><code class="language-cpp">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
//  图结构定义:顶点数组、边数组和访问标记数组
typedef struct gr {//图
    int ver[4]; //  顶点数组
    int edge[4][3]; //  边数组
    int vertex; //  顶点数
    int ed; //  边数
} gr;
//  初始化图结构
void init_graph(gr* g) {
    g-&gt;vertex = 0;
    g-&gt;ed = 0;
    for (int i = 0; i &lt; 4; i++) {
        g-&gt;ver[i] = 0;
        for (int j = 0; j &lt; 3; j++) {
            g-&gt;edge[i][j] = 0;
        }
    }
}
//  添加顶点
void add_vertex(gr* g, int v) {
    g-&gt;ver[v] = 1;
    g-&gt;vertex++;
}
//  添加边
void add_edge(gr* g, int u, int v) {
    g-&gt;edge[u][v] = 1;
    g-&gt;edge[v][u] = 1;
    g-&gt;ed++;
}
//  深度优先遍历
void dfs(gr* g, int v, int visited[], int\* count) {//dfs 用于遍历给定图结构中的所有顶点。 
    visited[v] = 1; //  标记顶点 v 为已访问
    printf("%d ", v); //  输出访问过的顶点 v
    for (int i = 0; i &lt; g-&gt;vertex; i++) {
        if (g-&gt;edge[v][i] == 1 &amp;&amp; visited[i] == 0) { //  如果顶点 i 与顶点 v 相邻且未访问
            dfs(g, i, visited, count); //  递归调用 dfs 函数，以访问顶点 i
        }
    }
}

int main() {
    gr g;
    init_graph(&amp;g);//  初始化图结构
    int vertex, ed;//  顶点数和边数
    printf("请输入顶点数和边数：");
    scanf("%d%d", &amp;vertex, &amp;ed);//  输入顶点数和边数
    for (int i = 0; i &lt; vertex; i++) {
        int v;
        printf("请输入第%d 个顶点：", i + 1);
        scanf("%d", &amp;v); //  输入顶点 u 和顶点 v
        add_vertex(&amp;g, v);//  添加边(u, v)和(v, u)
    }
    for (int i = 0; i &lt; ed; i++) {
        int u, v;
        printf("请输入第%d 条边的两个顶点（用空格隔开）：", i + 1);
        scanf("%d%d", &amp;u, &amp;v);
        add_edge(&amp;g, u, v);
    }
    int visited[4];//  访问标记数组
    for (int i = 0; i &lt; 4; i++) {
        visited[i] = 0;
    }
    int count = 0;//  访问过的顶点数
    dfs(&amp;g, 0, visited, &amp;count);//  从顶点 0 开始深度优先遍历图结构
    printf("\n 访问过的顶点数为：%d\n", count);
    return 0;
}</code></pre>

<blockquote>
<p style="margin-left:.0001pt;text-align:justify;">
知识点：折半查找法
</p>
<p style="margin-left:.0001pt;text-align:justify;">
别名：二分查找
</p>
<p style="margin-left:.0001pt;text-align:justify;">
要求线性表是有序表。按关键字有序排序
</p>
<p style="margin-left:.0001pt;text-align:justify;">
基本过程：中间记录关键字与查找关键字比较；如果两者相同，则查找成功；
</p>
<p style="margin-left:.0001pt;text-align:justify;">
如果查找关键字小于中间位置关键字，则查前部子表，否则查后部子表。
</p>

int main() {

        printf("元素  %d  不在列表中\n", k2);
    }
    return 0;
}

int BinSrch(RecordList l, KeyType k) {
    int low=1;
    int high=l.length;
    while (low&lt;=high) {
        int mid=(low+high) / 2;
        if (k == l.r[mid].key) {
            return(mid); //  找到待查元素
        } else if (k &lt; l.r[mid].key) {
            high=mid-1; //  未找到,则继续在前半区间进行查找
        } else {
            low= mid+1; //  继续在后半区间进行查找
        }
    }
    return(0);
}

<p style="margin-left:.0001pt;text-align:justify;">
Dijkstra(迪杰斯特拉)算法是典型的单源最短路径算法（一种贪心算法），用于计算一个节点到其他所有节点的最短路径。主要特点是以起始点为中心向外层层扩展，直到扩展到终点为止。Dijkstra 算法是很有代表性的最短路径算法，注意该算法要求图中不存在负权边。

#include &lt;limits.h&gt;
#define V 5 //  图中的顶点数

//定义函数来查找具有最小距离值的顶点
int mind(int dist[], int sptSet[]) {//最小距离 mind，dist[]存储每个节点的最短路径（权重），sptSet[]存储每个节点的父节点
    int min = INT_MAX, min_index;//定义的常量 INT_MAX 表示整数类型的最大值，min_index 索引最小值

//寻找图中未被访问过的节点中距离最短的节点，找到下一个要添加到树中的边所连接的节点
    for (int v = 0; v &lt; V; v++) {//遍历所有节点，检查它们是否被访问过
        if (sptSet[v] == 0 &amp;&amp; dist[v] &lt;= min) {//寻找图中未访问过的节点中，距离起点最近的节点。
            min = dist[v];                // （sptSet[v] == 0）。如果节点未被访问过，并且其距离起点（dist[v]）小于等于当前找到的最小距离（min），
            min_index = v;               //那么就更新最小距离（min）和最小距离对应的节点索引（min_index
        }
    }

return min_index;
}

//  打印构造的距离数组的函数

    printf("终点         路径长度\n");

        printf("%d \t\t %d\n", i, dist[i]);
}

//  实现图的 Dijkstra 单源最短路径算法的函数
void dijkstra(int g[V][V], int src) {//g 为图
    int dist[V]; //  输出数组。dist【i】保持从 src 到 i 的最短距离

int sptSet[V]; 
    //  确定路径树或从 src 到 i 的最短距离

//  将所有距离初始化为无穷大，并将 stpSet【】初始化为 0.//  如果最短路径中包含顶点 i，则 sptSet【i】为真
    for (int i = 0; i &lt; V; i++) { //  用于寻找一个有权重图中的最短路径。
        dist[i] = INT_MAX;                       
        sptSet[i] = 0;
    }
        //  源顶点与自身的距离始终为 0
    dist[src] = 0;

//  求所有顶点的最短路径
//用一个循环来迭代 V-1 次，每次迭代都会选择一个未被处理过的节点 u，将其标记为已处理，并更新其相邻节点的距离.
    for (int count = 0; count &lt; V - 1; count++) {
        int u = mind(dist, sptSet);

//  将拾取的顶点标记为已处理
        sptSet[u] = 1;

//  更新选中顶点的相邻顶点的分布值，确保找到的是最短路径
        for (int v = 0; v &lt; V; v++) {//遍历
            if (!sptSet[v] &amp;&amp; g[u][v] &amp;&amp; dist[u] != INT_MAX &amp;&amp;//检查它们是否与选定顶点之间的权重边更短，
                dist[u] + g[u][v] &lt; dist[v]) {         //如果是，就更新 dist【v】的值，
                dist[v] = dist[u] + g[u][v];        //让它等于 dist[u] + g[u][v]
            }
        }
    }
    printSolution(dist, V);//  打印构造的距离数组
}

    int g[V][V] = {<!-- -->{0, 1, 0, 0, 0},//图是一个 5x5 的矩阵

        {0, 0, 4, 0, 3},
        {0, 5, 6, 3, 0}};

dijkstra(g, 0); //寻找从顶点 0 开始的最短路径
    return 0;
}
