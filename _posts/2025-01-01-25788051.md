    <p>
     <span style="color:#009900;">
      <strong>
       --说明：本次创建的是Bank_ATM数据库（
      </strong>
     </span>
     <span style="color:#f33b45;">
      <strong>
       SQLserver数据库语法
      </strong>
     </span>
     <span style="color:#009900;">
      <strong>
       ）
      </strong>
     </span>
    </p>
    <p>
     <span style="color:#009900;">
      <strong>
       --1、创建数据库前首先判断是否已存在本数据库，然后删除已有数据库(若原有数据库没有大用).
      </strong>
     </span>
    </p>
    <p>
     <span style="color:#009900;">
      <strong>
       --2、其次依次创建主数据库文件(*.mdf)、次要文件(*.ndf)和日志文件(*.ldf).
      </strong>
     </span>
    </p>
    <p>
    </p>
    <p>
    </p>
    <p>
     <strong>
      <span style="color:#009900;">
       --使用master数据库
      </span>
     </strong>
    </p>
    <p>
    </p>
    <p>
     <strong>
      use master
      <br/>
      go
      <br/>
      <span style="color:#009900;">
       --判断是否存在Bank_ATM
      </span>
      <br/>
      if exists (select * from sysdatabases where name='Bank_ATM')
      <br/>
      drop database Bank_ATM
      <br/>
      <br/>
      <br/>
      <span style="color:#009900;">
       --创建BANK_ATM数据库
      </span>
      <br/>
      create database Bank_ATM
      <br/>
      ON PRIMARY
      <br/>
      (
      <br/>
      <span style="color:#009900;">
       --主数据库文件的描述
      </span>
      <br/>
      NAME ='BANK_ATM1',
      <br/>
      FILENAME ='E:\Project\BANK_ATM1.MDF',
      <br/>
      SIZE = 10,
      <br/>
      FILEGROWTH =10%
      <br/>
      ),
      <br/>
      (
      <br/>
      <span style="color:#009900;">
       --创建次要数据文件
      </span>
      <br/>
      NAME ='BANK_ATM2',
      <br/>
      FILENAME ='E:\Project\BANK_ATM2.ndf',
      <br/>
      SIZE = 5,
      <br/>
      MAXSIZE = 100,
      <br/>
      FILEGROWTH =1
      <br/>
      )
      <br/>
      LOG ON
      <br/>
      (
      <br/>
      <span style="color:#009900;">
       --创建日志文件
      </span>
      <br/>
      NAME ='BANK_ATM3',
      <br/>
      FILENAME ='E:\Project\BANK_ATM3.ldf',
      <br/>
      SIZE = 10,
      <br/>
      MAXSIZE = 50,
      <br/>
      FILEGROWTH =1
      <br/>
      )
      <br/>
      GO
     </strong>
    </p>
