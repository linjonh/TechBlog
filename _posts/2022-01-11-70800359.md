    <p>
     <strong>
      二、调用的简单实现
     </strong>
    </p>
    <p>
     <strong>
      <span style="white-space:pre">
      </span>
      2.1 调用静态方法
     </strong>
    </p>
    <p>
     <span style="white-space:pre">
     </span>
     实现的功能：在java类中实现一个静态方法，这个方法向终端打印一个字符串，这个字符串是作为参数传入进来的。
    </p>
    <p>
     <span style="white-space:pre">
     </span>
     java代码的实现如下：
    </p>
    <p>
    </p>
    <pre><code class="language-java">/* define a class for JNI */
public class JNIDemo {

	/* Implement a static method */
	public static void printHello(String str){
		System.out.println(str);
	}
	
}</code></pre>
    <span style="white-space:pre">
    </span>
    C代码的实现：
    <p>
    </p>
    <p>
    </p>
    <pre><code class="language-cpp">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;jni.h&gt;	/* 位于/usr/lib/jvm/java-1.7.0-openjdk-amd64/include */

/* 这个函数用于创建java虚拟机
 *	jvm : java虚拟机
 *	env	: JNI运行环境
 */
jint create_VM(JavaVM **jvm, JNIEnv** env) {  
    JavaVMInitArgs args;  
    JavaVMOption options[1]; 
	
    args.version = JNI_VERSION_1_6;  //	java版本
    args.nOptions = 1;  
    options[0].optionString = "-Djava.class.path=./";  	// 配置java类的目录为当前目录
    args.options = options;  
    args.ignoreUnrecognized = JNI_FALSE;  
	
    return JNI_CreateJavaVM(jvm, (void **)env, &amp;args);  // 创建java虚拟机
}  


/* 程序的入口函数 */
int main(int argc, char *argv[])
{
	jint ret = 0;
	
	JavaVM *jvm;
	JNIEnv *env;

	jclass cls;
	jmethodID mid;

	jstring jstr;

	/* 1、创建java虚拟机 */
	ret = create_VM(&amp;jvm, &amp;env);
	if(ret &lt; 0)
	{
		printf("create_VM error!\n");
		return -1;
	}

	/* 2、获得class，要调用的java类 */
	cls = (*env)-&gt;FindClass(env, "JNIDemo");
	if(cls == NULL)
	{
		printf("FindClass error!\n");
		return -1;
	}

	/* 3、实例化java方法 */

	/* 4、调用java方法 */
	jstr = (*env)-&gt;NewStringUTF(env, "Hello,world!");
	mid = (*env)-&gt;GetStaticMethodID(env, cls, "printHello", "(Ljava/lang/String;)V");	// 获取方法的id
	(*env)-&gt;CallStaticVoidMethod(env, cls, mid, jstr);	// 调用方法

	return 0;
}
</code></pre>

LD_LIBRARY_PATH=/usr/lib/jvm/java-7-openjdk-amd64/jre/lib/amd64/server ./c_demo</code></pre>

      <strong>
       <span style="white-space:pre">
       </span>
       2.2 调用非静态方法
      </strong>
     </p>
     <p>
      <span style="white-space:pre">
      </span>
      这个例子实现的功能是：C调用java实现的方法向终端打印一个字符串并接收java方法的返回值，把这个返回值也通过终端打印出来。
     </p>
     <p>
      <span style="white-space:pre">
      </span>
      java代码的实现如下：
     </p>
     <p>
     </p>
     <pre><code class="language-java">/* define a class for JNI */
public class JNIDemo {

	/* Implement a method */
	public int printHello(String str){
		System.out.println(str);
		return 128;
	}
	
}</code></pre>
     <span style="white-space:pre">
     </span>
     C代码的实现如下：
     <br/>
     <pre><code class="language-cpp">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;jni.h&gt;	/* 位于/usr/lib/jvm/java-1.7.0-openjdk-amd64/include */

/* 这个函数用于创建java虚拟机
 *	jvm : java虚拟机
 *	env	: JNI运行环境
 */
jint create_VM(JavaVM **jvm, JNIEnv** env) {  
    JavaVMInitArgs args;  
    JavaVMOption options[1]; 
	
    args.version = JNI_VERSION_1_6;  //	java版本
    args.nOptions = 1;  
    options[0].optionString = "-Djava.class.path=./";  	// 配置java类的目录为当前目录
    args.options = options;  
    args.ignoreUnrecognized = JNI_FALSE;  
	
    return JNI_CreateJavaVM(jvm, (void **)env, &amp;args);  // 创建java虚拟机
}  


/* 程序的入口函数 */
int main(int argc, char *argv[])
{
	jint ret = 0;
	
	JavaVM *jvm;
	JNIEnv *env;

	jclass cls;
	jmethodID cid;

	jobject obj;
	jmethodID mid;

	jstring jstr;

	/* 1、创建java虚拟机 */
	ret = create_VM(&amp;jvm, &amp;env);
	if(ret &lt; 0)
	{
		printf("create_VM error!\n");
		return -1;
	}

	/* 2、获得class，要调用的java类 */
	cls = (*env)-&gt;FindClass(env, "JNIDemo");
	if(cls == NULL)
	{
		printf("FindClass error!\n");
		return -1;
	}

	/* 3、实例化java方法 */
	cid = (*env)-&gt;GetMethodID(env, cls, "&lt;init&gt;", "()V");	// 获取JNIDemo类的构造方法
	if(cid == NULL)
	{
		printf("GetMethodID for constructor!\n");
		return -1;
	}

	obj = (*env)-&gt;NewObject(env, cls, cid);		// 构造一个JNIDemo类的实例
	if(obj == NULL)
	{
		printf("NewObject error!\n");
		return -1;
	}

	/* 4、调用java方法 */
	jstr = (*env)-&gt;NewStringUTF(env, "Hello,world!");

	mid = (*env)-&gt;GetMethodID(env, cls, "printHello", "(Ljava/lang/String;)I");	// 获得要调用的java方法
	ret = (*env)-&gt;CallIntMethod(env, obj, mid, jstr);	// 调用这个java方法
	printf("The return value is %d\n", ret);	// 打印java方法的返回值

	return 0;
}

       <strong>
        <span style="white-space:pre">
        </span>
        2.3 设置java类的成员属性
       </strong>
      </p>
      <p>
       <span style="white-space:pre">
       </span>
       这个liz实现的功能：在java类中实现两个属性，在实现一个打印属性的方法，C代码调用这个类，设置它的两个属性，并把属性值打印出来。
      </p>
      <p>
       <span style="white-space:pre">
       </span>
       java代码的实现如下：
      </p>
      <p>
      </p>
      <pre><code class="language-java">/* define a class for JNI */
public class JNIDemo {

	/* Define two member variables */
	private int age = 0;
	private String name = null;

	/* Implement a method */
	public void printInfo(){
		System.out.println("age  :  " + this.age);
		System.out.println("name :  " + this.name);
	}
	
}</code></pre>
      <span style="white-space:pre">
      </span>
      C代码的实现如下：
      <p>
      </p>
      <p>
       <span style="white-space:pre">
       </span>
      </p>
      <pre><code class="language-cpp">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;jni.h&gt;	/* 位于/usr/lib/jvm/java-1.7.0-openjdk-amd64/include */

/* 这个函数用于创建java虚拟机
 *	jvm : java虚拟机
 *	env	: JNI运行环境
 */
jint create_VM(JavaVM **jvm, JNIEnv** env) {  
    JavaVMInitArgs args;  
    JavaVMOption options[1]; 
	
    args.version = JNI_VERSION_1_6;  //	java版本
    args.nOptions = 1;  
    options[0].optionString = "-Djava.class.path=./";  	// 配置java类的目录为当前目录
    args.options = options;  
    args.ignoreUnrecognized = JNI_FALSE;  
	
    return JNI_CreateJavaVM(jvm, (void **)env, &amp;args);  // 创建java虚拟机
}  


/* 程序的入口函数 */
int main(int argc, char *argv[])
{
	jint ret = 0;
	
	JavaVM *jvm;
	JNIEnv *env;

	jclass cls;
	jmethodID cid;

	jobject obj;
	jmethodID mid;

	jfieldID fid_name;
	jfieldID fid_age;

	jstring jstr;

	/* 1、创建java虚拟机 */
	ret = create_VM(&amp;jvm, &amp;env);
	if(ret &lt; 0)
	{
		printf("create_VM error!\n");
		return -1;
	}

	/* 2、获得class，要调用的java类 */
	cls = (*env)-&gt;FindClass(env, "JNIDemo");
	if(cls == NULL)
	{
		printf("FindClass error!\n");
		return -1;
	}

	/* 3、实例化java方法 */
	cid = (*env)-&gt;GetMethodID(env, cls, "&lt;init&gt;", "()V");	// 获取JNIDemo类的构造方法
	if(cid == NULL)
	{
		printf("GetMethodID for constructor!\n");
		return -1;
	}

	obj = (*env)-&gt;NewObject(env, cls, cid);		// 构造一个JNIDemo类的实例
	if(obj == NULL)
	{
		printf("NewObject error!\n");
		return -1;
	}

	/* 4、设置类的属性 */
	fid_name = (*env)-&gt;GetFieldID(env, cls, "name", "Ljava/lang/String;");	// 获得JNIDemo类的name成员
	if(fid_name == NULL)
	{
		printf("GetFieldID for name is failed!\n");
		return -1;
	}

	fid_age = (*env)-&gt;GetFieldID(env, cls, "age", "I");		// 获得JNIDemo类的age成员
	if(fid_age == NULL)
	{
		printf("GetFieldID for age is failed!\n");
		return -1;
	}

	jstr = (*env)-&gt;NewStringUTF(env, "TECH-PRO");	// 初始化字符串的值

	(*env)-&gt;SetObjectField(env, obj, fid_name, jstr);	// 设置name成员为jstr
	(*env)-&gt;SetIntField(env, obj, fid_age, 25);		// 设置age成员为25

	/* 5、调用java方法 */
	mid = (*env)-&gt;GetMethodID(env, cls, "printInfo", "()V");	// 获得要调用的java方法
	(*env)-&gt;CallVoidMethod(env, obj, mid);	// 调用这个java方法

	return 0;
