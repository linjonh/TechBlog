    <blockquote>
     <p>
      &lt;template&gt;
     </p>
     <p>
      &lt;a-select
     </p>
     <p>
      v-model="value"
     </p>
     <p>
      :placeholder="placeholder"

      :filter-option="false"
     </p>
     <p>
      @search="handleSearch"
     </p>
     <p>
      &gt;
     </p>
     <p>
      &lt;div slot="dropdownRender" slot-scope="menu"&gt;

     <p>

     </p>
     <p>

     </p>
     <p>
      &lt;div @mousedown="e =&gt; e.preventDefault()" @click="changePage(0)" style="cursor:pointer"&gt;
     </p>
     <p>
      上一页
     </p>
     <p>
      &lt;/div&gt;
     </p>
     <p>
      &lt;a-spin :spinning="spinning" class="my-spin" @mousedown="e =&gt; e.preventDefault()"/&gt;
     </p>
     <p>
      &lt;div v-if="total &amp;&amp; pageSize &amp;&amp; page"&gt;{
      <!-- -->
      { nowPage }}/{
      <!-- -->
      { maxPage }}&lt;/div&gt;
     </p>
     <p>
      &lt;div @mousedown="e =&gt; e.preventDefault()" @click="changePage(1)" style="cursor:pointer"&gt;
     </p>
     <p>
      下一页
     </p>
     <p>
      &lt;/div&gt;
     </p>
     <p>
      &lt;/div&gt;
     </p>
     <p>
      &lt;/div&gt;
     </p>
     <p>
     </p>
     <p>
      &lt;a-select-option v-for="(item, index) in selectList" :key="index" :value="item[valueKey]"
     </p>
     <p>
      &gt;{
      <!-- -->
      { item[labelKey] }}
     </p>
     <p>
      &lt;/a-select-option&gt;
     </p>
     <p>
      &lt;/a-select&gt;
     </p>
     <p>
      &lt;/template&gt;
     </p>
     <p>
      &lt;script&gt;
     </p>
     <p>
      export default {
      <!-- -->
     </p>
     <p>
      name: 'selectPage',
     </p>
     <p>
      props: {
      <!-- -->
     </p>
     <p>
      // 下拉框总数据
     </p>
     <p>
      list: {
      <!-- -->
     </p>
     <p>
      type: Array,
     </p>
     <p>
      require: true
     </p>
     <p>
      },
     </p>
     <p>
      total: {
      <!-- -->
     </p>
     <p>
      type: Number,
     </p>
     <p>
      require: true
     </p>
     <p>
      },
     </p>
     <p>
      page: {
      <!-- -->
     </p>
     <p>
      type: Number,
     </p>
     <p>
      require: true
     </p>
     <p>
      },
     </p>
     <p>
      pageSize: {
      <!-- -->
     </p>
     <p>
      type: Number
     </p>
     <p>
      },
     </p>
     <p>
      // 提示
     </p>
     <p>
      placeholder: {
      <!-- -->
     </p>
     <p>
      type: String,
     </p>
     <p>
      default: '请选择'
     </p>
     <p>
      },
     </p>
     <p>
      // 下拉框显示值
     </p>
     <p>
      labelKey: {
      <!-- -->
     </p>
     <p>
      type: String
     </p>
     <p>
      },
     </p>
     <p>
      // 下拉框value
     </p>
     <p>
      valueKey: {
      <!-- -->
     </p>
     <p>
      type: String
     </p>
     <p>
      },
     </p>
     <p>
      spinning: {
      <!-- -->
     </p>
     <p>
      type: Boolean,
     </p>
     <p>
      default: false
     </p>
     <p>
      }
     </p>
     <p>
      // 默认值
     </p>
     <p>
      // value: {
      <!-- -->
     </p>
     <p>
      //   type: String
     </p>
     <p>
      // }
     </p>
     <p>
      },
     </p>
     <p>
      components: {
      <!-- -->
     </p>
     <p>
      VNodes: {
      <!-- -->
     </p>
     <p>
      functional: true,
     </p>
     <p>
      render: (h, ctx) =&gt; ctx.props.vnodes
     </p>
     <p>
      }
     </p>
     <p>
      },
     </p>
     <p>
      data() {
      <!-- -->
     </p>
     <p>
      return {
      <!-- -->
     </p>
     <p>
      value: undefined
     </p>
     <p>
      }
     </p>
     <p>
      },
     </p>
     <p>
      computed: {
      <!-- -->
     </p>
     <p>
      selectList() {
      <!-- -->
     </p>
     <p>
      return this.list
     </p>
     <p>
      },
     </p>
     <p>
      maxPage() {
      <!-- -->
     </p>
     <p>
      return Math.ceil(this.total / this.pageSize)
     </p>
     <p>
      },
     </p>
     <p>
      nowPage() {
      <!-- -->
     </p>
     <p>
      return this.page
     </p>
     <p>
      }
     </p>
     <p>
      },
     </p>
     <p>
      methods: {
      <!-- -->
     </p>
     <p>
      // 清空数据
     </p>
     <p>
      clearValue() {
      <!-- -->
     </p>
     <p>
      this.value = ''
     </p>
     <p>
      },
     </p>
     <p>
      // 换页事件，0切换到上一页，1切换到下一页
     </p>
     <p>
      changePage(status) {
      <!-- -->
     </p>
     <p>
      this.$emit('changePage', status)
     </p>
     <p>
      },
     </p>
     <p>
      // 选择下拉框时进行父传子
     </p>
     <p>
      chooseOption(value) {
      <!-- -->
     </p>
     <p>
      this.$emit('change', value)
     </p>
     <p>
      },
     </p>
     <p>
      handleSearch(value) {
      <!-- -->
     </p>
     <p>
      this.$emit('handleSearch', value)
     </p>
     <p>
      }
     </p>
     <p>
      }
     </p>
     <p>
      }
     </p>
