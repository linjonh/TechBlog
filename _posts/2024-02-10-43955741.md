    测试代码：
    <p>
    </p>
    <pre><code class="language-java">/**
 * 测试用Map接收参数
 * @author Xu
 *
 */
@RestController
@RequestMapping(value = "/mapParameter/")
public class MapParameter {
	
	/**
	 * 用Map接收前端提交的Form Data或Query String，如果有相同的参数名，只接收第一个
	 * @author XuJijun
	 * @param params
	 * @return
	 */
	@RequestMapping(value = "map")
	public Map&lt;String, Object&gt; test1(@RequestParam Map&lt;String, Object&gt; params) {
		Map&lt;String, Object&gt; resultMap = new HashMap&lt;String, Object&gt;();
		resultMap = params;
		return resultMap;
	}
	
	/**
	 * 用MultiValueMap接收前端提交的Form Data或Query String，可以接收相同的参数名的值到同一个list中
	 * @author XuJijun
	 * @param params
	 * @return
	 */
	@RequestMapping(value = "multiValueMap")
	public Map&lt;String, List&lt;Object&gt;&gt; test2(@RequestParam MultiValueMap&lt;String, Object&gt; params) {
		Map&lt;String, List&lt;Object&gt;&gt; resultMap = new HashMap&lt;&gt;();
		resultMap = params;
		return resultMap;
	}
	
	/**
	 * 用Map接收前端提交的json格式的Request Payload，如果有相同的参数名，只接收最后一个
	 * @author XuJijun
	 * @param params
	 * @return
	 */
	@RequestMapping(value = "jsonParams")
	public Map&lt;String, Object&gt; test3(@RequestBody Map&lt;String, Object&gt; params) {
		Map&lt;String, Object&gt; resultMap = new HashMap&lt;String, Object&gt;();
		resultMap = params;
		return resultMap;
	}
}
</code></pre>
