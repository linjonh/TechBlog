     <strong>
      小程序滚动条，滚到最底部解决方案1
     </strong>
    </p>
    <p>
     小程序滚动到底部使用Scorll-view实现方案:
    </p>
    <p>
     scrill-view：组件要固定高度。
    </p>
    <p>
     scroll-y:true
    </p>
    <p>
     scroll-top="{
     <!-- -->
     {scrollTop}}"
    </p>
    <p>
     scroll-with-animation="true"
    </p>
    <p>
     <strong>
      每次加载新的数据修改一下scroll-top的值即可。
     </strong>
    </p>
    <p>
     css代码：
    </p>
    <pre><code class="language-css">.list {
  border: 1px solid red;
  padding: 10px;
}
.list scroll-view {
  height: 350px;
}

.list .item {
  padding: 20px 10px;
  margin: 10px 0px;
  background: rgb(224, 224, 224);
}</code></pre>
    <p>
     wxml 代码：
    </p>
    <pre><code class="language-html">&lt;view class="list"&gt;
  &lt;scroll-view scroll-y="true" scroll-top="{<!-- -->{scrollTop}}" scroll-with-animation="true"&gt;
    &lt;view class="item" wx:for="{<!-- -->{list}}"&gt;{<!-- -->{item}}&lt;/view&gt;
  &lt;/scroll-view&gt;
&lt;/view&gt;</code></pre>
    <p>
     js：代码
    </p>
    <pre><code class="language-javascript">  /**
   * 页面的初始数据
   */
  data: {
    list: ['张三丰', '李四', '王五', '赵六'],
    scrollTop: 0
  },

  /**
   * 生命周期函数--监听页面加载
   */
  onLoad: function (options) {
    var _this = this;
    var index=0;
    setInterval(() =&gt; {
      index++;
      var list = _this.data.list;
      list.push('ceshi1');
      list.push('ceshi2');
      list.push('ceshi3');
      list.push('**********'+index);
      var top = _this.data.scrollTop + list.length * 60;
      console.info(top);
      _this.setData({
        list: list,
        scrollTop: top
      });
    }, 2000);
  },
</code></pre>
