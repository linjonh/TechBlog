     <span style="font-family: AR PL UMing CN,serif;">

      <span style="font-size: medium;">
       Linux
      </span>

     <span style="font-size: medium;">

     <span style="font-family: AR PL UMing CN,serif;">
      <span style="font-size: medium;">
