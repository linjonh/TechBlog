    <p>
     ① npm 下载依赖
    </p>
    <pre><code class="language-javascript">npm install vue-code-highlight --save</code></pre>
    <p>
     ②有两个种方式可以实现：
    </p>
    <p>
     <span style="color:#fe2c24;">
      <strong>
       第一种
      </strong>
     </span>
     ：
    </p>
    <p>
     在需要代码块高亮的vue页面引入：
    </p>
    <pre><code class="language-javascript">&lt;template&gt;
  &lt;div&gt;
    &lt;vue-code-highlight language="javascript"&gt;
      modifyUserGroup(data) {
        return fetch({
          url: "/mth-finance-capital-service/api/v1/userGroup/bcUsergroupUpdate",
          method: "POST +++++++++++++++",
          data
        });
      },
    &lt;/vue-code-highlight&gt;
  &lt;/div&gt;
&lt;/template&gt;

&lt;script&gt;
import "vue-code-highlight/themes/prism-tomorrow.css" // 直接引入即可
import "vue-code-highlight/themes/window.css"; // 同上
import { component as VueCodeHighlight } from 'vue-code-highlight'  // 引入
export default {
  components:{ VueCodeHighlight }, // 在模板里也要引入
  data(){
    return {

    }
  }
}
&lt;/script&gt;</code></pre>
    <p>
     <span style="color:#0d0016;">
      <strong>
       效果

    <p>
     <span style="color:#fe2c24;">
      <strong>
       第二种：
      </strong>
     </span>
    </p>
    <pre><code class="language-javascript">&lt;style lang="less" scoped&gt;

&lt;/style&gt;
&lt;template&gt;
  &lt;div&gt;
    &lt;div v-highlight &gt;
      &lt;pre class="language-javascript"&gt;
        &lt;code&gt;
          modifyUserGroup(data) {
            return fetch({
              url: "/mth-finance-capital-service/api/v1/userGroup/bcUsergroupUpdate",
              method: "POST",
              data
            });
          },
        &lt;/code&gt;
      &lt;/pre&gt;
    &lt;/div&gt;
  &lt;/div&gt;

&lt;/template&gt;

&lt;script&gt;
import "vue-code-highlight/themes/prism-tomorrow.css"; 
import "vue-code-highlight/themes/window.css";

export default {
  data(){
    return {

    }
  }
}
&lt;/script&gt;</code></pre>
