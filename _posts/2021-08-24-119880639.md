    <pre><code>@Test
public void test() {
    Map&lt;String, String&gt; map = new HashMap&lt;&gt;();
    map.put("k1", "k1");
    map.put("k2", "k2");

    System.out.println(map); // {k1=k1, k2=k2}

    if (map.containsKey("k1")) {
        map.remove("k1");
    }

    System.out.println(map); // {k2=k2}
}
</code></pre>
    <ul>
     <li>
      迭代器
     </li>
    </ul>
    <pre><code>@Test
public void test1() {
    Map&lt;String, String&gt; map = new HashMap&lt;&gt;();
    map.put("k1", "k1");
    map.put("k2", "k2");

    System.out.println(map); // {k1=k1, k2=k2}

    Iterator iter = map.keySet().iterator();
    while (iter.hasNext()) {
        String key = (String) iter.next();
        if ("k1".equals(key)) {
            iter.remove();
        }
    }

    System.out.println(map); // {k2=k2}
}
</code></pre>
