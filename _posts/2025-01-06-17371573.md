bool CScene::InitScene()
{
this-&gt;m_imgStt.Load(L"res\\bgStart.jpg");
this-&gt;m_imgNxt.Load(L"res\\bgNext.jpg");

    //如果加载失败, 返回false

    	this-&gt;m_imgNxt.IsNull())
    {
    	return false;
    }

    //开始为真, 背景起始坐标为0
    this-&gt;m_isStart = true;
    this-&gt;m_bgX = 0;

    //播放背景音乐

    mciSendString(L"play bgm repeat", NULL, 0, NULL);

//绘制场景
void CScene::StickScene(CDC &amp;bufferDC, CRect rClient)
{
//设置缩放图片的模式为:COLORONCOLOR, 以消除像素重叠
bufferDC.SetStretchBltMode(COLORONCOLOR);

    //如果到了左边界, 回到起点
    if(m_bgX &lt;= -rClient.Width())
    {
    	m_bgX = 0;

    	if(m_isStart)
    		m_isStart = false;
    }

    //客户区宽度
    int cltWth = rClient.Width();

    rClient.right = cltWth + m_bgX;
    rClient.left = m_bgX;

    //如果是开始就绘制起始背景
    if(m_isStart)
    {
    	this-&gt;m_imgStt.StretchBlt(bufferDC, rClient, SRCCOPY);
    }
    //将下一张背景作为起始背景
    else
    {
    	this-&gt;m_imgNxt.StretchBlt(bufferDC, rClient, SRCCOPY);
    }

    //绘制下一张背景
    rClient.left += cltWth;
    rClient.right += cltWth;
    m_imgNxt.StretchBlt(bufferDC, rClient, SRCCOPY);

}

//移动背景
