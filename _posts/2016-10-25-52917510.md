    <h2 id="%E4%B8%80%EF%BC%8C%E6%88%98%E8%88%B0">
     一，战舰
    </h2>
    <p>
     <a href="https://cn.puzzle-battleships.com/" rel="nofollow" title="在线play">
      在线play
     </a>
    </p>
    <p>
     找出藏在格子里的战船的位置，有些战船已经部分被揭示。
     <br/>
     战船是一条连续不断的由黑方块组成的直线，各种大小的战船的数量显示在格子下方的提示里。
     <br/>
     两条战船不能相邻（包括斜向），格子旁边的数字显示该行/列中被战船占据的格子数量。
    </p>

    <h3 id="%E9%9A%90%E8%97%8F%E8%A7%84%E5%88%99">
     隐藏规则
    </h3>
    <p>
     长为1的战舰由1个圆组成，长大于1的战舰除了两边的格子中间的都是正方形。
    </p>

     每个一共是三种不同的格子，初始局面给出的格子是哪一种是明确的。

    <p>
     代码：
    </p>
    <pre><code class="language-cpp">#include&lt;iostream&gt;
using namespace std;

需要硬编码的 3 行///
const int m4 = 2, m3 = 2, m2 = 3, m1 = 3; //战舰的个数，m4=2 代表有 2 个长度为 4 的战舰
int line[nn] = { 2, 3, 2, 0, 5, 1, 0, 6, 2, 2 }; //列和
int row[nn] = { 1, 3, 1, 1, 4, 1, 5, 1, 2, 4 }; //行和
需要硬编码的 3 行///

const int nn = 10; //游戏的维度
int avail[nn][nn]; //能否放战舰，一个是战舰不相邻的问题，一个是游戏开始的障碍物问题
int out[nn][nn];

void save(int avail[nn][nn], int line[nn], int row[nn], int save_avail[nn][nn], int save_line[nn], int save_row[nn]) //保存 3 个数组
{
for (int i = 0; i &lt; nn; i++)
{
for (int j = 0; j &lt; nn; j++)save_avail[i][j] = avail[i][j];
save_line[i] = line[i];
save_row[i] = row[i];
}
}

void re(int save_avail[nn][nn], int save_line[nn], int save_row[nn], int avail[nn][nn], int line[nn], int row[nn]) //还原 3 个数组
{
for (int i = 0; i &lt; nn; i++)
{
for (int j = 0; j &lt; nn; j++)avail[i][j] = save_avail[i][j];
line[i] = save_line[i];
row[i] = save_row[i];
}
}

bool g(int n);

bool f(int n, int l) //n 是递归深度，也是放了多少个战舰，l 是战舰长度
{
for (int pr = 0; pr &lt; nn; pr++) //横着放
{
for (int pl = 0; pl + l &lt;= nn; pl++) //(pr,pl)是战舰的最左那个格子
{
bool f = false;
for (int k = pl; k &lt; pl + l; k++)if (avail[pr][k] == 0)f = true;
if (f)continue;

    		int save_avail[nn][nn];	//回溯的时候保存avail
    		int save_line[nn];
    		int save_row[nn];
    		save(avail, line, row, save_avail, save_line, save_row);

    		row[pr] -= l;
    		if (row[pr] &lt; 0)
    		{
    			re(save_avail, save_line, save_row, avail, line, row);		//还原
    			continue;
    		}

    		f = false;
    		for (int k = pl; k &lt; pl + l; k++)
    		{
    			line[k]--;
    			if (line[k] &lt; 0)f = true;
    		}
    		if (f)
    		{
    			re(save_avail, save_line, save_row, avail, line, row);
    			continue;
    		}

    		for (int i = pr - 1; i &lt;= pr + 1; i++)
    		{
    			if (i &lt; 0 || i &gt;= nn)continue;
    			for (int j = pl - 1; j &lt;= pl + l; j++)
    			{
    				if (j&lt;0 || j &gt;= nn)continue;
    				avail[i][j] = 0;
    			}
    		}
    		if (g(n + 1))
    		{
    			for (int k = pl; k &lt; pl + l; k++)out[pr][k] = 1;
    			return true;
    		}
    		else re(save_avail, save_line, save_row, avail, line, row);
    	}
    }
    for (int pr = 0; pr + l &lt;= nn; pr++)		//竖着放
    {
    	for (int pl = 0; pl &lt; nn; pl++)		//(pr,pl)是战舰的最上那个格子
    	{
    		bool f = false;
    		for (int k = pr; k &lt; pr + l; k++)if (avail[k][pl] == 0)f = true;
    		if (f)continue;

    		int save_avail[nn][nn];	//回溯的时候保存avail
    		int save_line[nn];
    		int save_row[nn];
    		save(avail, line, row, save_avail, save_line, save_row);

    		line[pl] -= l;
    		if (line[pl] &lt; 0)
    		{
    			re(save_avail, save_line, save_row, avail, line, row);
    			continue;
    		}

    		f = false;
    		for (int k = pr; k &lt; pr + l; k++)
    		{
    			row[k]--;
    			if (row[k] &lt; 0)f = true;
    		}
    		if (f)
    		{
    			re(save_avail, save_line, save_row, avail, line, row);
    			continue;
    		}

    		for (int i = pr - 1; i &lt;= pr + l; i++)
    		{
    			if (i &lt; 0 || i &gt;= nn)continue;
    			for (int j = pl - 1; j &lt;= pl + 1; j++)
    			{
    				if (j&lt;0 || j &gt;= nn)continue;
    				avail[i][j] = 0;
    			}
    		}
    		if (g(n + 1))
    		{
    			for (int k = pr; k &lt; pr + l; k++)out[k][pl] = 1;
    			return true;
    		}
    		else re(save_avail, save_line, save_row, avail, line, row);
    	}
    }
    return false;

}

bool g(int n) //控制调用顺序
{
if (n &lt;= m4)return f(n, 4);
else if (n &lt;= m4 + m3)return f(n, 3);
else if (n &lt;= m4 + m3 + m2)return f(n, 2);
else if (n &lt;= m4 + m3 + m2 + m1)return f(n, 1);
return true;
}

int main()
{
for (int i = 0; i &lt; nn; i++)
{
for (int j = 0; j &lt; nn; j++)
{
avail[i][j] = 1; //初始化，1 表示可以放，0 表示不可以
out[i][j] = 0;
}
}

    g(1);
    for (int i = 0; i &lt; nn; i++)
    {
    	for (int j = 0; j &lt; nn; j++)cout &lt;&lt; out[i][j] &lt;&lt; "  ";
    	cout &lt;&lt; endl;
    }
    cout &lt;&lt; "end";
    system("pause&gt;nul");
    return 0;

}</code></pre>
