function getDataForParam(url, params,doSuccess, doFail) {
  wx.request({
    url: app.globalData.req_url + url,
    data: params,
    header: {
      "content-type": "application/json;charset=UTF-8"
    },
    method: 'GET',
    success: function (res) {
      doSuccess(res.data);
    },
    fail: function () {
      doFail();
    },
  })
}</code></pre>
