#include&lt;stdio.h&gt;
#include&lt;math.h&gt;
#include&lt;time.h&gt;
#include&lt;stdlib.h&gt;
//#include&lt;cstddef&gt;   //是C++的头文件不能用在C中
#include&lt;stddef.h&gt;
void menu();//菜单函数
void play();//开始游戏
int main()
{
	int input = 0;
	do {
		menu();
		printf("请选择-&gt;");
		scanf("%d", &amp;input);
		switch (input) {
		case 1:
			play();
			break;
		case 2:
			printf("退出游戏");
			break;
		default:
			printf("你输入错了,请重新输入");
			break;

		}
	} while (input);
	return 0;
}
void menu() {
	printf("**************猜字游戏********************\n");
	printf("*****1.开始游戏***********2.退出游戏*******\n");
	printf("******************************************\n");
}
void play() {
	//1.生成1-100随机数
	int ret = 0, guess = 0;
	//避免第二次玩的时候生成与第一次生成相同的数字
	//通过时间戳来设置随机数的生成起点
	srand((unsigned int)time(NULL));
	ret = rand() % 100 + 1;//取模最大为99，因此要+1
	//printf("%d", ret);
	//2.猜数字
	while (1) {
		printf("请猜数字\n");
		scanf("%d", &amp;guess);
		if (guess &gt; ret)
			printf("你猜大了\n");
		else if (guess &lt; ret)
			printf("你猜小了\n");
		else
		{
			printf("Congratulations!你猜正确咯\n");
			break;
		}
	}
}
