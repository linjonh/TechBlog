    <h2>

     搭建先电云平台
    </h2>
    <p>
     一、需求
     <br/>
     1、两台虚拟机------controller、compute
     <br/>
     2、centos与xiandian镜像
     <br/>
     3、一台内存8G的电脑
     <br/>
     4、CRT远程链接软件
     <br/>
     二、controller节点安装步骤
     <br/>
     #mkdir /opt/centos //创建centos文件夹
     <br/>
     #mkdir /opt/iaas //创建iaas文件夹
     <br/>
     #mount /dev/sr0 /opt/centos //将centos镜像挂载到centos文件夹
     <br/>
     #mount /dev/sr1 /opt/iaas //将xiandian镜像挂载到iaas文件夹
     <br/>
     #rm -rf /etc/yum.repo/* //清空yum文件夹
     <br/>
     #vi /etc/yum.repo/local.repo //创建yum源
     <br/>
     [centos]
     <br/>
     name=centos
     <br/>
     baseurl=file:///opt/centos
     <br/>
     gpgcheck=0
     <br/>
     [iaas]
     <br/>
     name=iaas
     <br/>
     baseurl=file:///opt/iaas/iaas-repo
     <br/>
     gpgcheck=0
     <br/>
     #yum clean all //清空yum缓存
     <br/>
     #yum repolist //验证yum源
     <br/>
     #yum install -y vsftpd //安装ftp服务器
     <br/>
     #echo “anon_root=/” &gt;&gt; /etc/vsftpd/vsftpd.conf //追加代码到vsftp
     <br/>
     #systemctl stop firewalld //关闭防火墙
     <br/>
     #setenforce 0 //关闭seliunx防火墙
     <br/>
     #systemctl restart vsftpd //重启ftp服务器
     <br/>
     #yum install -y iaas-xiandian 安装xiandian包
     <br/>
     #sed -i -e ‘s/PASS=/PASS=000000/’ -e ‘s/^#//’ /etc/xiandian/openrc.sh
     <a href="//xn--openrc-hh8ixet7jtx1d8fo.sh" rel="nofollow">
      //交互式修改openrc.sh
     </a>
     <br/>
     #vi /etc/xiandian/openrc.sh
     <a href="//xn--openrc-hg9py86h.xn--sh-zg3cw96f" rel="nofollow">
      //编辑openrc.sh文件
     </a>
     <br/>
     HOST_IP=192.168.100.10

     HOST_NAME=controller

    <p>
     三、compute节点安装步骤
     <br/>
     #mkdir /opt/centos //创建centos文件夹
     <br/>
     #mkdir /opt/iaas //创建iaas文件夹
     <br/>
     #mount /dev/sr0 /opt/centos //将centos镜像挂载到centos文件夹
     <br/>
     #mount /dev/sr1 /opt/iaas //将xiandian镜像挂载到iaas文件夹
     <br/>
     #rm -rf /etc/yum.repo/* //清空yum文件夹
     <br/>
     #vi /etc/yum.repo/local.repo //创建yum源
     <br/>
     [centos]
     <br/>
     name=centos
     <br/>
     baseurl=ftp://192.168.100.10/opt/centos
     <br/>
     gpgcheck=0
     <br/>
     [iaas]
     <br/>
     name=iaas
     <br/>
     baseurl=ftp://192.168.100.10/opt/iaas/iaas-repo
     <br/>
     gpgcheck=0
     <br/>
     #yum clean all //清空yum缓存
     <br/>
     #yum repolist //创建yum源
