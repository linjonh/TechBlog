    <h3>
     <a id="_2">
     </a>
     安装
    </h3>
    <pre><code class="prism language-bash"><span class="token function">sudo</span> <span class="token function">apt-get</span> <span class="token function">install</span> tinyproxy
</code></pre>
    <h3>
     <a id="_8">
     </a>
     配置文件位置
    </h3>
    <pre><code class="prism language-bash">/etc/tinyproxy.conf 
或者
/etc/tinyproxy/tinyproxy.conf 
</code></pre>
    <pre><code class="prism language-bash"><span class="token comment"># 预设是8888 Port, 可以更改</span>
Port <span class="token number">8888</span> 
<span class="token comment"># 将127.0.0.1改成你自己的IP</span>
Allow <span class="token number">127.0</span>.0.1 
<span class="token comment"># 想任何IP都可以连到Proxy在Allow前面打#注释</span>
<span class="token comment"># 或者设置成下面这样</span>
Allow <span class="token number">0.0</span>.0.0/0

<span class="token comment"># 可以设置授权，这样只有那些被授权的用户才能访问tinyproxy。要设置授权凭证，请到#BasicAuth用户密码这一行。取消这一行，并在这一行的最后写上您的密码。</span>
BasicAuth <span class="token punctuation">[</span>Username<span class="token punctuation">]</span> <span class="token punctuation">[</span>Password<span class="token punctuation">]</span>
</code></pre>
    <h3>

    <h3>
     <a id="_36">
     </a>
     运行停止状态重启
    </h3>
    <pre><code class="prism language-bash"><span class="token function">sudo</span> <span class="token function">service</span> tinyproxy start
<span class="token function">sudo</span> <span class="token function">service</span> tinyproxy stop
<span class="token function">sudo</span> <span class="token function">service</span> tinyproxy status
<span class="token function">sudo</span> <span class="token function">service</span> tinyproxy restart

<span class="token comment"># 设置开机启动</span>
systemctl <span class="token builtin class-name">enable</span> tinyproxy.service
</code></pre>
    <h3>
     <a id="_50">
     </a>
     连接测试
    </h3>
    <pre><code class="prism language-bash"><span class="token function">curl</span> -x <span class="token operator">&lt;</span>IP<span class="token operator">&gt;</span>:<span class="token operator">&lt;</span>PORT<span class="token operator">&gt;</span> www.baidu.com
<span class="token comment"># 用户名 + 密码</span>
<span class="token function">curl</span> -x http://user:123456@127.0.0.1:8888 www.baidu.com
<span class="token comment"># 有密码的话不加验证参数不会正常返回</span>
$ <span class="token function">curl</span> -x http://127.0.0.1:8888 www.baidu.com
Proxy Authentication Required
</code></pre>
