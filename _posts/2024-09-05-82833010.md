    <p>
     主要的原理，也是通过对微信小程序自带的方法进行一次二次的封装
     <br/>
     三种情况：
    </p>
    <ol>
     <li>
      0&lt;当前未读消息&lt;99，那么显示为多少多少+，
     </li>
     <li>
      当前未读消息&gt;99：显示为省略号
     </li>
     <li>
      当前未读消息== 0 ，取消角标的设置
     </li>
    </ol>
    <pre><code>  getUserNotReadNum(){
    this.xAjax({
      url: api.getUserNotReadNumUrl,
      callBack(res){
        if (99 &gt;= res.data.notReadNum &amp;&amp; res.data.notReadNum!=0){
          wx.setTabBarBadge({
            index: 1,
            text: res.data.notReadNum+''
          })
        } else if (99&lt;res.data.notReadNum){
          wx.setTabBarBadge({
            index: 1,
            text: '···'
          })
        } else if(res.data.notReadNum == 0) {
          wx.removeTabBarBadge({
            index: 1
          })
        }
        
      }
    })
  }
</code></pre>
