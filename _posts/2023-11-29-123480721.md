    <h4>
     数据库创建
    </h4>
    <p>
    </p>
    <pre><code class="language-sql">--创建数据库
--整个数据库的名字
CREATE DATABASE MYDb ON PRIMARY
(
 name='My_Db_Main',--主文件的引用名称
--  引用物理文件时所使用的名称
 filename='C:\My_Db\Main_My_Db.mdf',--包含路径的物理文件名称

 size=20mb,
 MAXSIZE=200mb,
 filegrowth=10mb
 
)

LOG ON
(
 name='My_Db_Log',--日志文件的引用名称
filename='C:\My_Db\Log_My_Db.ldf',--包含物理       路径的文件名称

 size=10mb,
 MAXSIZE=100mb,
 filegrowth=10mb
)</code></pre>
    <p>
    </p>
    <h4>
     数据库的修改
    </h4>
    <pre><code class="language-sql">--数据库的修改

--exec sp_helpdb MYDb;
--上面这条语句用于显示数据库的信息

ALTER DATABASE MYDb 
modify name= MYDB;
--修改数据库名字
--刷新数据库之后使用新名字，否则使用旧名字
ALTER DATABASE MYDB
modify file
(

  name='My_Db_Main',
--这个name是引用名称，用于定位旧文件，所以名称沿袭旧文件
filename='C:\My_Db\Main__My_Db.mdf',
--物理文件名是可以修改的，下次进入时会刷新。

 size=25mb,
--主文件的初始大小不能小于等于修改之前的主文件大小
 MAXSIZE=220mb,
 filegrowth=10mb
);
</code></pre>
