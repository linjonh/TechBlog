    <p>
     解决方法：
    </p>
    <ol>
     <li>
      在实体类的Date 属性加上两个必要的注解：
      <br/>
      @DateTimeFormat(pattern = “yyyy-MM-dd HH:mm:ss”)
      <br/>
      @JSONField(format = “yyyy-MM-dd HH:mm:ss”)
     </li>
     <li>
      在spring-mvc.xml中添加以下内容：
     </li>
    </ol>
    <pre><code>
&lt;mvc:annotation-driven&gt;
                 &lt;!-- 添加消息转换器 解决json数据传递过程中的乱码问题 --&gt;
                 &lt;mvc:message-converters&gt;
                     &lt;bean class="org.springframework.http.converter.StringHttpMessageConverter"&gt;
                         &lt;!-- 设置相应的属性 --&gt;
                         &lt;property name="supportedMediaTypes"&gt;
                             &lt;list&gt;
                                 &lt;value&gt;application/json;charset=UTF-8&lt;/value&gt;
                             &lt;/list&gt;
                        &lt;/property&gt;
                     &lt;/bean&gt;
                    &lt;!-- 日期格式的转化 --&gt;
                     &lt;bean
                         class="com.alibaba.fastjson.support.spring.FastJsonHttpMessageConverter"&gt;
                         &lt;property name="supportedMediaTypes"&gt;
                             &lt;list&gt;
                               &lt;value&gt;text/html;charset=UTF-8&lt;/value&gt;
                                &lt;value&gt;application/json&lt;/value&gt;
                            &lt;/list&gt;
                       &lt;/property&gt;
                         &lt;property name="features"&gt;
                             &lt;list&gt;
                                 &lt;!--WriteDateUseDateFormat这是fastjson默认的日期转换格式 --&gt;
                                &lt;value&gt;WriteDateUseDateFormat&lt;/value&gt;
                             &lt;/list&gt;
                         &lt;/property&gt;
                    &lt;/bean&gt;
                 &lt;/mvc:message-converters&gt;
       &lt;/mvc:annotation-driven&gt;



</code></pre>
