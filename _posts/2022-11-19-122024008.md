    <p>
     <strong>
      vs2019软件下运行代码如下：有颜色部分表示注释
     </strong>
    </p>
    <p>
     <span style="color:#79c6cd;">
      #define _CRT_SECURE_NO_WARNINGS 1//改行是为了用scanf（）函数不警告使用的第一行，必须放第一行
     </span>
     <br/>
     #include &lt;stdio.h&gt;
     <br/>
     #include &lt;stdlib.h&gt;
     <span style="color:#79c6cd;">
      //srand()和rand()函数头文件
     </span>
     <br/>
     #include &lt;time.h&gt;
     <span style="color:#79c6cd;">
      //time()函数头文件
     </span>
     <br/>
     void menu()
     <br/>
     {
     <!-- -->
     <br/>
     printf("猜0--100之间的数字\n");
     <br/>
     printf("输入1进入游戏\n");
     <br/>
     printf("输入0退出游戏\n");
     <br/>
     printf("请选择开始游戏还是退出游戏:\n");
     <br/>
     }
     <br/>
     void game()
     <br/>
     {
     <!-- -->
     <br/>
     int guss = 0;
     <br/>
     int random = rand()%100+1;
     <span style="color:#79c6cd;">
      //rand()函数生成伪随机也就是固定的一系列或叫做一组不一样的数
     </span>
     <br/>
     printf("请输入你猜的数字\n");
     <span style="color:#79c6cd;">
      //srand()函数放在rand()函数前，而且srand()括号里的数值只有是不同的时候，rand()函数才会生成随机数
     </span>
     <br/>
     while (1)
     <span style="color:#79c6cd;">
      //死循环
     </span>
     <span style="color:#79c6cd;">
      //rand（）可取0--32767，用%100取余数余数在0--99之间,实际上想生成0-999那就%999就可以了
     </span>
     <br/>
     {
     <!-- -->
     <br/>
     scanf("%d", &amp;guss);
     <br/>
     if(guss &lt; random)
     <br/>
     {
     <!-- -->
     <br/>
     printf("猜小了\n");
     <br/>
     <br/>
     }
     <br/>
     else if (guss &gt; random)
     <br/>
     {
     <!-- -->
     <br/>
     printf("猜大了\n");
     <br/>
     printf("请重新输入\n");
     <br/>
     }
     <br/>
     else
     <br/>
     {
     <!-- -->
     <br/>
     printf("恭喜你猜对了\n");
     <br/>
     break;
     <span style="color:#79c6cd;">
      //输入正确就跳出while循环
     </span>
     <br/>
     }
     <br/>
     }
     <br/>
     }
     <br/>
     int main()
     <br/>
     {
     <!-- -->
     <br/>
     <br/>
     int select = 0;
     <br/>
     srand((unsigned int)time(NULL));
     <span style="color:#79c6cd;">
      //time()函数是时间戳，读取电脑运行时间换算的数是一直在变化的数
     </span>
     <br/>
     do
     <span style="color:#79c6cd;">
      //srand()函数内部要求是unsigned类型而time()函数是int类型
     </span>
     <br/>
     {
     <span style="color:#79c6cd;">
      //因此用unsigned int，time()函数里放NULL表示不含参数，只获取当前时间
     </span>
     <br/>
     menu();
     <br/>
     scanf("%d", &amp;select);
     <br/>
     switch (select)
     <br/>
     {
     <!-- -->
     <br/>
     case 1:
     <br/>
     game();
     <br/>
     break;
     <br/>
     case 0:
     <br/>
     {
     <!-- -->
     <br/>
     printf("退出游戏\n");
     <br/>
     break;
     <span style="color:#79c6cd;">
      //跳出switch
     </span>
     <br/>
     }
     <br/>
     default:
     <br/>
     printf("输入错误，请重新输入：\n");
     <br/>
     }
    </p>
    <p>
     <br/>
     } while (select);
    </p>
    <p>
     return 0;
     <br/>
     }
    </p>
    <p>
     <strong>
      程序功能：输入1进入游戏，输入0则退出游戏，输入其他则需要重新输入；
     </strong>
    </p>
    <p>
     <strong>
      系统随机生成一个数，然后慢慢去猜，猜小了则提醒猜小了，猜大了则提醒猜大了，知道猜正确为止；如果再想玩继续输入1进入，不想玩就输入0；
     </strong>
    </p>
