    <h3>
     <a id="1bean_7">
     </a>
     1.在bean层中抽象出类来映射数据库
    </h3>
    <p>
     在这里设计一个类来映射数据库，写出各项属性，get和set方法，构造函数，并重写toString方法，以备在之后使用
    </p>
    <pre><code>package com.whut.bean;

/**
 *  实体类模型 -User
 */
public class User {
    private int id;     //用户id
    private String username;   //用户名
    private String password;   //密码


    // get and ser
    //User user = new User();


    public int getId() {
        return id;
    }

    public void setId(int id) {
        this.id = id;
    }

    public String getUsername() {
        return username;
    }

    public void setUsername(String username) {
        this.username = username;
    }

    public String getPassword() {
        return password;
    }

    public void setPassword(String password) {
        this.password = password;
    }

    //构造方法  - User user =new User("","","")

    public User(int id, String username, String password) {
        this.id = id;
        this.username = username;
        this.password = password;
    }

    public User() {
    }

    //toString

    @Override
    public String toString() {
        return "User{" +
                "id=" + id +
                ", username='" + username + '\'' +
                ", password='" + password + '\'' +
                '}';
    }
}
</code></pre>
    <h3>
     <a id="DaoMybatisDao_71">
     </a>
     Dao层接口和用Mybatis实现Dao层接口
    </h3>
    <p>
     建立一个接口，命名为"userDao"，在其中写入模糊查询方法，参数为String类型的username,因为查询结果可能有多个，所以返回值应为集合类型
    </p>
    <pre><code>package com.whut.dao;

import com.whut.bean.User;

import java.util.List;

/**
 * UserDao - 接口，与数据库直接交互
 */
public interface UserDao {
    //模糊查询
    List&lt;User&gt; FuzzyQueryByUsername(String username);

}
</code></pre>
    <p>
     在mapper文件下实现这个方法。注：parameterType即方法的参数类型，因为我们设计的方法参数为String类型，因此这个的值要为“java.lang.String”。返回值是User类的集合，所以应该为我们自己设计的User类。
    </p>
    <pre><code>&lt;?xml version="1.0" encoding="UTF-8" ?&gt;
&lt;!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd" &gt;
&lt;mapper namespace="com.whut.dao.UserDao" &gt;
    &lt;!--模糊查询--&gt;
    &lt;select id="FuzzyQueryByUsername" parameterType="java.lang.String" resultType="com.whut.bean.User"&gt;
        select * from tb_user where username like "%"#{username}"%"
    &lt;/select&gt;



&lt;/mapper&gt;
</code></pre>
    <h3>
     <a id="Service_101">
     </a>
     Service层接口和实现
    </h3>
    <p>
     Service层可以看成是一个桥梁，连接Dao层和Controller层，依次实现接口时，可以直接生成一个Dao类的对象，然后直接调用这个类的方法。
    </p>
    <pre><code>package com.whut.service;

import com.whut.bean.User;


import java.util.List;

/**
 * UserService -- 业务逻辑层
 */
public interface UserService {
    //模糊查询
    List&lt;User&gt; FuzzyQueryByUsername(String username);
}
</code></pre>
    <p>
     …
    </p>
    <pre><code>package com.whut.service.implate;

import com.whut.bean.User;
import com.whut.dao.UserDao;
import com.whut.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;

    /**
     * UserServoc 的实现
     */
    @Service
    public class UserServiceImpl implements UserService {
    
    
        @Autowired
        private UserDao userDao;
    
        @Override
        public List&lt;User&gt; FuzzyQueryByUsername(String username) {
            return userDao.FuzzyQueryByUsername(username);
        }
    
    
    }
</code></pre>
    <h3>
     <a id="Controller_148">
     </a>
     Controller层代码
    </h3>
    <pre><code>package com.whut.controller;

import com.whut.bean.User;
import com.whut.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.servlet.ModelAndView;

import javax.servlet.http.HttpServletRequest;
import java.util.List;

@Controller
@RequestMapping("user")
public class UserController {

    @Autowired
    private UserService userService;

    //模糊查询
    @RequestMapping("/FuzzyQueryByUsername")
    public ModelAndView FuzzyQueryByUsername(String username){
        //1.调用service方法
        List&lt;User&gt; userList = userService.FuzzyQueryByUsername(username);
        //2.实例化一个ModelAndView对象
        ModelAndView mv=new ModelAndView();
        mv.addObject("userList",userList);
        mv.setViewName("user-list");
        return mv;
    }

}
</code></pre>
