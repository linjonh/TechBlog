   loadmore: {
        inserted(el, binding) {
            const dom = el.querySelector('.el-select-dropdown .el-select-dropdown__wrap'); // 获取下拉框元素
            dom.addEventListener('scroll', function() { // 监听元素触底
                const condition = this.scrollHeight - this.scrollTop &lt;= this.clientHeight;
                if (condition) {
                    binding.value();
                }
            });
        },
    },
},

    <pre><code>// 下拉框滚动分页加载
loadmore() {
    console.log('触底，请求数据')
    this.curCount = this.studentOption.length
    if (this.curCount &lt; this.totalCount) {
        this.current = this.current + 1
        this.getStudentList(this.searchField)
    }
},
// 选择学员下拉框搜索功能
filterMethod(value) {
	this.searchField = value
    this.current = 1
    this.studentOption = []
    clearTimeout(this.timer)
    this.timer = setTimeout(() =&gt; {
        this.getStudentList(value)
    }, 1000)
},
// 获取学员列表
getStudentList(value = '') {
    var obj = {
        searchField: value,
        currPage: this.current,
        pageSize: this.pageSize,
    }
    getStuInformListPageAPI(obj).then((res) =&gt; { // 调接口
       if (res.resultCode === 200 &amp;&amp; res.data) {
           console.log("获取学员列表成功: ", res)
           this.totalCount = res.data.totalCount
           if (res.data.dataList &amp;&amp; res.data.dataList.length &gt; 0) {
               res.data.dataList.forEach(ele =&gt; {
                   var idx = this.studentOption.findIndex(item =&gt; { return item.uid == ele.uid })
                   if (idx == -1) {
                       this.studentOption.push(ele)
                   }
               })
           }
       } else {
           console.log(res.resultStr)
       }
   })
},
