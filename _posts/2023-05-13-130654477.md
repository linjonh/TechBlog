      <div>
       <div>
        <div>
         <span style="color:#0a1022;">
          <strong>
           OO view:
          </strong>
         </span>
        </div>
        <div>
         <span style="color:#0a1022;">
          <strong>
           implies that a component or
          </strong>

          <strong>
           class encapsulates only attributes and
          </strong>

         <span style="color:#0a1022;">
          <strong>
           operations that are closely related to one
          </strong>
         </span>
         <span style="color:#0a1022;">
          <strong>
           another and to the class or component
          </strong>
         </span>
         <span style="color:#0a1022;">
          <strong>
           itself
          </strong>
         </span>
        </div>
       </div>
