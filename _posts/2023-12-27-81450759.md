    <pre class="has"><code>public class C201_02_01比较数的大小 {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int num1 = sc.nextInt();
        int num2 = sc.nextInt();
        int num3 = sc.nextInt();
        if (num1&gt;=num2&amp;&amp;num1&gt;=num3){
            System.out.println("最大的数为："+num1);
        }else if (num2&gt;=num1&amp;&amp;num2&gt;=num3){
            System.out.println("最大的数为："+num2);
        }else {
            System.out.println("最大的数为："+num3);
        }
    }
}</code></pre>
