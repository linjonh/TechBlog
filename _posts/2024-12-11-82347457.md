    var that = this;
    let d = new Date();
    let nowtime = d.getTime();//获取点击时间
    if (nowtime - lasttime &gt; 1500) {//1500ms内无法识别再点击
      //添加自己的代码段
      })
      lasttime = nowtime;
    }
  },</code></pre>
