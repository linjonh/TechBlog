    <h3>
     1、先准备好要实现粘贴复制的代码
    </h3>
    <pre><code class="language-html"> &lt;view class="form_left"&gt;
          &lt;text class="form_one"&gt;订单编号&lt;/text&gt;
          &lt;text class="form_two"&gt;{<!-- -->{ orderNumber }}&lt;/text&gt;
          &lt;text class="form_copy" @click="copyChange(orderNumber)"&gt;复制&lt;/text&gt;
        &lt;/view&gt;</code></pre>
    <pre><code class="language-javascript">&lt;script setup&gt;
import { onLoad } from '@dcloudio/uni-app';
import { ref } from 'vue';
const orderNumber = ref('C2023020115CZ293829109819');
// 复制订单编号

const copyChange = (data) =&gt; {
  console.log('data', data);
  console.log('复制订单编号', 1111111);
  uni.setClipboardData({
    data: data,
    success: function (res) {
      console.log('success', res);
      uni.getClipboardData({
        success: function (res) {
          console.log('粘贴', res);
        }
      });
      uni.showToast({
        title: '复制成功',
        icon: 'none'
      });
    },
    fail: (error) =&gt; {
      console.log('失败', error);
      uni.showToast({
        title: '复制失败',
        icon: 'none'
      });
    }
  });
};

};
&lt;/script&gt;</code></pre>
