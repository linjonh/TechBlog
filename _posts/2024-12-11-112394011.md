    @ResponseBody
    public ResultResponse updateUerInfo(String id, String email, String nickName, String phoneNum, String roles) {

        List&lt;Role&gt; roleList = JSONObject.parseArray(roles, Role.class);
        List&lt;Long&gt; roleIdList = roleList.stream().map(role -&gt; role.getId()).collect(Collectors.toList());

        // 修改用户的角色
        iUserService.updateUserRole(roleIdList);
        // 修改用户信息

        return null;
