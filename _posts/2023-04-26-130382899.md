    <h2>
     <a id="_1">
     </a>
     检查时间是否会回跳
    </h2>
    <pre><code class="prism language-bash"><span class="token comment">#需要多执行几次，检查local time和rtc time是否一致，如果不一致就有可能导致时间回跳</span>
timedatectl status
</code></pre>
    <h2>
     <a id="_8">
     </a>
     安装时间同步服务
    </h2>
    <pre><code class="prism language-bash"><span class="token comment">#centos</span>
yum <span class="token function">install</span> chrony
systemctl <span class="token builtin class-name">enable</span> chronyd.service
</code></pre>
    <h2>
     <a id="_15">
     </a>
     配置时间同步服务
    </h2>
    <pre><code class="prism language-bash"><span class="token function">vim</span> /etc/chrony.conf
</code></pre>
    <h3>
     <a id="_20">
     </a>
     配置文件样例
    </h3>
    <pre><code class="prism language-bash"><span class="token comment"># 同步使用的端口</span>
acquisitionport <span class="token number">1123</span>

<span class="token comment"># 存储 Server 时间的本地目录</span>
dumpdir /var/run/chrony

<span class="token comment"># Ignore stratum in source selection.</span>
stratumweight <span class="token number">0.01</span>

<span class="token comment"># Record the rate at which the system clock gains/losses time.</span>
driftfile /var/lib/chrony/drift

<span class="token comment"># 闰秒配置,17h34m 消化 1s</span>
leapsecmode slew
maxslewrate <span class="token number">1000</span>
smoothtime <span class="token number">400</span> <span class="token number">0.001</span> leaponly

<span class="token comment"># In first three updates step the system clock instead of slew</span>
<span class="token comment"># if the adjustment is larger than 10 seconds.</span>
<span class="token comment"># makestep 0.1 3</span>

<span class="token comment">## Server config</span>
<span class="token comment"># Allow NTP client access from local network.</span>
allow <span class="token number">0.0</span>.0.0/0

<span class="token comment"># Listen for commands only on localhost.</span>
bindaddress <span class="token number">0.0</span>.0.0
port <span class="token number">123</span>
clientloglimit <span class="token number">1073741824</span>
<span class="token comment">#ratelimit interval 1</span>
<span class="token comment">#ratelimit burst 16</span>
<span class="token comment"># Serve time even if not synchronized to any NTP server.</span>
<span class="token builtin class-name">local</span> stratum <span class="token number">5</span>

<span class="token comment">## Command config</span>
bindcmdaddress <span class="token number">127.0</span>.0.1
bindcmdaddress /var/run/chrony/chronyd.sock
cmdallow all

<span class="token comment">## Real Time clock(RTC)</span>
hwclockfile /etc/adjtime
rtcautotrim <span class="token number">10</span>
rtcsync

keyfile /etc/chrony.keys

<span class="token comment"># Specify the key used as password for chronyc.</span>
commandkey <span class="token number">1</span>

<span class="token comment"># Generate command key if missing.</span>
generatecommandkey

<span class="token comment">## Log</span>
<span class="token comment"># Send a message to syslog if a clock adjustment is larger than 0.5 seconds.</span>
logchange <span class="token number">0.1</span>
log measurements statistics tracking
logdir /var/log/chrony

<span class="token comment"># Server 配置</span>
<span class="token comment">#ip address is the stable clock source which customer provided</span>
server ntp.aliyun.com iburst minpoll <span class="token number">4</span> maxpoll <span class="token number">6</span> prefer
</code></pre>
<h2>
<a id="_85">
</a>
检查时间同步服务是否正常
</h2>
<pre><code class="prism language-bash"><span class="token comment">#输出内容里的 IP 前边是\*代表服务正常，?代表有问题</span>
chronyc sources <span class="token parameter variable">-v</span>
</code></pre>
