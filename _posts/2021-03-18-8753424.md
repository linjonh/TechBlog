     Problem Description

    <div class="panel_content">
     判断两序列是否为同一二叉搜索树序列
    </div>
    <div class="panel_bottom">
    </div>
    <br/>
    <div align="left" class="panel_title">
     Input
    </div>
    <div class="panel_content">
     开始一个数n，(1&lt;=n&lt;=20) 表示有n个需要判断，n= 0 的时候输入结束。
     <br/>
     接下去一行是一个序列，序列长度小于10，包含(0~9)的数字，没有重复数字，根据这个序列可以构造出一颗二叉搜索树。
     <br/>
     接下去的n行有n个序列，每个序列格式跟第一个序列一样，请判断这两个序列是否能组成同一颗二叉搜索树。
    </div>

    <pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;cstring&gt;
#include &lt;cstdio&gt;
#include &lt;stdlib.h&gt;
using namespace std;

int a[30],b[30],cnt = 0;

typedef struct tree
{
    tree *l,*r;
    int num;
} tree;
tree *root;
tree *creat(int x)
{
    tree *t = (tree*)malloc(sizeof(tree));
    t-&gt;l = 0;
    t-&gt;r = 0;
    t-&gt;num = x;
    return t;
}

tree *inster(tree *s,int x)
{
    tree *t;
    if(s == NULL)
    {
        t = creat(x);
        s = t;
    }
    else
    {
        if(x &lt;= s-&gt;num)
            s-&gt;l = inster(s-&gt;l,x);
        else
            s-&gt;r = inster(s-&gt;r,x);
    }
    return s;
}

void libian(tree *root)
{
    if(root!=NULL)
    {
        b[cnt++] = root-&gt;num;
        libian(root-&gt;l);
        libian(root-&gt;r);
    }
}

int main()
{
    int n;
    while(cin &gt;&gt; n,n)
    {
        cnt = 0;
        root = NULL;
        int tem;
        char str[30];
        cin &gt;&gt; str;
        int len = strlen(str),i;
        for(i = 0; i&lt;len; i++)
        {
            tem = str[i] - '0';
            root = inster(root,tem);
        }
        libian(root);
        for(i = 0; i&lt;len; i++)
            a[i] = b[i];
        while(n--)
        {
            cnt = 0;
            cin &gt;&gt; str;
            root = NULL;
            for(i = 0; i&lt;len; i++)
            {
                tem = str[i] - '0';
                root = inster(root,tem);
            }
            libian(root);
            for(i = 0; i&lt;len; i++)
                if(a[i]!=b[i])
                {
                    cout &lt;&lt; "NO" &lt;&lt; endl;
                    break;
                }
            if(i&gt;=len)
                cout &lt;&lt; "YES" &lt;&lt; endl;
        }
    }
    return 0;
}
