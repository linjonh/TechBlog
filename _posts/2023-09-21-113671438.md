      from collections import Counter

      def PrimeNum(num):
     </p>
     <p>
      r_value =[]
     </p>
     <p>
      for i inrange(2,num+1):
     </p>
     <p>
      for jin range(2,i):
     </p>
     <p>
      if i % j == 0:
     </p>
     <p>
      break
     </p>
     <p>
      else:
     </p>
     <p>
      r_value.append(i)
     </p>
     <p>
      return r_value
     </p>
     <p>
      def PrimeFactorSolve(num,prime_list):
     </p>
     <p>
      for n inprime_list:
     </p>
     <p>
      if num % n == 0:
     </p>
     <p>
      return [n,num / n]
     </p>

     </p>
     <p>

     <p>
