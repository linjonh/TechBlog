     一、Spring Boot 启动与配置相关注解
    </h3>
    <h4 id="1.1%C2%A0%40SpringBootApplication" name="1.1%C2%A0%40SpringBootApplication">
     1.1
     <code>
      @SpringBootApplication
     </code>
    </h4>
    <p>
     这是 Spring Boot 应用中最核心的注解，通常位于主应用类上。它是一个组合注解，实际上包含了
     <code>
      @Configuration
     </code>
     、
     <code>
      @EnableAutoConfiguration
     </code>
     和
     <code>
      @ComponentScan
     </code>
     三个注解的功能。
    </p>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：
      <ul>
       <li>
        <code>
         @Configuration
        </code>
        ：表明该类是一个配置类，相当于传统 Spring 中的 XML 配置文件。
       </li>
       <li>
        <code>
         @EnableAutoConfiguration
        </code>
        ：开启 Spring Boot 的自动配置功能，Spring Boot 会根据类路径下的依赖和配置自动配置应用。
       </li>
       <li>
        <code>
         @ComponentScan
        </code>
        ：指定 Spring 扫描组件的范围，默认扫描主应用类所在包及其子包下带有
        <code>
         @Component
        </code>
        、
        <code>
         @Service
        </code>
        、
        <code>
         @Repository
        </code>
        、
        <code>
         @Controller
        </code>
        等注解的类。
       </li>
      </ul>
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;

@SpringBootApplication
public class MySpringBootApp {
    public static void main(String[] args) {
        SpringApplication.run(MySpringBootApp.class, args);
    }
}</code></pre>
    <h4 id="1.2%C2%A0%40EnableAutoConfiguration" name="1.2%C2%A0%40EnableAutoConfiguration">
     1.2
     <code>
      @EnableAutoConfiguration
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：Spring Boot 会根据类路径中的依赖自动配置应用。例如，如果类路径中存在 H2 数据库的依赖，Spring Boot 会自动配置一个嵌入式的 H2 数据库。不过，有时候自动配置可能不符合我们的需求，这时可以使用
      <code>
       exclude
      </code>
      属性排除某些自动配置类。
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.boot.autoconfigure.jdbc.DataSourceAutoConfiguration;

@SpringBootApplication(exclude = DataSourceAutoConfiguration.class)
public class MyAppWithoutDataSourceAutoConfig {
    public static void main(String[] args) {
        SpringApplication.run(MyAppWithoutDataSourceAutoConfig.class, args);
    }
}</code></pre>
    <h4 id="1.3%C2%A0%40Configuration" name="1.3%C2%A0%40Configuration">
     1.3
     <code>
      @Configuration
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于定义配置类，配置类可以包含多个
      <code>
       @Bean
      </code>
      注解的方法，这些方法会返回一个对象，该对象会被 Spring 容器管理。
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import java.util.ArrayList;
import java.util.List;

@Configuration
public class AppConfig {
    @Bean
    public List&lt;String&gt; myList() {
        return new ArrayList&lt;&gt;();
    }
}</code></pre>
    <h4 id="1.4%C2%A0%40ComponentScan" name="1.4%C2%A0%40ComponentScan">
     1.4
     <code>
      @ComponentScan
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：指定 Spring 扫描组件的范围。可以通过
      <code>
       basePackages
      </code>
      属性指定要扫描的包，也可以通过
      <code>
       basePackageClasses
      </code>
      指定要扫描的类所在的包。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.context.annotation.ComponentScan;
import org.springframework.context.annotation.Configuration;

@Configuration
@ComponentScan(basePackages = "com.example.demo.service")
public class AppConfig {
    // 配置类内容
}</code></pre>
    <h3 id="%E4%BA%8C%E3%80%81%E4%BE%9D%E8%B5%96%E6%B3%A8%E5%85%A5%E4%B8%8E%E7%BB%84%E4%BB%B6%E7%AE%A1%E7%90%86%E6%B3%A8%E8%A7%A3" name="%E4%BA%8C%E3%80%81%E4%BE%9D%E8%B5%96%E6%B3%A8%E5%85%A5%E4%B8%8E%E7%BB%84%E4%BB%B6%E7%AE%A1%E7%90%86%E6%B3%A8%E8%A7%A3">
     二、依赖注入与组件管理注解
    </h3>
    <h4 id="2.1%C2%A0%40Component" name="2.1%C2%A0%40Component">
     2.1
     <code>
      @Component
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：这是一个通用的组件注解，用于标记一个类为 Spring 组件，被 Spring 容器管理。它是
      <code>
       @Service
      </code>
      、
      <code>
       @Repository
      </code>
      、
      <code>
       @Controller
      </code>
      的父注解。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.stereotype.Component;

@Component
public class MyComponent {
    public void doSomething() {
        System.out.println("Doing something in MyComponent");
    }
}</code></pre>
    <h4 id="2.2%C2%A0%40Service" name="2.2%C2%A0%40Service">
     2.2
     <code>
      @Service
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：通常用于标记业务逻辑层的类，是
      <code>
       @Component
      </code>
      的一种特殊形式，语义上更明确表示这是一个服务类。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.stereotype.Service;

@Service
public class UserService {
    public void saveUser() {
        // 保存用户的业务逻辑
    }
}</code></pre>
    <h4 id="2.3%C2%A0%40Repository" name="2.3%C2%A0%40Repository">
     2.3
     <code>
      @Repository
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：主要用于标记数据访问层的类，如 DAO 类。Spring 会自动处理
      <code>
       @Repository
      </code>
      注解类抛出的异常，将其转换为 Spring 的数据访问异常。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.stereotype.Repository;

@Repository
public class UserRepository {
    public void saveUserToDb() {
        // 将用户保存到数据库的逻辑
    }
}</code></pre>
    <h4 id="2.4%C2%A0%40Controller" name="2.4%C2%A0%40Controller">
     2.4
     <code>
      @Controller
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于标记控制器类，处理 HTTP 请求。通常与
      <code>
       @RequestMapping
      </code>
      等注解配合使用。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.ResponseBody;

@Controller
public class MyController {
    @RequestMapping("/hello")
    @ResponseBody
    public String sayHello() {
        return "Hello, World!";
    }
}</code></pre>
    <h4 id="2.5%C2%A0%40RestController" name="2.5%C2%A0%40RestController">
     2.5
     <code>
      @RestController
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：是
      <code>
       @Controller
      </code>
      和
      <code>
       @ResponseBody
      </code>
      的组合注解，用于创建 RESTful 风格的控制器，方法返回的对象会自动转换为 JSON 或 XML 等格式返回给客户端。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class MyRestController {
    @GetMapping("/data")
    public String getData() {
        return "{\"message\": \"This is some data\"}";
    }
}</code></pre>
    <h4 id="2.6%C2%A0%40Autowired" name="2.6%C2%A0%40Autowired">
     2.6
     <code>
      @Autowired
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于自动装配 Bean，默认按照类型进行装配。可以用于构造器、字段、方法上。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

@Service
public class UserService {
    private UserRepository userRepository;

    @Autowired
    public UserService(UserRepository userRepository) {
        this.userRepository = userRepository;
    }

    public void saveUser() {
        userRepository.saveUserToDb();
    }
}</code></pre>
    <h4 id="2.7%C2%A0%40Qualifier" name="2.7%C2%A0%40Qualifier">
     2.7
     <code>
      @Qualifier
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：当存在多个相同类型的 Bean 时，
      <code>
       @Autowired
      </code>
      无法确定要注入哪个 Bean，此时可以使用
      <code>
       @Qualifier
      </code>
      指定要注入的 Bean 的名称。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.stereotype.Service;

@Service
public class MyService {
    private MyComponent myComponent;

    @Autowired
    public MyService(@Qualifier("specificComponent") MyComponent myComponent) {
        this.myComponent = myComponent;
    }
}</code></pre>
    <h4 id="2.8%C2%A0%40Resource" name="2.8%C2%A0%40Resource">
     2.8
     <code>
      @Resource
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：也是用于依赖注入，默认按照名称进行装配，如果找不到匹配的名称，则按照类型进行装配。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import javax.annotation.Resource;
import org.springframework.stereotype.Service;

@Service
public class MyService {
    @Resource
    private MyComponent myComponent;

    public void doSomething() {
        myComponent.doSomething();
    }
}</code></pre>
    <h4 id="2.9%C2%A0%40Value" name="2.9%C2%A0%40Value">
     2.9
     <code>
      @Value
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于从配置文件中读取属性值并注入到 Bean 的字段中。支持 SpEL 表达式。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

@Component
public class MyConfig {
    @Value("${app.name}")
    private String appName;

    public String getAppName() {
        return appName;
    }
}</code></pre>
    <h3 id="%E4%B8%89%E3%80%81Web%20%E5%BC%80%E5%8F%91%E7%9B%B8%E5%85%B3%E6%B3%A8%E8%A7%A3" name="%E4%B8%89%E3%80%81Web%20%E5%BC%80%E5%8F%91%E7%9B%B8%E5%85%B3%E6%B3%A8%E8%A7%A3">
     三、Web 开发相关注解
    </h3>
    <h4 id="3.1%C2%A0%40RequestMapping" name="3.1%C2%A0%40RequestMapping">
     3.1
     <code>
      @RequestMapping
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于映射 HTTP 请求到控制器的处理方法上。可以指定请求的 URL、请求方法（GET、POST 等）、请求参数等。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.ResponseBody;

@Controller
public class MyController {
    @RequestMapping(value = "/hello", method = RequestMethod.GET)
    @ResponseBody
    public String sayHello() {
        return "Hello, World!";
    }
}</code></pre>
    <h4 id="3.2%C2%A0%40GetMapping%E3%80%81%40PostMapping%E3%80%81%40PutMapping%E3%80%81%40DeleteMapping%E3%80%81%40PatchMapping" name="3.2%C2%A0%40GetMapping%E3%80%81%40PostMapping%E3%80%81%40PutMapping%E3%80%81%40DeleteMapping%E3%80%81%40PatchMapping">
     3.2
     <code>
      @GetMapping
     </code>
     、
     <code>
      @PostMapping
     </code>
     、
     <code>
      @PutMapping
     </code>
     、
     <code>
      @DeleteMapping
     </code>
     、
     <code>
      @PatchMapping
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：这些注解是
      <code>
       @RequestMapping
      </code>
      的简化形式，分别对应 HTTP 的 GET、POST、PUT、DELETE、PATCH 请求方法。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class MyRestController {
    @GetMapping("/data")
    public String getData() {
        return "{\"message\": \"This is some data\"}";
    }
}</code></pre>
    <h4 id="3.3%C2%A0%40PathVariable" name="3.3%C2%A0%40PathVariable">
     3.3
     <code>
      @PathVariable
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于获取 URL 中的路径变量。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class UserController {
    @GetMapping("/users/{id}")
    public String getUser(@PathVariable Long id) {
        return "User with ID: " + id;
    }
}</code></pre>
    <h4 id="3.4%C2%A0%40RequestParam" name="3.4%C2%A0%40RequestParam">
     3.4
     <code>
      @RequestParam
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于获取 HTTP 请求中的查询参数。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class SearchController {
    @GetMapping("/search")
    public String search(@RequestParam String keyword) {
        return "Searching for: " + keyword;
    }
}</code></pre>
    <h4 id="3.5%C2%A0%40RequestBody" name="3.5%C2%A0%40RequestBody">
     3.5
     <code>
      @RequestBody
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于将 HTTP 请求的主体内容绑定到方法的参数上，通常用于处理 JSON 或 XML 格式的数据。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class UserController {
    @PostMapping("/users")
    public String createUser(@RequestBody User user) {
        // 处理用户创建逻辑
        return "User created: " + user.getName();
    }
}

class User {
    private String name;

    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }
}</code></pre>
    <h4 id="3.6%C2%A0%40RequestHeader" name="3.6%C2%A0%40RequestHeader">
     3.6
     <code>
      @RequestHeader
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于获取 HTTP 请求的头部信息。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestHeader;

@RestController
public class HeaderController {

    <h4 id="3.7%C2%A0%40CookieValue" name="3.7%C2%A0%40CookieValue">

    </h4>

    <h4 id="3.8%C2%A0%40ModelAttribute" name="3.8%C2%A0%40ModelAttribute">
     3.8
     <code>
      @ModelAttribute
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>

    <h4 id="4.7%C2%A0%40Transactional" name="4.7%C2%A0%40Transactional">

     <code>

     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于声明事务边界，可以应用在类或方法上。在方法上使用时，该方法会在事务中执行；在类上使用时，该类的所有公共方法都会在事务中执行。可以设置事务的传播行为、隔离级别等属性。

     <li>
      <strong>

      </strong>

    @Transactional
    public void saveUser(User user) {
        userRepository.save(user);
    }

    <h4 id="5.7%C2%A0%40Around" name="5.7%C2%A0%40Around">
     5.7
     <code>
      @Around
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：环绕通知，环绕目标方法执行，可以在目标方法执行前后进行增强处理，还可以控制目标方法是否执行。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.aspectj.lang.ProceedingJoinPoint;

import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Pointcut;
import org.springframework.stereotype.Component;

@Aspect
@Component
public class LoggingAspect {
    @Pointcut("execution(* com.example.demo.service.*.*(..))")
    public void serviceMethods() {}

    @Around("serviceMethods()")
    public Object aroundServiceMethod(ProceedingJoinPoint joinPoint) throws Throwable {
        System.out.println("Before service method execution");
        Object result = joinPoint.proceed();
        System.out.println("After service method execution");
        return result;
    }
}</code></pre>

    <h4 id="7.5%C2%A0%40Scheduled" name="7.5%C2%A0%40Scheduled">
     7.5
     <code>
      @Scheduled
     </code>
    </h4>
    <ul>
     <li>
      <strong>
       作用
      </strong>
      ：用于创建定时任务，可指定任务的执行时间间隔、固定延迟等。
     </li>
     <li>
      <strong>
       示例
      </strong>
      ：
     </li>
    </ul>
    <pre><code class="language-java">import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;

@Component
public class MyScheduledTask {
    @Scheduled(fixedRate = 5000)
    public void doTask() {
        System.out.println("Task executed every 5 seconds");
    }
}</code></pre>
