      <br/>

      <br/>
      format: string
      <br/>
      ...args: any[ ]

    <p id="u93aa04fa">
     %{public } s    s指的是String 表示后面是以字符串的格式输出的
    </p>
    <p id="uee89777d">
     意思是 这个是以明文形式字符串输出
    </p>
    <p id="u9f4b0727">
     public 私有的 加密写死的
    </p>
    <p>
    </p>

     <img alt="" height="900" id="u77a7f1c0" src="https://i-blog.csdnimg.cn/img_convert/157e62e02ba7e72117586e7422345e68.png" width="1774"/>
    </p>
    <h4>
     封装项目日志文件 ：common/utils/Logger.ets
    </h4>
    <p id="uc787e6d2">
     日志类的封装思路：
    </p>
    <p>
     使用类的静态方法进行封装四个方法
    </p>
    <p>
     导出这个类
    </p>
    <p class="img-center">
     <img alt="" height="540" src="https://i-blog.csdnimg.cn/direct/669ac622c1684fe5816ef5bf1a545d04.png" width="760"/>
    </p>
    <pre><code class="hljs">import  { hilog }  from  '@kit.PerformanceAnalysisKit'

const domain =0x0000
const tag ='mylog'
const format ='%{public}s %{public}s'

// 按需导出
export class Logger {
static info(...args: string[ ]){
 hilog.info(domain,tag,format, ...args)
}
static debug(...args: string[]){
hilog.debug(domain, tag,format,...args)
}
static warn(...args: string[]){
hilog.warn(domain,tag,format,...args)
}
static error(...args: string[]){
hilog.error(domain,tag,format, ...args)
}
}</code></pre>
    <p>
     用的时候 直接导入即可
