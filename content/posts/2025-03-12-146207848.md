    <h3>
     思路一：
    </h3>
    <pre><code class="language-cpp">#include &lt;iostream&gt;
using namespace std;

bool isGoodNumber(int n) {
    for(int i = 1; n &gt; 0; i++) // 从最低位开始，i表示当前是第几位，i++位数加1
    {
        int digit = n % 10;  // 取出当前位的数字
        if ((i % 2 == 1 &amp;&amp; digit % 2 != 1) || (i % 2 == 0 &amp;&amp; digit % 2 != 0)) {
            return false;  // 如果当前位不满足条件，返回false
        }
        n = n / 10;  // 去掉最低位
         // 位数加1
    }
    return true;  // 所有位都满足条件，返回true
}

int main() {
    int N;
    cin &gt;&gt; N;

    int count = 0;
    for (int n = 1; n &lt;= N; n++) {
        if (isGoodNumber(n)) {
            count++;  // 如果是好数，计数器加1
        }
    }

    cout &lt;&lt; count;
    return 0;
}</code></pre>
    <h3>
     思路二：
    </h3>
    <pre><code class="language-cpp">#include &lt;iostream&gt;
using namespace std;

int main() {
    int N;
    cin &gt;&gt; N;

    int count = 0;  // 记录“好数”的数量

    for (int n = 1; n &lt;= N; n++) {
        int temp = n;  // 使用临时变量，避免修改外层循环的 n
        bool isGood = true;  // 假设当前数字是“好数”
        int i = 1;  // 位数计数器，从最低位开始

        while (temp &gt; 0) {
            int digit = temp % 10;  // 取出当前位的数字

            // 检查当前位是否满足条件
            if ((i % 2 == 1 &amp;&amp; digit % 2 != 1) || (i % 2 == 0 &amp;&amp; digit % 2 != 0)) {
                isGood = false;  // 如果某一位不满足条件，标记为不是“好数”
                break;  // 跳出内层循环
            }

            temp = temp / 10;  // 去掉最低位
            i++;  // 位数加1
        }

        if (isGood) {
            count++;  // 如果当前数字是“好数”，计数器加1
        }
    }

    cout &lt;&lt; count;  // 输出“好数”的数量
    return 0;
}</code></pre>
    <p>
     解释代码
    </p>
    <ol>
     <li>
      <p>
       <strong>
        临时变量
        <code>
         temp
        </code>
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         在内层循环中，使用
         <code>
          temp
         </code>
         来保存当前正在检查的数字，避免修改外层循环的
         <code>
          n
         </code>
         。
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        标志变量
        <code>
         isGood
        </code>
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         初始值为
         <code>
          true
         </code>
         ，假设当前数字是“好数”。
        </p>
       </li>
       <li>
        <p>
         如果某一位不满足条件，将
         <code>
          isGood
         </code>
         设置为
         <code>
          false
         </code>
         并跳出内层循环。
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        位数计数器
        <code>
         i
        </code>
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         从最低位开始，逐位检查数字的每一位是否满足“好数”的条件。
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        条件判断
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         如果当前位是奇数位（
         <code>
          i % 2 == 1
         </code>
         ），则检查数字是否为奇数（
         <code>
          digit % 2 == 1
         </code>
         ）。
        </p>
       </li>
       <li>
        <p>
         如果当前位是偶数位（
         <code>
          i % 2 == 0
         </code>
         ），则检查数字是否为偶数（
         <code>
          digit % 2 == 0
         </code>

#include &lt;cmath&gt;      // 提供pow()和round()数学函数
using namespace std;

int main() {
    int n;            // 转换参数n，必须是整数
    double d;         // 待转换的浮点数d，题目保证d&gt;0
    
    // 输入处理：按顺序读取整数n和浮点数d
    cin &gt;&gt; n &gt;&gt; d;    // 示例输入：3 9.14
    
    // 计算 d * 2^n（使用pow函数计算2的n次方）
    double m = d * pow(2, n); // 示例计算：9.14 * 8 = 73.12
    
    // 四舍五入到最近的整数，使用long long避免溢出
    long long ans = round(m); // 示例结果：round(73.12) = 73
    
    // 输出最终结果
    cout &lt;&lt; ans &lt;&lt; endl; // 正确输出应为整数73
    
    return 0;
}</code></pre>

     五，宝石组合
    </h2>
    <p>
     <img alt="" height="731" src="https://i-blog.csdnimg.cn/direct/cf23d342b2034e4db18756a8423c348e.png" width="1144"/>
    </p>
    <p>
    </p>
    <pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;algorithm&gt;
#include &lt;cmath&gt;
using namespace std;

const int N = 1e5 + 1; //定义一个常数 N,，表示数组大小的上线为100001
int s[N]; //定义整数数组s，用于存储输入的宝石的闪亮度
int n; //存储宝石的数量
vector&lt;int&gt; d[N]; 

int main() {
    cin &gt;&gt; n;
//遍历0到n-1，读取宝石的闪亮度，存储到数组s中
    for (int i = 0; i &lt; n; i++) {
        cin &gt;&gt; s[i];
    }
    sort(s, s + n);//对数组s中的宝石闪亮度进行升序排序
//找出数组s中每个元素的约数，将具有相同约数的数组元素vector数组 d中
    for (int i = 0; i &lt; n; i++) {
        for (int j = 1; j &lt;= sqrt(s[i]); j++) {
            if (s[i] % j == 0) {
                d[j].push_back(s[i]);
                if (s[i] / j != j) {
                    d[s[i] / j].push_back(s[i]);
                }
            }
        }
    }
//在存储具有相同约数的宝石闪亮度的vector数组d中，寻找第一个元素个数大于等于3的
    for (int i = N - 1; i &gt;= 0; i--) {
        if (d[i].size() &gt;= 3) {
            cout &lt;&lt; d[i][0];
            for (int j = 1; j &lt; 3; j++) {
                cout &lt;&lt; " " &lt;&lt; d[i][j];
            }
            break;
        }
    }
    return 0;
}</code></pre>
