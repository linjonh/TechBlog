    <pre><code class="hljs">#define _CRT_SECURE_NO_WARNINGS
#include&lt;stdio.h&gt;
#include&lt;string.h&gt;//库函数strcmp头文件
//使用qsort排序结构体
struct Stu
{
    char name[20];
    int age;
};
//按照年龄排序
int cmp_stu_by_age(const void* e1,const void* e2)
{
    return ((struct Stu*)e1)-&gt;age - ((struct Stu*)e2)-&gt;age;
    //错误写法：return *((struct Stu*)e1)-&gt;age - *((struct Stu*)e2)-&gt;age;
    //错误原因：((struct Stu*)e1)-&gt;age，((struct Stu*)e2)-&gt;age 已经拿到结构体数组元素了，
    //            不需要再用解引用 * 通过地址去找它们。
}
//按照名字排序
int cmp_stu_by_name(const void* e1, const void* e2)
{
    return strcmp(((struct Stu*)e1)-&gt;name, ((struct Stu*)e2)-&gt;name);
    //库函数：strcmp
    //比较字符串大小
}
int main()
{
    struct Stu s[3] = { {"wanger",45},{"zhangsan",20},{"lisi",19} };
    int sz = sizeof(s) / sizeof(s[0]);
    qsort(s, sz, sizeof(s[0]), cmp_stu_by_age);
    int i = 0;
    for(i=0;i&lt;sz;i++)
    {
        printf("name: %s,age: %d\n", s[i].name, s[i].age);
    }
    printf("\n");
    qsort(s, sz, sizeof(s[0]), cmp_stu_by_name);
    for (i = 0; i &lt; sz; i++)
    {
        printf("name: %s,age: %d\n", s[i].name, s[i].age);
    }
    return 0;
}</code></pre>

      思考：

    <p>
     <span style="color:#0d0016">
      <strong>
       为什么compare函数返回时用的是-&gt;，而main函数内打印时用的是.呢？
      </strong>
     </span>
    </p>
    <p>
     在C语言中， -&gt;运算符和.运算符都是用于访问结构体成员的，但它们的使用场景不同。
    </p>
    <p>
     <span style="color:#be191c">
      <strong>
       -&gt;运算符用于通过指针访问结构体成员。在  compare  函数中，参数  a  和  b  是  const void* 类型的指针，
       <br/>
       在函数内部将它们强制转换为  struct MyStruct* 类型的指针  s1  和  s2 ，所以使用-&gt;来访问结构体成员  num ，
       <br/>
       如  s1-&gt;num 。
      </strong>
     </span>
    </p>
    <p>
     <span style="color:#be191c">
      <strong>
       .运算符用于通过结构体变量直接访问结构体成员。在  main  函数中， arr  是结构体数组，
       <br/>
       arr[i]  是一个结构体变量，因此使用.来访问结构体成员  num  和  str ，如  arr[i].num  和  arr[i].str 。
      </strong>
     </span>
    </p>
