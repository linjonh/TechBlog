# mysql查询某个表属于哪个数据库

#查找某个表属于哪个数据库

> SELECT table\_schema FROM information\_schema.TABLES WHERE table\_name =
>   
> ‘表名’;

#查找某个字段在哪个数据库和表中有使用到：

> select table\_schema,table\_name from information\_schema.columns where
>   
> column\_name = ‘字段名’
