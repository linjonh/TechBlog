    <p>
     在使用 Nuxt.js 构建项目时，为了提高性能，通常会考虑对静态资源进行压缩。
     <code>
      compression-webpack-plugin
     </code>
     是一个常用的 Webpack 插件，用于在生产环境中对文件进行 Gzip 压缩。这对于减少网络传输时间和提高页面加载速度非常有帮助。下面是如何在 Nuxt.js 项目中配置
     <code>
      compression-webpack-plugin
     </code>
     的步骤：
    </p>
    <h4>
     1. 安装
     <code>
      compression-webpack-plugin
     </code>
    </h4>
    <p>
     首先，你需要安装
     <code>
      compression-webpack-plugin
     </code>
     。在你的项目根目录下打开终端，运行以下命令：
    </p>
    <p>
     npm install --save-dev compression-webpack-plugin@4.0.1
    </p>
    <h4>
     2. 配置
     <code>
      compression-webpack-plugin
     </code>
    </h4>
    <p>
     接下来，你需要在 Nuxt.js 的配置文件
     <code>
      uxt.config.js
     </code>
     中配置
     <code>
      compression-webpack-plugin，
     </code>
     添加或修改
     <code>
      build
     </code>
     配置部分：
    </p>
    <pre><code>//nuxt.config.js

const CompressionWebpackPlugin = require('compression-webpack-plugin')

module.exports = {
 build: {
    extend(config, { isDev, isClient }) {
      if (!isDev &amp;&amp; isClient) {
        config.plugins.push(
          new CompressionWebpackPlugin({
            filename: '[path].gz', 
            algorithm: 'gzip',
            test: /\.(js|css|html|svg)$/,  // 匹配文件名
            threshold: 10240, // 对超过10kb的数据进行压缩
            minRatio: 0.8,
            deleteOriginalAssets: false, // 是否删除原文件
          })
        )
      }
    },
  }
}

</code></pre>
