			"UDDLUULRUL",
			"UURLLLRRRU",
			"RRUURLDLRD",
			"RUDDDDUUUU",
			"URUDLLRRUU",
			"DURLRLDLRL",

			"UUDDUDUDLL",

	};
	static int count  = 0;
	static char res2 [][] = new char[10][10];
	static char res3[][] = new char [10][10];
	public static void main(String []args) {
	
	for(int i = 0;i&lt;10;i++) {
		res2[i] = res1[i].toCharArray();
	}
	for(int i = 0;i&lt;10;i++) {
		for(int j = 0;j&lt;10;j++) {
			res3= new char [10][10];//每次都置空一下
			cc(i,j);
			 
		}
	}
	  System.out.println(count);
	}
	static int cc(int i,int j) {
		if(i&lt;0||j&lt;0||i&gt;9||j&gt;9) {
			count++;
			return 1;
		}
		if(res3[i][j]==1) {//又走到上次那个位置了
			return 0;
		}
		res3[i][j]=1;//标记位置，防止一直循环
		if(res2[i][j]== 'L') {
			cc(i,j-1);
		}
		
		else if(res2[i][j]== 'R') {
			cc(i,j+1);
		}
	else if(res2[i][j]== 'U') {
		cc(i-1,j);
		}
	else if(res2[i][j]== 'D') {
		cc(i+1,j);
	}
		return 0;
	}</code></pre>
