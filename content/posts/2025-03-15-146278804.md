    <p>
     对于
     <img alt="M\times M" class="mathcode" src="https://latex.csdn.net/eq?M%5Ctimes%20M">
      的矩阵，使用两个指针变量，可以方便实现(i,j)处元素与(j,i)处元素交换位置。令指针Arow=&amp;A[i][0]，则Arow[j]可实现对第i行j列元素的访问。令指针Bptr=&amp;A[0][i]，则*Bptr就可以访问(0,i)处元素，然后，令Bptr+=M，就可访问同列不同行的元素。
     </img>
    </p>
    <p>
     代码实现
    </p>
    <p>
     #define M  3
     <br/>
     int A[3][3]={
     <!-- -->
     {1,2,3},{4,5,6},{7,8,9}};
     <br/>
     void transpose(int A[M][M]);
    </p>
    <p>
     void transpose(int A[M][M])
     <br/>
     {
     <!-- -->
     <br/>
     int i,j;
     <br/>
     int t;
     <br/>
     int *Arow;
     <br/>
     int *Bptr;
     <br/>
     for(i=0;i&lt;M;i++)
     <br/>
     {
     <!-- -->
     <br/>
     Arow=&amp;A[i][0];
     <br/>
     Bptr=&amp;A[0][i];
     <br/>
     for(j=0;j&lt;i;j++)
     <br/>
     {
     <!-- -->
     <br/>
     t=Arow[j];
     <br/>
     Arow[j]=*Bptr;
     <br/>
     *Bptr=t;
     <br/>
     Bptr+=M;
     <br/>
     }
     <br/>
     }
     <br/>
     }
    </p>
    <p>
     void main()
     <br/>
     {
     <!-- -->
     <br/>
     int B[M][M];
     <br/>
     int i,j;
     <br/>
     for(i=0;i&lt;M;i++)          //把A矩阵元素存储在B矩阵中
     <br/>
     for(j=0;j&lt;M;j++)
     <br/>
     B[i][j]=A[i][j];
     <br/>
     transpose(A);            //把A矩阵转置
     <br/>
     }
    </p>
