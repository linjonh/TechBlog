    <h4>
     1.
     <strong>
      <code>
       onLoad
      </code>
     </strong>
    </h4>
    <ul>
     <li>
      <p>
       <strong>
        触发时机
       </strong>
       ：页面
       <strong>
        首次加载
       </strong>
       时触发，且只会触发一次。
      </p>
     </li>
     <li>
      <p>
       <strong>
        适用场景
       </strong>
       ：适合在页面加载时执行一次性操作，比如获取页面参数、初始化数据、发送请求等。
      </p>
     </li>
     <li>
      <p>
       <strong>
        特点
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         可以接收页面传递的参数（通过
         <code>
          options
         </code>
         对象）。
        </p>
       </li>
       <li>
        <p>
         是 UniApp 页面生命周期钩子，不是 Vue 组件的钩子。
        </p>
       </li>
      </ul>
     </li>
    </ul>
    <h5>
     示例代码
    </h5>
    <p>
     javascript
    </p>
    <pre><code class="hljs">export default {
  onLoad(options) {
    console.log('页面加载，参数为:', options);
    // 初始化数据
    this.loadData();
  },
  methods: {
    loadData() {
      console.log('加载数据...');
    }
  }
}</code></pre>
    <hr/>
    <h4>
     2.
     <strong>
      <code>
       onShow
      </code>
     </strong>
    </h4>
    <ul>
     <li>
      <p>
       <strong>
        触发时机
       </strong>
       ：页面
       <strong>
        显示
       </strong>
       时触发，包括页面首次加载和从其他页面返回时。
      </p>
     </li>
     <li>
      <p>
       <strong>
        适用场景
       </strong>
       ：适合在页面每次显示时执行操作，比如刷新数据、更新状态等。
      </p>
     </li>
     <li>
      <p>
       <strong>
        特点
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         无法直接接收参数，但可以通过全局变量或缓存传递数据。
        </p>
       </li>
       <li>
        <p>
         是 UniApp 页面生命周期钩子，不是 Vue 组件的钩子。
        </p>
       </li>
      </ul>
     </li>
    </ul>
    <h5>
     示例代码
    </h5>
    <p>
     javascript
    </p>
    <pre><code class="hljs">export default {
  onShow() {
    console.log('页面显示');
    // 刷新数据
    this.refreshData();
  },
  methods: {
    refreshData() {
      console.log('刷新数据...');
    }
  }
}</code></pre>
    <hr/>
    <h4>
     3.
     <strong>
      <code>
       mounted
      </code>
     </strong>
    </h4>
    <ul>
     <li>
      <p>
       <strong>
        触发时机
       </strong>
       ：Vue 组件
       <strong>
        挂载到 DOM 中
       </strong>
       后触发。
      </p>
     </li>
     <li>
      <p>
       <strong>
        适用场景
       </strong>
       ：适合在组件挂载完成后操作 DOM 或执行依赖 DOM 的逻辑，比如初始化第三方库、获取 DOM 元素等。
      </p>
     </li>
     <li>
      <p>
       <strong>
        特点
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         是 Vue 组件的生命周期钩子，不是 UniApp 页面的钩子。
        </p>
       </li>
       <li>
        <p>
         在 UniApp 中，
         <code>
          mounted()
         </code>
         通常用于组件，而不是页面。
        </p>
       </li>
      </ul>
     </li>
    </ul>
    <h5>
     示例代码
    </h5>
    <p>
     vue
    </p>
    <pre><code class="hljs">&lt;template&gt;
  &lt;view&gt;
    &lt;text ref="myText"&gt;Hello UniApp&lt;/text&gt;
  &lt;/view&gt;
&lt;/template&gt;

&lt;script&gt;
export default {
  mounted() {
    console.log('组件已挂载');
    // 操作 DOM
    const textElement = this.$refs.myText;
    console.log('文本内容:', textElement.textContent);
  }
}
&lt;/script&gt;</code></pre>
    <hr/>

    <h4>
     5.
     <strong>
      综合示例
     </strong>
    </h4>
    <p>
     假设我们有一个页面，需要实现以下功能：
    </p>
    <ol>
     <li>
      <p>
       页面加载时获取参数并初始化数据。
      </p>
     </li>
     <li>
      <p>
       页面显示时刷新数据。
      </p>
     </li>
     <li>
      <p>
       组件挂载后操作 DOM。
      </p>
     </li>
    </ol>
    <h5>
     代码实现
    </h5>
    <p>
     vue
    </p>
    <pre><code class="hljs">&lt;template&gt;
  &lt;view&gt;
    &lt;text ref="myText"&gt;{<!-- -->{ message }}&lt;/text&gt;
    &lt;button @click="navigateToOtherPage"&gt;跳转到其他页面&lt;/button&gt;
  &lt;/view&gt;
&lt;/template&gt;

&lt;script&gt;
export default {
  data() {
    return {
      message: 'Hello UniApp'
    };
  },
  onLoad(options) {
    console.log('页面加载，参数为:', options);
    this.loadData();
  },
  onShow() {
    console.log('页面显示');
    this.refreshData();
  },
  mounted() {
    console.log('组件已挂载');
    const textElement = this.$refs.myText;
    console.log('文本内容:', textElement.textContent);
  },
  methods: {
    loadData() {
      console.log('加载数据...');
      // 模拟初始化数据
      this.message = '数据已加载';
    },
    refreshData() {
      console.log('刷新数据...');
      // 模拟刷新数据
      this.message = '数据已刷新';
    },
    navigateToOtherPage() {
      uni.navigateTo({
        url: '/pages/otherPage/otherPage'
      });
    }
  }
};
&lt;/script&gt;</code></pre>

    <h4>
     6.
     <strong>
      运行逻辑
     </strong>
    </h4>
    <ol>
     <li>
      <p>
       <strong>
        页面加载时
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         触发
         <code>
          onLoad
         </code>
         ，获取参数并初始化数据。
        </p>
       </li>
       <li>
        <p>
         触发
         <code>
          mounted
         </code>
         ，组件挂载完成，操作 DOM。
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        页面显示时
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         触发
         <code>
          onShow
         </code>
         ，刷新数据。
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        从其他页面返回时
       </strong>
       ：
      </p>
      <ul>
       <li>
        <p>
         再次触发
         <code>
          onShow
         </code>
         ，刷新数据。
        </p>
       </li>
      </ul>
     </li>
    </ol>
    <hr/>
    <h4>
     7.
     <strong>
      总结
     </strong>
    </h4>
    <ul>
     <li>
      <p>
       <strong>
        <code>
         onLoad
        </code>
       </strong>
       ：页面加载时执行一次性操作。
      </p>
     </li>
     <li>
      <p>
       <strong>
        <code>
         onShow
        </code>
       </strong>
       ：页面显示时执行刷新操作。
      </p>
     </li>
     <li>
      <p>
       <strong>
        <code>
         mounted
        </code>
       </strong>
       ：组件挂载后操作 DOM 或执行依赖 DOM 的逻辑。
      </p>
     </li>
    </ul>
