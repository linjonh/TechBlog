    <h3 id="%E6%A6%82%E5%BF%B5" name="%E6%A6%82%E5%BF%B5">
     概念
    </h3>
    <p>
     为了实现前后端交互，在前端我们需要向后端接口发送
     <strong>
      http请求
     </strong>
     ，并根据反馈的结果进行数据的渲染。在Vue3中，我们可以使用第三方库
     <strong>
      Axios
     </strong>
     实现这一功能。
    </p>
    <p>
     Axios是一个基于
     <strong>
      Promise
     </strong>
     的http客户端，用于浏览器和 Node.js 环境。它提供了简单易用的API，用于发送http请求并处理响应。它提供了
     <strong>
      请求拦截器
     </strong>
     和
     <strong>
      响应拦截器
     </strong>
     ，可以在请求发送前或响应到达后对其进行处理，如添加认证头、处理错误等。此外，Axios会自动将请求和响应的
     <strong>
      JSON数据
     </strong>
     转换为JavaScript对象，无需手动解析。
    </p>
    <p>
     本文中的开发环境为Windows10，使用npm安装Axios。
    </p>
    <h3 id="%E5%B0%81%E8%A3%85%E5%B7%A5%E5%85%B7%E5%8C%85" name="%E5%B0%81%E8%A3%85%E5%B7%A5%E5%85%B7%E5%8C%85">
     封装工具包
    </h3>
    <p>
     对于项目开发来说，发送Http是一种非常常见的需求，因此我们可以将发送请求的共同代码封装为工具包，以降低代码冗余度，同时方便开发。
    </p>
    <p>
     新建一个utils工具包（如果没有创建的话），新建两个js文件——
     <strong>
      http.js
     </strong>
     和
     <strong>
      postApi.js
     </strong>
     。
    </p>
    <p>
     在http.js中，添加如下代码——
    </p>
    <pre><code class="language-javascript">import axios from "axios";

const instance = axios.create({
  timeout: 3000,
  headers: { "X-Custom-Header": "foobar" },
});

// 添加请求拦截器
instance.interceptors.request.use(
  function (config) {
    // 在发送请求之前做些什么
    return config;
  },
  function (error) {
    // 对请求错误做些什么
    return Promise.reject(error);
  }
);

// 添加响应拦截器
instance.interceptors.response.use(
  function (response) {
    // 对响应数据做点什么
    return response.data;
  },
  function (error) {
    // 对响应错误做点什么
    return Promise.reject(error);
  }
);

export default instance;
</code></pre>
    <p>
     然后，在postApi.js中，添加如下的代码——
    </p>
    <pre><code class="language-javascript">import http from "@/utils/http"

export const postApi = (routePath, params = null) =&gt; {
    // 接口路径添加/api以配置跨域，根据后端给的接口来
    const path = '/api' + routePath;
    if (params != null) {
        // 需要FormData类型数据作为参数
        const formData = new FormData();
        Object.entries(params).forEach(([key, value]) =&gt; {
            formData.append(key, value)
        })
        return http.post(path, formData)
    } else {
        // 不需要参数
        return http.post(path)
    }
}</code></pre>
    <p>
     这里演示的是发送
     <strong>
      post
     </strong>
     请求的方式，如果需要发送get请求，则可以类比使用http.get(path)。
    </p>
    <h3 id="%E7%A4%BA%E4%BE%8B" name="%E7%A4%BA%E4%BE%8B">
     示例
    </h3>
    <p>
     在项目中，一个实际的发送Axios的示例如下——
    </p>
    <pre><code class="language-javascript">const handleLogin = () =&gt; {
    formLogin.value.validate().then(async () =&gt; {
        try {
            // 向登录接口发送登录请求信息
            const res = await postApi('/login', data.user);
            if (res.code == 200) {
                ElMessage({
                    message: '登录成功',
                    type: 'success',
                    duration: 1000
                });
                setTimeout(async () =&gt; {
                    // 解构用户信息并存储
                    const { userId, userName, superAdmin, menuList, permissionCodeList } = res.data;
                    const user = { userId, userName, superAdmin };
                    await store.dispatch('setUserAction', { user, menuList, permissionCodeList });
                    // 跳转主页
                    if (menuList.length &gt; 0) {
                        router.push("/main");
                    }
                }, 1200);
            } else {
                // 返回登录失败信息
                ElMessage.error(res.info);
                // 重新获取验证码
                data.src = '/api/checkCode?' + (Math.random().toString().replace(".", ""));
                data.user.checkCode = "";
            }
        } catch (err) {
            ElMessage.error(err);
        }
    }).catch(err =&gt; {
        ElMessage.error(JSON.stringify(err));
    })
}</code></pre>
    <p>
     该示例结合了Element-Plus组件，简单实现了发送用户账户密码信息完成登录请求的功能。
    </p>
