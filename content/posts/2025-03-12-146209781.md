    <pre><code class="language-cpp">#include "bits/stdc++.h"
using namespace std;
const int N = 103;
const int inf = 0x3f3f;
int dir[4][2] = {-1, 0, 1, 0, 0, -1, 0, 1}; 
int n, m;
int g[103][103];
int dis[N][N];
int done[N][N];
struct node{
	int x, y, dis;
	node(int a, int b, int c){
		x = a, y = b, dis = c;
	}
	bool operator &lt; (const node &amp;t) const{
		return dis &gt; t.dis;
	}
};
struct n_node{
	int id, dis;
	n_node(int a, int b){
		id = a, dis = b;
	}
	bool operator &lt; (const n_node &amp; t) const{
		return dis &gt; t.dis;
	}
};
void bfs(int x, int y){
	priority_queue&lt;node&gt; q;
	dis[x][y] = 0;
	memset(dis, inf, sizeof(dis));
	q.push(node(x, y, 0));
	while(q.size()){
		node t = q.top();
		done[t.x][t.y] = 1;
		q.pop();
		int xx = t.x, yy = t.y, diss = t.dis;
		for(int i = 0; i &lt; 4; i ++){
			int x1 = xx + dir[i][0], y1 = yy + dir[i][1];
			if(x1 &lt; 1 || x1 &gt; n || y1 &lt; 1 || y1 &gt; m || g[x1][y1] == 0) continue;
			if(done[x1][y1] ) continue;
			if(dis[x1][y1] &gt; diss + 1) dis[x1][y1] = diss + 1,done[x1][y1] = 1, q.push(node(x1, y1, diss + 1));			
		}
	}
	
}
int main(){	
	cin&gt;&gt;n&gt;&gt;m;
	int x, y;
	for(int i = 1; i &lt;= n; i ++)
		for(int j = 1; j &lt;= m; j ++){
			cin&gt;&gt;g[i][j];
			if(g[i][j] == 2) x = i, y = j;
		}
	bfs(x, y);
	int k;
	cin&gt;&gt;k;
	priority_queue&lt;n_node&gt; qq;
	for(int i = 1; i &lt;= k; i ++){
		cin&gt;&gt;y&gt;&gt;x;
		if(!done[x][y]) continue;
		qq.push(n_node(i, dis[x][y]));
	}
	n_node t(n_node(-1, 0));
	int flag = 0;
	while(qq.size()){
		t = qq.top();
		qq.pop();
		if(!qq.size()) break;
		if(t.dis == qq.top().dis){
			while(qq.size() &amp;&amp; qq.top().dis == t.dis) qq.pop();
			if(qq.size() == 0) flag = 1;
		}
		else {
			break;
		} 
	}
	if(t.id != -1  &amp;&amp; !flag)
		cout&lt;&lt;t.id&lt;&lt;" "&lt;&lt;t.dis;
	else cout&lt;&lt;"No winner.";	
	
	return 0;
}</code></pre>
