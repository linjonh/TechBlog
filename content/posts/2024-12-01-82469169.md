    <h4>
     使用Vue.js编写一个很小的筛选搜索案例。
    </h4>
    <h4>
     根据姓名搜索，性别筛选。
    </h4>
    <h4>
     代码
    </h4>
    <pre class="has"><code class="language-html">&lt;!DOCTYPE html&gt;
&lt;html lang="en"&gt;
&lt;head&gt;
    &lt;meta charset="UTF-8"&gt;
    &lt;title&gt;筛选小案例&lt;/title&gt;
&lt;/head&gt;
&lt;body&gt;
&lt;div id="app"&gt;
    &lt;input type="text" v-model="inputName"&gt;
    &lt;button @click="filterByName = inputName"&gt;搜索&lt;/button&gt;
    &lt;br&gt;
    性别:
        &lt;input type="radio" value=""   name="gender" v-model="filterByGender"&gt;全部
        &lt;input type="radio" value="男" name="gender" v-model="filterByGender"&gt;男
        &lt;input type="radio" value="女" name="gender" v-model="filterByGender"&gt;女
    &lt;br&gt;
    &lt;br&gt;
    &lt;table&gt;
        &lt;thead&gt;
        &lt;th&gt;id&lt;/th&gt;
        &lt;th&gt;姓名&lt;/th&gt;
        &lt;th&gt;性别&lt;/th&gt;
        &lt;/thead&gt;
        &lt;tbody&gt;
        &lt;tr v-for="item in filterUsers"&gt;
            &lt;td&gt;{<!-- -->{item.id}}&lt;/td&gt;
            &lt;td&gt;{<!-- -->{item.name}}&lt;/td&gt;
            &lt;td&gt;{<!-- -->{item.gender}}&lt;/td&gt;
        &lt;/tr&gt;
        &lt;/tbody&gt;
    &lt;/table&gt;

&lt;/div&gt;
&lt;script src="https://cdn.jsdelivr.net/npm/vue/dist/vue.js"&gt;&lt;/script&gt;
&lt;script&gt;
    new Vue({
        el: '#app',
        data: {
            filterByName: '',
            filterByGender: '',
            inputName: '',
            users: [
                {
                    id: 1,
                    name: 'Jack',
                    gender: '男'
                },
                {
                    id: 2,
                    name: 'abc',
                    gender: '女'
                },
                {
                    id: 3,
                    name: 'vaxc',
                    gender: '男'
                },
                {
                    id: 4,
                    name: 'awe2',
                    gender: '女'
                },
                {
                    id: 5,
                    name: 'ave',
                    gender: '男'
                }
            ]
        },
        computed:{
            filterUsers(){
                return this.users.filter(u =&gt; u.name.includes(this.filterByName) &amp;&amp; u.gender.includes(this.filterByGender))
            }
        }
    })


&lt;/script&gt;
&lt;/body&gt;
&lt;/html&gt;</code></pre>
