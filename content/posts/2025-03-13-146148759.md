* t值即为需要的值，但t值在Promise对象中
* 分析整个方法，发现代码可修改为

```js
function _(e, t, a) {

    const o = m(e, t, a);
    s().call("$encode", [o, {
        suc: function (t) {
            window.result = t
        },
        err: function (e) {
            t(e)
        }
    }]);
    return window.result
}

```
