    <h2>
     思路和时间复杂度
    </h2>
    <ol>
     <li>
      思路：利用二分，确定区间是左闭右闭，然后根据大小进行二分
     </li>
     <li>
      时间复杂度：
      <img alt="O(logn)" class="mathcode" src="https://latex.csdn.net/eq?O%28logn%29">

    <h2>
     代码
    </h2>
    <pre><code class="language-cpp">class Solution {
public:
    bool isPerfectSquare(int num) {
        bool flag = false;
        if(num == 0 || num == 1) return true;
        long long left = 0, right = num;
        // 左闭右闭
        while(left &lt;= right){
            long long mid = left + (right - left) / 2;
            if(mid * mid == num){
                return true;
            }else if(mid * mid &gt; num){
                right = mid-1;
            }else{
                left = mid+1;
            }
        }
        return flag;
    }
};</code></pre>
