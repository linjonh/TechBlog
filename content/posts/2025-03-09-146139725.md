    <h3>
     一、Web安全核心防御机制
    </h3>
    <h4>
     （一）XSS攻击防御（跨站脚本攻击）
    </h4>
    <h5>
     1. 原理与分类
    </h5>
    <ul>
     <li>
      ​
      <strong>
       存储型XSS
      </strong>
      ：恶意脚本被持久化存储在服务端（如数据库）
     </li>
     <li>
      ​
      <strong>
       反射型XSS
      </strong>
      ：脚本通过URL参数或表单提交触发执行
     </li>
     <li>
      ​
      <strong>
       DOM型XSS
      </strong>
      ：通过修改DOM节点动态插入脚本
     </li>
    </ul>
    <h5>
     2. 防御方案
    </h5>
    <pre><code>// Express中间件：全局XSS防护
const xss = require('xss');
app.use((req, res, next) =&gt; {
  // 对所有请求参数进行过滤
  req.cleanedParams = xss(req.params);
  req.cleanedQuery = xss(req.query);
  req.cleanedBody = xss(req.body);
  next();
});

// React组件安全渲染示例
function SafeComponent({ userInput }) {
  // 使用dangerouslySetInnerHTML需谨慎
  return (
    &lt;div&gt;
      {/* 静态内容直接渲染 */}
      &lt;p&gt;{userInput}&lt;/p&gt;
      
      {/* 动态内容必须经过转义 */}
      &lt;div dangerouslySetInnerHTML={<!-- -->{ __html: escapeHtml(userInput) }} /&gt;
    &lt;/div&gt;
  );
}

// 自定义转义函数
function escapeHtml(str) {
  return str.replace(/[&amp;&lt;&gt;"']/g, (match) =&gt; ({
    '&amp;': '&amp;amp;',
    '&lt;': '&amp;lt;',
    '&gt;': '&amp;gt;',
    '"': '&amp;quot;',
    "'": '&amp;#039;'
  })[match]);
}</code></pre>

    <h4>
     （二）CSRF攻击防御（跨站请求伪造）
    </h4>
    <h5>
     1. 攻击流程剖析
    </h5>
    <ol>
     <li>
      用户登录受信任站点A
     </li>
     <li>
      网站B诱导用户访问包含恶意请求的页面
     </li>
     <li>
      浏览器携带A的Cookie自动发送请求
     </li>
     <li>
      服务端误认为请求合法执行操作
     </li>
    </ol>
    <h5>
     2. 防御策略
    </h5>
    <pre><code>// Express CSRF保护配置
const csrf = require('csurf');
const csrfProtection = csrf({ cookie: true });

app.use(csrfProtection);

// AJAX请求携带CSRF Token示例
fetch('/api/submit', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-CSRFToken': document.querySelector('meta[name="csrf-token"]').getAttribute('content')
  },
  body: JSON.stringify(formData)
});

// CSRF Token meta标签配置
&lt;meta name="csrf-token" content="&lt;%= csrfToken %&gt;"&gt;</code></pre>

     二、进阶防护策略
    </h3>
    <h4>
     （一）CORS安全配置
    </h4>
    <pre><code># Nginx CORS安全配置示例
server {
  location /api {
    add_header 'Access-Control-Allow-Origin' 'https://trusted-domain.com';
    add_header 'Access-Control-Allow-Methods' 'GET,POST,PUT,DELETE';
    add_header 'Access-Control-Allow-Headers' 'Origin,X-Requested-With,Content-Type,Authorization';
    
    # 限制预检请求频率
    limit_req_zone $binary_remote_addr zone=api_limit:10m rate=1r/s;
    
    if ($request_method = 'OPTIONS') {
      add_header 'Access-Control-Max-Age' 1728000;
      add_header 'Content-Length' 0;
      return 204;
    }
  }
}</code></pre>

     三、日常开发最佳实践

    <h4>
     （一）密码安全规范
    </h4>
    <pre><code>// 密码哈希存储示例（Node.js）
const bcrypt = require('bcrypt');
const saltRounds = 12;

async function hashPassword(password) {
  // 生成盐值
  const salt = await bcrypt.genSalt(saltRounds);
  // 加密密码
  return await bcrypt.hash(password, salt);
}

// 密码验证
async function comparePassword(userPassword, hashedPassword) {
  return await bcrypt.compare(userPassword, hashedPassword);
}</code></pre>
    <h4>
     （二）文件上传安全
    </h4>
    <pre><code># Flask文件上传验证中间件
from werkzeug.utils import secure_filename
import os

def allowed_file(filename):
    ALLOWED_EXTENSIONS = {'png', 'jpg', 'jpeg', 'gif'}
    return '.' in filename and filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

@app.route('/upload', methods=['POST'])
def upload_file():
    file = request.files['file']
    if file and allowed_file(file.filename):
        filename = secure_filename(file.filename)
        file.save(os.path.join(app.config['UPLOAD_FOLDER'], filename))
    else:
        return 'Invalid file type', 400</code></pre>
