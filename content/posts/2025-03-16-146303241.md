# 蓝桥杯学习-11栈
## 11栈
先进后出
## 例题–蓝桥19877
用数组来设置栈
1.向栈顶插入元素--top位置标记元素
2.删除栈顶元素--top指针减减
3.输出栈顶元素--输出top位置元素
### 使用arraylist
import java.util.ArrayList;
import java.util.Scanner;
public class Main {
public static void main(String[] args) {
Scanner sc=new Scanner(System.in);
ArrayList array=new ArrayList<>();
int m=sc.nextInt();
for (int i = 0; i < m; i++){
String str=sc.next();
switch (str){
case "push":
int x=sc.nextInt();
array.add(x);
break;
case "pop":
if (array.size()==0){
break;
}else {
array.remove(array.size()-1);
}
break;
case "empty":
if (array.size()==0){
System.out.println("YES");
}else {
System.out.println("NO");
}
break;
case "query":
if (array.size()==0){
System.out.println("empty");
}else {
System.out.println(array.get(array.size()-1));
}
break;
default:
System.out.println("error");
}
}
}
}
