      <h3 id="HTML5">
       HTML5
      </h3>
     </li>
    </ul>
    <p>
     用HTML5编写网页简单的框架，导入css和js文件。
    </p>
    <pre><code class="language-html">&lt;!DOCTYPE html&gt;
&lt;html lang="en"&gt;
&lt;head&gt;
    &lt;meta charset="UTF-8"&gt;
    &lt;title&gt;播放器&lt;/title&gt;
    &lt;!--引入css文件--&gt;
    &lt;link rel="stylesheet" href="xin.css"&gt;
    &lt;script type="text/javascript"  src="yi.js"&gt;  //src:引入js文件的路径
    &lt;/script&gt;

&lt;/head&gt;
&lt;!--设置图片路径--&gt;
&lt;body&gt;
&lt;div class="videoContainer" align="center"&gt;
    &lt;video id="videoPlayer" ontimeupdate="progressUpdate()" 
    width="640" height="440" controls="controls"&gt;
        &lt;source src="图片/lv_0_20220609221451.mp4"  type="video/mp4" &gt;
        &lt;source src="图片/mixkit-beach-front-with-children-playing-2176.webm"             
         type="video/webm" &gt;
    &lt;/video&gt;
&lt;/div&gt;
&lt;div class="barContainer" align="center"&gt;
    &lt;div id="durationBar"&gt;
        &lt;div id="positionBar"&gt;&lt;span id="displayStatus"&gt;进度条.&lt;/span&gt;&lt;/div&gt;
    &lt;/div&gt;
&lt;/div&gt;
&lt;div class="top" align="center"&gt;
    &lt;button  class="button-3d-1" onclick="play()"&gt;播放&lt;/button&gt;
    &lt;button onclick="pause()"&gt;暂停&lt;/button&gt;
    &lt;button onclick="stop()"&gt;重新开始&lt;/button&gt;
&lt;/div&gt;
&lt;/body&gt;
&lt;/html&gt;</code></pre>

     <li>
      <h3 id="JavaScript">
       JavaScript
      </h3>
     </li>
    </ul>
    <p>
     JavaScript是一个脚本语言，能直接在web上运行，用Js来给按钮添加动作。
    </p>
    <pre><code class="language-javascript">var video;
var display;
window.onload = function(){
    video = document.getElementById("videoPlayer");
    display = document.getElementById("displayStatus");
}
// 定义播放按钮
function play(){
    video.play();
}
// 定义暂停按钮
function pause() {
    video.pause();
}
// 定义重新开始按钮
function stop() {
    video.pause();
    video.currentTime=0;
}
// 设置进度条
function progressUpdate() {
    var positionBar = document.getElementById("positionBar");
    positionBar.style.width = (video.currentTime / video.duration * 100) +"&amp;nbsp";
    display.innerHTML = (Math.round(video.currentTime *100)/100) + "能量值";
}</code></pre>
