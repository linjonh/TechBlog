	int head=1,tail=2;

	while(head&lt;tail)
	{
		head++;
		for(int k=0;k&lt;12;k++)
		{
			int tx=a[head].x+nextp[k][0];
			int ty=a[head].y+nextp[k][1];
			if(g[tx][ty]!=0||tx&lt;1||ty&lt;1)
				continue;
			tail++;
			g[tx][ty]=g[a[head].x][a[head].y]+1;
			if(tx==1&amp;&amp;ty==1)
			{
				cout&lt;&lt;g[tx][ty]&lt;&lt;endl;
				break;
				}	
			a[tail].x=tx;
			a[tail].y=ty;
		}
	}
}
int main()
{
	cin&gt;&gt;n&gt;&gt;m&gt;&gt;x&gt;&gt;y;
	bfs(n,m);
	bfs(x,y);
	return 0;
