     1 选择、转到Select、 Go To

    <p>
     <strong>
      Description
     </strong>
    </p>
    <p>
     <strong>
      描述
     </strong>
    </p>
    <p>
     <strong>
      VBA Code
     </strong>
    </p>
    <p>
     <strong>
      Backspace
     </strong>
    </p>
    <p>
     <strong>
      退格
     </strong>
    </p>
    <p>
     <strong>
      Selection.TypeBackspace
     </strong>
    </p>
    <p>
     <strong>
      Select Entire Document
     </strong>
    </p>
    <p>
     <strong>
     </strong>
    </p>
    <p>
     <strong>
      选择整个文档
     </strong>
    </p>
    <p>
     <strong>
     </strong>
    </p>
    <p>
     <strong>
      Selection.HomeKey Unit:=wdStory
     </strong>
    </p>
    <p>
     <strong>
      Selection.Extend
     </strong>
    </p>
    <p>
     <strong>
      Copy
     </strong>

     <strong>

     </strong>
    </p>
    <p>
     <strong>
      Selection.Copy
     </strong>
    </p>
    <p>
     <strong>
      Delete
     </strong>
    </p>
    <p>
     <strong>
      删除
     </strong>
    </p>
    <p>
     <strong>
      Selection.Delete Unit:=wdCharacter, Count:=1
     </strong>
    </p>
    <p>
     <strong>
      Insert After
     </strong>
    </p>
    <p>
     <strong>
      插入之后
     </strong>
    </p>

      Beginning of Line
     </strong>
    </p>
    <p>
     <strong>
      行首
     </strong>
    </p>
    <p>
     <strong>
      Selection.HomeKey Unit:=wdLine
     </strong>
    </p>
    <p>
     <strong>
      End of Line
     </strong>
    </p>
    <p>
     <strong>
      行结束
     </strong>
    </p>
    <p>
     <strong>
      Selection.EndKey Unit:=wdLine
     </strong>
    </p>
    <p>
     <strong>
      Paste
     </strong>
    </p>
    <p>
     <strong>
      粘贴
     </strong>
    </p>
    <p>
     <strong>
      Selection.Paste
     </strong>
    </p>
    <p>
     <strong>
      Select All
     </strong>
    </p>
    <p>
     <strong>
      全选
     </strong>
    </p>
    <p>
     <strong>
      Selection.WholeStory
     </strong>
    </p>
    <p>
     <strong>
      Select Entire Line
     </strong>
    </p>
    <p>
     <strong>
      选择整行
     </strong>
    </p>
    <p>
     <strong>
      Selection.EndKey Unit:=wdLine, Extend:=wdExtend

    </p>
    <p>
     <strong>
      Move Up Paragraph

    </p>
    <p>

    </p>
    <p>
     <strong>
      Selection.MoveUp Unit:=wdParagraph, Count:=1

    </p>
    <p>
     <strong>
      Move Right One Character
     </strong>
    </p>
    <p>
     <strong>
      向右移动一个字符
     </strong>
    </p>
    <p>
