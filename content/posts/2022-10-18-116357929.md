```
var ids = JSON.stringify(checkStatus.data); //该数组被转化成了字符串
$.post({
   url: '/multilinedel',  //要传递过去的路由名称
   data: {
       "ids":ids
         },
   traditional:true,    //这个必须有
   success: function (data) {
        //do nothing
                         }
});
```

```
data = request.form['ids']  # 此时data为一个字符串
```
