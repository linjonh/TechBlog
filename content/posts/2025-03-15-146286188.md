    <pre><code class="language-javascript">// 防止异步延迟
const Noop = () =&gt; { }
const lhl = (resDayaCallback) =&gt; {
//这边会重新赋值的
    let condd = () =&gt; { }
    return (...args) =&gt; {
        return new Promise((resolve, reject) =&gt; {
            // 每次进来的时候都执行一次Noop()，防止异步延迟
            condd();
            condd = () =&gt; {
                resolve = reject = Noop
            }
            resDayaCallback(...args).then(
                (res) =&gt; resolve(res),
                (err) =&gt; reject(err)
            )

        })
    }

}</code></pre>
