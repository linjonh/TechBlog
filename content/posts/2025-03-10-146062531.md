    <h4>
     1.
     <code>
      http
     </code>
     模块概述
    </h4>
    <p>
     <code>
      http
     </code>
     模块是 Node.js 内置的核心模块之一，它用于创建 HTTP 服务器和客户端，支持处理 HTTP 请求和响应。通过
     <code>
      http
     </code>
     模块，我们可以搭建 Web 服务器，接收浏览器或其他客户端的请求，并返回相应的数据。
    </p>
    <h4>
     2.
     <code>
      http
     </code>
     模块的作用
    </h4>
    <ul>
     <li>
      <strong>
       创建 HTTP 服务器
      </strong>
      ：监听指定端口，处理客户端请求，并返回 HTTP 响应。
     </li>
     <li>
      <strong>
       创建 HTTP 客户端
      </strong>
      ：向其他服务器发送 HTTP 请求并处理响应数据。
     </li>
     <li>
      <strong>
       处理 HTTP 请求
      </strong>
      ：解析请求路径、请求方法、请求头和请求体。
     </li>
     <li>
      <strong>
       发送 HTTP 响应
      </strong>
      ：返回 HTML、JSON、文件等数据给客户端。
     </li>
    </ul>
    <hr/>
    <h4>
     3.
     <code>
      http
     </code>
     服务器代码示例
    </h4>
    <p>
     <strong>
      文件名：server.js
     </strong>
     <br/>
     <strong>
      说明
     </strong>
     ：该代码创建了一个简单的 HTTP 服务器，监听
     <code>
      3000
     </code>
     端口，并根据不同的 URL 返回不同的内容。
    </p>
    <pre><code class="language-javascript">// 引入 http 模块
const http = require('http');

// 创建 HTTP 服务器
const server = http.createServer((req, res) =&gt; {
    // 设置响应头，指定返回内容类型
    res.writeHead(200, { 'Content-Type': 'text/plain; charset=utf-8' });

    // 解析请求 URL
    if (req.url === '/') {
        res.end('欢迎访问我的 Node.js 服务器！');
    } else if (req.url === '/about') {
        res.end('这是关于页面');
    } else {
        res.writeHead(404, { 'Content-Type': 'text/plain; charset=utf-8' });
        res.end('404 页面未找到');
    }
});

// 监听端口号 3000
server.listen(3000, () =&gt; {
    console.log('服务器运行在 http://localhost:3000');
});
</code></pre>
    <p>
     运行代码
    </p>
    <ol>
     <li>
      <strong>
       确保已安装 Node.js
      </strong>
     </li>
     <li>
      <strong>
       执行命令启动服务器
      </strong>
      ：
      <pre><code class="language-bash">node server.js
</code></pre>
     </li>
     <li>
      <strong>
       打开浏览器访问以下地址
      </strong>
      ：
      <ul>
       <li>
        <code>
         http://localhost:3000/
        </code>
        → 显示 "欢迎访问我的 Node.js 服务器！"
       </li>
       <li>
        <code>
         http://localhost:3000/about
        </code>
        → 显示 "这是关于页面"
       </li>
       <li>
        <code>
         http://localhost:3000/other
        </code>
        → 显示 "404 页面未找到"
       </li>
      </ul>
     </li>
    </ol>
    <hr/>
    <h4>
     4.
     <code>
      http
     </code>
     客户端代码示例
    </h4>
    <p>
     <strong>
      文件名：client.js
     </strong>
     <br/>
     <strong>
      说明
     </strong>
     ：该代码使用
     <code>
      http.request
     </code>
     方法向服务器发送 GET 请求并接收响应数据。
    </p>
    <pre><code class="language-javascript">// 引入 http 模块
const http = require('http');

// 发送 GET 请求
http.get('http://localhost:3000/', (res) =&gt; {
    let data = '';

    // 监听数据流
    res.on('data', chunk =&gt; {
        data += chunk;
    });

    // 响应结束后打印数据
    res.on('end', () =&gt; {
        console.log('服务器响应:', data);
    });

}).on('error', (err) =&gt; {
    console.log('请求出错:', err.message);
});
</code></pre>
    <p>
     运行代码
    </p>
    <ol>
     <li>
      <strong>
       确保服务器已运行 (
       <code>
        node server.js
       </code>
       )
      </strong>
     </li>
     <li>
      <strong>
       执行命令发送请求
      </strong>
      ：
      <pre><code class="language-bash">node client.js
</code></pre>
     </li>
     <li>
      <strong>
       终端输出
      </strong>
      ：
      <pre><code class="language-bash">服务器响应: 欢迎访问我的 Node.js 服务器！
</code></pre>
     </li>
    </ol>
    <hr/>
    <h4>
     5. 总结
    </h4>
    <ul>
     <li>
      <code>
       http
      </code>
      模块是 Node.js 处理 HTTP 请求和响应的核心模块。
     </li>
     <li>
      <code>
       http.createServer()
      </code>
      方法用于创建 HTTP 服务器。
     </li>
     <li>
      <code>
       server.listen(port)
      </code>
      让服务器监听指定端口。
     </li>
     <li>
      <code>
       http.get()
      </code>
      或
      <code>
       http.request()
      </code>
      可用于发送 HTTP 请求。
     </li>
    </ul>
    <p>
     通过
     <code>
      http
     </code>
     模块，我们可以轻松地在 Node.js 中创建 Web 服务器，并与前端或其他服务进行数据交互。
    </p>
