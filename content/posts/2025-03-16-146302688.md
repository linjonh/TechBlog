   <div class="htmledit_views" id="content_views">
    <p>
     如何在
     <code>
      ElementUI
     </code>
     的表格组件
     <code>
      Table
     </code>
     中插入图片缩略图，通过鼠标悬停显示大图？介绍以下2种方式：
    </p>
    <p>
     方法1：直接在模板元素中插入
    </p>
    <pre><code class="language-html">&lt;template&gt;
  &lt;el-table :data="tableData"&gt;
    &lt;el-table-column label="图片"&gt;
      &lt;template slot-scope="scope"&gt;
        &lt;el-popover placement="right" trigger="hover"&gt;
          &lt;img :src="scope.row.picture" style="max-width: 500px; max-height: 500px;"&gt;
          &lt;img slot="reference" :src="scope.row.thumbnail" style="width: 50px; height: 50px; vertical-align: middle;"&gt;
        &lt;/el-popover&gt;
      &lt;/template&gt;
    &lt;/el-table-column&gt;
  &lt;/el-table&gt;
&lt;/template&gt;

&lt;script&gt;
  export default {
    data() {
      return {
        tableData: [{
          thumbnail: 'https://www.lervor.com/thumbnail.jpg',
          picture: 'https://www.lervor.com/picture.jpg'
        }]
      }
    }
  }
&lt;/script&gt;</code></pre>
    <p>
     方法2：通过表格列属性格式化
    </p>
    <pre><code class="language-html">&lt;template&gt;
  &lt;el-table :data="tableData"&gt;
    &lt;el-table-column
      v-for="(column, index) in columns"
      :key="index"
      :label="column.label"
      :formatter="column.formatter"
    /&gt;
  &lt;/el-table&gt;
&lt;/template&gt;

&lt;script&gt;
  export default {
    data() {
      return {
        tableData: [{
          thumbnail: 'https://www.lervor.com/thumbnail.jpg',
          picture: 'https://www.lervor.com/picture.jpg'
        }],
        columns: [
          {
            label: '图片',
            formatter: row =&gt; {
              return this.$createElement('el-popover', {
                attrs: {
                  trigger: 'hover',
                  placement: 'right'
                }
              }, [
                this.$createElement('img', {
                  attrs: {
                    style: 'max-width: 500px; max-height: 500px;',
                    src: `${row.picture}`
                  }
                }),
                this.$createElement('img', {
                  attrs: {
                    style: 'width: 50px; height: 50px; vertical-align: middle;',
                    src: `${row.thumbnail}`
                  },
                  slot: 'reference'
                })
              ])
            }
          }
        ]
      }
    }
  }
&lt;/script&gt;</code></pre>
