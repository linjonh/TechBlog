    <pre><code class="hljs">//绝对值
//输入一个整数x，请你编写一个函数，int abs(int x)，输出x的绝对值,

//数据范围 - 100 &lt; x &lt; 100
//输入样例: -3
//输出样例:  3

int abs1(int x) {
	if (x &lt; 0) return -x;
	return x;
}

int main() {

	int x;
	cin &gt;&gt; x;

	int ret = abs1(x);
	cout &lt;&lt; "绝对值为: " &lt;&lt; ret &lt;&lt; endl;

	return 0;
}</code></pre>

    <p>
     本题需要求出区间[l,r]内所有整数的和，很简单，代码如下：
    </p>
    <pre><code class="hljs">//区间求和
//输入两个整数l和r，请你编写一个函数，int sum(intl,intr)，计算并输出区间1,内所有整数的和。
//输入格式共一行，包含两个整数 l 和r。
//输出共一行，包含一个整数，表示所求的和。
//数据范围 1≤l≤r &lt; 1000

int sum(int l, int r) {
	int ret = 0;
	for (int i = l; i &lt;= r; i++) {
		ret += i;
	}
	return ret;
}

int main() {
	int l, r;
	cin &gt;&gt; l &gt;&gt; r;

	int ret = sum(l, r);
	cout &lt;&lt; "所求的和为: " &lt;&lt; ret &lt;&lt; endl;

	return 0;

    <h5>
     第4题  最小公倍数
    </h5>
    <p>
     <img alt="" height="774" src="https://i-blog.csdnimg.cn/direct/3e7f1929d689489abfd683e2feccd72d.png" width="1152"/>
    </p>
    <p>
     求最小公倍数，就是找出一个数能同时整除a和b，并且这个数在所有公倍数中是最小值
    </p>
    <p>
     代码如下：
    </p>
    <pre><code class="hljs">//最小公倍数
//输入两个整数a和b，请你编写一个函数，int lcm(int a,int b)，计算并输出a和b的最小公倍数。
//输入共一行，包含两个整数a和b。
//输出共一行，包含一个整数，表示a和b的最小公倍数。
//数据范围 1 ≤ a, b≤ 1000
//输入样例:
//6  8
//输出样例 :
//24

int lcm(int a, int b) {
	for (int i = 1; i &lt;= a * b; i++) {     //最小公倍数不可能超过2个数相乘
		if (i % a == 0 &amp;&amp; i % b == 0) {
			return i;
		}
	}
	return -1; //一定不会执行
}

int main() {
	int a, b;
	cin &gt;&gt; a &gt;&gt; b;

	int ret = lcm(a, b);
	cout &lt;&lt; "最小公倍数为: " &lt;&lt; ret &lt;&lt; endl;

	return 0;
}</code></pre>

	int b[100];

							//m:b数组的长度
							//size:前size个数字

	for (int i = 0; i &lt; m; i++) {
		cout &lt;&lt; b[i] &lt;&lt; " ";

    <h5>
     第7题  数组翻转
    </h5>
    <p>
     <img alt="" height="925" src="https://i-blog.csdnimg.cn/direct/18dfd0717f71402fbc4abd4249c33088.png" width="1133"/>
    </p>
    <p>
     这道题，考的是翻转数组。其实就是双指针。定义变量 i 和变量 j，变量 i 指向下标为0的位置，变量 j 指向 size-1 的位置，交换它们之间的值，交换完毕后，i++，j--，当变量 i 和变量 j 错过时，循环结束。
    </p>
    <p>
     代码如下：
    </p>
    <pre><code class="hljs">//数组翻转
//给定一个长度为n的数组a和一个整数size，请你编写一个函数，
//void reverse(int a[],int size)
//实现将数组a中的前size个数翻转。
//输出翻转后的数组a。
//输入第一行包含两个整数n和size。
//第二行包含n个整数，表示数组a。
//输出共一行，包含n个整数，表示翻转后的数组a。
//数据范围 1≤ size &lt; n &lt; 1000

const int N = 1000;

void reverse(int a[], int size) {
	for (int i = 0, j = size - 1; i &lt; j; i++, j--) {
		swap(a[i], a[j]);
	}
}


int main() {
	int a[N];

	int n, size;	 //长度为n的数组a
	cin &gt;&gt; n &gt;&gt; size;

	for (int i = 0; i &lt; n; i++) {
		cin &gt;&gt; a[i];
	}

	reverse(a, size);

	for (int i = 0; i &lt; n; i++) {
		cout &lt;&lt; a[i] &lt;&lt; " ";
	}
	cout &lt;&lt; endl;

	return 0;
}</code></pre>

    <h5>
     第8题  数组去重
    </h5>
    <p>
     <img alt="" height="899" src="https://i-blog.csdnimg.cn/direct/848e5536ef2743188f653fd91c72953d.png" width="1120"/>
    </p>
    <p>
     去重之后的数组长度  = 前size个数中的不同数的个数 + 剩余的数（n-size)
    </p>
    <p>
     <img alt="" height="372" src="https://i-blog.csdnimg.cn/direct/03cbf5e589a04da5a595ff07ffd9e2d4.png" width="1135"/>
    </p>
    <blockquote>
     <p>
      数组去重，顾名思义，去掉重复出现的数。
     </p>
     <p>
      ①我们可以定义2个for循环，外层是下标 i 从0~size-1，也就是所有数，内层下标 j 从当前的下标 i 的下一个位置开始，一直到size-1结束，寻找后面的数是否与 a[i] 的值相同。
     </p>
     <p>
     </p>
     <p>
      ②定义bool型变量flag，假设每个数都只出现1次，flag初始值为true。当出现与 a[i] 的值相同的情况，flag的值为false。跳出当前循环，不再执行后续代码。
     </p>
     <p>
     </p>
     <p>
      ③如果遍历完整个数组，发现没有和 a[i] 的值相同的情况，flag的值仍然为true，那么count++; （count变量表示前size个数中的不同数的个数）
     </p>
     <p>
     </p>
     <p>
      ④去重之后的数组总长度 = 前size个数中的不同数的个数（count）+ 剩余的数（n-size)
     </p>
    </blockquote>
    <p>
     代码如下：
    </p>
    <pre><code class="hljs">//数组去重
//给定一个长度为n的数组a，请你编写一个函数:
//int unique(int a[]，int size);//返回数组前size个数中的不同数的个数。
//输入第一行包含两个整数n和size,
//第二行包含n个整数，表示数组a。
//输出共一行，包含一个整数表示去重完成之后的数组长度。
//数据范围 1 &lt; size &lt; n &lt; 1000

int unique(int a[], int size) {
	int count = 0;//前size个数中的不同数的个数

	for (int i = 0; i &lt; size; i++) {
		bool flag = true;//假设每个数只出现1次
		for (int j = i + 1; j &lt; size; j++) {
			if (a[j] == a[i]) //如果后面的数和当前的数重复
			{
				flag = false; //表明这个数不止出现1次
				break;		  //不再执行后面的循环
			}
		}
		if (flag) count++;    //如果这个数唯一,那么count++;
	}

	return count;
}

int main() {
	int a[1000];
	int n, size;
	cin &gt;&gt; n &gt;&gt; size;

	for (int i = 0; i &lt; n; i++) {
		cin &gt;&gt; a[i];
	}

	int ret = unique(a, size);
	cout &lt;&lt; "去重完成之后的数组长度为: " &lt;&lt; ret + (n - size) &lt;&lt; endl;

	return 0;
}</code></pre>

    <h5>
     第9题  数组排序
    </h5>
    <p>
     <img alt="" height="970" src="https://i-blog.csdnimg.cn/direct/bec09ac79b2d4aeba26798f9a0b759ce.png" width="1151"/>
    </p>
    <p>
     这里考一个排序算法，我们就选择排序吧~
    </p>
    <p>
     代码如下：
    </p>
    <pre><code class="hljs">//数组排序
//给定一个长度为n的数组a以及两个整数l和r，请你编写一个函数，
// void sort(int a[],int l,int r)，将a[l]~a[r]从小到大排序。
//输出排好序的数组a。
//输入第一行包含三个整数n，l，r。
//第二行包含n个整数，表示数组a。
//输出共一行，包含n个整数，表示排序完成后的数组a。
//数据范围 0≤l≤r &lt; n≤1000

//选择排序
//将a[l]~a[r]从小到大排序
void sort(int a[], int l, int r) {
	for (int i = l; i &lt;= r - 1; i++) {
		for (int j = i + 1; j &lt;= r; j++) {
			if (a[i] &gt; a[j]) swap(a[i], a[j]);
		}
	}
}

int main() {
	int a[1000];

	int n, l, r;       //长度为n的数组a以及两个整数l和r
	cin &gt;&gt; n &gt;&gt; l &gt;&gt; r;

	for (int i = 0; i &lt; n; i++) {
		cin &gt;&gt; a[i];
	}

	sort(a, l, r);

	for (int i = 0; i &lt; n; i++) {
		cout &lt;&lt; a[i] &lt;&lt; " ";
	}
	cout &lt;&lt; endl;

	return 0;
}</code></pre>

    <pre><code class="hljs">//跳台阶
//1个楼梯共有n级台阶，每次可以走一级或者两级，问从第0级台阶走到第n级台阶一共有多少种方案。
//输入共一行，包含一个整数n。
//输出共一行，包含一个整数，表示方案数。
//数据范围 1 &lt; n &lt; 15

int n1;
int ans1;

void Stairs(int k) {
	if (k == n1) ans1++;
	else if (k &lt; n1) {
		Stairs(k + 1);
		Stairs(k + 2);
	}
}

int main() {
	cin &gt;&gt; n1;
	Stairs(0);  //每次k的初始值均为0
	cout &lt;&lt; ans1 &lt;&lt; endl;

	return 0;
}</code></pre>

    <pre><code class="hljs">//走方格
//给定一个n*m的方格阵，沿着方格的边线走，从左上角(0,0)开始，
//每次只能往右或者往下走一个单位距离，问走到右下角(n, m)一共有多少种不同的走法。
//输入共一行，包含两个整数n和m。
//输出共一行，包含一个整数，表示走法数量。
//数据范围  1 &lt; n, m &lt; 10

int ans;
int n, m;

void dfs(int x, int y) {
	if (x == n &amp;&amp; y == m) ans++;
	else {
		if (y &lt; m) dfs(x, y + 1);
		if (x &lt; n) dfs(x + 1, y);
	}
}

int main() {
	cin &gt;&gt; n &gt;&gt; m;

	dfs(0, 0);
	cout &lt;&lt; ans &lt;&lt; endl;

	return 0;
}</code></pre>
