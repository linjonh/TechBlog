    <pre><code class="language-python">extends Node2D

# 用于累加 delta 的变量
var elapsed_time = 0
# 设定计时周期，单位为秒
var interval = 3

func _process(delta):
    # 累加 delta 到 elapsed_time
    elapsed_time += delta
    # 检查是否达到了设定的时间间隔
    if elapsed_time &gt;= interval:
        # 执行每 3 秒要做的逻辑
        print("每 3 秒执行一次的逻辑被触发")
        # 重置 elapsed_time，以便开始下一个计时周期
        elapsed_time = 0</code></pre>
