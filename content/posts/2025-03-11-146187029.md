    <h2>
     <a id="_1">
     </a>
     思路
    </h2>
    <h3>
     <a id="_2">
     </a>
     二分查找算法
    </h3>
    <h3>
     <a id="_3">
     </a>
     注意二段性
    </h3>
    <p>
     两段性为peak前arr[mid] &lt; arr[mid + 1]
     <br/>
     peak后arr[mid] &gt;arr[mid + 1]
    </p>
    <pre><code class="prism language-c++">#define _CRT_SECURE_NO_WARNINGS 1
#include&lt;vector&gt;
using namespace std;
class Solution {
public:
    int peakIndexInMountainArray(vector&lt;int&gt;&amp; arr) {
        int left = 0, right = arr.size() - 1, mid = 0;    //置二分查找左右指针
        while (left &lt; right)
        {
    mid = left + (right - left) / 2;
    if (arr[mid] &lt; arr[mid + 1])left = mid + 1; //两段性为peak前arr[mid] &lt; arr[mid + 1]
    else right = mid;        //peak后arr[mid] &gt;arr[mid + 1]
           
        }
        return left;
    }
};
</code></pre>
