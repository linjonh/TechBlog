	function getCurDate() {
		var date = new Date()
		var y = date.getFullYear();
		var m = date.getMonth() + 1;
		var d = date.getDate();
		var h = date.getHours();
		var min = date.getMinutes();
		var s = date.getSeconds();
		s = s<10 ? "0" + s : s;
