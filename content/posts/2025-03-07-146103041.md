     我们可以借助fmt函数来将一个整数以不同进制形式展示。
    </h4>
    <p>
     <span style="color:#fe2c24">
      <strong>
       占位符%b表示二进制
      </strong>
     </span>
    </p>
    <pre><code class="language-Go">package main
 
import "fmt"
 
func main(){
	// 十进制
	var a int = 10
	fmt.Printf("%d \n", a)  // 10
	fmt.Printf("%b \n", a)  // 1010  占位符%b表示二进制
 
	// 八进制  以0开头
	var b int = 077
	fmt.Printf("%o \n", b)  // 77
 
	// 十六进制  以0x开头
	var c int = 0xff
	fmt.Printf("%x \n", c)  // ff
	fmt.Printf("%X \n", c)  // FF
}
</code></pre>
