     int16_t Num;            //定义待被旋转编码器调节的变量
    </p>
    <p>
     int main(void)
     <br/>
     {
     <!-- -->
     <br/>
     /*模块初始化*/
     <br/>
     OLED_Init();        //OLED初始化
     <br/>
     Encoder_Init();        //旋转编码器初始化
     <br/>
     <br/>
     /*显示静态字符串*/
     <br/>
     OLED_ShowString(1, 1, "Num:");            //1行1列显示字符串Num:
     <br/>
     <br/>
     while (1)
     <br/>
     {
     <!-- -->
     <br/>
     Num += Encoder_Get();                //获取自上此调用此函数后，旋转编码器的增量值，并将增量值加到Num上
     <br/>
     OLED_ShowSignedNum(1, 5, Num, 5);    //显示Num
     <br/>
     }
     <br/>
     }
