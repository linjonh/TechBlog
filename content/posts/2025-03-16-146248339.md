    <h4 id="%E9%A2%98%E7%9B%AE%E8%A7%A3%E6%9E%90" name="%E9%A2%98%E7%9B%AE%E8%A7%A3%E6%9E%90">
     题目解析
    </h4>
    <p>
     <img alt="" height="1146" src="https://i-blog.csdnimg.cn/direct/d8f6556e595341b588ef552f23841816.png" width="966"/>

    <p>
     题意很容易理解
    </p>
    <h4 id="%E4%BB%A3%E7%A0%81%E5%8E%9F%E7%90%86" name="%E4%BB%A3%E7%A0%81%E5%8E%9F%E7%90%86">
     代码原理
    </h4>
    <p>
     法一：
    </p>
    <p>
     先将输入的数字使用sort进行一次排序，再遍历数组，依次与最大值比较，大于最大值则count+1
    </p>
    <p>
     法二：
    </p>
    <p>
     直接遍历数组，与最大值进行比较大于最大值则count+1
    </p>
    <p>
     法一是博主的第一个想法，但事后博主与同学交流后发现好像不需要排序也可以搞定
    </p>
    <h4 id="%E4%BB%A3%E7%A0%81%E7%BC%96%E5%86%99" name="%E4%BB%A3%E7%A0%81%E7%BC%96%E5%86%99">
     代码编写
    </h4>
    <p>
     法一：
    </p>
    <p>
     #include&lt;iostream&gt;
     <br/>
     #include&lt;bits/stdc++.h&gt;
     <br/>
     using namespace std;
     <br/>
     typedef long long ll;
    </p>
    <p>
     const ll N = 1000;
     <br/>
     ll a[N];
     <br/>
     int main()
     <br/>
     {
     <!-- -->
     <br/>
     int n = 0;
     <br/>
     cin &gt;&gt; n;
     <br/>
     for(ll i = 1; i &lt;= n; i++)
     <br/>
     {
     <!-- -->
     <br/>
     cin &gt;&gt; a[i];
     <br/>
     }
     <br/>
     sort(a + 1, a + 1 + n);
     <br/>
     ll max_val = -0x3f3f3f3f, sum = 0;
     <br/>
     for(ll i = 1; i &lt;= n; i++)
     <br/>
     {
     <!-- -->
     <br/>
     if(max_val &lt; a[i]) {
     <!-- -->
     <br/>
     sum++;
     <br/>
     max_val = a[i];
     <br/>
     }
     <br/>
     }
     <br/>
     cout &lt;&lt; sum;
     <br/>
     return 0;
     <br/>
     }
    </p>
