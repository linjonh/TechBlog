     代码一，对应力扣二叉搜索树中的检索，代码见下
    </p>
    <pre><code class="language-cpp">/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    TreeNode* searchBST(TreeNode* root, int val) {
        if(root == NULL){
            return NULL;
        }
        if(root-&gt;val &lt; val){
            return searchBST(root-&gt;right, val);
        }else if(root-&gt;val &gt; val){
            return searchBST(root-&gt;left, val);
        }else{
            return root;
        }
    }
};</code></pre>

     代码二，对应力扣 二叉搜索树的范围和，代码见下
    </p>
    <pre><code class="language-cpp">/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int rangeSumBST(TreeNode* root, int low, int high) {
        if(root == NULL){
            return 0;
        }
        int sum = 0;
        if(root-&gt;val &gt;= low &amp;&amp; root-&gt;val &lt;= high){
            sum += root-&gt;val;
        }
        sum += rangeSumBST(root-&gt;left, low, high);
        sum += rangeSumBST(root-&gt;right, low, high);
        return sum;
    }
};</code></pre>

     代码三，将有序数组转化为二叉搜索树，代码见下：
    </p>
    <pre><code class="language-cpp">/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    TreeNode* sortArray(vector&lt;int&gt;&amp; nums, int l, int r){
        if(l &gt; r){
            return NULL;
        }
        int mid = (l + r)/2;
        TreeNode* node = new TreeNode(nums[mid]);
        node-&gt;left = sortArray(nums, l, mid-1);
        node-&gt;right = sortArray(nums, mid+1, r);
        return node;

    }
public:
    TreeNode* sortedArrayToBST(vector&lt;int&gt;&amp; nums) {
        return sortArray(nums, 0, nums.size()-1);
    }
};</code></pre>

     代码四，对应力扣 验证二叉搜索树，代码见下
    </p>
    <pre><code class="language-cpp">/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
    vector&lt;int&gt; ret;
    void inOrder(TreeNode* root){
        if(root){
            inOrder(root-&gt;left);
            ret.push_back(root-&gt;val);
            inOrder(root-&gt;right);
        }
    }

public:
    bool isValidBST(TreeNode* root) {
        ret.clear();
        inOrder(root);
        for(int i=1; i&lt;ret.size(); ++i){
            if(ret[i] &lt; ret[i-1]){
                return false;
            }
        }
        return true;
    }
};</code></pre>
