public class Main{
    
    public static int count = 0;
    private static final int[][] dir = {<!-- -->{0,1},{1,0},{0,-1},{-1,0}};

    public static void bfs(int[][] matrix,int r,int c, int aim){
        // queue, linkedlist
        // method: poll. add. isEmpty
        Queue&lt;int[]&gt; que = new LinkedList&lt;&gt;();
        que.add(new int[]{r,c});
        matrix[r][c] = aim;
        count++;
        while(!que.isEmpty()){
            int[] cur = que.poll();
            int row = cur[0];
            int col = cur[1];
            for(int i=0; i&lt;4; i++){
                int nr = row + dir[i][0];
                int nc = col + dir[i][1];
                if(nr&lt;0||nc&lt;0||nr&gt;=matrix.length||nc&gt;=matrix[0].length)
                continue;
                if(matrix[nr][nc]==1){
                    que.add(new int[]{nr,nc});
                    count++;
                    matrix[nr][nc] = aim;
                }
            }

        }

    }

         417. 太平洋大西洋水流问题

        <span class="link-link">

     <li>
      思路：
      <ul>
       <li>
        这一题题目有一点难以理解，当时大体上有了前两题的铺垫还是比较好做的。
       </li>
       <li>
        题意大概是：左上两条边连接pacific，右下两条边连接Atlantic，雨水可以流向小于等于高度的方向（东南西北），求同时可以流向太平洋和大西洋的方块。
       </li>
       <li>
        这题可以分成两部分来看，首先找到可以流向pacific的点，然后找到可以流向Atlantic的点，求它们的交集。
        <ul>
         <li>
          存储方式：我一开始想到的是用三维数组进行存储，当时两个二维数组的方式可能更好写一些（没有太大区别）
         </li>
        </ul>
       </li>
       <li>
        遍历方式：采用深度优先遍历，遍历过的点都设置为true（有一点像岛屿问题），如果不能延伸则return（返回条件，即相邻岛屿高度&lt;目前岛屿高度）
       </li>
      </ul>
     </li>
     <li>
      代码：
      <ul>
       <li>
        <pre><code class="language-java">class Solution {
    static int[][] dirs = {<!-- -->{-1,0},{1,0},{0,-1},{0,1}};
    int[][] heights;
    int m,n;

    public List&lt;List&lt;Integer&gt;&gt; pacificAtlantic(int[][] heights) {
        this.heights = heights;
        this.m = heights.length;
        this.n = heights[0].length;
        boolean[][] pacific = new boolean[m][n];
        boolean[][] atlantic = new boolean[m][n];
        for(int i=0; i&lt;m; i++){
            dfs(i, 0, pacific);
        }    
         for(int i=0; i&lt;m; i++){
            dfs(i, n-1, atlantic);
        }
        for(int j=1; j&lt;n;j++){
            dfs(0,j,pacific);
        }           
        for(int j=0; j&lt;n-1; j++){
            dfs(m-1,j,atlantic);
        }
        List&lt;List&lt;Integer&gt;&gt; result = new ArrayList&lt;&gt;();
        for(int i=0; i&lt;m; i++){
            for(int j=0; j&lt;n; j++){
                if(pacific[i][j] &amp;&amp; atlantic[i][j]){
                    List&lt;Integer&gt; cell = new ArrayList&lt;&gt;();
                    cell.add(i);
                    cell.add(j);
                    result.add(cell);
                }
            }
        }
        return result;
    }

    public void dfs(int row, int col, boolean[][] ocean){
        if(ocean[row][col]) return;
        ocean[row][col] = true;
        for(int[] dir:dirs){
            int newRow = row + dir[0];
            int newCol = col + dir[1];
            if(newRow&gt;=0 &amp;&amp; newCol&gt;=0 &amp;&amp; newRow&lt;m &amp;&amp; newCol&lt;n &amp;&amp; heights[newRow][newCol]&gt;=heights[row][col]){
                dfs(newRow, newCol, ocean);
            }
        }
    }
}</code></pre>
