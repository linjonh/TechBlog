    <h2>
     2、题解
    </h2>
    <pre><code class="language-cpp">#include&lt;iostream&gt;
#include&lt;cstring&gt;
using namespace std;
const int N = 15;
int T, n;
int t[N], d[N], l[N];
bool st[N];
bool dfs(int pos, int end)//第几加以及结束时间
{
	if (pos &gt; n) return true;

	for (int i = 1;i &lt;= n;i++)
	{
		if (st[i] == true) continue;
		if (end &gt; t[i] + d[i]) continue;
		int newend = max(t[i], end) + l[i];
		st[i] = true;
		if (dfs(pos + 1, newend)) return true;
		st[i] = false;
	}
	return false;
}
int main()
{
	cin &gt;&gt; T;
	while (T--)
	{
		memset(st, 0, sizeof st);
		cin &gt;&gt; n;
		for (int i = 1;i &lt;= n;i++)
		{
			cin &gt;&gt; t[i]&gt;&gt;d[i]&gt;&gt;l[i];
		}
		if (dfs(1, 0)) cout &lt;&lt; "YES" &lt;&lt; endl;
		else cout &lt;&lt; "NO" &lt;&lt; endl;
	}
	
	return 0;
}</code></pre>
