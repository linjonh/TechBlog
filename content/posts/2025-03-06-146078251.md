#include &lt;QDebug&gt;

myThread::myThread(QObject* parent)
    : QThread(parent)

    qDebug() &lt;&lt; "线程创建";
}

void myThread::setRunning(bool running)
{
    QMutexLocker locker(&amp;m_mutex);
    m_running = running;
}

void myThread::run()
{
    qDebug() &lt;&lt; "线程启动";
    {
        QMutexLocker locker(&amp;m_mutex);
        m_running = true;
    }

    while(true) {
        { // 检查运行状态
            QMutexLocker locker(&amp;m_mutex);
            if(!m_running) break;
        }

        // 更新滑块值
        m_currentValue += m_direction;

        // 边界检测
        if(m_currentValue &gt;= 100) {
            m_currentValue = 100;
            m_direction = -1;
        }
        else if(m_currentValue &lt;= 0) {
            m_currentValue = 0;
            m_direction = 1;
        }

        // 发送更新信号
        emit updateSlider(m_currentValue);

        // 线程休眠（控制移动速度）
        msleep(50);
    }
    qDebug() &lt;&lt; "线程退出";
}

    <pre><code class="language-cpp">#include "mythread2.h"
#include &lt;QDir&gt;
#include &lt;QDebug&gt;

mythread2::mythread2(const QString &amp;source, const QString &amp;dest, QObject* parent)
    : QThread(parent), m_source(source), m_dest(dest)
{
    qDebug() &lt;&lt; "文件复制线程创建，源文件:" &lt;&lt; source;
}

void mythread2::run()
{
    qDebug() &lt;&lt; "开始文件复制操作...";

    QFile srcFile(m_source);
    QFile destFile(m_dest);

    // 打开源文件（只读模式）
    if (!srcFile.open(QIODevice::ReadOnly)) {
        qWarning() &lt;&lt; "无法打开源文件:" &lt;&lt; m_source;
        emit copyFinished(false, "无法打开源文件");
        return;
    }

    // 创建目标文件（只写模式）
    if (!destFile.open(QIODevice::WriteOnly)) {
        qWarning() &lt;&lt; "无法创建目标文件:" &lt;&lt; m_dest;
        srcFile.close();
        emit copyFinished(false, "无法创建目标文件");
        return;
    }

    const qint64 bufferSize = 1024 * 1024; // 1MB缓冲区
    char *buffer = new char[bufferSize];
    qint64 totalBytes = srcFile.size();
    qint64 bytesCopied = 0;

    qDebug() &lt;&lt; "开始复制，文件总大小:" &lt;&lt; totalBytes &lt;&lt; "bytes";

    while (!srcFile.atEnd()) {
        // 检查线程中断请求
        if (this-&gt;isInterruptionRequested()) {
            qDebug() &lt;&lt; "收到中断请求，停止复制";
            break;
        }

        // 读取文件内容
        qint64 readBytes = srcFile.read(buffer, bufferSize);
        if (readBytes == -1) {
            qWarning() &lt;&lt; "读取文件错误，位置:" &lt;&lt; bytesCopied;
            emit copyFinished(false, "读取文件错误");
            break;
        }

        // 写入目标文件
        qint64 writeBytes = destFile.write(buffer, readBytes);
        if (writeBytes != readBytes) {
            qWarning() &lt;&lt; "写入文件错误，预期写入:" &lt;&lt; readBytes &lt;&lt; "实际写入:" &lt;&lt; writeBytes;
            emit copyFinished(false, "写入文件错误");
            break;
        }

        // 更新进度
        bytesCopied += readBytes;

        // 计算进度百分比（使用浮点运算避免溢出）
        int progress = 0;
        if (totalBytes &gt; 0) {
            progress = static_cast&lt;int&gt;((static_cast&lt;double&gt;(bytesCopied) / totalBytes) * 100);
        }
        emit progressChanged(progress);

        // 调试输出（生产环境应注释掉）
        qDebug() &lt;&lt; "已复制:" &lt;&lt; bytesCopied &lt;&lt; "bytes (" &lt;&lt; progress &lt;&lt; "%)";
    }

    // 清理资源
    delete[] buffer;
    srcFile.close();
    destFile.close();

    // 处理最终结果
    if (bytesCopied == totalBytes) {
        qDebug() &lt;&lt; "文件复制完成，总字节:" &lt;&lt; bytesCopied;
        emit copyFinished(true, QFileInfo(m_source).fileName() + " 文件已存放成功");
    } else {
        qWarning() &lt;&lt; "复制中断，已复制:" &lt;&lt; bytesCopied &lt;&lt; "/" &lt;&lt; totalBytes;
        QFile::remove(m_dest); // 删除不完整文件
        emit copyFinished(false, "文件复制中断");
    }
}</code></pre>

#include "ui_widget.h"
#include &lt;QFileDialog&gt;

#include &lt;QDir&gt;

Widget::Widget(QWidget *parent)
    : QWidget(parent)
    , ui(new Ui::Widget)
    , m_copyThread(nullptr)

    // 初始化目标目录
    QDir dir(m_targetDir);
    if (!dir.exists()) {
        qDebug() &lt;&lt; "创建目标目录:" &lt;&lt; m_targetDir;
        dir.mkpath(".");
    }

    // 配置进度条
    ui-&gt;horizontalSlider-&gt;setRange(0, 100);
    ui-&gt;horizontalSlider-&gt;setValue(0);

    // 设置进度条样式
    QString qss = R"(

            border-radius: 7px;

        QSlider::handle:horizontal {
            background: qradialgradient(cx:0.5, cy:0.5, radius: 0.5,
                fx:0.5, fy:0.5,
                stop:0 white, stop:0.5 #FFD700,
                stop:1 #FFA500);
            width: 25px;
            height: 25px;
            margin: -5px 0;
            border-radius: 12px;
            border: 2px solid #FFFFFF;

    ui-&gt;horizontalSlider-&gt;setStyleSheet(qss);

Widget::~Widget()
{
    // 安全停止线程
    if (m_copyThread &amp;&amp; m_copyThread-&gt;isRunning()) {
        qDebug() &lt;&lt; "正在停止文件复制线程...";
        m_copyThread-&gt;requestInterruption();
        m_copyThread-&gt;wait(3000); // 最多等待3秒
        delete m_copyThread;
    }
    delete ui;
}

void Widget::on_pushButton_clicked()
{
    // 获取源文件路径
    QString filePath = QFileDialog::getOpenFileName(
        this,
        "选择要复制的文件",
        QDir::homePath(),
        "所有文件 (*.*)"
    );

    if (filePath.isEmpty()) {
        qDebug() &lt;&lt; "用户取消文件选择";
        return;
    }

    QFileInfo fileInfo(filePath);
    QString destPath = m_targetDir + "/" + fileInfo.fileName();

    qDebug() &lt;&lt; "目标文件路径:" &lt;&lt; destPath;

    // 检查文件是否已存在
    if (QFile::exists(destPath)) {
        QMessageBox::warning(this, "警告", "目标文件已存在!");
        qWarning() &lt;&lt; "目标文件已存在:" &lt;&lt; destPath;
        return;
    }

    // 清理旧线程
    if (m_copyThread) {
        qDebug() &lt;&lt; "终止之前的复制线程...";
        m_copyThread-&gt;requestInterruption();
        m_copyThread-&gt;wait();
        delete m_copyThread;
        m_copyThread = nullptr;
    }

    // 创建新线程
    m_copyThread = new mythread2(filePath, destPath, this);

    // 连接信号槽
    connect(m_copyThread, &amp;mythread2::progressChanged,
            this, &amp;Widget::updateProgress);
    connect(m_copyThread, &amp;mythread2::copyFinished,
            this, &amp;Widget::handleCopyResult);
    connect(m_copyThread, &amp;QThread::finished,
            m_copyThread, &amp;QObject::deleteLater);

    // 初始化UI状态
    ui-&gt;pushButton-&gt;setEnabled(false);
    ui-&gt;horizontalSlider-&gt;setValue(0); // 重置进度条
    ui-&gt;textEdit-&gt;append("开始复制文件: " + fileInfo.fileName());

    // 启动线程
    qDebug() &lt;&lt; "启动文件复制线程...";
    m_copyThread-&gt;start();
}

void Widget::updateProgress(int value)
{
    // 确保在主线程更新UI
    Q_ASSERT(QThread::currentThread() == this-&gt;thread());

    // 限制更新频率（每5%更新一次）
    static int lastValue = -5;
    if (value - lastValue &gt;= 5 || value == 100) {
        ui-&gt;horizontalSlider-&gt;setValue(value);
        lastValue = value;
        qDebug() &lt;&lt; "当前进度:" &lt;&lt; value &lt;&lt; "%";
    }
}

void Widget::handleCopyResult(bool success, const QString &amp;message)
{
    // 恢复按钮状态
    ui-&gt;pushButton-&gt;setEnabled(true);

    // 显示结果信息
    ui-&gt;textEdit-&gt;append(message);
    if (!success) {
        QMessageBox::critical(this, "错误", message);
        ui-&gt;horizontalSlider-&gt;setValue(0); // 失败时重置进度
    } else {
        ui-&gt;horizontalSlider-&gt;setValue(100); // 成功时显示100%
    }

    // 清理线程指针
    m_copyThread = nullptr;
    qDebug() &lt;&lt; "文件复制操作完成，结果:" &lt;&lt; message;
