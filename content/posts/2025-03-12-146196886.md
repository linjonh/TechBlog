public:
    int maxProfit(vector&lt;int&gt;&amp; prices) {
      int result=0;
      for(int i=1;i&lt;prices.size();i++){
        result+=max(0,prices[i]-prices[i-1]);
      }  
      return result;
    }

    <pre><code class="language-cpp">class Solution {
public:
    bool canJump(vector&lt;int&gt;&amp; nums) {
        int cover=0;
        if(nums.size()==1) return true;
        for(int i=0;i &lt;= cover;i++){
            cover=max(i+nums[i],cover);
            if(cover&gt;=nums.size()-1) return true;
        }
        return false;
    }
};</code></pre>
    <h2>
     <a href="https://leetcode.cn/problems/jump-game-ii/description/" rel="nofollow" title="45. 跳跃游戏 II - 力扣（LeetCode）">
      45. 跳跃游戏 II - 力扣（LeetCode）
     </a>
    </h2>
    <pre><code class="language-cpp">class Solution {
public:
    int jump(vector&lt;int&gt;&amp; nums) {
        if(nums.size()==1) return 0;
        int curDistance=0;
        int nextDistance=0;
        int ans=0;
        for(int i=0;i&lt;nums.size();i++){
            //更新最远下标
            nextDistance=max(nums[i]+i,nextDistance);
            if(i==curDistance){
                ans++;
                curDistance=nextDistance;
                if(nextDistance&gt;=nums.size()-1) break;
            }
        }
        return ans;
    }
};</code></pre>

      1005. K 次取反后最大化的数组和 - 力扣（LeetCode）

    <p>
     需要注意的是如何使用sort函数实现绝对值降序排列，需要自定义cmp函数来确定排序规则。
    </p>
    <pre><code class="language-cpp">class Solution {
    static bool cmp(int a,int b){
        return abs(a) &gt; abs(b);
}
public:
    int largestSumAfterKNegations(vector&lt;int&gt;&amp; nums, int k) {
        sort(nums.begin(),nums.end(),cmp);
        for(int i=0;i&lt;nums.size();i++){
            if(nums[i]&lt;0 &amp;&amp; k&gt;0){
                nums[i]*=-1;
                k--;
            }
        }
        //如果k依旧不为0
        if(k%2==1) nums[nums.size()-1]*=-1;
        int result=0;
        for(int i=0;i&lt;nums.size();i++){
            result+=nums[i];
        }
        return result;
    }
};</code></pre>
