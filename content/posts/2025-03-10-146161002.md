public int orangesRotting(int[][] grid) {
    Queue<int[]> queue = new LinkedList<>();
    int fresh = 0, time = 0;
  
    // ğŸŒ‹ åˆå§‹åŒ–ï¼šæ‰€æœ‰è…çƒ‚æ©˜å­å…¥é˜Ÿ
    for (int i = 0; i < grid.length; i++) {
        for (int j = 0; j < grid[0].length; j++) {
            if (grid[i][j] == 2) queue.offer(new int[]{i, j});
            else if (grid[i][j] == 1) fresh++;
        }
    }
  
    int[][] dirs = {{1,0}, {-1,0}, {0,1}, {0,-1}};
    while (!queue.isEmpty() && fresh > 0) {
        int size = queue.size();
        for (int i = 0; i < size; i++) {
            int[] pos = queue.poll();
            for (int[] d : dirs) {
                int x = pos[0] + d[0];
                int y = pos[1] + d[1];
                if (x >= 0 && x < grid.length && y >= 0 && y < grid[0].length 
                    && grid[x][y] == 1) {
                    grid[x][y] = 2;
                    queue.offer(new int[]{x, y});
                    fresh--;
                }
            }
        }
        time++; // â³ æ—¶é—´æµé€
    }
    return fresh == 0 ? time : -1;
}

public int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
    Map<Integer, List<int[]>> graph = new HashMap<>();
    for (int[] f : flights) {
        graph.computeIfAbsent(f[0], key -> new ArrayList<>()).add(new int[]{f[1], f[2]});
    }
  
    // ä¼˜å…ˆé˜Ÿåˆ—ï¼šæŒ‰ä»·æ ¼æ’åºï¼ˆä¸‰å…ƒç»„ï¼šå½“å‰èŠ‚ç‚¹ï¼Œå‰©ä½™æ¬¡æ•°ï¼Œç´¯è®¡ä»·æ ¼ï¼‰
    PriorityQueue<int[]> pq = new PriorityQueue<>((a,b) -> a[2]-b[2]);

    while (!pq.isEmpty()) {
        int[] curr = pq.poll();

        if (node == dst) return price;
        if (stops == 0) continue;

            pq.offer(new int[]{neighbor[0], stops-1, price + neighbor[1]});

public int minJumps(int[] arr) {
    Map<Integer, List<Integer>> valueMap = new HashMap<>();
    for (int i = 0; i < arr.length; i++) {
        valueMap.computeIfAbsent(arr[i], k -> new ArrayList<>()).add(i);
    }
  
    Queue<Integer> queue = new LinkedList<>();
    boolean[] visited = new boolean[arr.length];
    queue.offer(0);
    visited[0] = true;
    int steps = 0;
  
    while (!queue.isEmpty()) {
        int size = queue.size();
        for (int i = 0; i < size; i++) {
            int curr = queue.poll();
            if (curr == arr.length - 1) return steps;
    
            // ä¸‰ç§è·³æ³•ï¼šå·¦è·³ã€å³è·³ã€ç­‰å€¼è·³
            if (curr - 1 >= 0 && !visited[curr-1]) {
                visited[curr-1] = true;
                queue.offer(curr-1);
            }
            if (curr + 1 < arr.length && !visited[curr+1]) {
                visited[curr+1] = true;
                queue.offer(curr+1);
            }
            if (valueMap.containsKey(arr[curr])) {
                for (int same : valueMap.get(arr[curr])) {
                    if (!visited[same]) {
                        visited[same] = true;

            }
