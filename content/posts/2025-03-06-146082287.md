#include &lt;iostream&gt;

#include &lt;osgViewer/Viewer&gt;
#include &lt;osgDB/ReadFile&gt;
#include &lt;osgGA/TrackballManipulator&gt;
#include &lt;osgGA/GUIEventAdapter&gt;

int main() {
	osg::ref_ptr&lt;osgViewer::Viewer&gt; viewer = new osgViewer::Viewer;

	viewer-&gt;setSceneData(osgDB::readNodeFile("glider.osg"));
	// 设置默认操作器
	viewer-&gt;setCameraManipulator(new osgGA::TrackballManipulator);
	
	viewer-&gt;realize();
	
	viewer-&gt;setKeyEventSetsDone(osgGA::GUIEventAdapter::KEY_KP_Space);

	// 线程渲染模式
	std::cout &lt;&lt; "Threading model: " &lt;&lt; viewer-&gt;getThreadingModel() &lt;&lt; std::endl;

	viewer-&gt;run();
	

	/*  替代 run()的作用

			std::cout &lt;&lt; "No need to Continue " &lt;&lt; frameNum &lt;&lt; std::endl;// 300帧渲染完了，就推出渲染
		}
		frameNum--;
		viewer-&gt;frame();
	}

	return 0;
