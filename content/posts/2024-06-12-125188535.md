    if (element.category_name.indexOf(value) &gt; -1) {
      const ab = rebuildData(value, element.children)
      const obj = {
        ...element,
        children: ab
      }
      newArr.push(obj)
    } else {
      // 判断当前节点知否有子节点，并且子节点中有数据，有数据继续递归查找
      if (element.children &amp;&amp; element.children.length &gt; 0) {
        const ab = rebuildData(value, element.children)
        const obj = {
          ...element,
          children: ab
        }

          newArr.push(obj)
        }
      }
    }
  })
