    <p>
     正则表达式是一种进行模式匹配和文本操纵的复杂而又强大的工具。虽然正则表达式比纯粹的文本匹配效率低，但是它却更灵活。按照它的语法规则，随需构造出的匹配模式就能够从原始文本中筛选出几乎任何你想要得到的字符组合。
    </p>
    <p>
     Go语言通过regexp标准包为正则表达式提供了官方支持，如果你已经使用过其他编程语言提供的正则相关功能，那么你应该对Go语言版本的不会太陌生，但是它们之间也有一些小的差异，因为Go实现的是RE2标准，除了\C，详细的语法描述参考：
     <a href="http://code.google.com/p/re2/wiki/Syntax" rel="nofollow">
      http://code.google.com/p/re2/wiki/Syntax
     </a>
    </p>
    <p>
     其实字符串处理我们可以使用strings包来进行搜索(Contains、Index)、替换(Replace)和解析(Split、Join)等操作，但是这些都是简单的字符串操作，他们的搜索都是大小写敏感，而且固定的字符串，如果我们需要匹配可变的那种就没办法实现了，当然如果strings包能解决你的问题，那么就尽量使用它来解决。因为他们足够简单、而且性能和可读性都会比正则好。
    </p>
    <p>
     正则表达式匹配规则：
     <br/>
     <img alt="这里写图片描述" src="https://img-blog.csdn.net/20180117093647206?watermark/2/text/aHR0cDovL2Jsb2cuY3Nkbi5uZXQvdGVubnlzb25za3k=/font/5a6L5L2T/fontsize/400/fill/I0JBQkFCMA==/dissolve/70/gravity/SouthEast" title=""/>
    </p>
    <p>
     示例代码：
    </p>
    <pre class="prettyprint"><code class="hljs livecodeserver">package main

import (
    <span class="hljs-string">"fmt"</span>
    <span class="hljs-string">"regexp"</span>
)

func main() {
    context1 := <span class="hljs-string">"3.14 123123 .68 haha 1.0 abc 6.66 123."</span>

   <span class="hljs-comment"> //MustCompile解析并返回一个正则表达式。如果成功返回，该Regexp就可用于匹配文本。</span>
   <span class="hljs-comment"> //解析失败时会产生panic</span>
   <span class="hljs-comment"> // \d 匹配数字[0-9]，d+ 重复&gt;=1次匹配d，越多越好（优先重复匹配d）</span>
    <span class="hljs-built_in">exp1</span> := regexp.MustCompile(`\d+\.\d+`)

   <span class="hljs-comment"> //返回保管正则表达式所有不重叠的匹配结果的[]string切片。如果没有匹配到，会返回nil。</span>
   <span class="hljs-comment"> //result1 := exp1.FindAllString(context1, -1) //[3.14 1.0 6.66]</span>
    result1 := <span class="hljs-built_in">exp1</span>.FindAllStringSubmatch(context1, -<span class="hljs-number">1</span>)<span class="hljs-comment"> //[[3.14] [1.0] [6.66]]</span>

    fmt.Printf(<span class="hljs-string">"%v\n"</span>, result1)
    fmt.Printf(<span class="hljs-string">"\n------------------------------------\n\n"</span>)

    context2 := `
        &lt;title&gt;标题&lt;/title&gt;
        &lt;<span class="hljs-operator">div</span>&gt;你过来啊&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;<span class="hljs-operator">div</span>&gt;hello mike&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;<span class="hljs-operator">div</span>&gt;你大爷&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;body&gt;呵呵&lt;/body&gt;
    `
   <span class="hljs-comment"> //(.*?)被括起来的表达式作为分组</span>
   <span class="hljs-comment"> //匹配&lt;div&gt;xxx&lt;/div&gt;模式的所有子串</span>
    <span class="hljs-built_in">exp2</span> := regexp.MustCompile(`&lt;<span class="hljs-operator">div</span>&gt;(.*?)&lt;/<span class="hljs-operator">div</span>&gt;`)
    result2 := <span class="hljs-built_in">exp2</span>.FindAllStringSubmatch(context2, -<span class="hljs-number">1</span>)

   <span class="hljs-comment"> //[[&lt;div&gt;你过来啊&lt;/div&gt; 你过来啊] [&lt;div&gt;hello mike&lt;/div&gt; hello mike] [&lt;div&gt;你大爷&lt;/div&gt; 你大爷]]</span>
    fmt.Printf(<span class="hljs-string">"%v\n"</span>, result2)
    fmt.Printf(<span class="hljs-string">"\n------------------------------------\n\n"</span>)

    context3 := `
        &lt;title&gt;标题&lt;/title&gt;
        &lt;<span class="hljs-operator">div</span>&gt;你过来啊&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;<span class="hljs-operator">div</span>&gt;hello 
        mike
        go&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;<span class="hljs-operator">div</span>&gt;你大爷&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;body&gt;呵呵&lt;/body&gt;
    `
    exp3 := regexp.MustCompile(`&lt;<span class="hljs-operator">div</span>&gt;(.*?)&lt;/<span class="hljs-operator">div</span>&gt;`)
    result3 := exp3.FindAllStringSubmatch(context3, -<span class="hljs-number">1</span>)

   <span class="hljs-comment"> //[[&lt;div&gt;你过来啊&lt;/div&gt; 你过来啊] [&lt;div&gt;你大爷&lt;/div&gt; 你大爷]]</span>
    fmt.Printf(<span class="hljs-string">"%v\n"</span>, result3)
    fmt.Printf(<span class="hljs-string">"\n------------------------------------\n\n"</span>)

    context4 := `
        &lt;title&gt;标题&lt;/title&gt;
        &lt;<span class="hljs-operator">div</span>&gt;你过来啊&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;<span class="hljs-operator">div</span>&gt;hello 
        mike
        go&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;<span class="hljs-operator">div</span>&gt;你大爷&lt;/<span class="hljs-operator">div</span>&gt;
        &lt;body&gt;呵呵&lt;/body&gt;
    `
    exp4 := regexp.MustCompile(`&lt;<span class="hljs-operator">div</span>&gt;(?s:(.*?))&lt;/<span class="hljs-operator">div</span>&gt;`)
    result4 := exp4.FindAllStringSubmatch(context4, -<span class="hljs-number">1</span>)

    <span class="hljs-comment">/*
        [[&lt;div&gt;你过来啊&lt;/div&gt; 你过来啊] [&lt;div&gt;hello
            mike
            go&lt;/div&gt; hello
            mike
            go] [&lt;div&gt;你大爷&lt;/div&gt; 你大爷]]
    */</span>
    fmt.Printf(<span class="hljs-string">"%v\n"</span>, result4)
    fmt.Printf(<span class="hljs-string">"\n------------------------------------\n\n"</span>)

    <span class="hljs-keyword">for</span> _, <span class="hljs-keyword">text</span> := range result4 {
        fmt.Println(<span class="hljs-keyword">text</span>[<span class="hljs-number">0</span>])<span class="hljs-comment"> //带有div</span>
        fmt.Println(<span class="hljs-keyword">text</span>[<span class="hljs-number">1</span>])<span class="hljs-comment"> //不带带有div</span>
        fmt.Println(<span class="hljs-string">"================\n"</span>)
    }
}</code></pre>
