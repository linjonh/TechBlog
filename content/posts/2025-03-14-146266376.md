     深度优先搜索
    </p>
    <pre><code class="language-cpp">class Solution {
private:
    vector&lt;vector&lt;int&gt;&gt; result;
    vector&lt;int&gt; temp;

    void allPathsSourceTarget(vector&lt;vector&lt;int&gt;&gt; &amp;graph,int v){
        if(v==graph.size()-1)
            result.push_back(temp);
        else{
            for(auto&amp; adjVertex:graph[v]){
                temp.push_back(adjVertex);
                allPathsSourceTarget(graph,adjVertex);
                temp.pop_back();
            }
        }
    }

public:
    vector&lt;vector&lt;int&gt;&gt; allPathsSourceTarget(vector&lt;vector&lt;int&gt;&gt;&amp; graph) {
        result.clear();
        temp.push_back(0);
        allPathsSourceTarget(graph,0);
        temp.pop_back();
        return result;
    }
};</code></pre>
