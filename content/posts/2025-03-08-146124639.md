    <h2>
     for 循环
    </h2>
    <h3>
     <strong>
      遍历列表
     </strong>
    </h3>
    <blockquote>
     <p>
      for fruit in apple banana orange
      <br/>
      do
      <br/>
      echo "I like $fruit"
      <br/>
      done
     </p>
    </blockquote>
    <h3>
     <strong>
      遍历文件
     </strong>
    </h3>
    <blockquote>
     <p>
      <strong>
       for file in *.txt
       <br/>
       do
       <br/>
       echo "Processing $file"
       <br/>
       cp "$file" backup/
       <br/>
       done
      </strong>
     </p>
    </blockquote>
    <h3>
     <strong>
      数字范围
     </strong>
     （使用
     <code>
      seq
     </code>
     或
     <code>
      {start..end}
     </code>
     ）
    </h3>
    <blockquote>
     <p>
      # 1到5
      <br/>
      for i in {1..5}
      <br/>
      do
      <br/>
      echo "Number: $i"
      <br/>
      done
     </p>
     <p>
      # 步长2（需要 Bash 4+）
      <br/>
      for i in {1..10..2}
      <br/>
      do
      <br/>
      echo "Odd: $i"
      <br/>
      done
     </p>
     <p>
      # 使用 seq
      <br/>
      for i in $(seq 1 5)
      <br/>
      do
      <br/>
      echo "Seq: $i"
      <br/>
      done
     </p>
    </blockquote>
    <h3>
     <strong>
      C 语言风格
     </strong>
    </h3>
    <blockquote>
     <p>
      for ((i=0; i&lt;5; i++))
      <br/>
      do
      <br/>
      echo "Index: $i"
      <br/>
      done
     </p>
    </blockquote>
