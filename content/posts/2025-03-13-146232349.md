    <h5>

    <h5>
     <strong>
      VisualSVN Server批量添加用户，批量追加用户到svn路径下 passwd 文件
     </strong>
    </h5>
    <hr/>
    <ul>
     <li>
      <p>
       <strong>
        使用
       </strong>
       <strong>
        htpasswd.exe程序，如果未安装 Apache，可以下载
        <a class="link-info" href="https://www.apachelounge.com/download/" rel="nofollow" title="Apache">
         Apache
        </a>
       </strong>
      </p>
      <pre><a href="https://www.apachelounge.com/download/VS17/binaries/httpd-2.4.63-250207-win64-VS17.zip" rel="nofollow" title="httpd-2.4.63-250207-win64-VS17.zip">httpd-2.4.63-250207-win64-VS17.zip</a> 下载解压直接复制路径即可</pre>
     </li>
     <li>
      <strong>
       使用Windows 系统脚本（PowerShell）运行以下内容
      </strong>
     </li>
    </ul>
    <pre><code class="language-bash">#方法一（各自独立密码写法）
$htpasswdPath = "C:\svn\auth\htpasswd"  # 替换为实际路径（svn安装路径）
$users = @{
    "user1" = "password1"
    "user2" = "password2"
    "user3" = "password3"
}

foreach ($user in $users.Keys) {
    $password = $users[$user]
    &amp; "C:\Program Files\Apache\bin\htpasswd.exe" -bB $htpasswdPath $user $password
}

#方法二（统一密码写法）
$htpasswdPath = "C:\svn\auth\htpasswd"  # 替换为实际路径（svn安装路径）
$users = @{
    "user1", "user2", "user3"
}

foreach ($user in $users.Keys) {
    $password = $users[$user]
    &amp; "C:\Program Files\Apache\bin\htpasswd.exe" -bB $htpasswdPath $user $password
}</code></pre>
