     代码实现
    </h3>
    <pre><code>@RestController
public class HomeController {

    @GetMapping("/backend/proxy")
    public ResponseEntity&lt;?&gt; proxy(@RequestParam MultiValueMap&lt;String, String&gt; params, @RequestHeader MultiValueMap&lt;String, String&gt; heads) {

        if (!params.containsKey("target")) {
            return ResponseEntity.status(HttpStatus.NOT_FOUND).body("{\"message\": \"target param is required\"}");
        }

        String url = params.remove("target").get(0);
        System.out.println("url: " + url);

        URI uri;
        try {
            uri = new URI(url);
        } catch (URISyntaxException ex) {
            return ResponseEntity.status(HttpStatus.NOT_FOUND).body("{\"message\": \"target value invalid\"}");
        }
        String result = WebClient.create().get().uri(uriBuilder -&gt; uriBuilder.scheme(uri.getScheme())
                .host(uri.getHost()).path(uri.getPath()).queryParams(params).build())
                .headers(headers -&gt; headers.addAll(heads)).retrieve()
                .bodyToMono(String.class).block();
        return ResponseEntity.ok(result);
    }
}
