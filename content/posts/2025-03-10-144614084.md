```cpp
#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
int check_sys()
{
	int a = 1;
	return *(char*)&a;//如果是小端存储第一位会为1，大端则相反。
	
}
int main()
{   
	int ret = check_sys();//判断函数返回1为小端，返回0为大端
	if (ret == 1)
	{
		printf("小端\n");
	}
	else
	{
		printf("大端\n");
	}
	return 0;
}
```
