    <p>
     今天实现一个基于element-plus表单组件的二次封装，什么是二次封装？查看以下表单，传统表单组件是不是用&lt;el-form&gt;嵌套几个&lt;el-form-item&gt;即可实现，那么一个表单可不可以实现，传入一个对象给封装组件，通过对象的type值来进行表单的实现。当一个项目有多个不同的表单内容时，能够实现代码的复用性、可读性。

&lt;script setup&gt;
import { ref } from 'vue'
// 接收参数
defineProps({
    formData: {
        type: Object,
        required: true
    },
    formConfig: {
        type: Array,
        required: true,
        validator: value =&gt; value.every(row =&gt; Array.isArray(row))
    },
    labelWidth: {
        type: String,
        default: '80px'
    }
})
// 提交表单的方法
const formRef = ref(null);

const submitForm = () =&gt; {
    formRef.value.validate((valid) =&gt; {
        if (valid) {
            console.log('表单提交成功:', props.formData)
        } else {
            console.error('请完善表单!')
            return false;
        }
    })
}
// 重置表单的方法
const resetForm = () =&gt; {
    // 清空校验信息
    formRef.value.clearValidate();
}
// 暴露给父组件使用
defineExpose({
    submitForm,
    resetForm
})

&lt;/script&gt;
