    <h5>
     <a id="_130">
     </a>
     变量
    </h5>
    <h6>
     <a id="1_var_131">
     </a>
     1.
     <code>
      var
     </code>
    </h6>
    <ul>
     <li>
      <code>
       var
      </code>
      定义的变量可以被覆盖.
     </li>
     <li>
      具有函数作用域, 在函数内声明的
      <code>
       var
      </code>
      变量在函数内有效:
     </li>
    </ul>
    <pre><code class="prism language-JavaScript">function testVar() {
    if (true) {
        var x = 10; // 函数作用域
    }
    console.log(x); // 输出: 10
}
testVar();

//在函数外不能访问:
&lt;script&gt;  
    function testVar() {  
        if (true) {  
            var x = 10; // 函数作用域  
        }   
    }  
    alert(x);  
&lt;/script&gt;
</code></pre>
    <ul>
     <li>
      不受块级作用域
      <code>
       {}
      </code>
      影响，即使声明在块内，块外也可以访问。
     </li>
    </ul>
    <pre><code class="prism language-JavaScript">{
    var x = 10;
}
console.log(x); // 输出: 10

</code></pre>
    <h6>
     <a id="2_let_165">
     </a>
     2.
     <code>
      let
     </code>
    </h6>
    <ul>
     <li>
      具有块级作用域:变量只在声明的
      <code>
       {}
      </code>
      内有效,超出作用域不能访问
     </li>
    </ul>
    <pre><code class="prism language-JavaScript">&lt;script&gt;

        let a = 1;  
    }  
    alert(a);  //不会弹出警告框
&lt;/script&gt;
</code></pre>
    <ul>
     <li>
      不能重复定义
     </li>
    </ul>
    <pre><code class="prism language-JavaScript">{  
    let a = 1;  
    let a="s";  //报错
}
</code></pre>
    <h6>
     <a id="3_const_185">
     </a>
     3.
     <code>
      const
     </code>
    </h6>
    <ul>
     <li>
      <p>
       作用域:和
       <code>
        let
       </code>
       一样,也不能重复定义
      </p>
      <pre><code class="prism language-javascript"><span class="token punctuation">{<!-- --></span>

     </li>

      <p>

        const

      </p>
     </li>
     <li>
      <p>
       <code>
        const

    <pre><code class="prism language-javascript">console<span class="token punctuation">.</span><span class="token function">log</span><span class="token punctuation">(</span><span class="token function">Boolean</span><span class="token punctuation">(</span><span class="token number">1</span><span class="token punctuation">)</span><span class="token punctuation">)</span><span class="token punctuation">;</span> <span class="token comment">// 输出 true</span>
console<span class="token punctuation">.</span><span class="token function">log</span><span class="token punctuation">(</span><span class="token function">Boolean</span><span class="token punctuation">(</span><span class="token number">0</span><span class="token punctuation">)</span><span class="token punctuation">)</span><span class="token punctuation">;</span> <span class="token comment">// 输出 false</span>
console<span class="token punctuation">.</span><span class="token function">log</span><span class="token punctuation">(</span><span class="token function">Boolean</span><span class="token punctuation">(</span><span class="token string">""</span><span class="token punctuation">)</span><span class="token punctuation">)</span><span class="token punctuation">;</span> <span class="token comment">// 输出 false</span>
console<span class="token punctuation">.</span><span class="token function">log</span><span class="token punctuation">(</span><span class="token function">Boolean</span><span class="token punctuation">(</span><span class="token string">"hello"</span><span class="token punctuation">)</span><span class="token punctuation">)</span><span class="token punctuation">;</span> <span class="token comment">// 输出 true</span>
</code></pre>
    <p>
     不要使用
     <code>
      ==
     </code>
     比较，使用
     <code>
      ===
     </code>
     等号,因为
     <code>
      ===
     </code>
     不仅比较值,还会比较类型是否相等.
    </p>
    <h5>
     <a id="_251">
     </a>
     循环
    </h5>
    <h6>
     <a id="forin_253">
     </a>
     <code>
      for...in
     </code>
     循环:
    </h6>
    <ul>
     <li>
      可用于遍历对象,数组的属性，不能遍历字符串、
      <code>
       Set
      </code>
      、
      <code>
       Map
      </code>
      .
     </li>
     <li>
      用于数组时，遍历的是数组的索引，不是数组的值。
     </li>
    </ul>
    <pre><code class="prism language-JavaScript">let obj = { name: "Alice", age: 25, city: "NY" };
for (const key in obj) {
  console.log(key, obj[key]);
}
// 输出：
// name Alice
// age 25
// city NY

</code></pre>
    <h6>
     <a id="forof_271">
     </a>
     <code>
      for...of
     </code>
     循环:
    </h6>
    <ul>
     <li>
      直接遍历值,可用于数组、字符串、
      <code>
       Set
      </code>
      、
      <code>
       Map
      </code>
      等,不能遍历对象
     </li>
    </ul>
    <pre><code class="prism language-JavaScript">let arr = ["a", "b", "c"];

for (const value of arr) {
  console.log(value); // 输出 "a", "b", "c"（值）
}

</code></pre>
    <h6>
     <a id="forEach__284">
     </a>
     <code>
      forEach()
     </code>
     循环
    </h6>
    <p>
     用于遍历数组.
    </p>
    <p>
     <strong>
      语法
     </strong>
     :
    </p>
    <pre><code class="prism language-JavaScript">array.forEach(function(currentValue, index, array) {

});

//currentValue（必需）：当前正在处理的元素。
//index（可选）：当前元素的索引
//array（可选）：调用 forEach() 的原数组
</code></pre>
    <p>
     <strong>
      示例
     </strong>
     :
    </p>
    <pre><code class="prism language-javascript"><span class="token keyword">let</span> fruits <span class="token operator">=</span> <span class="token punctuation">[</span><span class="token string">"apple"</span><span class="token punctuation">,</span> <span class="token string">"banana"</span><span class="token punctuation">,</span> <span class="token string">"orange"</span><span class="token punctuation">]</span><span class="token punctuation">;</span>

  console<span class="token punctuation">.</span><span class="token function">log</span><span class="token punctuation">(</span>index <span class="token operator">+</span> <span class="token string">": "</span> <span class="token operator">+</span> fruit<span class="token punctuation">)</span><span class="token punctuation">;</span>
<span class="token punctuation">}</span><span class="token punctuation">)</span><span class="token punctuation">;</span>

<span class="token comment">//输出:</span>
<span class="token comment">/*0: apple
1: banana
2: orange*/</span>

</code></pre>
    <pre><code class="prism language-javascript"><span class="token keyword">let</span> numbers <span class="token operator">=</span> <span class="token punctuation">[</span><span class="token number">1</span><span class="token punctuation">,</span> <span class="token number">2</span><span class="token punctuation">,</span> <span class="token number">3</span><span class="token punctuation">,</span> <span class="token number">4</span><span class="token punctuation">]</span><span class="token punctuation">;</span>

<span class="token comment">//输出:2 4 6 8</span>

    <p>
     <strong>
      注意
     </strong>
     :
    </p>

      return

    </p>
    <pre><code class="prism language-javascript"><span class="token keyword">let</span> arr <span class="token operator">=</span> <span class="token punctuation">[</span><span class="token number">1</span><span class="token punctuation">,</span> <span class="token number">2</span><span class="token punctuation">,</span> <span class="token number">3</span><span class="token punctuation">,</span> <span class="token number">4</span><span class="token punctuation">,</span> <span class="token number">5</span><span class="token punctuation">]</span><span class="token punctuation">;</span>
