    <h2>
     一、二叉搜索树（BST）基本原理与设计总结
    </h2>
    <p>
     <strong>
      注：基本原理的详细分析可以在数据结构第六节中查看，这里是简单描述。
     </strong>
    </p>
    <blockquote>
     <p>
      二叉搜索树（Binary Search Tree）是一种特殊的二叉树结构，具有以下核心性质：
     </p>
     <ol>
      <li>
       <p>
        <strong>
         有序性
        </strong>
        ：每个节点的左子树所有节点值均小于当前节点值
       </p>
      </li>
      <li>
       <p>
        <strong>
         递归结构
        </strong>
        ：每个节点的右子树所有节点值均大于当前节点值
       </p>
      </li>
      <li>
       <p>
        <strong>
         动态维护
        </strong>
        ：通过插入和删除操作维护树的有序性
       </p>
      </li>
     </ol>
     <p>
      其时间复杂度特性为：在平衡状态下，插入、删除、查找操作的平均时间复杂度为O(log n)；最坏情况下退化为链表结构，时间复杂度为O(n)
     </p>
    </blockquote>
    <blockquote>
     <p>
      在这里本设计
      <strong>
       使用了两个版本来实现二叉搜索树：K版本和KV版本
      </strong>
     </p>
     <p>
      <span style="color:#fe2c24">
       <strong>
        K版本：
       </strong>
      </span>
     </p>
     <ul>
      <li>
       声明并定义命名空间 K。实现了 BSTree 类，这个类主要用于存储和操作单一的关键字（Key）。
      </li>
      <li>
       每个节点包含一个键 _key，并且提供了插入（Insert）、查找（Find）和删除（Erase）等操作
      </li>
      <li>
       主要用于简单的关键字搜索和管理。
      </li>
     </ul>
     <p>
      <span style="color:#fe2c24">
       <strong>
        KV版本：
       </strong>
      </span>
     </p>
     <ul>
      <li>
       实现了 BSTree 类，这个类用于存储和操作键值对（Key-Value Pair）。
      </li>
      <li>
       每个节点包含一个键 _key 和一个值 _value，并且提供了插入键值对（Insert）、通过键查找节点（Find）等操作。
      </li>
      <li>
       主要用于需要关联数据存储和检索的场景，例如统计单词出现次数、存储配置信息等。
      </li>
     </ul>
    </blockquote>
    <p>
     同时本设计
     <strong>
      使用了两个版本来实现增删查改与遍历的操作：迭代版和递归版
     </strong>
    </p>

    <h3>
     2.1 节点结构定义（K版本）
    </h3>
    <blockquote>
     <p>
      在实现二叉搜索树之前，首先需要定义树的结构。基本的二叉搜索树由节点组成，每个节点包含一个键值（Key）、一个指向左子节点的指针（Left）和一个指向右子节点的指针（Right）。
     </p>
    </blockquote>
    <p>
     下面是二叉搜索树节点的定义示例：
    </p>
    <pre><code class="language-cpp">template&lt;class K&gt;
struct BSTreeNode {
    BSTreeNode&lt;K&gt;* _left;   // 左子树指针
    BSTreeNode&lt;K&gt;* _right;  // 右子树指针
    K _key;                 // 存储的关键字
    
    BSTreeNode(const K&amp; key) // 构造函数
        :_left(nullptr), _right(nullptr), _key(key)
    {}
};</code></pre>

    <h3>
     3.1 插入操作（迭代版）
    </h3>
    <blockquote>
     <p>
      <strong>
       实现要点
      </strong>
      ：
     </p>
     <ol>
      <li>
       <p>
        空树直接创建根节点
       </p>
      </li>
      <li>
       <p>
        循环查找插入位置，维护parent指针
       </p>
      </li>
      <li>
       <p>
        比较节点值决定插入方向
       </p>
      </li>
      <li>
       <p>
        时间复杂度：O(h)，h为树高度
       </p>
      </li>
     </ol>
    </blockquote>
    <pre><code class="language-cpp">bool Insert(const K&amp; key) {
    if (_root == nullptr) { // 空树处理
        _root = new Node(key);
        return true;
    }
    
    Node* parent = nullptr;
    Node* cur = _root;
    while (cur) { // 搜索插入位置
        parent = cur;
        if (cur-&gt;_key &lt; key) cur = cur-&gt;_right;
        else if (cur-&gt;_key &gt; key) cur = cur-&gt;_left;
        else return false; // 已存在相同key
    }
    
    cur = new Node(key); // 创建新节点
    // 链接到父节点
    if (parent-&gt;_key &lt; key) 
        parent-&gt;_right = cur;
    else 
        parent-&gt;_left = cur;
    return true;
}</code></pre>
    <hr/>
    <h3>
     3.2 删除操作（迭代版）
    </h3>
    <blockquote>
     <p>
      <strong>
       删除策略
      </strong>
      ：
     </p>
     <ul>
      <li>
       <p>
        <strong>
         单子树节点
        </strong>
        ：直接提升子树
       </p>
      </li>
      <li>
       <p>
        <strong>
         双子树节点
        </strong>
        ：寻找右子树最小节点进行值替换
       </p>
      </li>
      <li>
       <p>
        <strong>
         根节点处理
        </strong>
        ：需要特殊处理根指针
       </p>
      </li>
     </ul>
    </blockquote>
    <pre><code class="language-cpp">		bool Erase(const K&amp; key)
		{
			Node* parent = nullptr;
			Node* cur = _root;
			while (cur)
			{
				if (cur-&gt;_key &lt; key)
				{
					parent = cur;
					cur = cur-&gt;_right;
				}
				else if (cur-&gt;_key &gt; key)
				{
					parent = cur;
					cur = cur-&gt;_left;
				}
				else
				{
					// 1、左为空
					// 2、右为空
					// 3、左右都不为空，替换删除
					if (cur-&gt;_left == nullptr)
					{
						//if (parent == nullptr)
						if (cur == _root)
						{
							_root = cur-&gt;_right;
						}
						else
						{
							if (parent-&gt;_left == cur)
							{
								parent-&gt;_left = cur-&gt;_right;
							}
							else
							{
								parent-&gt;_right = cur-&gt;_right;
							}
						}

						delete cur;
					}
					else if (cur-&gt;_right == nullptr)
					{
						//if (parent == nullptr)
						if (cur == _root)
						{
							_root = cur-&gt;_left;
						}
						else
						{
							if (parent-&gt;_left == cur)
							{
								parent-&gt;_left = cur-&gt;_left;
							}
							else
							{
								parent-&gt;_right = cur-&gt;_left;
							}
						}

						delete cur;
					}
					else
					{
						// 右子树的最小节点
						Node* parent = cur;
						Node* minRight = cur-&gt;_right;
						while (minRight-&gt;_left)
						{
							parent = minRight;
							minRight = minRight-&gt;_left;
						}

						cur-&gt;_key = minRight-&gt;_key;
						if (minRight == parent-&gt;_left)
						{
							parent-&gt;_left = minRight-&gt;_right;
						}
						else
						{
							parent-&gt;_right = minRight-&gt;_right;
						}
						delete minRight;
					}

					return true;
				}
			}
			return false;
		}</code></pre>

     <p>
      <strong>
       边界情况处理
      </strong>
     </p>
     <ol>
      <li>
       <p>
        <strong>
         空树处理
        </strong>
        ：初始时
        <code>
         cur = nullptr
        </code>
        ，直接返回
        <code>
         nullptr
        </code>
       </p>
      </li>
      <li>
       <p>
        <strong>
         键不存在处理
        </strong>
        ：循环自然结束返回
        <code>
         nullptr
        </code>
       </p>
      </li>
      <li>
       <p>
        <strong>
         重复键处理
        </strong>
        ：由于插入时禁止重复键，查找时不会遇到重复情况
       </p>
      </li>
     </ol>
    </blockquote>
    <pre><code class="language-cpp">// KV版本二叉搜索树的查找操作（迭代实现）
Node* Find(const K&amp; key) {
    Node* cur = _root;          // 从根节点开始搜索
    while (cur != nullptr) {    // 当当前节点存在时循环
        if (key &gt; cur-&gt;_key) { // 目标键大于当前节点键
            cur = cur-&gt;_right;  // 转向右子树搜索
        } 
        else if (key &lt; cur-&gt;_key) { // 目标键小于当前节点键
            cur = cur-&gt;_left;  // 转向左子树搜索
        }
        else {                  // 找到匹配键
            return cur;         // 返回目标节点指针
        }
    }
    return nullptr;             // 遍历结束未找到返回空指针
}</code></pre>

    <h3>
     4.1 中序遍历
    </h3>
    <blockquote>
     <p>
      中序遍历是一种遍历二叉树的方式，按照访问左子树——根节点——右子树的顺序进行。对于二叉搜索树，中序遍历会按照节点键值的升序访问所有节点，这是一种非常有用的遍历方式，因为它可以生成树的有序列表。
     </p>
    </blockquote>
    <blockquote>
     <p>
      以下是中序遍历的递归实现代码：
     </p>
     <pre><code class="language-cpp">// 中序遍历（递归实现）
void InOrder()
{
	_InOrder(_root);
	cout &lt;&lt; endl;</code></pre>
     <p>
      在这段代码中，
      <code>
       InOrder
      </code>
      函数通过调用辅助函数
      <code>
       _InOrder
      </code>
      来实现递归遍历。
      <code>
       _InOrder
      </code>
      函数首先递归遍历左子树，然后打印当前节点的键值，最后递归遍历右子树。通过这种方式，可以按照从小到大的顺序打印树中的所有节点。
     </p>
    </blockquote>
    <h3>
     4.2 树的销毁与拷贝
    </h3>
    <blockquote>
     <p>
      销毁二叉搜索树是指释放树中所有节点占用的内存空间。销毁操作通常在二叉搜索树不再需要时调用，以确保不浪费内存资源。销毁操作可以通过递归实现，从根节点开始，先递归销毁左子树和右子树，然后再销毁根节点。
     </p>
    </blockquote>
    <blockquote>
     <p>
      以下是销毁二叉搜索树的代码实现：
     </p>
     <pre><code class="language-cpp">template&lt;class K&gt;
void BSTree&lt;K&gt;::Destroy(BSTreeNode&lt;K&gt;* root) {
    if (root == nullptr) {
        return;
    }
    Destroy(root-&gt;_left);
    Destroy(root-&gt;_right);
    delete root;
}</code></pre>
     <p>
      在这段代码中，
      <code>
       Destroy
      </code>
      函数通过递归调用自身，先销毁左子树，再销毁右子树，最后销毁根节点，从而释放整个树的内存空间。
     </p>
    </blockquote>
    <blockquote>
     <p>
      拷贝二叉搜索树是指创建一个新的二叉搜索树，其结构与原树完全相同，节点的值也相同。拷贝操作通常在需要复制二叉搜索树时使用。拷贝操作同样可以通过递归实现，从根节点开始，先递归拷贝左子树和右子树，然后再创建新节点，并将其键值设置为原节点的键值。
     </p>
    </blockquote>
    <blockquote>
     <p>
      以下是拷贝二叉搜索树的代码实现：
     </p>
     <pre><code class="language-cpp">template&lt;class K&gt;
BSTreeNode&lt;K&gt;* BSTree&lt;K&gt;::Copy(BSTreeNode&lt;K&gt;* root) {
    if (root == nullptr) {
        return nullptr;
    }
    BSTreeNode&lt;K&gt;* newRoot = new BSTreeNode&lt;K&gt;(root-&gt;_key);
    newRoot-&gt;_left = Copy(root-&gt;_left);
    newRoot-&gt;_right = Copy(root-&gt;_right);
    return newRoot;
}</code></pre>
     <p>
      在这段代码中，
      <code>
       Copy
      </code>
      函数通过递归调用自身，先拷贝左子树，再拷贝右子树，最后创建新节点，并将其键值设置为原节点的键值，从而生成与原树结构相同的新树。
     </p>
    </blockquote>

    <h3>
     6.1 BSTree.h
    </h3>
    <pre><code class="language-cpp">#pragma once
#define _CRT_SECURE_NO_WARNINGS
#include&lt;iostream&gt;
using namespace std;
/*
* 声明并定义 命名空间 K
* 实现了 BSTree 类，这个类主要用于存储和操作单一的关键字（Key）。
* 每个节点包含一个键 _key，并且提供了插入（Insert）、查找（Find）和删除（Erase）等操作。
* 主要用于简单的关键字搜索和管理。
*/  
namespace K
{
	// 定义二叉搜索树结点模板结构体 K
	template&lt;class K&gt;
	struct BSTreeNode
	{
		BSTreeNode&lt;K&gt;* _left;  // 指针：左孩子
		BSTreeNode&lt;K&gt;* _right; // 指针：右孩子
		K _key; // 数据：存储该结点实际数据(单一的关键字)

		// 构建函数
		BSTreeNode(const K&amp; key)
			:_left(nullptr)
			, _right(nullptr)
			,_key(key)
		{}
	};

	// 定义二叉搜索树模板类 K
	template&lt;class K&gt;
	class BSTree
	{
		// 使用Node作为二叉搜索树结点的别名
		typedef BSTreeNode&lt;K&gt; Node;
	public:
		//K 结构体结点 需要默认构造函数，以便在没有提供关键字的情况下创建节点。
		一、默认成员函数
		// 1.1 构造函数
		BSTree()
			:_root(nullptr)
		{}
		// 1.2 拷贝构造
		BSTree(const BSTree&lt;K&gt;&amp; t)
		{
			_root = Copy(t._root);
		}
		// 1.3 操作符重载
		BSTree&lt;K&gt;&amp; operator=(BSTree&lt;K&gt;&amp; t)
		{
			swap(_root, t._root);
			return *this;
		}
		// 1.4 析构函数
		~BSTree()
		{
			Destroy(_root);
			_root = nullptr;
		}
		二、二叉搜索树操作
		// 2.1 插入（迭代实现）
		// 根节点为空，创建新结点；键值已存在，返回false；键值不存在，找到合适位置插入新结点
		bool Insert(const K&amp; key)
		{
			if (_root == nullptr)
			{
				_root = new Node(key);
				return true;
			}
			Node* parent = nullptr;
			Node* cur = _root;
			while (cur)
			{
				// 找到插入的位置
				if (cur-&gt;_key &lt; key)
				{
					parent = cur;
					cur = cur-&gt;_right;
				}
				else if (cur-&gt;_key &gt; key)
				{
					parent = cur;
					cur = cur-&gt;_left;
				}
				else return false;
			}
			// 插入结点
			cur = new Node(key);
			if (parent-&gt;_key &lt; key)	parent-&gt;_right = cur;
			else parent-&gt;_left = cur;

			return true;
		}
		// 2.2 删除（迭代实现）
		bool Erase(const K&amp; key)
		{
			Node* parent = nullptr;
			Node* cur = _root;
			while (cur)
			{
				if (cur-&gt;_key &lt; key)
				{
					parent = cur;
					cur = cur-&gt;_right;
				}
				else if (cur-&gt;_key &gt; key)
				{
					parent = cur;
					cur = cur-&gt;_left;
				}
				else
				{
					// 1、左为空
					// 2、右为空
					// 3、左右都不为空，替换删除
					if (cur-&gt;_left == nullptr)
					{
						//if (parent == nullptr)
						if (cur == _root)
						{
							_root = cur-&gt;_right;
						}
						else
						{
							if (parent-&gt;_left == cur)
							{
								parent-&gt;_left = cur-&gt;_right;
							}
							else
							{
								parent-&gt;_right = cur-&gt;_right;
							}
						}

						delete cur;
					}
					else if (cur-&gt;_right == nullptr)
					{
						//if (parent == nullptr)
						if (cur == _root)
						{
							_root = cur-&gt;_left;
						}
						else
						{
							if (parent-&gt;_left == cur)
							{
								parent-&gt;_left = cur-&gt;_left;
							}
							else
							{
								parent-&gt;_right = cur-&gt;_left;
							}
						}

						delete cur;
					}
					else
					{
						// 右子树的最小节点
						Node* parent = cur;
						Node* minRight = cur-&gt;_right;
						while (minRight-&gt;_left)
						{
							parent = minRight;
							minRight = minRight-&gt;_left;
						}

						cur-&gt;_key = minRight-&gt;_key;
						if (minRight == parent-&gt;_left)
						{
							parent-&gt;_left = minRight-&gt;_right;
						}
						else
						{
							parent-&gt;_right = minRight-&gt;_right;
						}
						delete minRight;
					}

					return true;
				}
			}
			return false;
		}
		// 2.3 查找（迭代实现）
		// 从根节点开始找，找到返回true，否则返回 false
		bool Find(const K&amp; key)
		{
			Node* cur = _root;
			while (cur)
			{
				if (cur-&gt;_key &lt; key) cur = cur-&gt;_right;
				else if (cur-&gt;_key &gt; key) cur = cur-&gt;_left;
				else return true;
			}
			return false;
		}
		// 2.4 中序遍历（递归实现）
		void InOrder()
		{
			_InOrder(_root);
			cout &lt;&lt; endl;
		}
		// 2.5 插入（递归实现）
		bool InsertR(const K&amp; key)
		{
			return _InsertR(_root, key);
		}
		// 2.6 删除（递归实现）
		bool EraseR(const K&amp; key)
		{
			return _EraseR(_root, key);
		}
		// 2.7 查找（递归实现）
		bool FindR(const K&amp; key)
		{
			return _EraseR(_root, key);
		}
	private:
		三、二叉搜索树结点操作
		// 3.1 销毁结点
		void Destroy(Node* root)
		{
			if (root == nullptr)
				return;

			Destroy(root-&gt;_left);
			Destroy(root-&gt;_right);
			delete root;
		}
		// 3.2 拷贝结点
		Node* Copy(Node* root)
		{
			if (root == nullptr)
				return nullptr;

			Node* newRoot = new Node(root-&gt;_key);
			newRoot-&gt;_left = Copy(root-&gt;_left);
			newRoot-&gt;_right = Copy(root-&gt;_right);

			return newRoot;
		}
		// 3.3 删除结点
		bool _EraseR(Node*&amp; root, const K&amp; key)
		{
			if (root == nullptr)
			{
				return false;
			}

			if (root-&gt;_key &lt; key)
			{
				return _EraseR(root-&gt;_right, key);
			}
			else if (root-&gt;_key &gt; key)
			{
				return _EraseR(root-&gt;_left, key);
			}
			else
			{
				Node* del = root;
				if (root-&gt;_right == nullptr)
				{
					root = root-&gt;_left;
				}
				else if (root-&gt;_left == nullptr)
				{
					root = root-&gt;_right;
				}
				else
				{
					Node* minRight = root-&gt;_right;
					while (minRight-&gt;_left)
					{
						minRight = minRight-&gt;_left;
					}

					swap(root-&gt;_key, minRight-&gt;_key);

					// 转换成在子树中去删除节点
					return _EraseR(root-&gt;_right, key);
				}

				delete del;

				return true;
			}
		}
		// 3.4 插入结点
		bool _InsertR(Node*&amp; root, const K&amp; key)
		{
			if (root == nullptr)
			{
				root = new Node(key);
				return true;
			}

			if (root-&gt;_key &lt; key)
				return _InsertR(root-&gt;_right, key);
			else if (root-&gt;_key &gt; key)
				return _InsertR(root-&gt;_left, key);
			else
				return false;
		}
		// 3.5 查找结点
		bool _FindR(Node* root, const K&amp; key)
		{
			if (root == nullptr)
				return false;

			if (root-&gt;_key &lt; key)
				return _FindR(root-&gt;_right, key);
			else if (root-&gt;_key &gt; key)
				return _FindR(root-&gt;_left, key);
			else
				return true;
		}
		// 3.6 中序遍历结点
		void _InOrder(Node* root)
		{
			if (root == nullptr)
				return;

			_InOrder(root-&gt;_left);
			cout &lt;&lt; root-&gt;_key &lt;&lt; " ";
			_InOrder(root-&gt;_right);
		}
	private:
		Node* _root = nullptr;
	};

}

/*
* 实现了 BSTree 类，这个类用于存储和操作键值对（Key-Value Pair）。
*每个节点包含一个键 _key 和一个值 _value，并且提供了插入键值对（Insert）、通过键查找节点（Find）等操作。
*主要用于需要关联数据存储和检索的场景，例如统计单词出现次数、存储配置信息等。
*/

namespace KV
{
	// 定义二叉搜索树结点模板结构体 KV
	template&lt;class K,class V&gt;
	struct BSTreeNode
	{
		BSTreeNode&lt;K, V&gt;* _left;
		BSTreeNode&lt;K, V&gt;* _right;
		K _key;   // 用于确定节点在二叉搜索树中的位置和排序，是树的索引和查找依据
		V _value; // 用于存储与 _key 相关的数据，提供额外的信息存储和检索能力

		// 构建函数
		BSTreeNode(const K&amp; key,const V&amp; value)
			:_left(nullptr)
			,_right(nullptr)
			,_key(key)
			,_value(value)
		{}
	};
	/*这里KV不需要默认成员函数
	* KV 结构体结点 由于总是需要提供键值对来创建节点，因此不需要默认构造函数。
	* 在实际编程中，构造函数的目的是为了确保对象在创建时能够正确初始化其成员变量。
	*对于包含多个成员变量的类来说，如果没有提供所有成员变量的初始化参数，则需要默认构造函数来确保对象能够被正确创建。
	*/
	// 定义二叉搜索树模板类 KV
	template&lt;class K, class V&gt;
	class BSTree
	{
		typedef BSTreeNode&lt;K, V&gt; Node;
	public:
		二叉搜索树操作
		// 插入
		bool Insert(const K&amp; key, const V&amp; value)
		{
			if (_root == nullptr)
			{
				_root = new Node(key, value);
				return true;
			}

			Node* parent = nullptr;
			Node* cur = _root;
			while (cur)
			{
				if (cur-&gt;_key &lt; key)
				{
					parent = cur;
					cur = cur-&gt;_right;
				}
				else if (cur-&gt;_key &gt; key)
				{
					parent = cur;
					cur = cur-&gt;_left;
				}
				else
				{
					return false;
				}
			}

			cur = new Node(key, value);
			if (parent-&gt;_key &lt; key)
			{
				parent-&gt;_right = cur;
			}
			else
			{
				parent-&gt;_left = cur;
			}

			return true;
		}

		Node* Find(const K&amp; key)
		{
			Node* cur = _root;
			while (cur)
			{
				if (cur-&gt;_key &lt; key)
				{
					cur = cur-&gt;_right;
				}
				else if (cur-&gt;_key &gt; key)
				{
					cur = cur-&gt;_left;
				}
				else
				{
					return cur;
				}
			}

			return nullptr;
		}

		void Inorder()
		{
			_Inorder(_root);
		}

		void _Inorder(Node* root)
		{
			if (root == nullptr)
				return;

			_Inorder(root-&gt;_left);
			cout &lt;&lt; root-&gt;_key &lt;&lt; ":" &lt;&lt; root-&gt;_value &lt;&lt; endl;
			_Inorder(root-&gt;_right);
		}
	private:
		Node* _root = nullptr;

	};
}</code></pre>
