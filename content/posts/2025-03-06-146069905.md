     踩坑：1、缩放在x、y方向只输出正值，为负值时的翻转与对称的成像需要通过剪切值的正反去修正；

    <pre><code>            // 提取线性变换部分
            double a = homMat[0];
            double b = homMat[1];
            double c = homMat[3];
            double d = homMat[4];

            // 计算X轴缩放和旋转角度
            Sx = Math.Sqrt(a * a + c * c);
            if (Sx &lt; 1e-10)
            { return -1; }
            // 计算旋转角度
            Phi = Math.Atan2(c, a); // 弧度

            // 计算Y轴缩放
            double cosPhi = Math.Cos(Phi);
            double sinPhi = Math.Sin(Phi);
            Sy = Math.Sqrt(Math.Pow((b * cosPhi + d * sinPhi), 2) + Math.Pow((d * cosPhi - b * sinPhi), 2));

            // 计算水平剪切参数
            double sd1 = (cosPhi * b) + (sinPhi * d);
            double sd2 = (cosPhi * d - sinPhi * b);
            Theta = Math.Atan(sd1 / sd2);

            // 根据缩放只输出正值对水平剪切的影响，修改剪切值
            if (sd2 &lt;= 0)
            {
                Theta = sd1 &lt;= 0 ? (Math.PI + Theta) : (-1 * (Math.PI + Theta));
            }
            else
            {
                Theta = (Theta * -1);
            }

            // 将弧度标准化到[-π, π]范围
            if (Phi &gt; Math.PI)
            {
                Phi -= (2 * Math.PI);
            }
            else if (Phi &lt; -Math.PI)
            {
                Phi += (2 * Math.PI);
            }
            if (Theta &gt; Math.PI)
            {
                Theta -= (2 * Math.PI);
            }
            else if (Theta &lt; -Math.PI)
            {
                Theta += (2 * Math.PI);
            }
            // 将弧度转为角度
            if (AngType == 1)
            {
                Phi = Phi * (180 / Math.PI);
                Theta = Theta * (180 / Math.PI);
            }
            return 0;
        }
        catch (Exception e)
        {
            return -1;
        }
    }
