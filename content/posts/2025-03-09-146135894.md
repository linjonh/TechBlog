     509. 斐波那契数
    </h3>
    <pre><code>class Solution {
public:
    int fib(int n) {
        if(n==0){
            return 0;
        }else if(n==1){
            return 1;
        }
        vector&lt;int&gt; v(n+1, 0);
        v[1] = 1;
        for(int i{2}; i&lt;v.size(); i++){
            v[i] = v[i-1] + v[i-2];
        }
        return v.back();
    }
};
</code></pre>

    <pre><code>class Solution {
public:
   int climbStairs(int n) {
       vector&lt;int&gt; dp(n+1, 0);
       dp[0] = 1;
       dp[1] = 1;
       for(int i{2}; i&lt;=n; i++){
           dp[i] = dp[i-1] + dp[i-2];
       }
       return dp[n];
   }
};
</code></pre>
    <h3>
     <a id="746__45">
     </a>
     746. 使用最小花费爬楼梯
    </h3>
    <ol>
     <li>
      动态规划五步曲
     </li>
    </ol>
    <pre><code>class Solution {
public:
    int minCostClimbingStairs(vector&lt;int&gt;&amp; cost) {
        vector&lt;int&gt; dp(cost.size()+1, 0);
        dp[0] = 0;
        dp[1] = 0;
        for(int i{2}; i&lt;dp.size(); i++){
            dp[i] = min(dp[i-1]+cost[i-1], dp[i-2]+cost[i-2]);
        }
        return dp.back();
    }
};
</code></pre>
