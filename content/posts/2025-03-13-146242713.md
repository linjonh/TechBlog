    <pre><code class="language-cpp">class Solution {
public:
    string makeSmallestPalindrome(string s) {
        int left=0;
        int right=s.size()-1;
        while(left&lt;right){
            if(s[left]&lt;s[right])s[right]=s[left];
            else if(s[left]&gt;s[right])s[left]=s[right];
            left++;
            right--;
        }
        return s;
    }
};</code></pre>
