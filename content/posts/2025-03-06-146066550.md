```TypeScript
// ClickButtonComponent.ets

import { Component, State } from '@ohos/arkui';

@Component
struct ClickButtonComponent {
  @State message: string = 'Hello, ArkUI!';

  build() {
    Column() {
      Text(this.message)
        .fontSize(20)
        .margin({ top: 20 })
      Button('Click Me')
        .width(200)
        .height(60)
        .margin({ top: 20 })
        .onClick(() => {
          this.showMessage();
        });
    }
  }

  showMessage() {
    this.message = 'Button Clicked!';
  }
}

export default ClickButtonComponent;

```
