    <h3>
     题目描述
    </h3>
    <p>
     给出两个序列 pushed 和 poped 两个序列，其取值从 1 到 n(n≤100000)。已知入栈序列是 pushed，如果出栈序列有可能是 poped，则输出
     <code>
      Yes
     </code>
     ，否则输出
     <code>
      No
     </code>
     。为了防止骗分，每个测试点有多组数据，不超过 5 组。
    </p>
    <h3>
     输入格式
    </h3>
    <p>
     第一行一个整数 q，询问次数。
    </p>
    <p>
     接下来 q 个询问，对于每个询问：
    </p>
    <p>
     第一行一个整数 n 表示序列长度；
    </p>
    <p>
     第二行 n 个整数表示入栈序列；
    </p>
    <p>
     第三行 n 个整数表示出栈序列；
    </p>
    <h3>
     输出格式
    </h3>
    <p>
     对于每个询问输出答案。
    </p>
    <h3>
     输入输出样例
    </h3>
    <p>
     <strong>
      输入 #1
     </strong>
     复制
    </p>
    <pre>2
5
1 2 3 4 5
5 4 3 2 1
4
1 2 3 4
2 4 1 3</pre>
    <p>
     <strong>
      输出 #1
     </strong>
     复制
    </p>
    <pre>Yes
No</pre>
    <p>
    </p>
    <p>
     代码实现：
    </p>
    <p>
     #include &lt;iostream&gt;
     <br/>
     #include &lt;stack&gt;
     <br/>
     #include &lt;vector&gt;
     <br/>
     /* run this program using the console pauser or add your own getch, system("pause") or input loop */
     <br/>
     using namespace std;
     <br/>
     int main(int argc, char** argv)
     <br/>
     {
     <!-- -->
     <br/>
     int m,n,i,j,c[100001];
     <br/>
     cin&gt;&gt;m;
     <br/>
     stack&lt;int&gt; sta;
     <br/>
     for(j=0;j&lt;m;j++)
     <br/>
     {
     <!-- -->
     <br/>
     cin&gt;&gt;n;
     <br/>
     int a[n],b[n];
     <br/>
     for(i=0;i&lt;n;i++)
     <br/>
     {
     <!-- -->
     <br/>
     cin&gt;&gt;a[i];
     <br/>
     }
     <br/>
     for(i=0;i&lt;n;i++)
     <br/>
     {
     <!-- -->
     <br/>
     cin&gt;&gt;b[i];
     <br/>
     }
     <br/>
     int head=0;
     <br/>
     for(i=0;i&lt;n;i++)
     <br/>
     {
     <!-- -->
     <br/>
     sta.push(a[i]);
     <br/>
     while(sta.top()==b[head])
     <br/>
     {
     <!-- -->
     <br/>
     sta.pop();
     <br/>
     head++;
     <br/>
     if(sta.empty())
     <br/>
     {
     <!-- -->
     <br/>
     break;
     <br/>
     }
     <br/>
     }
     <br/>
     }
     <br/>
     if(sta.empty())
     <br/>
     {
     <!-- -->
     <br/>
     c[j]=1;
     <br/>
     }
     <br/>
     else
     <br/>
     {
     <!-- -->
     <br/>
     c[j]=0;
     <br/>
     }
     <br/>
     while (!sta.empty()) {
     <!-- -->
     <br/>
     sta.pop();
     <br/>
     }
     <br/>
     }
     <br/>
     for(i=0;i&lt;m;i++)
     <br/>
     {
     <!-- -->
     <br/>
     if(c[i]==0)
     <br/>
     {
     <!-- -->
     <br/>
     cout&lt;&lt;"No"&lt;&lt;endl;
     <br/>
     }
     <br/>
     else
     <br/>
     {
     <!-- -->
     <br/>
     cout&lt;&lt;"Yes"&lt;&lt;endl;
     <br/>
     }
     <br/>
     }
     <br/>
     return 0;
     <br/>
     }
