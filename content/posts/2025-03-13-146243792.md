     下面是基于
     <strong>
      Spring Boot
     </strong>
     使用
     <strong>
      RabbitMQ
     </strong>
     实现这七种模式的示例代码。假设已经引入了以下依赖：
    </p>
    <h4>
     Maven 依赖
    </h4>
    <pre><code class="language-XML">&lt;dependency&gt;
    &lt;groupId&gt;org.springframework.boot&lt;/groupId&gt;
    &lt;artifactId&gt;spring-boot-starter-amqp&lt;/artifactId&gt;
&lt;/dependency&gt;
</code></pre>
    <p>
    </p>
    <h3>
     <strong>
      1. Hello World! 模式
     </strong>
    </h3>
    <p>
     生产者和消费者直接发送和接收消息。
    </p>
    <p>
     配置类
    </p>
    <pre><code class="language-java">@Configuration
public class RabbitConfig {
    public static final String QUEUE_NAME = "hello";

    @Bean
    public Queue helloQueue() {
        return new Queue(QUEUE_NAME);
    }
}
</code></pre>
    <p>
     生产者
    </p>
    <pre><code class="language-java">@RestController
@RequestMapping("/hello")
public class HelloProducer {

    @Autowired
    private AmqpTemplate rabbitTemplate;

    @GetMapping("/send")
    public String send() {
        rabbitTemplate.convertAndSend(RabbitConfig.QUEUE_NAME, "Hello RabbitMQ!");
        return "Message sent!";
    }
}
</code></pre>
    <p>
     消费者
    </p>
    <pre><code class="language-java">@Component
public class HelloConsumer {

    @RabbitListener(queues = RabbitConfig.QUEUE_NAME)
    public void receive(String message) {
        System.out.println("Received: " + message);
    }
}
</code></pre>
    <h3>
     <strong>
      2. Work Queues（工作队列）
     </strong>
    </h3>
    <p>
     多个消费者从同一个队列中获取任务，进行任务分发。
    </p>
    <p>
     配置类
    </p>
    <pre><code class="language-java">@Configuration
public class WorkQueueConfig {
    public static final String WORK_QUEUE = "work_queue";

    @Bean
    public Queue workQueue() {
        return new Queue(WORK_QUEUE);
    }
}
</code></pre>
    <p>
     生产者
    </p>
    <pre><code class="language-java">@RestController
@RequestMapping("/work")
public class WorkQueueProducer {

    @Autowired
    private AmqpTemplate rabbitTemplate;

    @GetMapping("/send/{msg}")
    public String send(@PathVariable String msg) {
        rabbitTemplate.convertAndSend(WorkQueueConfig.WORK_QUEUE, msg);
        return "Work message sent!";
    }
}
</code></pre>
    <p>
     消费者（多个）
    </p>
    <pre><code class="language-java">@Component
public class WorkConsumer {

    @RabbitListener(queues = WorkQueueConfig.WORK_QUEUE)
    public void receive(String message) throws InterruptedException {
        System.out.println("Worker received: " + message);
        Thread.sleep(1000);  // 模拟任务耗时
    }
}
</code></pre>
    <h3>
     <strong>
      3. Publish/Subscribe（发布/订阅）
     </strong>
    </h3>
    <p>
     使用
     <strong>
      Fanout Exchange
     </strong>
     实现广播。
    </p>
    <p>
     配置类
    </p>
    <pre><code class="language-java">@Configuration
public class FanoutConfig {
    public static final String FANOUT_EXCHANGE = "fanout_exchange";

    @Bean
    public FanoutExchange fanoutExchange() {
        return new FanoutExchange(FANOUT_EXCHANGE);
    }

    @Bean
    public Queue queue1() {
        return new Queue("fanout.queue1");
    }

    @Bean
    public Queue queue2() {
        return new Queue("fanout.queue2");
    }

    @Bean
    public Binding binding1() {
        return BindingBuilder.bind(queue1()).to(fanoutExchange());
    }

    @Bean
    public Binding binding2() {
        return BindingBuilder.bind(queue2()).to(fanoutExchange());
    }
}
</code></pre>
    <p>
     生产者
    </p>
    <pre><code class="language-java">@RestController
@RequestMapping("/fanout")
public class FanoutProducer {

    @Autowired
    private AmqpTemplate rabbitTemplate;

    @GetMapping("/send")
    public String send() {
        rabbitTemplate.convertAndSend(FanoutConfig.FANOUT_EXCHANGE, "", "Fanout message!");
        return "Fanout message sent!";
    }
}
</code></pre>
    <h3>
     <strong>
      4. Routing（路由模式）
     </strong>
    </h3>
    <p>
     使用
     <strong>
      Direct Exchange
     </strong>
     和路由键实现定向投递。
    </p>
    <p>
     配置类
    </p>
    <pre><code class="language-java">@Configuration
public class DirectConfig {
    public static final String DIRECT_EXCHANGE = "direct_exchange";

    @Bean
    public DirectExchange directExchange() {
        return new DirectExchange(DIRECT_EXCHANGE);
    }

    @Bean
    public Queue errorQueue() {
        return new Queue("direct.error");
    }

    @Bean
    public Queue infoQueue() {
        return new Queue("direct.info");
    }

    @Bean
    public Binding errorBinding() {
        return BindingBuilder.bind(errorQueue()).to(directExchange()).with("error");
    }

    @Bean
    public Binding infoBinding() {
        return BindingBuilder.bind(infoQueue()).to(directExchange()).with("info");
    }
}
</code></pre>
    <h3>
     <strong>
      5. Topics（主题模式）
     </strong>
    </h3>
    <p>
     使用
     <strong>
      Topic Exchange
     </strong>
     实现多级路由。
    </p>
    <p>
     配置类
    </p>
    <pre><code class="language-java">@Configuration
public class TopicConfig {
    public static final String TOPIC_EXCHANGE = "topic_exchange";

    @Bean
    public TopicExchange topicExchange() {
        return new TopicExchange(TOPIC_EXCHANGE);
    }

    @Bean
    public Queue topicQueue1() {
        return new Queue("topic.queue1");
    }

    @Bean
    public Queue topicQueue2() {
        return new Queue("topic.queue2");
    }

    @Bean
    public Binding binding1() {
        return BindingBuilder.bind(topicQueue1()).to(topicExchange()).with("*.error");
    }

    @Bean
    public Binding binding2() {
        return BindingBuilder.bind(topicQueue2()).to(topicExchange()).with("log.#");
    }
}
</code></pre>
    <h3>
     <strong>
      6. RPC（远程过程调用）
     </strong>
    </h3>
    <p>
     生产者发送请求，消费者处理后返回响应。
    </p>
    <h4>
     RPC 服务端
    </h4>
    <pre><code class="language-java">@Component
public class RpcServer {

    @RabbitListener(queues = "rpc_queue")
    public String process(String message) {
        return "Processed: " + message;
    }
}
</code></pre>
    <h4>
     RPC 客户端
    </h4>
    <pre><code class="language-java">@RestController
@RequestMapping("/rpc")
public class RpcClient {

    @Autowired
    private AmqpTemplate rabbitTemplate;

    @GetMapping("/send/{msg}")
    public String send(@PathVariable String msg) {
        Object response = rabbitTemplate.convertSendAndReceive("rpc_queue", msg);
        return "RPC response: " + response;
    }
}
</code></pre>
    <h3>
     <strong>
      7. Publisher Confirms（发布者确认）
     </strong>
    </h3>
    <p>
     确保消息成功发送到 RabbitMQ 服务器。
    </p>
    <h4>
     配置类
    </h4>
    <pre><code class="language-java">@Configuration
public class ConfirmConfig {

    @Bean
    public CachingConnectionFactory connectionFactory() {
        CachingConnectionFactory factory = new CachingConnectionFactory("localhost");
        factory.setPublisherConfirms(true);
        return factory;
    }

    @Bean
    public RabbitTemplate rabbitTemplate(CachingConnectionFactory connectionFactory) {
        RabbitTemplate template = new RabbitTemplate(connectionFactory);
        template.setConfirmCallback((correlationData, ack, cause) -&gt; {
            if (ack) {
                System.out.println("Message confirmed");
            } else {
                System.err.println("Message failed: " + cause);
            }
        });
        return template;
    }
}
</code></pre>
