    {
        if (a &gt; max)//判定最高分
        {
            max = a;
        }
        if (a &lt; small)//判定最低分
        {
            small = a;

        count++;//计数器
        if (count == 7)//计数器=7时代表一组的分数好了可以进行计算

            printf("%.2f\n", (sum - max - small) / 5.0);
            count = 0;//重置
            max = 0;//重置
            small = 100;//重置
            sum = 0;//重置
        }

}

int main()
{
    int n = 0;
    int arr[50] = { 0 };
    int m = 0;//要插入的数据
    //输入数据
    scanf("%d", &amp;n);
    int i = 0;
    for (i = 0; i &lt; n; i++)
    {
        scanf("%d", &amp;arr[i]);
    }
    scanf("%d", &amp;m);
    //处理
    for (i = n; i &gt; 0; i--)
    {
        if (arr[i - 1] &gt; m)
        {
            arr[i] = arr[i - 1];
        }
        else
        {
            arr[i] = m;
            break;
        }
    }
    //如果插入数字小于所有的数字
    if (i == 0)
    {
        arr[i] = m;
    }
    //打印
    for (i = 0; i &lt; n + 1; i++)
    {
        printf("%d ", arr[i]);

        return 0;

    //输入
    scanf("%d %d", &amp;m, &amp;n);
    for (int i = 0; i &lt; m; i++)
    {
        for (int j = 0; j &lt; n; j++)

            scanf("%d ", &amp;arr1[i][j]);
