    </p>
    <p class="img-center">
     <img alt="" height="514" src="https://i-blog.csdnimg.cn/direct/917b06e1c4d84c1cbd4a6e8cfaa628e0.png" width="472"/>

    <p>
     思路：
    </p>
    <p>
     <span style="background-color:#ffd900">
      我们利用栈来解答，首先遍历数组我们找左括号把他放到栈里，然后再找游括号来看看是否与之匹配。如果遍历完之后栈为空代表匹配完成返回true，反之返回false。
     </span>
    </p>
    <p>
     注意如果没学习过栈的可以看看这篇博客：
     <a href="https://blog.csdn.net/2403_84958571/article/details/146190436?fromshare=blogdetail&amp;sharetype=blogdetail&amp;sharerId=146190436&amp;sharerefer=PC&amp;sharesource=2403_84958571&amp;sharefrom=from_link" title="【C++】数据结构 栈的实现-CSDN博客">
      【C++】数据结构 栈的实现-CSDN博客
     </a>
    </p>
    <p>
     代码实例：
    </p>
    <pre><code class="language-cpp">typedef char STDataType;
typedef struct Stack
{
	STDataType* arr;
	int top;//指向栈顶的结构
	int capacity;//容量
}ST;

//栈的初始化
void STInit(ST* ps)
{
	ps-&gt;arr = NULL;
	ps-&gt;top = ps-&gt;capacity = 0;
}
//入栈-栈顶
void StackPush(ST* ps, STDataType x)
{
	assert(ps);
	if (ps-&gt;top == ps-&gt;capacity)
	{
		//空间不够-增容
		int newCapacity = ps-&gt;capacity == 0 ? 4 : 2*ps-&gt;capacity;
		STDataType* tmp = (STDataType*)realloc(ps-&gt;arr, newCapacity * sizeof(STDataType));
		if (tmp == NULL)
		{
			perror("realloc fail!");
			exit(1);
		}
		ps-&gt;arr = tmp;
		ps-&gt;capacity = newCapacity;
	}
	//空间够了
	ps-&gt;arr[ps-&gt;top++] = x;
}
//判断栈是否为空
bool StackEmpty(ST* ps)
{
	assert(ps);
	return ps-&gt;top == 0;
}
//拿栈顶的数据
STDataType StackTop(ST* ps)
{
	assert(!StackEmpty(ps));
	return ps-&gt;arr[ps-&gt;top - 1];
}
//栈的销毁
void STDestroy(ST* ps)
{
	if (ps-&gt;arr)
		free(ps-&gt;arr);
	ps-&gt;arr = NULL;
	ps-&gt;top = ps-&gt;capacity = 0;
}
//出栈-栈顶
void StackPop(ST* ps)
{
	assert(!StackEmpty(ps));
	--ps-&gt;top;
}
bool isValid(char* s) 
{
    ST st;
    STInit(&amp;st);
    char* pi=s;
    //遍历所有的字符串s
    while(*pi !='\0')
    {
        if(*pi=='(' ||*pi=='[' ||*pi=='{')
        {
            StackPush(&amp;st,*pi);
        }
        else
        {
            //找右括号并且与之匹配
            if(StackEmpty(&amp;st))//判断栈是否为空
            {
                STDestroy(&amp;st);
                return false;
            }
            char top=StackTop(&amp;st);
            if((top=='(' &amp;&amp; *pi != ')')
            || (top=='[' &amp;&amp; *pi !=']')
            || (top=='{' &amp;&amp; *pi !='}'))
            {
                STDestroy(&amp;st);
                return false;
            }
            StackPop(&amp;st);
        }
        pi++;
    }
    //栈为空，说明所有的左括号都已经匹配完，反之非有效字符
    bool ret =StackEmpty(&amp;st)?true:false;
    return ret;
}</code></pre>
