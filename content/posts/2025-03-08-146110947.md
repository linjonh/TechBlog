    <pre><code>class Solution {
public:
    int fib(int n) {
        std::vector&lt;int&gt; dp(n+1);//没有考虑n=0的时候不能初始化dp【1】
        if(n==0){
            return 0;
        }
        dp[0] = 0;
        dp[1] = 1;
        if(n == 0||n==1){
            return dp[n];
        }
        for(int i = 2;i&lt;=n;i++){//for先i&lt;=n；然后i++
            dp[i] = dp[i-1]+dp[i-2];
        }
        return dp[n];

};</code></pre>

    <h3>
     第三题：杨辉三角
    </h3>
    <p>
     刚开始在怀疑怎么输出
    </p>
    <pre>[[1],[1,1],[1,2,1],[1,3,3,1],[1,4,6,4,1]]，</pre>
    <p>
     但是函数返回vector&lt;vector&lt;int&gt;&gt;，其实写[[]]的函数已经内置了
    </p>
    <p>
     缺陷：运行时报错，
     <strong>
      <span style="background-color:#ffd900">
       初始化数组numRows，所以i应该是&lt;numRows
      </span>
     </strong>
    </p>
    <pre><code>class Solution {
public:
    vector&lt;vector&lt;int&gt;&gt; generate(int numRows) {
        vector&lt;vector&lt;int&gt;&gt; dp(numRows);
        for(int i = 0;i&lt;numRows;i++){//i&lt;
            dp[i].resize(i + 1);
            dp[i][0] = 1;
            dp[i][i] = 1;
            for (int j = 1; j &lt; i; j++) {
                dp[i][j] = dp[i-1][j-1]+dp[i-1][j];
            }
        }
        return dp;
    }
};</code></pre>
