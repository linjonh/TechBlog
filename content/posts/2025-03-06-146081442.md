{

    QFile file1;

    QFile file2;

    QFileInfo fileInfo(filename);
    QString file_end = fileInfo.fileName();//获取文件名

    QString path="D:/QTproject/"+file_end;//拼接文件路径+文件名

    file2.setFileName(path);
    file2.open(QIODevice::WriteOnly);
    double size=0;
    while(1)
    {
        char arr[1024]="";
        int res=file1.read(arr,1024);
        if(res==0){qDebug()&lt;&lt;"传输数据大小: "&lt;&lt;size;return;}
        file2.write(arr,1024);
        size+=res;
       // qDebug()&lt;&lt;"传输数据大小: "&lt;&lt;size;
    }

    file1.close();
    file2.close();
}

    event-&gt;acceptProposedAction();

void Widget::dropEvent(QDropEvent *event)
{
    const QMimeData* mimedata=event-&gt;mimeData();//获取文件路径

    QList&lt;QUrl&gt; list=mimedata-&gt;urls();//获取所有的文件地址

    QUrl url=list[0];//获取第几个文件具体地址

    filename=url.toLocalFile();//转化成字符串类型

    emit send(filename);

    thread-&gt;start();
}
