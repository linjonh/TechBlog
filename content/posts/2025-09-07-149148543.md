    string optimalDivision(vector<int>& nums) 
    {

        if(n == 1) return to_string(nums[0]);
        else if(n == 2) return to_string(nums[0]) + "/" + to_string(nums[1]);
        string ans = to_string(nums[0]) + "/(";

        return ans;
