      10分钟实现学生管理系统并实现
      <strong>
       <span style="color:#fe2c24">

      功能

    <pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;string&gt;
#include &lt;algorithm&gt;

// 学生类
struct Student {
    int id;// 学号
    std::string name;// 名字
    double score;// 分数
    // 打印学生信息
    void print() const {
        std::cout &lt;&lt; "ID: " &lt;&lt; id &lt;&lt; ", Name: " &lt;&lt; name &lt;&lt; ", Score: " &lt;&lt; score &lt;&lt; std::endl;
    }
};

// 管理员类
class Admin {
public:
    // 添加学生
    void addStudent(int id, const std::string&amp; name, double score) {
        students.push_back({ id, name, score });// 会调用默认的构造函数生成匿名学生
        std::cout &lt;&lt; "添加学生成功!" &lt;&lt; std::endl;
    }
    // 删除学生
    void deleteStudent(int id) {
        auto it = std::remove_if(students.begin(), students.end(), [id](const Student&amp; s) {
            return s.id == id;
         });
        if (it != students.end()) {
            students.erase(it，students.end());
            std::cout &lt;&lt; "删除学生成功!" &lt;&lt; std::endl;
        }
        else {
            std::cout &lt;&lt; "学生不存在!" &lt;&lt; std::endl;
        }
    }
    // 查找学生
    void findStudent(int id) const {
        auto it = std::find_if(students.begin(), students.end(), [id](const Student&amp; s) {
            return s.id == id;
        });

        if (it != students.end()) {
            it-&gt;print();// 调用学生的打印
        }
        else {
            std::cout &lt;&lt; "学生不存在!" &lt;&lt; std::endl;
        }
    }
    // 更新学生
    void updateStudent(int id, const std::string&amp; newName, double newScore) {
        auto it = std::find_if(students.begin(), students.end(), [id](const Student&amp; s) {
            return s.id == id;
         });

        if (it != students.end()) {
            it-&gt;name = newName;
            it-&gt;score = newScore;
            std::cout &lt;&lt; "更新学生成功!" &lt;&lt; std::endl;
        }
        else {
            std::cout &lt;&lt; "学生不存在!" &lt;&lt; std::endl;
        }
    }
    // 列出所有学生信息
    void listAllStudents() const {
        for (const auto&amp; student : students) {
            student.print();
        }
    }
private:
    std::vector&lt;Student&gt; students;//用数组存储学生
};

int main() {
    Admin manager;
    std::string name;
    int id = 0;
    double score = 0.0;

    int choice = 0;
    while (true) {
        std::cout &lt;&lt; "1. 添加学生" &lt;&lt; std::endl;
        std::cout &lt;&lt; "2. 删除学生" &lt;&lt; std::endl;
        std::cout &lt;&lt; "3. 查找学生" &lt;&lt; std::endl;
        std::cout &lt;&lt; "4. 更新学生" &lt;&lt; std::endl;
        std::cout &lt;&lt; "5. 列出所有学生信息" &lt;&lt; std::endl;
        std::cout &lt;&lt; "6. 退出" &lt;&lt; std::endl;
        std::cout &lt;&lt; "Enter your choice: ";
        std::cin &gt;&gt; choice;

        switch (choice) {
        case 1:
            std::cout &lt;&lt; "请输入学号 姓名 分数";
            std::cin &gt;&gt; id &gt;&gt; name &gt;&gt; score;
            manager.addStudent(id, name, score);
            break;
        case 2:
            std::cout &lt;&lt; "请输入要删除学生的学号";
            std::cin &gt;&gt; id;
            manager.deleteStudent(id);
            break;
        case 3:
            std::cout &lt;&lt; "请输入要查找学生的学号";
            std::cin &gt;&gt; id;
            manager.findStudent(id);
            break;
        case 4:
            std::cout &lt;&lt; "请输入要更新学生的学号，姓名，分数";
            std::cin &gt;&gt; id &gt;&gt; name &gt;&gt; score;
            manager.updateStudent(id, name, score);
            break;
        case 5:
            manager.listAllStudents();
            break;
        case 6:
            return 0;// 表正常退出
        default:
            std::cout &lt;&lt; "输入不合法，请重新输入" &lt;&lt; std::endl;
        }
    }
    return 0;
}</code></pre>
    <blockquote>
     <p>
      remove_if
     </p>
    </blockquote>
    <p>
     <strong>
     </strong>
     remove_if是 C++
     <code>
      &lt;
      <span style="color:#4da8ee">
       <strong>
        algorithm
       </strong>
      </span>
      &gt;
     </code>
     头文件中的一个算法，
     <strong>
      将不符合条件的元素移动到前面，并返回新的结束迭代器
     </strong>
     ，需要配合容器的
     <code>
      erase()
     </code>
     方法来真正删除元素
     <span style="background-color:#ffd900">
      (再删除一段访问)
     </span>
    </p>
    <pre><code class="language-cpp">#include &lt;algorithm&gt;  // std::remove_if
#include &lt;vector&gt;     // std::vector
#include &lt;iostream&gt;   // std::cout
using namespace std;

int main() {
    std::vector&lt;int&gt; vec = {1,3, 4, 5, 6};

    auto end_begin =  std::remove_if(vec.begin(), vec.end(), [](int x){
        return x % 2 == 0;//移除所有偶数
    });
    vec.erase(end_begin, vec.end());// 删除一段访问

    for (int n : vec) {
        std::cout &lt;&lt; n &lt;&lt; " ";  // 输出：1 3 5
    }
    cout &lt;&lt; endl;
}
</code></pre>
    <blockquote>
     <p>
      find_if
     </p>
    </blockquote>
    <p>
     <code>
      find_if
     </code>
     是 C++
     <code>
      &lt;
      <span style="color:#4da8ee">
       <strong>
        algorithm
       </strong>
      </span>
      &gt;
     </code>
     头文件中的一个算法，用于
     <span style="color:#fe2c24">
      <strong>
       查找
      </strong>
     </span>
     容器中
     <span style="color:#fe2c24">
      <strong>
       第一个
      </strong>
     </span>
     符合指定条件的元素，并返回指向该元素的迭代器。如果找不到，则返回
     <code>
      end()
     </code>
     迭代器
    </p>
    <pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;vector&gt;
#include &lt;algorithm&gt; // std::find_if

int main() {
    std::vector&lt;int&gt; vec = {1, 3, 5, 4, 6, 7};

    auto it = std::find_if(vec.begin(), vec.end(), [](int x){
        return x % 2 == 0;
    });

    if (it != vec.end()) {
        std::cout &lt;&lt; "找到第一个偶数：" &lt;&lt; *it &lt;&lt; std::endl; // 输出：找到第一个偶数：4
    } else {
        std::cout &lt;&lt; "未找到偶数" &lt;&lt; std::endl;
    }
}
