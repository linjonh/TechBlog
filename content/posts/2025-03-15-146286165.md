#include&lt;vector&gt;
#include&lt;algorithm&gt;
using namespace std;
struct time {
	int begin;
	int end;
};
bool com(struct time a, struct time b) {
	return a.end &lt; b.end;
}
int main() {
	int n;
	cin &gt;&gt; n;
	vector&lt;struct time&gt; s(n);
	for (int i = 0; i &lt; n; i++)cin &gt;&gt; s[i].begin &gt;&gt; s[i].end;
	sort(s.begin(), s.end(), com);
	int count = 1;
	int t = s[0].end;
	for (int i = 0; i &lt; n - 1; i++) {
		if (s[i+1].begin &gt;= t) {
			count++;
			t = s[i + 1].end;
		}
	}
	cout &lt;&lt; count &lt;&lt; endl;
	return 0;
}</code></pre>
