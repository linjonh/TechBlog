    <h4>
     分类：
    </h4>
    <ul>
     <li>
      算术仿函数
     </li>
     <li>
      关系仿函数
     </li>
     <li>
      逻辑仿函数
     </li>
    </ul>
    <h4>
     用法：
    </h4>
    <ul>
     <li>
      这些仿函数所产生的对象，用法和一般函数完全相同
     </li>
     <li>
      使用内建函数对象，需要引入头文件 #include&lt;functional&gt;
     </li>
    </ul>
    <h4>
     1. 算数仿函数
    </h4>
    <ul>
     <li>
      功能描述：
      <ul>
       <li>
        实现四则运算
       </li>
       <li>
        其中negate是一元运算，其他都是二元运算
       </li>
      </ul>
     </li>
     <li>
      仿函数原型：
     </li>
    </ul>
    <pre><code>template&lt;class T&gt; T plus&lt;T&gt; //执行加法运算。
template&lt;class T&gt; T minus&lt;T&gt; //执行减法运算。
template&lt;class T&gt; T multiplies&lt;T&gt; //执行乘法运算。
template&lt;class T&gt; T divides&lt;T&gt;  //执行除法运算。
template&lt;class T&gt; T modulus&lt;T&gt;  //执行取模运算。
template&lt;class T&gt; T negate&lt;T&gt;   //执行取反运算。</code></pre>
    <ul>
     <li>
      示例：
     </li>
    </ul>
    <pre><code>#include&lt;iostream&gt;
using namespace std;
#include &lt;functional&gt;

//内建函数对象 算术运算符

//negate 一元仿函数 取反仿函数
void test01{
    negate&lt;int&gt;n;
    cout &lt;&lt; n(50) &lt;&lt; endl;
}

//plus 二元仿函数  加法
void test02(){
    plus&lt;int&gt;p;
    cout &lt;&lt; p(10,20) &lt;&lt; endl;
}

int main(){
    
    test01();  // -50
    test02();  // 30
    return 0;
}</code></pre>
    <h4>
     2. 关系仿函数
    </h4>
    <ul>
     <li>
      功能描述：
      <ul>
       <li>
        实现关系对比
       </li>
      </ul>
     </li>
     <li>
      仿函数原型：
     </li>
    </ul>
    <pre><code>template&lt;class T&gt; bool equal_to&lt;T&gt;   //检查两个值是否相等。(等于)
template&lt;class T&gt; bool not_equal_to&lt;T&gt;  //检查两个值是否不相等。(不等于)
template&lt;class T&gt; bool less&lt;T&gt;  //检查一个值是否小于另一个值。（小于）
template&lt;class T&gt; bool greater&lt;T&gt; //检查一个值是否大于另一个值。（大于）
template&lt;class T&gt; bool less_equal&lt;T&gt;  //检查一个值是否小于等于另一个值。（小于等于）
template&lt;class T&gt; bool greater_equal&lt;T&gt;  //检查一个值是否大于等于另一个值。（大于等于）</code></pre>
    <ul>
     <li>
      示例：
     </li>
    </ul>
    <pre><code>#include&lt;iostream&gt;
using namespace std;
#include &lt;vector&gt;
#include &lt;algorithm&gt;
#include &lt;functional&gt;

//内建函数对象 关系仿函数

// 大于 greater

void test01(){
    vector&lt;int&gt;v;
    v.push_back(10);
    v.push_back(20);
    v.push_back(30);
    v.push_back(40);
    v.push_back(50);
    for(vector&lt;int&gt;::iterator it = v.begin(); it != v.end(); it++){
        cout &lt;&lt; *it &lt;&lt; " ";
    }
    cout &lt;&lt; endl;
    //降序 greater&lt;int&gt;()内建函数对象
    sort(v.begin(),v.end(),greater&lt;int&gt;());
    for(vector&lt;int&gt;::iterator it = v.begin(); it != v.end(); it++){
        cout &lt;&lt; *it &lt;&lt; " ";
    }
    cout &lt;&lt; endl;
}


int mian(){

    test01();
    return 0;
}
// 10 20 30 40 50
// 50 40 30 20 10</code></pre>
    <h4>
     3. 逻辑仿函数
    </h4>
    <ul>
     <li>
      功能描述：
      <ul>
       <li>
        实现逻辑运算
       </li>
      </ul>
     </li>
     <li>
      函数原型：
     </li>
    </ul>
    <pre><code>template&lt;class T&gt; bool logical_and&lt;T&gt;   //逻辑与
template&lt;class T&gt; bool logical_or&lt;T&gt;  //逻辑或
template&lt;class T&gt; bool logical_not&lt;T&gt;  //逻辑非
</code></pre>
    <ul>
     <li>
      示例：
     </li>
    </ul>
