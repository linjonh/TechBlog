```cpp
#include <iostream>
using namespace std;
int main()
{
  // 请在此输入您的代码
  int n;
  cin>>n;
  string s;
  while(n>0){
    n--;
    char c='A'+(n%26);
    s = c+s;
    n/=26;
  }
  cout<<s;
  return 0;
}
```
