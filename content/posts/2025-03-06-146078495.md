{
    int n;
    cin&gt;&gt;n;
    int s1, s2;
    cin&gt;&gt;s1&gt;&gt;s2;
    s1--, s2--;
    vector&lt;vector&lt;int&gt;&gt; adj1(n), adj2(n);
    int m1, m2;
    cin&gt;&gt;m1;
    vector&lt;PII&gt; edj1(m1);
    for(int i=0, u, v; i&lt;m1; i++){
        cin&gt;&gt;u&gt;&gt;v;
        u--, v--;
        if(u&gt;v) swap(u,v);
        edj1[i]={u,v};
        adj1[u].emplace_back(v);
        adj1[v].emplace_back(u);
    }
    cin&gt;&gt;m2;
    vector&lt;PII&gt; edj2(m2);
    for(int i=0, u, v; i&lt;m2; i++){
        cin&gt;&gt;u&gt;&gt;v;
        u--, v--;
        if(u&gt;v) swap(u,v);
        edj2[i]={u,v};
        adj2[u].emplace_back(v);
        adj2[v].emplace_back(u);
    }
    vector&lt;vector&lt;int&gt;&gt; d(n,vector&lt;int&gt;(n,1e9+7));
    priority_queue&lt;array&lt;int,3&gt;,vector&lt;array&lt;int,3&gt;&gt;,greater&lt;array&lt;int,3&gt;&gt;&gt; q;
    for(auto [u1,v1]:edj1){
        for(auto [u2,v2]:edj2){
            if(u1==u2 &amp;&amp; v1==v2){
                q.push({0,u1,u1});
            }
        }
    }
    vector&lt;vector&lt;bool&gt;&gt; vis(n,vector&lt;bool&gt;(n, false));
    int res=-1;
    while(!q.empty()){
        auto [d, u1, u2]=q.top();
        q.pop();
        if(vis[u1][u2]) continue;
        vis[u1][u2]=true;
        if(u1== s1 &amp;&amp; u2== s2){
            res=d;
        }
        d+=abs(u1-u2);
        for(auto v1:adj1[u1]){
            for(auto v2: adj2[u2]){
                q.push({d,v1,v2});
            }
        }
    }
    cout&lt;&lt;res&lt;&lt;'\n';
}

{
    int n;

    vector&lt;vector&lt;int&gt;&gt; adj1(n), adj2(n);

        cin&gt;&gt;u&gt;&gt;v;

    priority_queue&lt;array&lt;int,3&gt;,vector&lt;array&lt;int,3&gt;&gt;,greater&lt;array&lt;int,3&gt;&gt;&gt; q;
    q.push({0,s1,s2});
    int res=-1;
    while(!q.empty()){
        auto [d,u1,u2]=q.top();
        q.pop();
        if(dis[u1][u2]!=-1) continue;
        dis[u1][u2]=d;

        for(int v1:adj1[u1]){
            for(int v2:adj2[u2]){
                if(u1==u2 &amp;&amp; v1==v2){
                    res=d;
                    cout&lt;&lt;res&lt;&lt;'\n';
                    return;
                }
                q.push({d+abs(v1-v2),v1,v2});
            }
        }
    }
    cout&lt;&lt;res&lt;&lt;'\n';
}
