    <blockquote>
     <p>
      作为后端语言开发自然要连接数据库,对数据的增删改查,目前
      <code>
       nest
      </code>
      提供可以连接的数据库有
      <code>
       mySQL
      </code>
      和
      <code>
       mongoDB
      </code>
      ,
      <a href="https://docs.nestjs.com/techniques/sql" rel="nofollow">
       官网地址
      </a>
      ,本文介绍使用
      <code>
       nest
      </code>
      连接
      <code>
       mongoDB
      </code>
      数据库,官网介绍也很简单,
      <a href="https://github.com/nestjs/nest/tree/master/examples/06-mongoose">
       官网提供的案例
      </a>
     </p>
    </blockquote>
    <h4 id="一使用步骤">
     一、使用步骤
    </h4>
    <ul>
     <li>
      <p>
       1、安装依赖包
      </p>
      <pre class="prettyprint"><code class="language-javascript hljs">npm install --save @nestjs/mongoose mongoose</code></pre>
     </li>
     <li>
      <p>
       2、在项目中创建一个
       <code>
        database
       </code>
       作为数据库连接的文件/或者叫包
      </p>
      <ul>
       <li>
        <p>
         <code>
          database.providers.ts
         </code>
         类似
         <code>
          angular
         </code>
         中使用工厂的方式把数据库连接池注入到项目中
        </p>
        <pre class="prettyprint"><code class="language-javascript hljs">import * as mongoose from <span class="hljs-string">'mongoose'</span>;

export <span class="hljs-keyword">const</span> databaseProviders = [
  {
    provide: <span class="hljs-string">'DbConnectionToken'</span>,
    useFactory: async (): Promise&lt;mongoose.Connection&gt; =&gt;
      await mongoose.connect(<span class="hljs-string">'mongodb://localhost/nest'</span>),
  },
];</code></pre>
       </li>
       <li>
        <p>
         <code>
          database.module.ts
         </code>
         就是一个导入连接池与导出的
         <code>
          module
         </code>
         文件
        </p>
        <pre class="prettyprint"><code class="language-javascript hljs">import { Module } from <span class="hljs-string">'@nestjs/common'</span>;
import { databaseProviders } from <span class="hljs-string">'./database.providers'</span>;

@Module({
  components: [...databaseProviders],
  exports: [...databaseProviders],
})
export <span class="hljs-keyword">class</span> DatabaseModule {}</code></pre>
       </li>
      </ul>
     </li>
     <li>
      3、根据官网的创建一个
      <code>
       cats
      </code>
      的文件夹存放关于
      <code>
       cat
      </code>
      的全部信息
     </li>
     <li>
      4、根据上文中介绍的要创建几个基础文件然后注入到
      <code>
       cats.module.ts
      </code>
      中,再将
      <code>
       cats.module.ts
      </code>
      注入到
      <code>
       app.module.ts
      </code>
      中
     </li>
     <li>
      <p>
       5、在文件夹下创建一个新的文件夹和文件(俗称
       <code>
        App
       </code>
       )
      </p>
      <ul>
       <li>
        <code>
         dto
        </code>
        文件夹
       </li>
       <li>
        <code>
         interfaces
        </code>
        文件夹
       </li>
       <li>
        <code>
         schemas
        </code>
        文件夹,创建本
        <code>
         App
        </code>
        的
        <code>
         schema
        </code>
       </li>
       <li>
        <code>
         cats.providers.ts
        </code>
        把
        <code>
         database
        </code>
        中的
        <code>
         database.providers.ts
        </code>
        与本文件夹下的
        <code>
         schema
        </code>
        相关联(使用
        <code>
         useFactory
        </code>
        依赖注入的方式)
       </li>
      </ul>
     </li>
     <li>
      <p>
       6、
       <code>
        schemas
       </code>
       的书写(数据建模)
      </p>
      <pre class="prettyprint"><code class="language-javascript hljs">import * as mongoose from <span class="hljs-string">'mongoose'</span>;

export <span class="hljs-keyword">const</span> CatSchema = <span class="hljs-keyword">new</span> mongoose.Schema({
  name: <span class="hljs-built_in">String</span>,
  age: <span class="hljs-built_in">Number</span>,
  breed: <span class="hljs-built_in">String</span>,
});</code></pre>
     </li>
     <li>
      <p>
       7、
       <code>
        interfaces
       </code>
       的书写(接口约束数据类型)
      </p>
      <pre class="prettyprint"><code class="language-javascript hljs">import { Document } from <span class="hljs-string">'mongoose'</span>;

export interface Cat extends Document {
  readonly name: string;
  readonly age: number;
  readonly breed: string;
}</code></pre>
     </li>
     <li>
      <p>
       8、
       <code>
        dto
       </code>
       的书写(也是约束数据类型的)
      </p>
      <pre class="prettyprint"><code class="language-javascript hljs">export <span class="hljs-keyword">class</span> CreateCatDto {
  readonly name: string;
  readonly age: number;
  readonly breed: string;
}</code></pre>
     </li>
     <li>
      <p>
       9、
       <code>
        cats.providers.ts
       </code>
       的书写(关键点)
      </p>
      <pre class="prettyprint"><code class="language-javascript hljs">import { Connection } from <span class="hljs-string">'mongoose'</span>;
<span class="hljs-comment">// 引入schema</span>
import { CatSchema } from <span class="hljs-string">'./schemas/cat.schema'</span>;

export <span class="hljs-keyword">const</span> catsProviders = [
  {
    <span class="hljs-comment">// 自己定义一个到时候在service.ts中注入</span>
    provide: <span class="hljs-string">'CatModelToken'</span>, 
    <span class="hljs-comment">// 使用CatSchema</span>
    useFactory: (connection: Connection) =&gt; connection.model(<span class="hljs-string">'Cat'</span>, CatSchema),
    <span class="hljs-comment">// DbConnectionToken是database.providers.ts里面的key</span>
    inject: [<span class="hljs-string">'DbConnectionToken'</span>],
  },
];</code></pre>
     </li>
     <li>
      <p>
       10、关于
       <code>
        cats.service.ts
       </code>
       的书写
      </p>
      <pre class="prettyprint"><code class="language-javascript hljs">import { Model } from <span class="hljs-string">'mongoose'</span>;
import { Component, Inject } from <span class="hljs-string">'@nestjs/common'</span>;
import { Cat } from <span class="hljs-string">'./interfaces/cat.interface'</span>;
import { CreateCatDto } from <span class="hljs-string">'./dto/create-cat.dto'</span>;

@Component()
export <span class="hljs-keyword">class</span> CatsService {
  <span class="hljs-comment">// 注入的CatModelToken要与cats.providers.ts里面的key一致就可以</span>
  constructor(@Inject(<span class="hljs-string">'CatModelToken'</span>) private readonly catModel: Model&lt;Cat&gt;) {}

  <span class="hljs-comment">// 创建数据</span>
  async create(createCatDto: CreateCatDto): Promise&lt;Cat&gt; {
    <span class="hljs-keyword">const</span> createdCat = <span class="hljs-keyword">new</span> <span class="hljs-keyword">this</span>.catModel(createCatDto);
    <span class="hljs-keyword">return</span> await createdCat.save();
  }

  <span class="hljs-comment">// 查询全部数据</span>
  async findAll(): Promise&lt;Cat[]&gt; {
    <span class="hljs-keyword">return</span> await <span class="hljs-keyword">this</span>.catModel.find().exec();
  }

  <span class="hljs-comment">// 根据id查询</span>
  async findById(_id): Promise&lt;Cat&gt; {
    <span class="hljs-keyword">return</span> await <span class="hljs-keyword">this</span>.catModel.findById(_id).exec()
  }
}</code></pre>
     </li>
     <li>
      <p>
       11、参考代码(组件
       <code>

       和
       <code>
        food
       </code>
       )
      </p>
     </li>
    </ul>
