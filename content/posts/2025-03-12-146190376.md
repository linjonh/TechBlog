    <h2>
     2.总览
    </h2>
    <p>
     回调函数我写了无参 一参 两参的版本 可以自行添加参数
    </p>
    <pre><code class="language-cs">using System;
using System.Collections;
using UnityEngine;

public class TimeManager : MonoBehaviour
{
    private static TimeManager instance;
    public static TimeManager Instance =&gt; instance;

    private void Awake()
    {
        if (instance == null)
        {
            instance = this;
        }
    }

    private bool CheckCount(int count)
    {
        if (count &lt; 0)
        {
            Debug.LogError("循环次数不能为负数！");
            return false;
        }
        return true;
    }

    private bool CheckTime(float time)
    {
        if (time &lt; 0)
        {
            Debug.LogError("等待时间不能为负数！");
            return false;
        }
        return true;
    }

    #region 等待固定时间秒
    /// &lt;summary&gt;
    /// 等待固定时间以后执行回调
    /// &lt;/summary&gt;
    /// &lt;param name="waitTime"&gt;等待时间（秒）&lt;/param&gt;
    /// &lt;param name="callback"&gt;回调函数&lt;/param&gt;
    public void WaitTime(float waitTime, Action callback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(WaitTimeHandle(waitTime, () =&gt; callback?.Invoke()));
        }
    }

    public void WaitTime&lt;T&gt;(float waitTime, T param, Action&lt;T&gt; callback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(WaitTimeHandle(waitTime, () =&gt; callback?.Invoke(param)));
        }
    }

    public void WaitTime&lt;T, K&gt;(float waitTime, T param1, K param2, Action&lt;T, K&gt; callback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(WaitTimeHandle(waitTime, () =&gt; callback?.Invoke(param1, param2)));
        }
    }

    private IEnumerator WaitTimeHandle(float waitTime, Action action)
    {
        yield return new WaitForSeconds(waitTime);
        action?.Invoke();
    }
    #endregion
    #region 等待固定时间秒（不受缩放影响）
    /// &lt;summary&gt;
    /// 等待固定时间以后执行回调（不受Time.timeScale影响）
    /// &lt;/summary&gt;
    /// &lt;param name="waitTime"&gt;等待时间（秒）&lt;/param&gt;
    /// &lt;param name="callback"&gt;回调函数&lt;/param&gt;
    public void WaitRealTime(float waitTime, Action callback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(WaitRealTimeHandle(waitTime, () =&gt; callback?.Invoke()));
        }
    }

    public void WaitRealTime&lt;T&gt;(float waitTime, T param, Action&lt;T&gt; callback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(WaitRealTimeHandle(waitTime, () =&gt; callback?.Invoke(param)));
        }
    }

    public void WaitRealTime&lt;T, K&gt;(float waitTime, T param1, K param2, Action&lt;T, K&gt; callback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(WaitRealTimeHandle(waitTime, () =&gt; callback?.Invoke(param1, param2)));
        }
    }

    private IEnumerator WaitRealTimeHandle(float waitTime, Action action)
    {
        yield return new WaitForSecondsRealtime(waitTime);
        action?.Invoke();
    }
    #endregion

    #region 按固定时间间隔循环执行
    public void LoopTime(float spacing, int overNumber, Action callback)
    {
        if (CheckTime(spacing) &amp;&amp; CheckCount(overNumber))
        {
            StartCoroutine(LoopTimeHandle(spacing, overNumber, () =&gt; callback?.Invoke()));
        }
    }

    public void LoopTime&lt;T&gt;(float spacing, int overNumber, T param, Action&lt;T&gt; callback)
    {
        if (CheckTime(spacing) &amp;&amp; CheckCount(overNumber))
        {
            StartCoroutine(LoopTimeHandle(spacing, overNumber, () =&gt; callback?.Invoke(param)));
        }
    }

    public void LoopTime&lt;T, K&gt;(float spacing, int overNumber, T param1, K param2, Action&lt;T, K&gt; callback)
    {
        if (CheckTime(spacing) &amp;&amp; CheckCount(overNumber))
        {
            StartCoroutine(LoopTimeHandle(spacing, overNumber, () =&gt; callback?.Invoke(param1, param2)));
        }
    }

    private IEnumerator LoopTimeHandle(float spacing, int overNumber, Action action)
    {
        for (int i = 0; i &lt; overNumber; i++)
        {
            yield return new WaitForSeconds(spacing);
            action?.Invoke();
        }
    }
    #endregion

    #region 等待固定帧执行一次
    public void WaitFrame(int frameCount, Action callback)
    {
        if (CheckCount(frameCount))
        {
            StartCoroutine(WaitFrameHandle(frameCount, () =&gt; callback?.Invoke()));
        }
    }

    public void WaitFrame&lt;T&gt;(int frameCount, T param, Action&lt;T&gt; callback)
    {
        if (CheckCount(frameCount))
        {
            StartCoroutine(WaitFrameHandle(frameCount, () =&gt; callback?.Invoke(param)));
        }
    }

    public void WaitFrame&lt;T, K&gt;(int frameCount, T param1, K param2, Action&lt;T, K&gt; callback)
    {
        if (CheckCount(frameCount))
        {
            StartCoroutine(WaitFrameHandle(frameCount, () =&gt; callback?.Invoke(param1, param2)));
        }
    }

    private IEnumerator WaitFrameHandle(int frameCount, Action action)
    {
        for (int i = 0; i &lt; frameCount; i++)
        {
            yield return null;
        }
        action?.Invoke();
    }
    #endregion

    #region 进度反馈
    public void WaitTimeWithProgress(float waitTime, Action&lt;float&gt; progressCallback, Action completeCallback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(ProgressTimer(waitTime, progressCallback, completeCallback));
        }
    }

    private IEnumerator ProgressTimer(float duration, Action&lt;float&gt; progress, Action complete)
    {
        float startTime = Time.time;
        while (Time.time - startTime &lt; duration)
        {
            progress?.Invoke((Time.time - startTime) / duration);
            yield return null;
        }
        complete?.Invoke();
    }

    public void WaitTimeWithProgress&lt;T&gt;(float waitTime, T param, Action&lt;float, T&gt; progressCallback, Action&lt;T&gt; completeCallback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(ProgressTimer(waitTime, param, progressCallback, completeCallback));
        }
    }

    private IEnumerator ProgressTimer&lt;T&gt;(float duration, T param, Action&lt;float, T&gt; progress, Action&lt;T&gt; complete)
    {
        float startTime = Time.time;
        while (Time.time - startTime &lt; duration)
        {
            progress?.Invoke((Time.time - startTime) / duration, param);
            yield return null;
        }
        complete?.Invoke(param);
    }

    public void WaitTimeWithProgress&lt;T, K&gt;(float waitTime, T param1, K param2, Action&lt;float, T, K&gt; progressCallback, Action&lt;T, K&gt; completeCallback)
    {
        if (CheckTime(waitTime))
        {
            StartCoroutine(ProgressTimer(waitTime, param1, param2, progressCallback, completeCallback));
        }
    }

    private IEnumerator ProgressTimer&lt;T, K&gt;(float duration, T param1, K param2, Action&lt;float, T, K&gt; progress, Action&lt;T, K&gt; complete)
    {
        float startTime = Time.time;
        while (Time.time - startTime &lt; duration)
        {
            progress?.Invoke((Time.time - startTime) / duration, param1, param2);
            yield return null;
        }
        complete?.Invoke(param1, param2);
    }
    #endregion

    #region 等待当前帧结束执行回调
    public void WaitForEndOfFrame(Action callback)
    {
        StartCoroutine(WaitForEndOfFrameHandle(callback));
    }

    private IEnumerator WaitForEndOfFrameHandle(Action callback)
    {
        yield return new WaitForEndOfFrame();
        callback?.Invoke();
    }

    public void WaitForEndOfFrame&lt;T&gt;(T param, Action&lt;T&gt; callback)
    {
        StartCoroutine(WaitForEndOfFrameHandle(param, callback));
    }

    private IEnumerator WaitForEndOfFrameHandle&lt;T&gt;(T param, Action&lt;T&gt; callback)
    {
        yield return new WaitForEndOfFrame();
        callback?.Invoke(param);
    }

    public void WaitForEndOfFrame&lt;T, K&gt;(T param1, K param2, Action&lt;T, K&gt; callback)
    {
        StartCoroutine(WaitForEndOfFrameHandle(param1, param2, callback));
    }

    private IEnumerator WaitForEndOfFrameHandle&lt;T, K&gt;(T param1, K param2, Action&lt;T, K&gt; callback)
    {
        yield return new WaitForEndOfFrame();
        callback?.Invoke(param1, param2);
    }
    #endregion
    public void Stop(IEnumerator func)
    {
        StopCoroutine(func);
    }

    public void StopAll()
    {
        StopAllCoroutines();
    }

    public TimeChainContext StartChain()
    {
        return new TimeChainContext(this);
    }
}</code></pre>
