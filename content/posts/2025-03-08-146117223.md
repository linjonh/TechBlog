      使用i18n
      <br/>
      LanguageSwitcher.vue
     </li>
    </ol>
    <pre><code class="prism language-vue">&lt;template&gt;
  &lt;div class="language-switcher"&gt;
    &lt;select v-model="locale" @change="switchLanguage"&gt;
      &lt;option 
        v-for="lang in availableLocales" 
        :key="lang.code" 
        :value="lang.code"
      &gt;
        {<!-- -->{ lang.name }}
      &lt;/option&gt;
    &lt;/select&gt;
  &lt;/div&gt;
&lt;/template&gt;

&lt;script setup lang="ts"&gt;
const { locale, locales } = useI18n()
const switchLocalePath = useSwitchLocalePath()
const router = useRouter()

const availableLocales = computed(() =&gt; {
  return (locales.value).map(l =&gt; {
    return {
      code: l.code,
      name: l.name
    }
  })
})

const switchLanguage = () =&gt; {
  const path = switchLocalePath(locale.value)
  if (path) {
    router.push(path).then(() =&gt; {
      window.location.reload() // 强制刷新页面
    })
  }
}
&lt;/script&gt;

&lt;style scoped lang="less"&gt;
