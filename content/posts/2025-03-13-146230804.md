    <p>
     import fs from 'fs';
    </p>
    <p>
     import path from 'path';
    </p>
    <p>
    </p>
    <p>
     import xl from "excel4node";
    </p>
    <p>
    </p>
    <p>
     // 创建一个新的工作簿
    </p>
    <p>
     const wb = new xl.Workbook();
    </p>
    <p>
    </p>
    <p>
     // 添加一个工作表
    </p>
    <p>
     const ws = wb.addWorksheet('Sheet 1');
    </p>
    <p>
    </p>
    <p>
     // 指定要读取的文件夹路径
    </p>
    <p>
     const directoryPath = 'G://测试'; // 替换为你的文件夹路径
    </p>
    <p>
    </p>
    <p>
     let i = 1;
    </p>
    <p>
    </p>
    <p>
     const putImg = (ws, file) =&gt; {
     <!-- -->
    </p>
    <p>
     ws.row(i).setHeight(200); // 设置每行的行高为50
    </p>
    <p>
    </p>
    <p>
     ws.column(i).setWidth(40);
    </p>
    <p>
     <br/>
    </p>
    <p>
     ws.addImage({
     <!-- -->
    </p>
    <p>
     path: 'G://测试//' + file,
    </p>
    <p>
     type: 'picture',
    </p>
    <p>
     position: {
     <!-- -->
    </p>
    <p>
     type: 'twoCellAnchor',
    </p>
    <p>
     from: {
     <!-- -->
    </p>
    <p>
     col: 1,
    </p>
    <p>
     colOff: 0,
    </p>
    <p>
     row: i,
    </p>
    <p>
     rowOff: 0,
    </p>
    <p>
     },
    </p>
    <p>
     to: {
     <!-- -->
    </p>
    <p>
     col: 2,
    </p>
    <p>
     colOff: 0,
    </p>
    <p>
     row: i + 1,
    </p>
    <p>
     rowOff: 0,
    </p>
    <p>
     },
    </p>
    <p>
     },
    </p>
    <p>
     });
    </p>
    <p>
     i++;
    </p>
    <p>
     }
    </p>
    <p>
    </p>
    <p>
     // 使用同步方法读取文件夹
    </p>
    <p>
     try {
     <!-- -->
    </p>
    <p>
     const files = fs.readdirSync(directoryPath);
    </p>
    <p>
    </p>
    <p>
     files.forEach(file =&gt; {
     <!-- -->
    </p>
    <p>
     // 获取文件的完整路径
    </p>
    <p>
     const filePath = path.join(directoryPath, file);
    </p>
    <p>
    </p>
    <p>
     // 使用fs.statSync检查是文件还是目录
    </p>
    <p>
     const stats = fs.statSync(filePath);
    </p>
    <p>
     if (stats.isFile() &amp;&amp; file.startsWith("1074118")) {
     <!-- -->
    </p>
    <p>
     console.log(`File: ${file}`);
    </p>
    <p>
     let name = file.split(".")[0]
    </p>
    <p>
     ws.cell(i,2).string(name)
    </p>
    <p>
     // 将图片放入excel
    </p>
    <p>
     putImg(ws, file)
    </p>
    <p>
    </p>
    <p>
    </p>
    <p>
     } else if (stats.isDirectory()) {
     <!-- -->
    </p>
    <p>
     // console.log(`Directory: ${file}`); // 你可以递归地读取子目录
    </p>
    <p>
     }
    </p>
    <p>
     });
    </p>
    <p>
     // 将工作簿保存到文件
    </p>
    <p>
     wb.write('document_with_image.xlsx', (err, stats) =&gt; {
     <!-- -->
    </p>
    <p>
     if (err) {
     <!-- -->
    </p>
    <p>
     console.error('出现错误:', err);
    </p>
    <p>
     } else {
     <!-- -->
    </p>
    <p>
     console.log('带有图片的Excel文档已创建');
    </p>
    <p>
     }
    </p>
    <p>
     });
    </p>
    <p>
     } catch (err) {
     <!-- -->
    </p>
    <p>
     console.error(`Unable to scan directory: ${err}`);
    </p>
    <p>
     }
    </p>
