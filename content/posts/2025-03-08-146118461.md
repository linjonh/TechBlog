    <p>

    <blockquote>
     <pre><code class="hljs">enum PromiseState {
  PENDING,
  FULFILLED,
  REJECTED
}

class MyPromise {
  private state: PromiseState = PENDING;
  private value: any;
  private handlers: Function[] = [];
  
  constructor(executor) {
    const resolve = (value) =&gt; {
      if (this.state !== PENDING) return;
      this.state = FULFILLED;
      this.value = value;
      this.handlers.forEach(h =&gt; h());
    };
    
    const reject = (reason) =&gt; {
      if (this.state !== PENDING) return;
      this.state = REJECTED;
      this.value = reason;
      this.handlers.forEach(h =&gt; h());
    };
    
    try {
      executor(resolve, reject);
    } catch (e) {
      reject(e);
    }
  }
  
  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) =&gt; {
      const handler = () =&gt; {
        try {
          const callback = this.state === FULFILLED ? onFulfilled : onRejected;
          const result = callback(this.value);
          resolve(result);
        } catch (e) {
          reject(e);
        }
      };
      
      if (this.state !== PENDING) {
        queueMicrotask(handler);
      } else {
        this.handlers.push(handler);
      }
    });
  }
}
</code></pre>

// 转译结果
function example() {
  return _asyncToGenerator(function* () {
    const a = yield getA();
    const b = yield getB(a);
    return a + b;
  })();
}

function _asyncToGenerator(fn) {
  return function () {
    const gen = fn.apply(this, arguments);
    return new Promise((resolve, reject) =&gt; {
      function step(key, arg) {
        try {
          const { value, done } = gen[key](arg);
          if (done) {
            resolve(value);
          } else {
            Promise.resolve(value).then(
              val =&gt; step("next", val),
              err =&gt; step("throw", err)
            );
          }
        } catch (error) {
          reject(error);
        }
      }
      step("next");
    });
  };
}

  constructor() {
    this.events = new Map();
  }

  on(event, listener) {
    if (!this.events.has(event)) {
      this.events.set(event, []);
    }
    this.events.get(event).push(listener);
  }

  async emit(event, ...args) {
    const listeners = this.events.get(event) || [];
    for (const listener of listeners) {
      await listener(...args);
    }
  }

  constructor(size = navigator.hardwareConcurrency || 4) {
    this.pool = Array(size).fill().map(() =&gt; ({
      worker: new Worker('worker.js'),
      busy: false
    }));
  }

  exec(taskData) {
    const freeWorker = this.pool.find(w =&gt; !w.busy);
    if (!freeWorker) {
      return Promise.reject('All workers busy');
    }

    freeWorker.busy = true;
    return new Promise((resolve, reject) =&gt; {
      freeWorker.worker.onmessage = e =&gt; {
        freeWorker.busy = false;
        resolve(e.data);
      };
      freeWorker.worker.postMessage(taskData);
    });
  }
}

// 使用示例

  .then(processed =&gt; updateImage(processed));
