    <pre><code class="language-cpp">class Solution {
private:
    void numIslands(vector&lt;vector&lt;char&gt;&gt;&amp; grid,vector&lt;vector&lt;bool&gt;&gt;&amp; visited,int x,int y){
        if(x&lt;0||y&lt;0||x&gt;=grid.size()||y&gt;=grid[0].size())
            return;
        else if(grid[x][y]=='0'||visited[x][y])
            return;
        else{
            visited[x][y]=true;
            numIslands(grid,visited,x,y+1);
            numIslands(grid,visited,x+1,y);
            numIslands(grid,visited,x,y-1);
            numIslands(grid,visited,x-1,y);
        }
    }
public:
    int numIslands(vector&lt;vector&lt;char&gt;&gt;&amp; grid) {
        int result=0;
        int m=grid.size(),n=grid[0].size();
        vector&lt;vector&lt;bool&gt;&gt; visited(m,vector&lt;bool&gt;(n,false));
        for(int i=0;i&lt;m;i++)
            for(int j=0;j&lt;n;j++){
                if(grid[i][j]=='1'&amp;&amp;!visited[i][j]){
                    result++;
                    numIslands(grid,visited,i,j);
                }
            }
        return result;
    }
};</code></pre>
