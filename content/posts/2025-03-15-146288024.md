#include<bits/stdc++.h>
using namespace std;
int main()
{
	string a,b;
	cin>>a;
	cin>>b;
	int lena,lenb,jia=1,jib=1;
	int ai[6],bi[6];
	lena=a.size();
	lenb=b.size();
	for(int i=0;i<a.size();i++)
	{
		ai[i]=a[i]-'@';
		jia=jia*ai[i];	 
	}
	for(int j=0;j<b.size();j++)
	{
		bi[j]=b[j]-'@';
		jib=jib*bi[j];
	}
	if(jia%47==jib%47) cout<<"GO"<<endl;
	else cout<<"STAY"<<endl;
	return 0;
}
