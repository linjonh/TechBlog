template&lt;typename T&gt;
struct TreeNode{
	T val;

	TreeNode():val(0), left(NULL), right(NULL){}
	TreeNode(T x):val(x), left(NULL), right(NULL){}

template&lt;typename T&gt;
class BinarySearchTree{
private:
	TreeNode&lt;T&gt; *root;
	
	TreeNode&lt;T&gt;* insertNode(TreeNode&lt;T&gt; *node, T value);
	TreeNode&lt;T&gt;* removeNode(TreeNode&lt;T&gt; *node, T value);
	bool searchNode(TreeNode&lt;T&gt; *node, T value);
	void inOrder(TreeNode&lt;T&gt; *node);
public:
	BinarySearchTree(): root(NULL){}
	~BinarySearchTree();
	
	void insert(T value){
		root = insertNode(root, value);
	}
	
	void remove(T value){
		root = removeNode(root, value);
	}
	
	bool search(T value){
		return searchNode(root, value);
	}
	void inOrderTraversal(){
		inOrder(root);
		cout &lt;&lt; endl;
	}
};

template&lt;typename T&gt;
BinarySearchTree&lt;T&gt;::~BinarySearchTree(){
	while(root){
		remove(root-&gt;val);
	}
}

template&lt;typename T&gt;
TreeNode&lt;T&gt;* BinarySearchTree&lt;T&gt;::insertNode(TreeNode&lt;T&gt; *node, T value){
	if(node == NULL){
		return new TreeNode&lt;T&gt;(value);
	}
	if(value &lt; node-&gt;val){
		node-&gt;left = insertNode(node-&gt;left, value);
	}else if(value &gt; node-&gt;val){
		node-&gt;right = insertNode(node-&gt;right, value);
	}
	return node;
}

template&lt;typename T&gt;
TreeNode&lt;T&gt;* BinarySearchTree&lt;T&gt;::removeNode(TreeNode&lt;T&gt; *node, T value){
	if(node == NULL){
		return NULL;
	}
	if(value &lt; node-&gt;val){
		node-&gt;left = removeNode(node-&gt;left, value);
	}else if(value &gt; node-&gt;val){
		node-&gt;right = removeNode(node-&gt;right, value);
	}else{
		if(node-&gt;left == NULL &amp;&amp; node-&gt;right == NULL){
			delete node;
			return NULL;
		}else if(node-&gt;left == NULL){
			TreeNode&lt;T&gt; *rightChild = node-&gt;right;
			delete node;
			return rightChild;
		}else if(node-&gt;right == NULL){
			TreeNode&lt;T&gt; *leftChild = node-&gt;left;
			delete node;
			return leftChild;
		}else{
			TreeNode&lt;T&gt; *replacement = node-&gt;right;
			while(replacement-&gt;left){
				replacement = replacement-&gt;left;
			}
			node-&gt;val = replacement-&gt;val;
			node-&gt;right = removeNode(node-&gt;right,replacement-&gt;val);
		}
	}
	return node;
}

template&lt;typename T&gt;
bool BinarySearchTree&lt;T&gt;::searchNode(TreeNode&lt;T&gt; *node, T value){
	if(node == NULL){
		return false;
	}
	if(value &lt; node-&gt;val){
		return searchNode(node-&gt;left, value);
	}else if(value &gt; node-&gt;val){
		return searchNode(node-&gt;right, value);
	}
    return true;
}
