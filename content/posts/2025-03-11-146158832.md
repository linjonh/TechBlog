    <pre>#include "bird.h"</pre>

    <pre>Bird::Bird(QWidget *parent) : QWidget(parent)</pre>
    <pre>{<!-- --></pre>
    <pre>    //加载所有图片</pre>
    <pre>    for(int i = 0 ; i &lt; this-&gt;max;i++)</pre>
    <pre>    {<!-- --></pre>
    <pre>        QString str = QString(":/Bird/bird%1.png").arg(i+1);</pre>
    <pre>        this-&gt;m_bird_Pix.load(str);</pre>
    <pre>    }</pre>
    <pre>    //设置鸟的大小</pre>
    <pre>    this-&gt;setFixedSize(this-&gt;m_bird_Pix.width(),this-&gt;m_bird_Pix.height());</pre>
    <pre></pre>
    <pre>    //创建定时器</pre>
    <pre>    this-&gt;timer = new QTimer(this);</pre>
    <pre></pre>
    <pre>    //监听定时器</pre>
    <pre>    connect(timer,&amp;QTimer::timeout,[=](){<!-- --></pre>
    <pre>        QString str = QString(":/Bird/bird%1.png").arg(this-&gt;min++);</pre>
    <pre>        this-&gt;m_bird_Pix.load(str);</pre>
    <pre>        if(this-&gt;min &gt; this-&gt;max)</pre>
    <pre>        {<!-- --></pre>
    <pre>            this-&gt;min = 1;</pre>
    <pre>        }</pre>
    <pre>        emit changePix(); //告诉窗口正在做切图</pre>
    <pre>    });</pre>
    <pre></pre>
    <pre>    //创建菜单</pre>
    <pre>    this-&gt;m_Menu = new QMenu();</pre>
    <pre>    //监听菜单下的退出菜单项</pre>
    <pre>    connect(this-&gt;m_Menu-&gt;addAction("退出"),&amp;QAction::triggered,[=](){<!-- --></pre>
    <pre>        exit(0);</pre>
    <pre>    });</pre>
    <pre>}</pre>
    <pre></pre>
    <pre>void Bird::running()</pre>
    <pre>{<!-- --></pre>
    <pre>    this-&gt;timer-&gt;start(70);</pre>
    <pre>}</pre>
    <pre></pre>
    <pre>void Bird::mousePressEvent(QMouseEvent *event)</pre>
    <pre>{<!-- --></pre>
    <pre>    this-&gt;m_Pos = event-&gt;globalPos() -   ((QWidget *) (this-&gt;parent()))-&gt;frameGeometry().topLeft();</pre>
    <pre>    this-&gt;mouseDown = true; //鼠标按下，将状态改为true</pre>
    <pre></pre>
    <pre>    //右键菜单弹出</pre>
    <pre>    if(event-&gt;button() == Qt::RightButton)</pre>
    <pre>    {<!-- --></pre>
    <pre>        this-&gt;m_Menu-&gt;popup(QCursor::pos());</pre>
    <pre>    }</pre>
    <pre></pre>
    <pre>    //窗口消失后 重置鼠标按下状态</pre>
    <pre>    connect(this-&gt;m_Menu,&amp;QMenu::aboutToHide,[=](){<!-- --></pre>
    <pre>        this-&gt;mouseDown = false;</pre>
    <pre>    });</pre>
    <pre>}</pre>
    <pre></pre>
    <pre>void Bird::mouseMoveEvent(QMouseEvent *event)</pre>
    <pre>{<!-- --></pre>
    <pre>    //发送自定义信号，告诉窗口移动的位置</pre>
    <pre>    emit this-&gt;moving( event-&gt;globalPos() - this-&gt;m_Pos );</pre>
    <pre>}</pre>
    <pre></pre>
    <pre>void Bird::mouseReleaseEvent(QMouseEvent *event)</pre>
    <pre>{<!-- --></pre>
    <pre>    //鼠标释放 ，视为鼠标没有按下</pre>
    <pre>    this-&gt;mouseDown = false;</pre>
    <pre>}</pre>
