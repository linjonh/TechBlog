class Solution {
public:
    int A(int a) {
        if (a <= 0) return 0;
        return (1 << a) - 1 - a;
    }
    int beautifulSubsets(vector<int>& nums, int k) {
        map<int, int> mp;
        for (int i = 0; i < nums.size(); i++) {
            mp[nums[i]]++;
        }
        int sum = nums.size(), ans = sum;
        for (pair<int, int> p : mp) {
            int t = sum;
            if (mp.find(p.first + k) != mp.end()) {
                t -= mp[p.first + k];
            }
            ans += A(t);
            cout << "sum=" << sum << "," << "t=" << t << endl;
            sum -= p.second;
        }
        return ans;
    }
};
