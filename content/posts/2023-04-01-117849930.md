    <h3>
     命令行实现
    </h3>
    <p>
     FFMPEG可以直接通过命令行获取音视频设备列表，命令如下：
    </p>
    <blockquote>
     <p>
      ffmpeg -list_devices true -f dshow -i dummy
     </p>
    </blockquote>

    <h3>
     程序实现
    </h3>
    <pre><code class="language-cpp">#include &lt;iostream&gt;

extern "C" 
{
#include &lt;libavcodec/avcodec.h&gt;
#include &lt;libavformat/avformat.h&gt;
#include &lt;libavutil/avutil.h&gt;
#include &lt;libswscale/swscale.h&gt;
#include &lt;libavdevice/avdevice.h&gt;
}

#pragma comment(lib, "avcodec.lib")
#pragma comment(lib, "avformat.lib")
#pragma comment(lib, "avutil.lib")
#pragma comment(lib, "swscale.lib")
#pragma comment(lib, "avdevice.lib")

using namespace std;

void showDevice()
{
	AVFormatContext* pFormat = avformat_alloc_context();
	AVDictionary *dict = NULL;
	av_dict_set(&amp;dict, "list_devices", "true", 0);
	AVInputFormat* fmt = av_find_input_format("dshow");
	avformat_open_input(&amp;pFormat, "video=dummy", fmt, &amp;dict);
}

void showOption()
{
	AVFormatContext* pFormat = avformat_alloc_context();
	AVDictionary* dict = NULL;
	av_dict_set(&amp;dict, "list_options", "true", 0);
	AVInputFormat* fmt = av_find_input_format("dshow");
	avformat_open_input(&amp;pFormat, "video=Integrated Camera", fmt, &amp;dict);

}


int main()
{
	av_register_all();
	avdevice_register_all();

	cout &lt;&lt; "device list:" &lt;&lt; endl;
	showDevice();

	cout &lt;&lt; "device option:" &lt;&lt; endl;
	showOption();

	return 0;
}</code></pre>
