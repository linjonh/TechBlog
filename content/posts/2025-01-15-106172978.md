    <pre><code class="prism language-powershell"><span class="token comment"># 这里配上监听前端项目的端口</span>
        listen       6080<span class="token punctuation">;</span>
        server_name  localhost<span class="token punctuation">;</span>

        <span class="token comment">#charset koi8-r;</span>

        <span class="token comment">#access_log  logs/host.access.log  main;</span>
		
		
        location <span class="token operator">/</span> <span class="token punctuation">{<!-- --></span>
		
			<span class="token comment">#这里指定你前端项目的目录</span>
            root   D:\VSCode\Workspace\<span class="token function">sl</span>\sl_mgr_web<span class="token punctuation">;</span>
			
			<span class="token comment">#这里是说默认打开目录下哪个页面，我这里打开登录页</span>
            index  login<span class="token punctuation">.</span>html login<span class="token punctuation">.</span>html<span class="token punctuation">;</span>
			
        <span class="token punctuation">}</span>
		
		<span class="token comment">#这里关键哦，是解决前端请求后台跨域的问题</span>
		<span class="token comment">#因为我们启动nginx时，已经占用了89端口，但是我们后端启动使用8080端口的</span>
		<span class="token comment">#这里的意思是，当你写的请求路径是 /api/login时，nginx其实帮我们代理去请求http://127.0.0.1:8080/sl/login接口的哦，所以实际我们就解决跨域问题了</span>
		location <span class="token operator">/</span>api <span class="token punctuation">{<!-- --></span>		
		
			<span class="token comment"># 后台服务路径（注意一下区别哦）</span>
			<span class="token comment"># 这是后端启动带项目名的配置</span>
			proxy_pass http:<span class="token operator">/</span><span class="token operator">/</span>127<span class="token punctuation">.</span>0<span class="token punctuation">.</span>0<span class="token punctuation">.</span>1:8080<span class="token operator">/</span><span class="token function">sl</span><span class="token punctuation">;</span>
			<span class="token comment"># 这是后端启动不带项目名的路径配置</span>
			<span class="token comment">#proxy_pass http://127.0.0.1:8080/sl/;</span>
		<span class="token punctuation">}</span>
</code></pre>
