    <p>
     提供分隔器组件，分隔不同内容块/内容元素。
    </p>
    <blockquote>
     <p>
      <strong>
       说明：
      </strong>
     </p>
     <p>
      该组件从API Version 7开始支持。
     </p>
    </blockquote>
    <h3 id="子组件">
     <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-basic-components-divider?id=%E5%AD%90%E7%BB%84%E4%BB%B6" rel="nofollow" title="子组件">
      子组件
     </a>
    </h3>
    <p>
     无
    </p>
    <h3 id="接口">
     <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-basic-components-divider?id=%E6%8E%A5%E5%8F%A3" rel="nofollow" title="接口">
      接口
     </a>
    </h3>
    <p>
     Divider()
    </p>
    <p>
     从API version 9开始，该接口支持在ArkTS卡片中使用。
    </p>
    <h3 id="属性">
     <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-basic-components-divider?id=%E5%B1%9E%E6%80%A7" rel="nofollow" title="属性">
      属性
     </a>
    </h3>
    <p>
     除支持
     <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-universal-attributes-size" rel="nofollow" title="通用属性">
      通用属性
     </a>
     外，还支持以下属性：
    </p>
    <table>
     <thead>
      <tr>
       <th>
        名称
       </th>
       <th>
        参数类型
       </th>
       <th>
        描述
       </th>
      </tr>
     </thead>
     <tbody>
      <tr>
       <td>
        vertical
       </td>
       <td>
        布尔
       </td>
       <td>
        使用水平分割线还是垂直分割线。false：水平分割线；true：垂直分割线。
        <br/>
        默认值：false
        <br/>
        从API version 9开始，该接口支持在ArkTS卡片中使用。
       </td>
      </tr>
      <tr>
       <td>
        颜色
       </td>
       <td>
        <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-types?id=resourcecolor" rel="nofollow" title="ResourceColor">
         ResourceColor
        </a>
       </td>
       <td>
        分割线颜色。
        <br/>
        默认值：'#33182431'
        <br/>
        从API version 9开始，该接口支持在ArkTS卡片中使用。
       </td>
      </tr>
      <tr>
       <td>
        strokeWidth
       </td>
       <td>
        数字 |字符串
       </td>
       <td>
        分割线宽度。
        <br/>
        默认值：1px
        <br/>
        单位：vp
        <br/>
        从API version 9开始，该接口支持在ArkTS卡片中使用。
        <br/>
        <strong>
         说明：
        </strong>
        <br/>
        分割线的宽度不支持百分比设置。优先级低于
        <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-universal-attributes-size" rel="nofollow" title="通用属性height">
         通用属性height
        </a>
        ，超过通用属性设置大小时，按照通用属性进行裁切。
       </td>
      </tr>
      <tr>
       <td>
        lineCap
       </td>
       <td>
        <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-appendix-enums?id=linecapstyle" rel="nofollow" title="LineCapStyle">
         LineCapStyle
        </a>
       </td>
       <td>
        分割线的端点样式。
        <br/>
        默认值：LineCapStyle.Butt
        <br/>
        从API version 9开始，该接口支持在ArkTS卡片中使用。
       </td>
      </tr>
     </tbody>
    </table>
    <h3 id="示例">
     <a href="https://open.weharmonyos.com/#/docs-OpenHarmony-4.0-Release/zh-cn/application-dev/reference/arkui-ts/ts-basic-components-divider?id=%E7%A4%BA%E4%BE%8B" rel="nofollow" title="示例">
      示例
     </a>
    </h3>
    <pre><code>// xxx.ets
@Entry
@Component
struct DividerExample {
  build() {
    Column() {
      // 使用横向分割线场景
      Text('Horizontal divider').fontSize(9).fontColor(0xCCCCCC)
      List() {
        ForEach([1, 2, 3], (item: number) =&gt; {
          ListItem() {
            Text('list' + item).width('100%').fontSize(14).fontColor('#182431').textAlign(TextAlign.Start)
          }.width(244).height(48)
        }, (item: number) =&gt; item.toString())
      }.padding({ left: 24, bottom: 8 })

      Divider().strokeWidth(8).color('#F1F3F5')
      List() {
        ForEach([4, 5], (item: number) =&gt; {
          ListItem() {
            Text('list' + item).width('100%').fontSize(14).fontColor('#182431').textAlign(TextAlign.Start)
          }.width(244).height(48)
        }, (item: number) =&gt; item.toString())
      }.padding({ left: 24, top: 8 })

      // 使用纵向分割线场景
      Text('Vertical divider').fontSize(9).fontColor(0xCCCCCC)
      Column() {
        Column() {
          Row().width(288).height(64).backgroundColor('#30C9F0').opacity(0.3)
          Row() {
            Button('Button')
              .width(136)
              .height(22)
              .fontSize(16)
              .fontColor('#007DFF')
              .fontWeight(500)
              .backgroundColor(Color.Transparent)
            Divider().vertical(true).height(22).color('#182431').opacity(0.6).margin({ left: 8, right: 8 })
            Button('Button')
              .width(136)
              .height(22)
              .fontSize(16)
              .fontColor('#007DFF')
              .fontWeight(500)
              .backgroundColor(Color.Transparent)
          }.margin({ top: 17 })
        }
        .width(336)
        .height(152)
        .backgroundColor('#FFFFFF')
        .borderRadius(24)
        .padding(24)
      }
      .width('100%')
      .height(168)
      .backgroundColor('#F1F3F5')
      .justifyContent(FlexAlign.Center)
      .margin({ top: 8 })
    }.width('100%').padding({ top: 24 })
  }
}</code></pre>
