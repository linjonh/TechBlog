    <h3>
     2.安装之前请确保你安装了
     <strong>
      Python 和
     </strong>
     pip  V5.2需要python3.8和pip
    </h3>
    <h4>
    </h4>
    <p>
    </p>
    <blockquote>
     <div>
      <span style="color:#000000">
       mkdir esp32
      </span>
     </div>
     <div>
      <span style="color:#000000">
       cd esp32
      </span>
     </div>
     <div>
      <div>
       <span style="color:#000000">
        git clone
       </span>
       <span style="color:#0000ff">
        https://gitee.com/EspressifSystems/esp-gitee-tools.git
       </span>
      </div>
      <div>
       <div>
        <span style="color:#000000">
         cd esp-gitee-tools
        </span>
       </div>
       <div>
        <span style="color:#000000">
         ./jihu-mirror.sh set
        </span>
       </div>
       <div>
        <div>
         <span style="color:#000000">
          cd ..
         </span>
        </div>
        <div>
         <div>
          <span style="color:#000000">
           git clone --recursive
          </span>
          <span style="color:#0000ff">
           https://github.com/espressif/esp-idf.git
          </span>
         </div>
         <div>
          <div>
           <span style="color:#000000">
            cd ~/esp32/esp-idf
           </span>
          </div>
          <div>
           <div>
            <span style="color:#000000">
             git checkout v5.2
            </span>
           </div>
           <div>
            <span style="color:#000000">
             git branch
            </span>
           </div>
           <div>
            <div>
             <span style="color:#000000">
              git submodule --init --recursive
             </span>
             <br/>
             git submodule update --init --recursive  //更新子模块
            </div>
            <div>
             <div>
              <span style="color:#000000">
               ../esp-gitee-tools/install.sh
              </span>
             </div>
             <div>
              <span style="color:#000000">
               cd ~/esp32/esp-idf
              </span>
             </div>
             <div>
              <div>
               <span style="color:#000000">
                source export.sh
               </span>
              </div>
             </div>
            </div>
           </div>
          </div>
         </div>
        </div>
       </div>
      </div>
     </div>
    </blockquote>
    <p>
     <span style="color:#000000">
      在
     </span>
     <span style="color:#000000">
      ~/.profile追加
     </span>
    </p>
    <blockquote>
     <p>
      cd ~
     </p>
     <p>
      nano .profile
     </p>
     <p>
      <span style="color:#000000">
       source esp32/esp-idf/export.sh   //在结尾加上这个
      </span>
     </p>
    </blockquote>
