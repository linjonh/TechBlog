def extract_urls(html_text):
    pattern=r'&lt;a\s+[^&gt;]*?href\s*=\s*["\'](.*?)["\']'
    urls =re.findall(pattern,html_text)
    return urls

def main():
    html_text=""    
    while True:
        try:
            line=input()
            if(line.strip()==""):

            html_text+=line+"\n"
        except EOFError:
            break
    urls=extract_urls(html_text)
