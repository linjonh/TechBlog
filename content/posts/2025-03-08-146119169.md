    <h4>
     ​
     <strong>
      RabbitMQ交换机（Exchange）的核心作用
     </strong>
    </h4>
    <p>
     在RabbitMQ中，​
     <strong>
      交换机
     </strong>
     是消息路由的核心组件，负责接收生产者发送的消息，并根据规则（如路由键、头信息等）将消息分发到对应的队列中。
     <br/>
     不同交换机类型决定了消息的
     <strong>
      路由逻辑
     </strong>
     ，使用不同的交换机在不同的场景下可以提高消息系统的高可用性。
    </p>
    <p>
    </p>
    <h4>
     <strong>
      1. 直连交换机（Direct Exchange）​
     </strong>
    </h4>
    <h5>
     ​
     <strong>
      路由机制
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       精确匹配路由键（Routing Key）​
      </strong>
      ：消息会被发送到与
      <code>
       Routing Key
      </code>
      ​
      <strong>
       完全匹配
      </strong>
      的队列。
     </li>
     <li>
      ​
      <strong>
       典型场景
      </strong>
      ：一对一或一对多的精确消息分发。
     </li>
    </ul>
    <h5>
     ​
     <strong>
      应用场景
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       任务分发
      </strong>
      ：如订单处理系统，根据订单类型（如
      <code>
       order.payment
      </code>
      、
      <code>
       order.shipping
      </code>
      ）分发到不同队列。
     </li>
     <li>
      ​
      <strong>
       日志分类
      </strong>
      ：将不同级别的日志（
      <code>
       log.error
      </code>
      、
      <code>
       log.info
      </code>
      ）路由到对应的处理服务。
     </li>
    </ul>
    <p>
     <strong>
      使用直连交换机实现消息发送和接收
     </strong>
    </p>

    <h4>
     <strong>
      2. 扇出交换机（Fanout Exchange）​
     </strong>
    </h4>
    <h5>
     ​
     <strong>
      路由机制（一个交换机转发到多个队列）
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       广播模式
      </strong>
      ：忽略
      <code>
       Routing Key
      </code>
      ，将消息
      <strong>
       发送到所有绑定的队列
      </strong>
      。
     </li>
     <li>
      ​
      <strong>
       典型场景
      </strong>
      ：消息的全局通知或并行处理。
     </li>
    </ul>
    <h5>
     ​
     <strong>
      应用场景
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       实时通知系统
      </strong>
      ：如用户注册成功后，同时发送邮件、短信、更新缓存。
     </li>
     <li>
      ​
      <strong>
       日志广播
      </strong>
      ：多个服务订阅同一日志源，各自独立处理。
     </li>
    </ul>
    <p>
     <strong>
      使用扇出交换机实现消息发送和接收
     </strong>
    </p>
    <p>
     <strong>
      1.扇出交换机配置
     </strong>
    </p>
    <pre><code class="language-java">package com.atguigu.demomq;

import org.springframework.amqp.core.Binding;
import org.springframework.amqp.core.BindingBuilder;
import org.springframework.amqp.core.FanoutExchange;
import org.springframework.amqp.core.Queue;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class FanoutExchangeConfig {

    // 定义扇出交换机
    @Bean
    public FanoutExchange fanoutExchange() {
        return new FanoutExchange("fanout.user.register", true, false);
    }

    // 定义邮件队列
    @Bean
    public Queue emailQueue() {
        return new Queue("fanout.user.email", true);
    }

    // 定义短信队列
    @Bean
    public Queue smsQueue() {
        return new Queue("fanout.user.sms", true);
    }

    // 绑定所有队列到扇出交换机（无需路由键）
    @Bean
    public Binding emailBinding() {
        return BindingBuilder.bind(emailQueue()).to(fanoutExchange());
    }

    @Bean
    public Binding smsBinding() {
        return BindingBuilder.bind(smsQueue()).to(fanoutExchange());
    }
}</code></pre>

    <h4>
     ​
     <strong>
      3. 主题交换机（Topic Exchange）​
     </strong>
    </h4>
    <p>
    </p>
    <h5>
     ​
     <strong>
      路由机制
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       模式匹配路由键
      </strong>
      ：使用
      <code>
       *
      </code>
      （匹配一个单词）和
      <code>
       #
      </code>
      （匹配多个单词）通配符。
     </li>
     <li>
      ​
      <strong>
       典型场景
      </strong>
      ：灵活的多条件消息路由。
     </li>
    </ul>
    <h5>
     ​
     <strong>
      应用场景
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       新闻订阅系统
      </strong>
      ：用户订阅特定主题（如
      <code>
       news.sports.*
      </code>
      、
      <code>
       news.tech.#
      </code>
      ）。
     </li>
     <li>
      ​
      <strong>
       设备状态监控
      </strong>
      ：根据设备类型和区域路由消息（如
      <code>
       sensor.temperature.room1
      </code>
      ）。
     </li>
    </ul>
    <p>
     <strong>
      1.配置主题交换机
     </strong>
    </p>
    <pre><code class="language-java">package com.atguigu.demomq;

import org.springframework.amqp.core.Binding;
import org.springframework.amqp.core.BindingBuilder;
import org.springframework.amqp.core.Queue;
import org.springframework.amqp.core.TopicExchange;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class TopicExchangeConfig {

    // 定义主题交换机
    @Bean
    public TopicExchange topicExchange() {
        return new TopicExchange("topic.news", true, false);
    }

    // 定义体育新闻队列
    @Bean
    public Queue sportsQueue() {
        return new Queue("topic.news.sports", true);
    }

    // 定义科技新闻队列
    @Bean
    public Queue techQueue() {
        return new Queue("topic.news.tech", true);
    }

    // 绑定体育队列：匹配 news.sports.*
    @Bean
    public Binding sportsBinding() {
        return BindingBuilder.bind(sportsQueue())
                .to(topicExchange())
                .with("news.sports.*");
    }

    // 绑定科技队列：匹配 news.tech.#
    @Bean
    public Binding techBinding() {
        return BindingBuilder.bind(techQueue())
                .to(topicExchange())
                .with("news.tech.#");
    }
}</code></pre>

    <h4>
     ​
     <strong>
      4. 头交换机（Headers Exchange）​
     </strong>
    </h4>
    <p>
    </p>
    <h5>
     ​
     <strong>
      路由机制（
     </strong>
     我的理解是一种基于 ​
     <strong>
      多条件组合
     </strong>
     的消息路由机制
     <strong>
      ）
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       基于消息头（Headers）匹配
      </strong>
      ：忽略
      <code>
       Routing Key
      </code>
      ，通过键值对（Headers）匹配队列绑定的条件。
     </li>
     <li>
      ​
      <strong>
       匹配规则
      </strong>
      ：
      <code>
       x-match
      </code>
      参数设为
      <code>
       all
      </code>
      （需全部匹配）或
      <code>
       any
      </code>
      （匹配任意一个）。
     </li>
    </ul>
    <h5>
     ​
     <strong>
      应用场景
     </strong>
    </h5>
    <ul>
     <li>
      ​
      <strong>
       复杂路由逻辑
      </strong>
      ：如根据消息的版本号、语言等元数据路由。
     </li>
     <li>
      ​
      <strong>
       多维度过滤
      </strong>
      ：如同时匹配用户类型（
      <code>
       user_type: vip
      </code>
      ）和地理位置（
      <code>
       region: asia
      </code>
      ）。
     </li>
    </ul>
    <p>
     <strong>
      1.头交换机配置
     </strong>
    </p>
    <pre><code class="language-java">package com.atguigu.demomq;

import org.springframework.amqp.core.Binding;
import org.springframework.amqp.core.BindingBuilder;
import org.springframework.amqp.core.HeadersExchange;
import org.springframework.amqp.core.Queue;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.HashMap;
import java.util.Map;

@Configuration
public class HeadersExchangeConfig {

    // 定义头交换机
    @Bean
    public HeadersExchange headersExchange() {
        return new HeadersExchange("headers.user", true, false);
    }

    // 定义VIP用户队列
    @Bean
    public Queue vipQueue() {
        return new Queue("headers.user.vip", true);
    }

    // 绑定VIP队列，要求同时匹配 userType=vip 和 region=asia
    @Bean
    public Binding vipBinding() {
        Map&lt;String, Object&gt; headers = new HashMap&lt;&gt;();
        headers.put("userType", "vip");
        headers.put("region", "asia");

                .to(headersExchange())
                .whereAll(headers).match(); // whereAll 表示需全部匹配

import org.springframework.amqp.core.Message;

import org.springframework.amqp.rabbit.core.RabbitTemplate;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

@RestController
public class HeaderUserVipService {

    @Autowired
    private RabbitTemplate rabbitTemplate;
    @GetMapping("/sendHeaderMessage")
    public String  sendVipMessage() {
        MessageProperties props = new MessageProperties();
        props.setHeader("userType", "vip");
        props.setHeader("region", "asia");
        Message msg = new Message("HeaderMessage".getBytes(), props);
        rabbitTemplate.send("headers.user", "", msg);

    }
