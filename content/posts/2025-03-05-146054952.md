#### 第四天

![](https://i-blog.csdnimg.cn/direct/67b5513151244fe2a518beed06840912.png)

```cs
#include <stdio.h>

// 判断是否为闰年
int isLeapYear(int year) {
    return (year % 4 == 0 && year % 100!= 0) || (year % 400 == 0);
}

int main() {
    int year, month, day;
    int days_in_month[] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    int total_days = 0;

    printf("请输入年月日：");
    scanf("%d %d %d", &year, &month, &day);

    // 判断闰年，更新二月天数
    if (isLeapYear(year)) {
        days_in_month[1] = 29;
    }

    // 计算天数
    for (int i = 0; i < month - 1; i++) {
        total_days += days_in_month[i];
    }
    total_days += day;

    printf("这是第%d天\n", total_days);

    return 0;
}
```

#### 第十二天

求素数，差点给我搞红温了，555555555555

```cs
#include <stdio.h>

int main() {
    // 遍历 101 到 199 之间的所有整数
    for (int i = 101; i < 200; i++) {
        int is_prime = 1;  // 每次判断新的数之前，假设 i 是素数
        // 检查 i 是否能被 2 到 i-1 之间的任何整数整除
        for (int j = 2; j < i; j++) {
            if (i % j == 0) {
                is_prime = 0;  // i 不是素数
                break;
            }
        }
        // 如果 i 是素数，则输出 i
        if (is_prime) {
            printf("%d\n", i);
        }
    }
    return 0;
}
```

#### 第十七天

求字符串里面存在多少数字，字母，空格和其他字符

```cs
#include <stdio.h>

int main() {
    char x[100];
    int shuzi = 0;
    int zimu = 0;
    int kongge =0 ;
    int qita = 0;
    printf("请输入字符串:");
    fgets(x,sizeof(x),stdin);
    for (int  i = 0; x[i] !='\0'; i++)
    {   
        if (x[i]=='\n')
        {
            break;
        }
        if (x[i]<='9' && x[i]>='0')
        {
            shuzi++;
        }
        else if ((x[i]>='a' && x[i]<='z') || (x[i]>='A' && x[i]<='Z'))
        {
            zimu ++;
        }
        else if (x[i] == ' ')
        {
            kongge++;
        }else{
            qita++;
        }
    }
    printf("%d %d %d %d\n",shuzi,zimu,kongge,qita);
    
    return 0;
}
```

#include <stdio.h>
#include <math.h>

int main() {
    int count;
    int bottom;
    int sum = 0;
    int i;

    printf("请输入底数和次数:");
    scanf("%d %d", &bottom, &count);

    for (i = 0; i <= count; i++) {
        sum += (int)pow(bottom, i);
    }

    printf("总数是:%d\n", sum);
