    <h2>
     2.配置nginx做负载均衡
    </h2>
    <blockquote>
     <p>
      <span style="background-color:#ffd900">
       upstream tomcat{
       <!-- -->
      </span>
      <br/>
      <span style="background-color:#ffd900">
       server 192.168.60.11:8081;
      </span>
      <br/>
      <span style="background-color:#ffd900">
       server 192.168.60.11:8082;
      </span>
      <br/>
      <span style="background-color:#ffd900">
       server 192.168.60.11:8083;
      </span>
     </p>
     <p>
      <span style="background-color:#ffd900">
       }
      </span>
     </p>
    </blockquote>
    <blockquote>
     <p>
      server {
      <!-- -->
      <br/>
      listen       80;
      <br/>
      server_name  localhost;
     </p>
     <p>
      #charset koi8-r;
     </p>
     <p>
      #access_log  logs/host.access.log  main;
     </p>
     <p>
      location / {
      <!-- -->
      <br/>
      root   html;
      <br/>
      index  index.php index.html index.htm;
      <br/>
      <span style="background-color:#ffd900">
       proxy_pass http://tomcat;
      </span>
      <br/>
      }
     </p>
    </blockquote>
