long long n,ans=1;
int main()
{
    scanf("%lld",&n);
    for(long long i=2;i*i<=n;i++)
        {
        	int cnt=0; //cnt计数，表示质因子pri[i]的指数
        	while(!(n%i))cnt++,n/=i;
        	if(cnt%2)ans*=i; //如果指数不是偶数，在x中要有一个这个质因子，保证指数为偶数
		}
	if(n!=1)ans*=n;//注意n没分尽的情况
	printf("%lld",ans);
	return 0;
}
