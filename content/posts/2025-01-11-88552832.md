# mysql导出数据库设计说明书

-- 导出某一确定数据库的确定表的相关信息

set @rownum=0;
  
SELECT
  
@rownum:=@rownum+1 AS 序号,
  
COLUMN\_COMMENT AS 列的描述,
  
COLUMN\_NAME AS 列名,
  
COLUMN\_TYPE AS 类型 ,
  
CHARACTER\_MAXIMUM\_LENGTH as 长度,
  
(CASE WHEN  IS\_NULLABLE = 'YES' THEN '是' ELSE '否' END) AS 是否为空
  
FROM
  
information\_schema.`COLUMNS`
  
WHERE
  
TABLE\_SCHEMA = (select database()) AND
  
TABLE\_NAME = ('t\_draft\_info') ORDER BY  ORDINAL\_POSITION ;

--导出确定数据库内的所有表的相关信息

set @rownum=0;
  
SELECT
  
@rownum:=@rownum+1 AS 序号 ,
  
t.TABLE\_COMMENT AS 描述 ,
  
t.table\_name AS 表名
  
FROM
  
information\_schema.`TABLES` t
  
WHERE
  
t.TABLE\_SCHEMA = (select database()) ORDER BY t.TABLE\_NAME;
