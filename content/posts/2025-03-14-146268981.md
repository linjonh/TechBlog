  &lt;!-- 1.引入依赖 --&gt;
  &lt;script src="https://unpkg.com/react@18/umd/react.development.js"&gt;&lt;/script&gt;
  &lt;script src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"&gt;&lt;/script&gt;

  &lt;!-- 编写React代码 --&gt;

    // React18之前：ReactDOM.render()
    // ReactDOM.render(&lt;h2&gt;Hello World&lt;/h2&gt;,document.getElementById('root'))

    // React18之后：createRoot
    const root = ReactDOM.createRoot(document.getElementById('root'))
    /* 
      要修改文本内容
      step1:将文本定义成变量
    */
    let message = 'Hello World'

    //step2:监听按钮的点击
    function btnClick() {
      // 2.1修改数据
      message = 'Hello React'

      // 2.2重新渲染界面
      rootRender()
    }

    rootRender()
    // step3.封装一个渲染函数
    function rootRender( ){
      // 第一个（）表示方法；第二个()表示它是一个整体
      root.render((
        &lt;div&gt;
          &lt;h1&gt;{message}&lt;/h1&gt;
          &lt;button onClick={btnClick}&gt;修改文本&lt;/button&gt;
        &lt;/div&gt;
      ))
    }
  &lt;/script&gt;

    <h4 id="P5RyA">
     案例-组件化开发
    </h4>
    <p id="u258841b2">
     可使用类的方式，将其封装成一个组件
    </p>
    <pre id="dNvnk"><code class="language-html">&lt;body&gt;
  &lt;div id="root"&gt;&lt;/div&gt;

  &lt;script src="../lib/react.js"&gt;&lt;/script&gt;
  &lt;script src="../lib/react-dom.js"&gt;&lt;/script&gt;
  &lt;script src="../lib/babel.js"&gt;&lt;/script&gt;
  &lt;script type="text/babel"&gt;
    // 使用组件重构代码
    // 1.类组件-类名必须大写！！
    // 2.函数式组件
    class App extends React.Component {
      // 组件数据
      constructor() {
        super()

        //this.state = {定义的数据}
        this.state = {
          message:"Hello World"
        }
      }
      // 组件方法(实例方法)
      btnClick() {
        //该函数默认this指向为undefined——改变其this指向
        // 内部完成两件事：1.将state中的值修改掉  2.自动重新执行render函数
        this.setState({
          message:"Hello React"
        })
      }

      // render中返回的jsx内容即root根节点渲染内容
      render() {
        // console.log("render",this);//此处this-App组件实例
        return (
          &lt;div&gt;
            &lt;h2&gt;{this.state.message}&lt;/h2&gt;
            &lt;button onClick={this.btnClick.bind(this)}&gt;修改文本&lt;/button&gt;
          &lt;/div&gt;
        )
      }
    }
    
    // 将组件渲染到界面上
    const root = ReactDOM.createRoot(document.getElementById('root'))
    root.render(&lt;App /&gt;)
  &lt;/script&gt;
&lt;/body&gt;</code></pre>

    <h4 id="mewAM">
     jsx书写规范
    </h4>
    <pre id="GbKd2"><code class="language-javascript">render(){
  const {message} = this.state
  /* 
  书写规范：
  1.JSX结构中只有一个根元素
  2.JSX结构通常包裹一个（）—— 将整个jsx当成一个整体，实现换行
  、3.jsx可以是单标签，也可以是双标签，但是单标签必须以/&gt;结尾
  */
  return (
    &lt;div&gt;
      &lt;h1&gt;{ message }&lt;/h1&gt;
      &lt;br/&gt;
    &lt;/div&gt;
  )
}</code></pre>

    <h4 id="BszwB">
     jsx插入内容
    </h4>
    <pre id="gQoZg"><code class="language-html">&lt;body&gt;
  &lt;div id="root"&gt;&lt;/div&gt;

  &lt;script src="../lib/react.js"&gt;&lt;/script&gt;
  &lt;script src="../lib/react-dom.js"&gt;&lt;/script&gt;
  &lt;script src="../lib/babel.js"&gt;&lt;/script&gt;

  &lt;script type="text/babel"&gt;
    //1.创建root
    const root = ReactDOM.createRoot(document.getElementById('root'))

    //2..定义App根组件
    class App extends React.Component {
      constructor() {
        super()

        this.state = {
          message:"Hello World",
          names:[ 'Jack', 'Tom', 'Lucy' ],
          count:100,

          aaa:undefined,
          bbb:null,
          ccc:true,

          friend:{name:'lucy', age:22},

          firstNmae:'张',
          lastName:'三',

          age:20,

          movies:["哪吒","唐探","三体"]
        }
      }

      render(){
        // 1.插入标识符
        const {message, names,count} = this.state
        const { aaa, bbb, ccc} = this.state
        const {friend} = this.state

        // 2.对内容进行运算后显示(插入表达式)
        const {firstNmae, lastName} = this.state
        const fullName = firstNmae + "" + lastName

        const {age} = this.state
        const ageStr = age &gt; 18 ? '成年人' : '未成年人'
        const {movies} = this.state
        const items = movies.map(item =&gt; &lt;li&gt;{item}&lt;/li&gt;)

        // 3.返回jsx的内容
        return (
          &lt;div&gt;
            {/* 1.Number/String/Array直接显示处理 */}
            &lt;h1&gt;{ message }&lt;/h1&gt;
            &lt;h2&gt; {names} &lt;/h2&gt;
            &lt;h2&gt; {count} &lt;/h2&gt;

            {/* 2.undefined/null/Boolean页面什么都不显示-取值为空 */}
            &lt;h2&gt;{aaa}&lt;/h2&gt;
            &lt;h2&gt;{bbb}&lt;/h2&gt;
            &lt;h3&gt;{ccc}&lt;/h3&gt;

            {/* 3.Object类型不能作为子元素显示 */}
            {/* &lt;h2&gt;{friend}&lt;/h2&gt; */}
            &lt;h2&gt;{Object.keys(friend)}&lt;/h2&gt;

            {/* 4.可插入对应的表达式 */}
            &lt;h2&gt;{ 10 + 20 }&lt;/h2&gt;
            &lt;h2&gt;{firstNmae + "" + lastName}&lt;/h2&gt;
            &lt;h2&gt;{fullName}&lt;/h2&gt;

            {/* 5.插入三元运算符 */}
            &lt;h2&gt;{ageStr}&lt;/h2&gt;
            &lt;h2&gt;{age &gt;=18 ? "成年" : "未成年"}&lt;/h2&gt;

            {/* 可以调用方法获取结果 */}
            &lt;ul&gt;{items}&lt;/ul&gt;
            &lt;ul&gt;
              {movies.map(item =&gt; &lt;li&gt;{item}&lt;/li&gt;)}
            &lt;/ul&gt;
            &lt;ul&gt;{this.getMovies()}&lt;/ul&gt;
          &lt;/div&gt;
        )
      }

      getMovies(){
        const liEls = this.state.movies.map(item =&gt; &lt;li&gt;{item}&lt;/li&gt;)
        return liEls
      }
    }

    // 3.将App组件渲染到root上
    root.render(&lt;App /&gt;)
  &lt;/script&gt;
&lt;/body&gt;</code></pre>

  &lt;script src="../lib/react.js"&gt;&lt;/script&gt;
  &lt;script src="../lib/react-dom.js"&gt;&lt;/script&gt;
  &lt;script src="../lib/babel.js"&gt;&lt;/script&gt;

  &lt;script type="text/babel"&gt;
    //1.创建root
    const root = ReactDOM.createRoot(document.getElementById('root'))

    //2..定义App根组件
    class App extends React.Component {
      constructor() {
        super()

        this.state = {
          title:"学习第一天",
          imgURL:"https://p1.ssl.qhmsg.com/t0143b426aee3bb03cb.jpg",
          link:"htttp://www.baidu.com",

          isActive:true,
          objStyle:{color:"red", fontSize:"30px"}
        }
      }

      render(){
        const {title, imgURL,link,isActive,objStyle} = this.state

        // 需求：isActive为true时，添加active类名
        // 写法一：
        const className = `abc cba ${isActive? "active" : ""}`
        // 写法二：
        const classList = ["abc","cba"]
        if(isActive) {
          classList.push("active")
        }
        return (
          &lt;div&gt;
            {/* 1.基本属性绑定 */}
            &lt;h1 title={title}&gt;我是h1元素&lt;/h1&gt;
            {/* &lt;img src={imgURL} alt=""/&gt; */}
            &lt;a href={link}&gt;百度一下&lt;/a&gt;

            {/* 2.绑定class属性 */}
            &lt;h2 className={className}&gt;哈哈哈哈哈&lt;/h2&gt;
            &lt;h2 className={classList.join(" ")}&gt;哈哈哈哈哈&lt;/h2&gt;

            {/* 3.绑定style属性:绑定对象类型 */}
            &lt;h2 style={<!-- -->{color:"red", fontSize:"30px"}}&gt;11111&lt;/h2&gt;
            &lt;h2 style= {objStyle}&gt;888&lt;/h2&gt;
          &lt;/div&gt;
        )
      }
    }

    // 3.将App组件渲染到root上
    root.render(&lt;App /&gt;)
  &lt;/script&gt;
