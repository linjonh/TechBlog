    <h2>
     代码

    <pre><code class="language-cpp">class Solution {
public:
    bool backspaceCompare(string s, string t) {
        stack&lt;char&gt; ss;
        stack&lt;char&gt; tt;

        for(int i = 0; i &lt; s.size(); i++){
            if(s[i] != '#'){
                ss.push(s[i]);
            }else{
                if(!ss.empty()){
                    ss.pop();
                }
            }
        }

        for(int i = 0; i &lt; t.size(); i++){
            if(t[i] != '#'){
                tt.push(t[i]);
            }else{
                if(!tt.empty()){
                    tt.pop();
                }
            }
        }

        if(ss.size() != tt.size()) return false;
        while(!ss.empty() &amp;&amp; !tt.empty()){
            if(ss.top() != tt.top()){
                return false;
            }
            ss.pop();
            tt.pop();
        }

        return true;
    }
};</code></pre>
