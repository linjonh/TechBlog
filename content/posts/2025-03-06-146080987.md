    <table>
     <thead>
      <tr>
       <th>

       </th>
       <th>
        优点
       </th>
       <th>

     <tbody>
      <tr>
       <td>
        <strong>
         JAXB
        </strong>
       </td>
       <td>

       <td>

       </td>

       </td>
      </tr>

       <td>

        </strong>
       </td>

     <strong>
      Jackson
     </strong>
