方法4: 利用数组的indexOf下标属性来查询。

function unique4(arr) {

var newArr = []

for (var i = 0; i < arr.length; i++) {

if (newArr.indexOf(arr[i])===-1) {

newArr.push(arr[i])

}

}

return newArr

}

console.log(unique4([1, 1, 2, 3, 5, 3, 1, 5, 6, 7, 4]));

function unique5(arr) {
