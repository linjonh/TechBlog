#### 返回多个值
def test_b(v1,v2):
    sum = v1 + v2 
    jf = v1 - v2
    cf = v1 * v2
    return sum,jf,cf
        
res = test_b(2,5)
print res
(7, -3, 10)  #返回是一个元组

va1,va2,va3 = test_b(2,5)
print va1,va2,va3 
7 -3 10     #返回多个值
