        {
            foreach (string item in Directory.GetFileSystemEntries(path))
            {
                if (File.GetAttributes(item).HasFlag(FileAttributes.Directory))
                {
                    // 递归删除子目录
                    DeleteDirectoryContents(item);

                else
                {
                    // 删除文件
