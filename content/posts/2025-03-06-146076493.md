    <pre><code class="language-Scala">package sparksessiondemo

object adress {
  def main(args: Array[String]): Unit = {
    val a:Integer = 10
    val b:Integer = 10

    println(a == b)    //true
    println(a.equals(b))  //true
    println(a.eq(b))   //true


    val a1: Integer = 128
    val b1: Integer = 128

    println(a1 == b1) //true
    println(a1.equals(b1)) //true
    println(a1.eq(b1)) //false


    val c: Int = 10
    val d: Int = 10

    println(c == d) //true
    println(c.equals(d)) //true
    //println(c.eq(d)) //the result type of an implicit conversion must be more specific than AnyRef


    val c1: Int = 128
    val d1: Int = 128

    println(c1 == d1) //true
    println(c1.equals(d1)) //true
   // println(c1.eq(d1)) //the result type of an implicit conversion must be more specific than AnyRef


    val e: String = "128"
    val f: String = "128"

    println(e == f) //true
    println(e.equals(f)) //true
    println(e.eq(f))  //true


    val e1: String = new String("128")
    val f1: String = new String("128")

    println(e1 == f1) //true
    println(e1.equals(f1)) //true
    println(e1.eq(f1))  //false

  }
}</code></pre>

    <hr/>
    <h4>
     <strong>
      1.
      <code>
       Integer
      </code>
      类型的比较
     </strong>
    </h4>
    <p>
     <strong>
      <code>
       Integer
      </code>
      是 Java 中的包装类，Scala 继承了 Java 的行为。
     </strong>
    </p>
    <h5>
     <strong>
      第一部分：
      <code>
       a
      </code>
      和
      <code>
       b
      </code>
      的比较
     </strong>
    </h5>
    <pre><code class="language-Scala">val a: Integer = 10
val b: Integer = 10

println(a == b)    // true
println(a.equals(b))  // true
println(a.eq(b))   // true</code></pre>
    <ul>
     <li>
      <p>
       <strong>
        <code>
         a == b
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          <code>
           ==
          </code>
          是值比较，
          <code>
           10 == 10
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         a.equals(b)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          <code>
           equals
          </code>
          是值比较，
          <code>
           10.equals(10)
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         a.eq(b)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          <code>
           eq
          </code>
          是引用比较。由于
          <code>
           10
          </code>
          在
          <code>
           -128
          </code>
          到
          <code>
           127
          </code>
          范围内，
          <code>
           a
          </code>
          和
          <code>
           b
          </code>
          指向缓存中的同一个对象，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
    </ul>
    <h5>
     <strong>
      第二部分：
      <code>
       a1
      </code>
      和
      <code>
       b1
      </code>
      的比较
     </strong>
    </h5>
    <pre><code class="language-Scala">val a1: Integer = 128
val b1: Integer = 128

println(a1 == b1) // true
println(a1.equals(b1)) // true
println(a1.eq(b1)) // false</code></pre>
    <ul>
     <li>
      <p>
       <strong>
        <code>
         a1 == b1
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           128 == 128
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         a1.equals(b1)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           128.equals(128)
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         a1.eq(b1)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          引用比较。由于
          <code>
           128
          </code>
          超出了
          <code>
           -128
          </code>
          到
          <code>
           127
          </code>
          的范围，
          <code>
           a1
          </code>
          和
          <code>
           b1
          </code>
          是两个不同的对象，结果为
          <code>
           false
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
    </ul>
    <hr/>
    <h4>
     <strong>
      2.
      <code>
       Int
      </code>
      类型的比较
     </strong>
    </h4>
    <p>
     <strong>
      <code>
       Int
      </code>
      是 Scala 中的基本类型（值类型），不是引用类型。
     </strong>
    </p>
    <h5>
     <strong>
      第一部分：
      <code>
       c
      </code>
      和
      <code>
       d
      </code>
      的比较
     </strong>
    </h5>
    <pre><code class="language-Scala">val c: Int = 10
val d: Int = 10

println(c == d) // true
println(c.equals(d)) // true
// println(c.eq(d)) // 编译错误</code></pre>
    <ul>
     <li>
      <p>
       <strong>
        <code>
         c == d
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           10 == 10
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         c.equals(d)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           10.equals(10)
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         c.eq(d)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          <code>
           eq
          </code>
          是引用比较，但
          <code>
           Int
          </code>
          是值类型，没有引用。Scala 会尝试将
          <code>
           Int
          </code>
          隐式转换为
          <code>
           AnyRef
          </code>
          ，但无法确定具体类型，因此编译错误。
         </strong>
        </p>
       </li>
      </ul>
     </li>
    </ul>
    <h5>
     <strong>
      第二部分：
      <code>
       c1
      </code>
      和
      <code>
       d1
      </code>
      的比较
     </strong>
    </h5>
    <pre><code class="language-Scala">val c1: Int = 128
val d1: Int = 128

println(c1 == d1) // true
println(c1.equals(d1)) // true
// println(c1.eq(d1)) // 编译错误</code></pre>
    <ul>
     <li>
      <p>
       <strong>
        <code>
         c1 == d1
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           128 == 128
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         c1.equals(d1)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           128.equals(128)
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         c1.eq(d1)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          同上，
          <code>
           Int
          </code>
          是值类型，无法进行引用比较，编译错误。
         </strong>
        </p>
       </li>
      </ul>
     </li>
    </ul>

    <h4>
     <strong>
      3.
      <code>
       String
      </code>
      类型的比较
     </strong>
    </h4>
    <p>
     <strong>
      <code>
       String
      </code>
      是引用类型，Scala 中的
      <code>
       String
      </code>
      行为与 Java 一致。
     </strong>
    </p>
    <h5>
     <strong>
      第一部分：
      <code>
       e
      </code>
      和
      <code>
       f
      </code>
      的比较
     </strong>
    </h5>
    <pre><code class="language-Scala">val e: String = "128"
val f: String = "128"

println(e == f) // true
println(e.equals(f)) // true
println(e.eq(f))  // true</code></pre>
    <ul>
     <li>
      <p>
       <strong>
        <code>
         e == f
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           "128" == "128"
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         e.equals(f)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           "128".equals("128")
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         e.eq(f)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          引用比较。由于字符串常量池的存在，
          <code>
           e
          </code>
          和
          <code>
           f
          </code>
          指向同一个对象，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
    </ul>
    <h5>
     <strong>
      第二部分：
      <code>
       e1
      </code>
      和
      <code>
       f1
      </code>
      的比较
     </strong>
    </h5>
    <pre><code class="language-Scala">val e1: String = new String("128")
val f1: String = new String("128")

println(e1 == f1) // true
println(e1.equals(f1)) // true
println(e1.eq(f1))  // false</code></pre>
    <ul>
     <li>
      <p>
       <strong>
        <code>
         e1 == f1
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           "128" == "128"
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         e1.equals(f1)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          值比较，
          <code>
           "128".equals("128")
          </code>
          ，结果为
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         e1.eq(f1)
        </code>
        ：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          引用比较。由于
          <code>
           new String("128")
          </code>
          创建了新的对象，
          <code>
           e1
          </code>
          和
          <code>
           f1
          </code>
          是两个不同的对象，结果为
          <code>
           false
          </code>
          。

        </p>

      </ul>

        <code>
         Integer
        </code>
        的缓存机制：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          Java 和 Scala 对
          <code>
           Integer
          </code>
          类型有一个缓存机制：对于值在
          <code>
           -128
          </code>
          到
          <code>
           127
          </code>
          之间的
          <code>
           Integer
          </code>
          对象，会从缓存中返回相同的对象，
          <code>
           eq
          </code>
          返回
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
       <li>
        <p>
         <strong>
          对于超出这个范围的值，每次都会创建新的
          <code>
           Integer
          </code>
          对象，
          <code>
           eq
          </code>
          返回
          <code>
           false
          </code>
          。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         Int
        </code>
        是值类型：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          <code>
           Int
          </code>
          没有引用，
          <code>
           eq
          </code>
          无法使用。
         </strong>
        </p>
       </li>
      </ul>
     </li>
     <li>
      <p>
       <strong>
        <code>
         String
        </code>
        的常量池：
       </strong>
      </p>
      <ul>
       <li>
        <p>
         <strong>
          直接赋值的字符串会使用常量池，
          <code>
           eq
          </code>
          返回
          <code>
           true
          </code>
          。
         </strong>
        </p>
       </li>
       <li>
        <p>
         <strong>
          使用
          <code>
           new String
          </code>
          创建的对象会生成新的引用，
          <code>
           eq
          </code>
          返回
          <code>
           false。
          </code>
         </strong>
        </p>
       </li>
       <li>
        <p>
         <strong>
          <code>
           关于String常量池的更详细说明，见：
          </code>
          <a href="https://blog.csdn.net/m0_63322122/article/details/146076342" title="Scala 中的String常量池-CSDN博客">
           Scala 中的String常量池-CSDN博客
          </a>
         </strong>
        </p>
       </li>
      </ul>
     </li>
    </ol>

     <strong>
      通过这些例子，可以清楚地理解 Scala 中不同类型的行为及其比较操作的区别。
     </strong>
