###### 2、读取pcap数据包文件

```python
import pyshark
import binascii

# 打开PCAP文件并设置过滤器
packets = pyshark.FileCapture('out.pcap', display_filter="icmp.type==0")

# 处理每个数据包
for each in packets:
    try:
        # 解码ICMP数据负载
        data = binascii.unhexlify(each.icmp.data).decode()
        if data.startswith('$$START$$'):
            data = data[len('$$START$$'):]
            print(data)
        
    except Exception as e:
        print(f"Error processing packet: {e}")

```
