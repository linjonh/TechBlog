    <pre><code class="language-java">@RestController
@RequestMapping("/")
public class RetrievalController {
    private static final String ORIGINAL_API_URL = "http://你的ragflowapi的主机和端口/api/v1/retrieval";

    @PostMapping("/retrieval")
    public ResponseEntity&lt;Map&lt;String, List&lt;Map&lt;String, Object&gt;&gt;&gt;&gt; retrieveChunks(
            @RequestBody RetrievalRequest requestData,
            @RequestHeader("Authorization") String authorization) {
        // 验证 Authorization 头
        if (!authorization.startsWith("Bearer ")) {
            return ResponseEntity.status(HttpStatus.UNAUTHORIZED).body(null);
        }
        String apiKey = authorization.split(" ")[1];

        // 构建请求体
        Map&lt;String, Object&gt; payload = new HashMap&lt;&gt;();
        payload.put("question", requestData.getQuery());
        payload.put("dataset_ids", Collections.singletonList(requestData.getKnowledge_id()));
        payload.put("top_k", requestData.getRetrieval_setting().getTop_k());
        payload.put("similarity_threshold", requestData.getRetrieval_setting().getScore_threshold());

        // 构建请求头
        HttpHeaders headers = new HttpHeaders();
        headers.setContentType(MediaType.APPLICATION_JSON);
        headers.set("Authorization", "Bearer " + apiKey);
        HttpEntity&lt;Map&lt;String, Object&gt;&gt; entity = new HttpEntity&lt;&gt;(payload, headers);

        // 发送请求
        RestTemplate restTemplate = new RestTemplate();
        try {
            ResponseEntity&lt;Map&gt; response = restTemplate.exchange(
                    ORIGINAL_API_URL,
                    HttpMethod.POST,
                    entity,
                    Map.class
            );

            if (response.getStatusCode() != HttpStatus.OK) {
                return ResponseEntity.status(response.getStatusCode()).body(null);
            }

            // 处理响应数据
            Map&lt;String, Object&gt; originalData = response.getBody();
            Map&lt;String, Object&gt; data = (Map&lt;String, Object&gt;) originalData.get("data");
            List&lt;Map&lt;String, Object&gt;&gt; chunks = (List&lt;Map&lt;String, Object&gt;&gt;) data.get("chunks");

            List&lt;Map&lt;String, Object&gt;&gt; records = new ArrayList&lt;&gt;();
            if (chunks != null) {
                for (Map&lt;String, Object&gt; chunk : chunks) {
                    Map&lt;String, Object&gt; record = new HashMap&lt;&gt;();
                    record.put("content", chunk.get("content"));
                    record.put("score", chunk.get("similarity"));
                    record.put("title", chunk.getOrDefault("document_keyword", "Unknown Document"));
                    Map&lt;String, Object&gt; metadata = new HashMap&lt;&gt;();
                    metadata.put("document_id", chunk.get("document_id"));
                    record.put("metadata", metadata);
                    records.add(record);
                }
            }

            Map&lt;String, List&lt;Map&lt;String, Object&gt;&gt;&gt; result = new HashMap&lt;&gt;();
            result.put("records", records);
            return ResponseEntity.ok(result);
        } catch (HttpClientErrorException e) {
            return ResponseEntity.status(e.getStatusCode()).body(null);
        }
    }
}</code></pre>
