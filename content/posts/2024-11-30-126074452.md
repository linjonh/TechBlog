    <p>
     今天分享音视频利用ffmpeg进行音视频解码方面的知识。
    </p>
    <p>
     我们都知道视频的存储方式一般都是MP4、avi、FLV等封装的格式，如果需要在设备进行播放视频，就需要对其进行相应的处理，图片一般需要yuv或rgb格式的图片数据才能进行显示，音频需要pcm的格式数据进行播放，所以这时候就需要对封装格式的视频进行解码，而ffmpeg提供了一套接口可以对视频进行处理。下面是解码的流程图。
    </p>
    <p>
     <img alt="" height="603" src="https://i-blog.csdnimg.cn/blog_migrate/d0e688cc5f9a2ab809edef2b196a32e3.png" width="865"/>
    </p>
    <p>
     接下来就可以利用ffmpeg提供的接口进行解码了。
    </p>
    <p>
     首先需要配置ffmpeg的环境，之前已经分享过环境搭建了，有问题的小伙伴可以查看我以前分享过的文章。
    </p>
    <p>
     由于ffmpeg是用c语言编写的库，没有类的概念，所以都是利用上下文的方式来进行参数之间的传递。
    </p>
    <p>
     首先需要调用 av_register_all()初始化所有的封装库，利用avformat_open_input（）初始化解封装器。
    </p>
    <p>
     一个视频一般都包含了视频流和音频流，这边介绍一下如何获取到这两个流的信息
    </p>
    <p>
     一般有两种方式进行获取，一种是通过遍历的方式来进行获取，另外一种是通过ffmpeg提供的接口来进行获取。
    </p>
    <p>
     这边的ic是调用上面的初始化解封装器后获得的解封装上下文，里面包含了所有流的信息，nb_streams是一个有多少个流。
    </p>
    <p>
     遍历的方式：
    </p>
    <pre><code>	//获取音视频流信息(遍历，函数获取)
	for (int i = 0;i &lt; ic-&gt;nb_streams;i++)
	{
		AVStream *as = ic-&gt;streams[i];
		cout &lt;&lt; "format=" &lt;&lt; as-&gt;codecpar-&gt;format &lt;&lt; endl;
		cout &lt;&lt; "codec_id=" &lt;&lt; as-&gt;codecpar-&gt;codec_id &lt;&lt; endl;
		//音频 AVMEDIA_TYPE_AUDIO
		if (as-&gt;codecpar-&gt;codec_type == AVMEDIA_TYPE_AUDIO)
		{
			audioStream = i;
			cout &lt;&lt; i &lt;&lt;"音频信息" &lt;&lt; endl;
			cout &lt;&lt; "sample_rate=" &lt;&lt; as-&gt;codecpar-&gt;sample_rate &lt;&lt; endl;
			cout &lt;&lt; "channels=" &lt;&lt; as-&gt;codecpar-&gt;channels &lt;&lt; endl;
			cout &lt;&lt; "aduio fps=" &lt;&lt; r2d(as-&gt;avg_frame_rate) &lt;&lt; endl;
			//一帧数据 单通道样本数
			cout &lt;&lt; "frame_size=" &lt;&lt; as-&gt;codecpar-&gt;frame_size &lt;&lt; endl;
			//1024 * 2 * 2(根据存储格式,2个字节) fps = sample_rate/frame_size		
		}
		//视频AVMEDIA_TYPE_VIDEO
		else if (as-&gt;codecpar-&gt;codec_type == AVMEDIA_TYPE_VIDEO)
		{
			videoStream = i;
			cout &lt;&lt; i &lt;&lt; "视频信息" &lt;&lt; endl;
			cout &lt;&lt; "width=" &lt;&lt; as-&gt;codecpar-&gt;width &lt;&lt; endl;
			cout &lt;&lt; "height=" &lt;&lt; as-&gt;codecpar-&gt;height &lt;&lt; endl;
			//帧率 fps 分数转换

			cout &lt;&lt; "video fps="&lt;&lt;r2d(as-&gt;avg_frame_rate) &lt;&lt; endl;
		}
	}</code></pre>
    <p>
     还有一种方式是通过av_find_best_stream来获取，通过传入要获取的流获取到对应的索引。
    </p>
    <p>
     解封装完后就会变成对应的音频压缩数据和视频压缩数据，接下来要进行解码操作
    </p>
    <p>
     主要调用api流程如下：
    </p>
    <p>
     1、avcodec_find_decoder    找到视频解码器
    </p>
    <p>
     2、avcodec_alloc_context3  创建解码器上下文
    </p>
    <p>
     3、avcodec_parameters_to_context 配置解码器上下文参数
    </p>
    <p>
     4、avcodec_open2  打开解码器上下文
    </p>
    <p>
     音频也是类似的。
    </p>
    <p>
     解码后图像格式和音频格式一般也不能在显示器和音频设备上进行播放，因为解码后的图像格式一般为yuv420p，显示器支持的一般为rgb格式的，所以需要进行像素格式转换，而解码后的音频格式为AV_SAMPLE_FMT_FLTP，即为32个字节的数据。
     <img alt="" height="435" src="https://i-blog.csdnimg.cn/blog_migrate/e9414e52e6b3e50a986fa4a253f40201.png" width="551"/>
    </p>
    <p>
    </p>
    <p>
     可能通过看自己的电脑配置就可以知道一般音频设备支持的都是16位或者24位大小的音频数据，所以也需要进行音频重采样，这边也主要是利用ffmpeg提供的api
    </p>
    <p>
     像素格式转换用到的函数为
    </p>
    <p>
     1、sws_getCachedContext，获取到格式转换上下文
    </p>
    <p>
     2、sws_scale，进行像素格式转换
    </p>
    <p>
     音频重采样用的函数为：
    </p>
    <p>
     1、swr_alloc，获取到重采样上下文的空间
    </p>
    <p>
     2、swr_alloc_set_opts，设置重采样参数
    </p>
    <p>
     3、 swr_init，初始化重采样上下文
    </p>
    <p>
     4、swr_convert，进行重采样
    </p>
    <p>
     下面是源码分享，主要是打印了流程中的一些信息，并没有对图像和音频进行显示。
    </p>
    <pre><code>#include &lt;iostream&gt;
#include &lt;thread&gt;
extern "C"
{
#include "libavformat/avformat.h"
#include "libavcodec/avcodec.h"
#include "libswscale/swscale.h"
#include "libswresample/swresample.h"
}
#pragma comment(lib,"avformat.lib")
#pragma comment(lib,"avutil.lib")
#pragma comment(lib,"avcodec.lib")
#pragma comment(lib,"swscale.lib")
#pragma comment(lib,"swresample.lib")
using namespace std;

static double r2d(AVRational r)
{
	return r.num == 0 ? 0:(double)r.num / (double)r.den;
}

void XSleep(int ms)
{
	chrono::milliseconds du(ms);
	this_thread::sleep_for(du);
}
int main(int argc,char *argv[])
{
	cout &lt;&lt; "Test Demux FFmpeg.club" &lt;&lt; endl;
	const char *path = "test.mp4";
	//初始化封装库
	av_register_all();

	//初始化网络库(可以打开rtsp rtmp http协议的流媒体视频)
	avformat_network_init();

	//注册解码器库
	avcodec_register_all();

	AVDictionary *opts = NULL;
	//设置rtsp流以tcp协议打开
	av_dict_set(&amp;opts, "rtsp_transport", "tcp", 0);

	//网络延时时间
	av_dict_set(&amp;opts, "max_delay", "500", 0);
	//解封装上下文
	AVFormatContext *ic = NULL;

	int re = avformat_open_input(&amp;ic,path,
		0,// 0表示自动选择解封装器
		&amp;opts//参数设置，比如rtsp的延时时间
		);
	if (re != 0)
	{
		char buf[1024] = { 0 };
		av_strerror(re,buf,sizeof(buf)-1);
		cout &lt;&lt; "open " &lt;&lt; path &lt;&lt; " failed!:" &lt;&lt; buf &lt;&lt; endl;
		getchar();
		return -1;
	}
	cout &lt;&lt; "open " &lt;&lt; path &lt;&lt; " success!" &lt;&lt;  endl;


	//获取流信息
	re = avformat_find_stream_info(ic, 0);

	//总时长 毫秒
	int totalMs = ic-&gt;duration / (AV_TIME_BASE/1000);
	cout &lt;&lt; "totalMs = " &lt;&lt; totalMs &lt;&lt; endl;

	//打印视频流详细信息
	av_dump_format(ic, 0, path, 0);

	//音视频索引，读取时区分音视频
	int videoStream = 0;
	int audioStream = 1;

	//获取音视频流信息(遍历，函数获取)
	for (int i = 0;i &lt; ic-&gt;nb_streams;i++)
	{
		AVStream *as = ic-&gt;streams[i];
		cout &lt;&lt; "format=" &lt;&lt; as-&gt;codecpar-&gt;format &lt;&lt; endl;
		cout &lt;&lt; "codec_id=" &lt;&lt; as-&gt;codecpar-&gt;codec_id &lt;&lt; endl;
		//音频 AVMEDIA_TYPE_AUDIO
		if (as-&gt;codecpar-&gt;codec_type == AVMEDIA_TYPE_AUDIO)
		{
			audioStream = i;
			cout &lt;&lt; i &lt;&lt;"音频信息" &lt;&lt; endl;
			cout &lt;&lt; "sample_rate=" &lt;&lt; as-&gt;codecpar-&gt;sample_rate &lt;&lt; endl;
			cout &lt;&lt; "channels=" &lt;&lt; as-&gt;codecpar-&gt;channels &lt;&lt; endl;
			cout &lt;&lt; "aduio fps=" &lt;&lt; r2d(as-&gt;avg_frame_rate) &lt;&lt; endl;
			//一帧数据 单通道样本数
			cout &lt;&lt; "frame_size=" &lt;&lt; as-&gt;codecpar-&gt;frame_size &lt;&lt; endl;
			//1024 * 2 * 2(根据存储格式,2个字节) fps = sample_rate/frame_size		
		}
		//视频AVMEDIA_TYPE_VIDEO
		else if (as-&gt;codecpar-&gt;codec_type == AVMEDIA_TYPE_VIDEO)
		{
			videoStream = i;
			cout &lt;&lt; i &lt;&lt; "视频信息" &lt;&lt; endl;
			cout &lt;&lt; "width=" &lt;&lt; as-&gt;codecpar-&gt;width &lt;&lt; endl;
			cout &lt;&lt; "height=" &lt;&lt; as-&gt;codecpar-&gt;height &lt;&lt; endl;
			//帧率 fps 分数转换

			cout &lt;&lt; "video fps="&lt;&lt;r2d(as-&gt;avg_frame_rate) &lt;&lt; endl;
		}
	}

	//获取视频流
	videoStream=av_find_best_stream(ic, AVMEDIA_TYPE_VIDEO,-1,-1,NULL,0);
	cout &lt;&lt; "av_find_best_stream videoStream=" &lt;&lt; videoStream &lt;&lt; endl;
	//
	//找到视频解码器
	AVCodec *vcodec=avcodec_find_decoder(ic-&gt;streams[videoStream]-&gt;codecpar-&gt;codec_id);
	if (!vcodec)
	{
		cout &lt;&lt; "can't find the codec id" &lt;&lt; ic-&gt;streams[videoStream]-&gt;codecpar-&gt;codec_id &lt;&lt; endl;
		getchar();
		return -1;
	}
	cout &lt;&lt; "find the codec id" &lt;&lt; ic-&gt;streams[videoStream]-&gt;codecpar-&gt;codec_id &lt;&lt; endl;
	
	//视频解码器打开
	//创建解码器上下文
	AVCodecContext *vc = avcodec_alloc_context3(vcodec);

	//配置解码器上下文参数
	avcodec_parameters_to_context(vc, ic-&gt;streams[videoStream]-&gt;codecpar);
	//8线程解码
	vc-&gt;thread_count = 8;

	//打开解码器上下文
	re = avcodec_open2(vc, 0, 0);
	if (re != 0)
	{
		char buf[1024] = { 0 };
		av_strerror(re, buf, sizeof(buf) - 1);
		cout &lt;&lt; "avcodec_open2 failed!:" &lt;&lt; buf &lt;&lt; endl;
		getchar();
		return -1;
	}
	cout &lt;&lt; "video avcodec_open2 success" &lt;&lt; endl;


	///
	//音频解码器打开
	AVCodec *acodec = avcodec_find_decoder(ic-&gt;streams[audioStream]-&gt;codecpar-&gt;codec_id);
	if (!acodec)
	{
		cout &lt;&lt; "can't find the codec id" &lt;&lt; ic-&gt;streams[audioStream]-&gt;codecpar-&gt;codec_id &lt;&lt; endl;
		getchar();
		return -1;
	}
	cout &lt;&lt; "find the codec id" &lt;&lt; ic-&gt;streams[audioStream]-&gt;codecpar-&gt;codec_id &lt;&lt; endl;
	//创建解码器上下文
	AVCodecContext *ac = avcodec_alloc_context3(acodec);

	//配置解码器上下文参数
	avcodec_parameters_to_context(ac, ic-&gt;streams[audioStream]-&gt;codecpar);
	//8线程解码
	ac-&gt;thread_count = 8;

	//打开解码器上下文
	re = avcodec_open2(ac, 0, 0);
	if (re != 0)
	{
		char buf[1024] = { 0 };
		av_strerror(re, buf, sizeof(buf) - 1);
		cout &lt;&lt; "avcodec_open2 failed!:" &lt;&lt; buf &lt;&lt; endl;
		getchar();
		return -1;
	}
	cout &lt;&lt; "audio avcodec_open2 success" &lt;&lt; endl;

	//malloc AVPacket并初始化
	AVPacket *pkt = av_packet_alloc();
	AVFrame *frame = av_frame_alloc();

	//创建像素格式和尺寸转换上下文
	SwsContext *vctx = NULL;
	unsigned char *rgb = NULL;

	//音频重采样上下文初始化
	SwrContext *actx = swr_alloc();
	actx = swr_alloc_set_opts(actx,
		av_get_default_channel_layout(2),               //输出格式
		AV_SAMPLE_FMT_S16,                              //输出样本格式
		ac-&gt;sample_rate,                                //采样率
		av_get_default_channel_layout(ac-&gt;channels),    //输入格式
		ac-&gt;sample_fmt,
		ac-&gt;sample_rate,
		0,0
		);
	re = swr_init(actx);
	if (re != 0)
	{
		char buf[1024] = { 0 };
		av_strerror(re, buf, sizeof(buf) - 1);
		cout &lt;&lt; "swr_init!:" &lt;&lt; buf &lt;&lt; endl;
		getchar();
		return -1;
	}
	unsigned char *pcm = NULL;

	for (;;)
	{
		int re = av_read_frame(ic,pkt);
		if (re != 0)
		{
			cout &lt;&lt; "===============================================end==================================" &lt;&lt; endl;
			int ms = 3000;//三秒位置
			long long pos = (double)ms / (double)1000 * r2d(ic-&gt;streams[pkt-&gt;stream_index]-&gt;time_base);
			av_seek_frame(ic,videoStream,pos,AVSEEK_FLAG_BACKWARD|AVSEEK_FLAG_FRAME);
			continue;
		}
		cout &lt;&lt; "pkt-&gt;size"&lt;&lt;pkt-&gt;size &lt;&lt; endl;
		//显示时间
		cout &lt;&lt; "pkt-&gt;pts" &lt;&lt; pkt-&gt;pts &lt;&lt; endl;
		//转换成ms 实际的显示时间
		cout &lt;&lt; "pkt-&gt;pts ms=" &lt;&lt; pkt-&gt;pts*(r2d(ic-&gt;streams[pkt-&gt;stream_index]-&gt;time_base)) &lt;&lt; endl;
		//解码时间
		cout &lt;&lt; "pkt-&gt;dts" &lt;&lt; pkt-&gt;dts &lt;&lt; endl;
		AVCodecContext *cc = NULL;
		if (pkt-&gt;stream_index == videoStream)
		{
			cout &lt;&lt; "图像" &lt;&lt; endl;
			cc = vc;
		}
		else if (pkt-&gt;stream_index == audioStream)
		{
			cout &lt;&lt; "音频" &lt;&lt; endl;
			cc = ac;
		}
		else
		{
			break;
		}
		///解码视频
		//发送packet到解码线程,不占用cpu
		//send传NULL后调用多次receive取出所有缓冲帧
		re = avcodec_send_packet(cc,pkt);
		if (re != 0)
		{
			char buf[1024] = { 0 };
			av_strerror(re, buf, sizeof(buf) - 1);
			cout &lt;&lt; "avcodec_send_packet failed!:" &lt;&lt; buf &lt;&lt; endl;
			continue;
		}
		 
		for (;;)
		{
			//不占用cpu，从线程中获取解码接口,一次send可能对应多次receive
			re = avcodec_receive_frame(cc, frame);
			if (re != 0)break;
			cout &lt;&lt; "recv frame" &lt;&lt; frame-&gt;format &lt;&lt; " " &lt;&lt; frame-&gt;linesize[0] &lt;&lt; endl;
			//视频
			if (cc == vc)
			{
				vctx = sws_getCachedContext(
					vctx,  //传NULL会新创建
					frame-&gt;width,frame-&gt;height,     //输入的宽、高、格式
					(AVPixelFormat)frame-&gt;format,  //输入的格式yuv420P
					frame-&gt;width,frame-&gt;height,   //输出的宽高
					AV_PIX_FMT_RGBA,             //输出格式
					SWS_BILINEAR,               //尺寸变化的算法
					0,0,0
					);
				if (vctx)
				{
					if (!rgb)rgb = new unsigned char[frame-&gt;height*frame-&gt;width * 4];
					uint8_t *data[2] = { 0 };
					data[0] = rgb;
					int lines[2] = { 0 };
					lines[0] = frame-&gt;width * 4;
					re = sws_scale(vctx,
						frame-&gt;data,          //输入数据
						frame-&gt;linesize, 0,   //输入行大小
						frame-&gt;height,        //输入高度
						data,                 //输出数据和大小
						lines
						);
					cout &lt;&lt; "sws_scale = " &lt;&lt; re &lt;&lt; endl;
				}
			}
			else   //音频
			{
				uint8_t *data[2] = { 0 };
				if (!pcm)pcm = new uint8_t[frame-&gt;nb_samples * 2 * 2];
				data[0] = pcm;
				re = swr_convert(actx,
					data, frame-&gt;nb_samples,   //输出
					(const uint8_t**)frame-&gt;data, frame-&gt;nb_samples  //输入
					);
				cout &lt;&lt; "swr_convert = " &lt;&lt; re &lt;&lt; endl;

			}
		}


		//释放，引用计数-1 为0释放空间
		av_packet_unref(pkt);

		//XSleep(500);
	}
	av_frame_free(&amp;frame);
	av_packet_free(&amp;pkt);
	if (ic)
	{
		//释放封装上下文，并且把ic置为0
		avformat_close_input(&amp;ic);
	}


	getchar();
	return 0;
}</code></pre>
