    <p>
     这道题我们可以用优先级队列来解决，我们的算法思路如下：
    </p>
    <p>
     先把n个数的和加起来 并把是偶数的值插入到优先级队列里面
    </p>
    <p>
     接下来如果队列有值并且你还有操作次数，你就不断的把最大的偶数除2，并且判断除2的结果是不是偶数，如果是偶数继续插入到堆里面
    </p>
    <p>
     <br/>
    </p>
    <pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;queue&gt;
typedef long long ll;
using namespace std;
ll sum;
ll n,k;
priority_queue &lt;int&gt; heap;
int main()
{
    cin &gt;&gt; n &gt;&gt; k;
    ll x;
    for(int i = 1;i&lt;=n;i++)
    {
        cin &gt;&gt; x;
        sum+=x;
        if(x%2 == 0) heap.push(x);
    }
    while(k--&amp;&amp;heap.size())
    {
       ll t = heap.top();heap.pop();
        sum-=t;
        t/=2;
        if(t%2==0) heap.push(t);
        sum+=t;
        
        
    }
    cout &lt;&lt; sum &lt;&lt; endl;
    
    return 0;
}</code></pre>
