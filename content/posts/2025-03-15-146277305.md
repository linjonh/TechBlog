    <h3>
     步骤
    </h3>
    <p>
     读书签文件，分割到QStringList，点击右键菜单获取序号，交换序号，写书签文件。
    </p>
    <p>
     QStringList.swap(int, int) 过时，QStringList.swapItemsAt(int, int) 代替。
    </p>
    <p>
    </p>
    <h3>
     书签文件路径
    </h3>
    <p>
     mainwindow.h
    </p>
    <pre><code class="language-cpp">const QString filepath_bookmark = QStandardPaths::writableLocation(QStandardPaths::ConfigLocation) + "/gtk-3.0/bookmarks";</code></pre>
    <p>
    </p>
    <h3>
     上移
    </h3>
    <pre><code class="language-cpp">connect(action_bookmark_move_up, &amp;QAction::triggered, [=](){
    QTreeWidgetItem *TWI = ui-&gt;treeWidget_nav-&gt;currentItem();
    int index = TWI_bookmark-&gt;indexOfChild(TWI);
    QFile file(filepath_bookmark);
    if (file.open(QFile::ReadWrite)) {
        QTextStream TS(&amp;file);
        QString s = TS.readAll();
        QStringList SL = s.split("\n");
        SL.swapItemsAt(index, index - 1);
        s = "";
        for (int i=0; i&lt;SL.length(); i++) {
            s += SL.at(i);
            if (i &lt; SL.length() - 1)
                s += "\n";
        }
        file.resize(0);
        TS &lt;&lt; s;
        file.close();
    }
});</code></pre>
    <h3>
     下移
    </h3>
    <p>
     swap -1 改成 +1，其他一样。
    </p>
    <pre><code class="language-cpp">connect(action_bookmark_move_down, &amp;QAction::triggered, [=](){
    QTreeWidgetItem *TWI = ui-&gt;treeWidget_nav-&gt;currentItem();
    int index = TWI_bookmark-&gt;indexOfChild(TWI);
    QFile file(filepath_bookmark);
    if (file.open(QFile::ReadWrite)) {
        QTextStream TS(&amp;file);
        QString s = TS.readAll();
        QStringList SL = s.split("\n");
        SL.swapItemsAt(index, index + 1);
        s = "";
        for (int i=0; i&lt;SL.length(); i++) {
            s += SL.at(i);
            if (i &lt; SL.length() - 1)
                s += "\n";
        }
        file.resize(0);
        TS &lt;&lt; s;
        file.close();
    }
});</code></pre>
