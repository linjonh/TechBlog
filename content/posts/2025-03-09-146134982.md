     前端 | CORS 跨域问题解决
    </h1>
   </div>

 </div>
 <article class="baidu_pl">
  <div class="article_content clearfix" id="article_content">
   <link href="../../assets/css/kdoc_html_views-1a98987dfd.css" rel="stylesheet"/>

   <div class="htmledit_views" id="content_views">
    <h4>
     <strong>
      问题
     </strong>
     ：
     <span style="color:#fe2c24">
      <span style="background-color:#fef2f0">
       Access to fetch at 'http://localhost:3000/save' from origin 'http://localhost:5174' has been blocked by CORS policy: Response to preflight request doesn't pass access control check: No 'Access-Control-Allow-Origin' header is present on the requested resource. If an opaque response serves your needs, set the request's mode to 'no-cors' to fetch the resource with CORS disabled.。
      </span>
     </span>
    </h4>
    <h4>
     <strong>
      分析
     </strong>
     ：
     <strong>
      CORS (跨域请求) 被拦截
     </strong>
     ——
     <code>
      http://localhost:5174
     </code>
     请求
     <code>
      http://localhost:3000
     </code>
     ，
     <span style="background-color:#fef2f0">
      但
     </span>
     <code>
      <span style="background-color:#fef2f0">
       Access-Control-Allow-Origin
      </span>
     </code>
     <span style="background-color:#fef2f0">
      头部没有配置，导致请求被浏览器拦截
     </span>
     。
    </h4>
    <p>
     <img alt="" height="364" src="https://i-blog.csdnimg.cn/direct/3ab6ca9cbcd54d44b0b8e0e700616c28.png" width="1667"/>
    </p>
    <hr/>
    <h4>
     <span style="color:#fe2c24">
      <strong>
       解决：
      </strong>
     </span>
     前端可以尝试进行跨域请求。
    </h4>
    <h5>
     <strong>
      ①修改后端
     </strong>
    </h5>
    <h5>
     （假设是 Express）： 在
     <code>
      server.js
     </code>
     （或
     <code>
      index.js
     </code>
     ）中添加：
    </h5>
    <pre><code class="language-javascript">// 记得ctrl+shift+y打开终端，在里头运行 npm install cors
const cors = require("cors"); 
app.use(cors());</code></pre>
    <p>
    </p>
    <h5>
     （假设包管理中的
     <strong>
      <span style="background-color:#fef2f0">
       "type": "module"
      </span>
     </strong>
     ）
    </h5>
    <pre><code class="language-javascript">// 记得ctrl+shift+y打开终端，在里头运行 npm install cors
import cors from 'cors';
app.use(cors());</code></pre>
    <blockquote>
     根据情况，是
     <span style="color:null">
      <span style="background-color:#fef2f0">
       <strong>
        express服务器
       </strong>
      </span>
     </span>
     还是
     <span style="color:null">
      <strong>
       <span style="background-color:#fef2f0">
        "type": "module"
       </span>
      </strong>
      ，判断是用require还是import
     </span>
    </blockquote>
    <hr/>
    <h5>
     ②手动设置响应头：
    </h5>
    <pre><code class="language-javascript">app.use((req, res, next) =&gt; {
    res.header('Access-Control-Allow-Origin', 'http://localhost:5174');
    res.header('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.header('Access-Control-Allow-Headers', 'Content-Type, Authorization');
    if (req.method === 'OPTIONS') {
      return res.status(200).end();
    }
    next();
  });</code></pre>
    <p>
     或者
    </p>
    <pre><code class="language-javascript">app.use((req, res, next) =&gt; {
  res.header("Access-Control-Allow-Origin", "*");  // 或指定特定前端地址
  res.header("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE, OPTIONS");
  res.header("Access-Control-Allow-Headers", "Content-Type, Authorization");
  next();
});
</code></pre>
    <hr/>
    <h5>
     ③如果后端无法修改，可以在
     <code>
      fetch
     </code>
     请求中加入
     <span style="color:#fe2c24">
      <code>
       <span style="background-color:#fef2f0">
        mode: "cors"
       </span>
      </code>
     </span>
     ：
    </h5>
    <pre><code class="language-javascript">const response = await fetch("http://localhost:3000/save", {
  method: "POST",
  mode: "cors",  // 添加 CORS 模式
  headers: {
    "Content-Type": "application/json"
  },
  body: JSON.stringify({ content: text })
});
</code></pre>
    <hr/>
    <h4>
     完整代码：
    </h4>
    <pre><code class="language-javascript">import cors from 'cors';
import express from 'express';

const app = express();
const PORT = 3000;  // 后端服务器端口

// 允许跨域请求
app.use(cors({
    origin: 'http://localhost:5174', // 允许的前端域名
    methods: ['GET', 'POST'],    // 允许的请求方法
    allowedHeaders: ['Content-Type', 'Authorization'],  // 允许的请求头
})); 

// 额外的跨域处理
app.use((req, res, next) =&gt; {
    res.header('Access-Control-Allow-Origin', 'http://localhost:5174');
    res.header('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
    res.header('Access-Control-Allow-Headers', 'Content-Type, Authorization');
    if (req.method === 'OPTIONS') {
      return res.status(200).end();
    }
    next();
  });

app.use(express.json());    // 解析JSON请求体

// 添加 get请求，检查后端是否正常运行
app.get('/',  (req, res) =&gt; {
  res.send('√后端运行正常！');
});</code></pre>
    <blockquote>
     <p>
      注： 注意顺序，press和cors的顺序
     </p>
    </blockquote>
