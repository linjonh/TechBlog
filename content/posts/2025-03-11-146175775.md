    <h5>
     讲解算法原理
    </h5>
    <p>
     解法：
     <br/>
     算法思路：
     <br/>
     简单滑动窗⼝~
    </p>
    <h5>
     编写代码
    </h5>
    <p>
     c++算法代码：
    </p>
    <pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;string&gt;
using namespace std;
int main()
{
 string s;
 cin &gt;&gt; s;
 int left = 0, right = 0, n = s.size();
 int hash[26] = { 0 }; // 统计窗⼝内每种字符出现了多少次
 int count = 0; // 统计窗⼝内⼀共有多少种字符 int ret = 0;
 while(right &lt; n)
 {
 if(hash[s[right] - 'a']++ == 0) count++; // 0-&gt;1, 窗⼝内多了⼀种字符 while(count &gt; 2) {
 if(hash[s[left++] - 'a']-- == 1) count--; // 1-&gt;0, 窗⼝内少了⼀种字符 }
 ret = max(ret, right - left + 1); right++;
 }
 cout &lt;&lt; ret &lt;&lt; endl;
 return 0;
}</code></pre>
    <p>
     java算法代码：
    </p>
    <pre><code class="language-java">import java.util.Scanner;
// 注意类名必须为 Main, 不要有任何 package xxx 信息
public class Main
{
 public static void main(String[] args) 
 {
 Scanner in = new Scanner(System.in); char[] s = in.next().toCharArray();
 int left = 0, right = 0, n = s.length; int count = 0; // 统计窗⼝内有多少种字符
 int[] hash = new int[26]; // 统计窗⼝内每种字符出现的次数 int ret = 0;
 while(right &lt; n)
 {
 if(hash[s[right] - 'a']++ == 0) // 0-&gt;1，窗⼝内多了⼀种字符 {
 count++;
 }
 while(count &gt; 2)
 {
 if(hash[s[left++] - 'a']-- == 1) // 1-&gt;0，窗⼝内少了⼀种字符 {
 count--;
 }
 }
 ret = Math.max(ret, right - left + 1); right++;
 }
 System.out.println(ret);
 }
}</code></pre>
