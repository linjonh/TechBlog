    <h4>
     1.前言
    </h4>
    <p>
     vue3中使用reactive函数创建一个响应式对象，当对象数据发生变化的时候，依赖这些数据的计算属性和模板会自动的更新。
    </p>
    <h4>
     2.实例
    </h4>
    <p>
     <img alt="" height="678" src="https://i-blog.csdnimg.cn/direct/cb7f5bd505f5460280036326064e4aa8.png" width="1250"/>
    </p>
    <h5>
     <strong>
      2.1 简写
     </strong>
    </h5>
    <pre><code class="language-html">&lt;template&gt;
  &lt;div&gt;
    &lt;p&gt;用户名: {<!-- -->{ userName }}&lt;/p&gt;
    &lt;p&gt;用户名的大写形式: {<!-- -->{ upperCaseName }}&lt;/p&gt;
    &lt;button @click="changeName"&gt;更改用户名&lt;/button&gt;
  &lt;/div&gt;
&lt;/template&gt;

&lt;script&gt;
import { reactive, computed } from 'vue';

export default {
  name: 'UserComponent',
  setup() {
    // 使用reactive创建响应式对象
    const state = reactive({
      name: 'messi'
    });

    // 定义计算属性upperCaseName，根据state.name计算得到大写形式
    const upperCaseName = computed(() =&gt; state.name.toUpperCase());

    // 定义一个方法来更改用户名
    const changeName = () =&gt; {
      state.name = 'ronaldo';
    };

    // 返回需要在模板中使用的变量和方法
    return {
      userName: state.name,
      upperCaseName,
      changeName
    };
  }
};
&lt;/script&gt;

&lt;style scoped&gt;
/* 样式部分可以根据需要添加 */
&lt;/style&gt;
</code></pre>
    <ul>
     <li>
      <p>
       <code>
        reactive
       </code>
       用于创建一个响应式的
       <code>
        state
       </code>
       对象，该对象包含一个
       <code>
        name
       </code>
       属性。
      </p>
     </li>
     <li>
      <p>
       <code>
        computed
       </code>
       用于定义一个计算属性
       <code>
        upperCaseName
       </code>
       ，它的值是基于
       <code>
        state.name
       </code>
       计算得出的大写形式。
      </p>
     </li>
     <li>
      <p>
       当
       <code>
        state.name
       </code>
       的值改变时，
       <code>
        upperCaseName
       </code>
       会自动更新，因为它是计算属性，会跟踪它的依赖并自动重新计算。
      </p>
     </li>
     <li>
      <p>
       <code>
        changeName
       </code>
       是一个方法，用于更改
       <code>
        state.name
       </code>
       的值。
      </p>
     </li>
     <li>
      <p>
       这些变量和方法通过
       <code>
        setup
       </code>
       函数的返回对象暴露给模板，使得模板可以访问和使用它们
      </p>
     </li>
    </ul>
    <h5>
     2.2 考虑读写
    </h5>
    <p>
    </p>
    <pre><code class="language-html">&lt;template&gt;
  &lt;div&gt;
    &lt;p&gt;姓名: {<!-- -->{ fullName }}&lt;/p&gt;
    &lt;p&gt;年龄: {<!-- -->{ age }}&lt;/p&gt;
    &lt;input v-model="firstName" placeholder="修改名"&gt;
    &lt;input v-model="lastName" placeholder="修改姓"&gt;
  &lt;/div&gt;
&lt;/template&gt;

&lt;script&gt;
import { reactive, computed } from 'vue';

export default {
  setup() {
    // 使用reactive创建响应式对象
    const state = reactive({
      firstName: 'messi',
      lastName: 'Lionel',
      age: 34,
    });

    // 定义计算属性，考虑读和写操作
    const fullName = computed({
      get: () =&gt; `${state.firstName} ${state.lastName}`,
      set: (newValue) =&gt; {
        [state.firstName, state.lastName] = newValue.split(' ');
      }
    });

    return {
      firstName: state.firstName,
      lastName: state.lastName,
      age: state.age,
      fullName,
    };
  },
};
&lt;/script&gt;

&lt;style scoped&gt;
  div {
    padding: 20px;
  }
  input {
    margin-top: 10px;
  }
&lt;/style&gt;
</code></pre>
    <ul>
     <li>
      <p>
       <code>
        reactive
       </code>
       用于创建响应式对象
       <code>
        state
       </code>
       ，它包含了组件的状态数据。
      </p>
     </li>
     <li>
      <p>
       <code>
        computed
       </code>
       定义了一个计算属性
       <code>
        fullName
       </code>
       ，它既有
       <code>
        get
       </code>
       方法也有
       <code>
        set
       </code>
       方法。
       <code>
        get
       </code>
       方法返回
       <code>
        firstName
       </code>
       和
       <code>
        lastName
       </code>
       组成的完整名字；
       <code>
        set
       </code>
       方法允许你通过设置
       <code>
        fullName
       </code>
       来修改
       <code>
        firstName
       </code>
       和
       <code>
        lastName
       </code>
       。
      </p>
     </li>
     <li>
      <p>
       <code>
        v-model
       </code>
       指令用于在输入框和计算属性
       <code>
        fullName
       </code>
       之间创建双向绑定，这意味着当用户在输入框中输入内容时，
       <code>
        fullName
       </code>
       会自动更新，并且反过来，
       <code>
        fullName
       </code>
       的变化也会反映在输入框中。
      </p>
     </li>
    </ul>
