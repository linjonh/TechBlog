		FirstName: "yc",
		LastName:  "z",
		Age:       18,
	}

		fmt.Println("error unmarshal XML:", err)
		return
	}

	// <?xml version="1.0" encoding="UTF-8"?>
	// <Person>
	//   <first_name>yc</first_name>
	//   <last_name>z</last_name>
	//   <age>18</age>
	// </Person>
	fmt.Println(xml.Header + string(output))
}
