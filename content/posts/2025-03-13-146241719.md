   <div class="htmledit_views" id="content_views">
    <p>
     <strong>
      首先用 express 搭建后端服务器，注意
     </strong>
     使用中间件解析json格式的请求体，才会获取到 post 参数
    </p>
    <pre><code>app.use(express.json());</code></pre>
    <p>
     app.js
    </p>
    <pre><code class="hljs">const express = require('express')
const app = express()
app.use(express.json());
const port = 3000

app.post('/api/vue2', (req, res) =&gt; {
  console.log(req.body)
  res.header('Access-Control-Allow-Origin','*')
  res.send({ss: req.body})
})

app.listen(port, () =&gt; {
  console.log(`Example app listening on port ${port}`)
})
</code></pre>
    <p>
     node app.js 启动服务
    </p>
    <p>
     vue.config.js
    </p>
    <pre><code class="hljs">const { defineConfig } = require('@vue/cli-service')
module.exports = defineConfig({
  transpileDependencies: true,
  devServer: {
    proxy: {
      '/api': {
        target: 'http://localhost:3000'
      }
    }
  }
})
</code></pre>
    <p>
     main.js
    </p>
    <pre><code class="hljs">import Vue from 'vue'
import App from './App.vue'
import axios from 'axios'

Vue.config.productionTip = false
Vue.prototype.$axios = axios

new Vue({
  render: h =&gt; h(App),
}).$mount('#app')
</code></pre>
    <p>
     App.vue
    </p>
    <pre><code class="hljs">export default {
  name: 'App',
  components: {
    HelloWorld
  },
  mounted() {
    console.log('mounted')
    this.$axios({
      method: 'post',
      url: '/api/vue2',
      data: {
        aa: 'test1',
        bb: 'test2'
      }
    }).then(res =&gt; {
      console.log('res', res)
    })
  }
}</code></pre>
    <p>
     运行后发现调用成功
    </p>
