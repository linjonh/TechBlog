package com.zuidaima.util;

import java.util.Comparator;
import java.util.HashMap;
import java.util.Map;
import java.util.TreeMap;

public class ValueComparator implements Comparator&lt;Long&gt; {

	Map&lt;Long, Double&gt; base;

	public ValueComparator(Map&lt;Long, Double&gt; base) {
		this.base = base;
	}

	public int compare(Long a, Long b) {
		if (base.get(a).doubleValue() &gt;= base.get(b).doubleValue()) {
			return -1;
		} else {
			return 1;
		}
	}

	public static &lt;K, V extends Comparable&lt;V&gt;&gt; Map&lt;K, V&gt; sortByValues(
			final Map&lt;K, V&gt; map) {
		Comparator&lt;K&gt; valueComparator = new Comparator&lt;K&gt;() {
			public int compare(K k1, K k2) {
				int compare = map.get(k2).compareTo(map.get(k1));
				if (compare == 0)
					return 1;
				else
					return compare;
			}
		};
		Map&lt;K, V&gt; sortedByValues = new TreeMap&lt;K, V&gt;(valueComparator);
		sortedByValues.putAll(map);
		return sortedByValues;
	}

	public static void main(String[] args) {
		HashMap&lt;Long, Double&gt; map = new HashMap&lt;Long, Double&gt;();
		map.put(1l, 99.5);
		map.put(2l, 67.2);
		map.put(3l, 67.5);
		map.put(4l, 67.6);

		ValueComparator bvc = new ValueComparator(map);
		TreeMap&lt;Long, Double&gt; sorted_map = new TreeMap&lt;Long, Double&gt;(bvc);

		System.out.println("unsorted map: " + map);

		sorted_map.putAll(map);

		System.out.println("results: " + sorted_map);

		Map&lt;Long, Double&gt; sorted_map2 = sortByValues(map);
		
		System.out.println("results2: " + sorted_map2);
	}
}
