   <div class="htmledit_views" id="content_views">
    <p>
     <span style="color:#fe2c24;">
      <strong>
       使用“ALTER DATABASE”命令对数据库可进行以下修改：
      </strong>
     </span>
    </p>
    <p>
     <strong style="color:#fe2c24;">
      语法格式：
     </strong>
    </p>
    <pre><code class="language-sql">ALTER DATABASE database_name
{	ADD FILE &lt;filespec&gt;[,…n][ TO FILEGROUP filegroup_name ]	/*在文件组中增加数据文件*/
	| ADD LOG FILE &lt;filespec&gt;[,…n]           				/*增加日志文件*/
	| REMOVE FILE logical_file_name           				/*删除数据文件*/
	| ADD FILEGROUP filegroup_name          				/*增加文件组*/
	| REMOVE FILEGROUP filegroup_name      					/*删除文件组*/
	| MODIFY FILE &lt;filespec&gt;                 				/*更改文件属性*/
	| MODIFY NAME = new_dbname           					/*数据库更名*/
	| MODIFY FILEGROUP filegroup_name {filegroup_property | NAME = new_filegroup_name }
	| SET &lt;optionspec&gt; [ ,...n ] [ WITH &lt;termination&gt; ]   	/*设置数据库属性*/
	| COLLATE &lt; collation_name &gt; 							/*指定数据库排序规则*/
}
GO
</code></pre>
    <p>
     <span style="color:#fe2c24;">
      <strong>
       删除数据库：
      </strong>
     </span>
    </p>
    <p>
     <span style="color:#fe2c24;">
      <strong>
       DROP DATABASE（）
      </strong>
     </span>
     <span style="color:#fe2c24;">
      <strong>
       GO--删除指定的数据库+要删除的数据库名称
      </strong>
     </span>
    </p>
    <p>
    </p>
    <pre><code>DROP DATABASE database_name[,…n][;]
GO
</code></pre>
    <p>
     <span style="color:#fe2c24;">
      <strong>
       创建数据库命令方式：
      </strong>
     </span>
    </p>
    <p>
     <span style="color:#fe2c24;">
      <strong>
       使用命令行的方式创建一个名字为”TEST1“的数据库，同时需要明确指定TEST1的数据文件和日志文件所存储的路径和大小等信息。
      </strong>
     </span>
    </p>
    <pre><code class="language-sql">CREATE DATABASE TEST1
ON
(
	NAME='TEST1_DATA',
	FILENAME='C:\Users\TEST1_DATA.MDF',
	SIZE=5MB,
	MAXSIZE=UNLIMITED,
	FILEGROWTH=10%
)
LOG ON
(
	NAME='TEST1_LOG',
	FILENAME='C:\Users\TEST1_LOG.LDF',
	SIZE=5MB,
	MAXSIZE=50MB,
	FILEGROWTH=1MB	
)
GO
</code></pre>
