    <h3>
     <strong>
      测试HDFS操作
     </strong>
    </h3>
    <p>
     # 创建HDFS目录
     <br/>
     hdfs dfs -mkdir /distributed-test
    </p>
    <p>
     # 上传文件到HDFS
     <br/>
     hdfs dfs -put /local/file.txt /distributed-test
    </p>
    <p>
     # 查看文件分布（副本位置）
     <br/>
     hdfs fsck /distributed-test/file.txt -files -blocks -locations
    </p>
