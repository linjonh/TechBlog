    <pre><code class="language-cpp">int myAtoi(char *str) {
    int index = 0;
    int sign = 1;
    int result = 0;
    
    // Step 1: Skip all leading whitespace characters
    // 1. 跳过前导空格
    while (isspace(str[index])) {
        index++;
    }
    
    // Step 2: Check if the next character is '-' or '+'
    // 2. 处理符号
    if (str[index] == '+' || str[index] == '-') {
        sign = (str[index] == '-') ? -1 : 1;
        index++;
    }
    
    // Step 3: Convert the consecutive digits to an integer
    // 3. 读取数字字符
    while (isdigit(str[index])) {
        int digit = str[index] - '0';
 
        // Step 4: Handle overflow and underflow
        // 4. 检查是否溢出
        if (result &gt; (INT_MAX - digit) / 10) {
            return (sign == 1) ? INT_MAX : INT_MIN;
        }
        
        result = result * 10 + digit;
        index++;
    }
    
    // Apply the sign
    // 返回最终结果
    return result * sign;
}</code></pre>
