# vite配置全局scss

```javascript
import { defineConfig } from 'vite'

export default defineConfig({
  server: {
    port: 5173,
    host: '0.0.0.0'
  },
  css: {
    preprocessorOptions: {
      scss: {
        additionalData: `
            @use "@/assets/style/element/index.scss" as *;
            @use "@/assets/style/base.scss" as *;
        `,
        api: 'modern-compiler',
      },
    }, // 全局scss样式
  },
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url))
    },
  }, // 路径别名
})

```
