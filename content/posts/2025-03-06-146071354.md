    <p>
     <strong>
      进入到conf
     </strong>
    </p>
    <pre><code>cd /usr/local/tomcat/conf/
</code></pre>
    <p>
     <strong>
      备份tomcat-users.xml
     </strong>
    </p>
    <pre><code>cp tomcat-users.xml{.,bak}
</code></pre>
    <p>
     <strong>
      编辑tomcat-users.xml
     </strong>
    </p>
    <pre><code> vim tomcat-users.xml
</code></pre>
    <p>
     <strong>
      增加以下内容 配置tomcat-users.xml
     </strong>
    </p>
    <p>
     <img alt="" height="265" src="https://i-blog.csdnimg.cn/direct/aeb3a1e557c843b2807365746439360d.png" width="731"/>
    </p>
    <pre><code>  &lt;role rolename="manager-gui"/&gt;
  &lt;role rolename="admin-gui"/&gt;
  &lt;user username="tomcat" password="123456" roles="manager-gui,admin-gui"/&gt;
</code></pre>
    <p>
     <strong>
      修改网段
     </strong>
    </p>
    <pre><code class="hljs">vim ../webapps/manager/META-INF/context.xml
</code></pre>
    <p>
     <img alt="" height="159" src="https://i-blog.csdnimg.cn/direct/03a261bb59c641e18b8bda95995951b8.png" width="1665"/>
    </p>
    <p>
     <strong>
      接着还需要修改一个文件
     </strong>
    </p>
    <pre><code class="hljs">vim ../webapps/host-manager/META-INF/context.xml
</code></pre>
    <pre><code class="hljs"> allow="192\.\d+\.\d+\.\d+|::1|0:0:0:0:0:0:0:1" /&gt;
</code></pre>
    <p>
    </p>
    <pre><code class="hljs">  allow="^.*$" /&gt;
</code></pre>
