export JAVA_HOME=$JAVA_20_HOME


# 切换 Java 版本的函数
switch_java() {
    if [ $1 = '8' ]; then
        export JAVA_HOME=$JAVA_8_HOME
    elif [ $1 = '20' ]; then
        export JAVA_HOME=$JAVA_20_HOME
    else
        echo "不支持的 Java 版本。请输入 8 或 20。"
    fi
    java -version
}
