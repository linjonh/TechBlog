# micropython的modbus 数据转换

import time,math,struct, json
  
a=1
  
b=struct.pack(">h", a)
  
print(a,b)        #打印结果 1 b'\x00\x01'
  
a=100
  
b=struct.pack(">h", a)
  
print(a,b)        #打印结果 100 b'\x00d'
  
print('%x %x' %(b[0],b[1]))         #打印结果 0 64
  
a=1000
  
b=struct.pack(">h", a)
  
print(a,b)        #打印结果 1000 b'\x03\xe8'
  
print('%x %x' %(b[0],b[1]))        #打印结果 3 e8
