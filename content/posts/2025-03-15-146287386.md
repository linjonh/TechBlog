    <h3>
     <a id="7__157">
     </a>
     7. 事务与锁
    </h3>
    <ul>
     <li>
      <strong>
       事务
      </strong>
      ：确保一系列操作要么全部成功，要么全部失败。
     </li>
    </ul>
    <pre><code class="prism language-sql"><span class="token keyword">BEGIN</span> <span class="token keyword">TRANSACTION</span><span class="token punctuation">;</span>

<span class="token keyword">BEGIN</span> TRY
    <span class="token comment">-- SQL操作</span>
    <span class="token keyword">UPDATE</span> YourTableName <span class="token keyword">SET</span> Age <span class="token operator">=</span> Age <span class="token operator">+</span> <span class="token number">1</span> <span class="token keyword">WHERE</span> Name <span class="token operator">=</span> <span class="token string">'Alice'</span><span class="token punctuation">;</span>
    <span class="token comment">-- 如果所有操作成功，提交事务</span>
    <span class="token keyword">COMMIT</span> <span class="token keyword">TRANSACTION</span><span class="token punctuation">;</span>
<span class="token keyword">END</span> TRY
<span class="token keyword">BEGIN</span> CATCH
    <span class="token comment">-- 如果出现错误，回滚事务</span>
    <span class="token keyword">ROLLBACK</span> <span class="token keyword">TRANSACTION</span><span class="token punctuation">;</span>
    <span class="token comment">-- 处理错误</span>
    <span class="token keyword">PRINT</span> ERROR_MESSAGE<span class="token punctuation">(</span><span class="token punctuation">)</span><span class="token punctuation">;</span>
<span class="token keyword">END</span> CATCH<span class="token punctuation">;</span>

      <strong>
