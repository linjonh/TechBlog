#include<stdio.h>

intmain(void) {

/*

书的价格

*/

double prices[10] = {28.9, 32.7, 45.6, 78,35, 86.2, 27.8, 43, 56, 65};

int i, j, n;

double sum, count;

scanf("%d", &n);

for(i = 0; i < n; ++i) {

sum = 0;

for(j = 0; j < 10; ++j) {

scanf("%lf", &count);

sum += count * prices[j];

}

printf("%.2f\n", sum);

}

return 0;

}
