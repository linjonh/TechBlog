    <h2>
     <a id="_2">
     </a>
     思路
    </h2>
    <h3>
     <a id="_3">
     </a>
     分情况讨论
    </h3>
    <h4>
     <a id="n_4">
     </a>
     n大于零
    </h4>
    <p>
     return 1.0/pow(x,-n);
    </p>
    <h4>
     <a id="n_6">
     </a>
     n小于零
    </h4>
    <p>
     return pow（x/n）；
    </p>
    <h4>
     <a id="n_8">
     </a>
     n为奇数
    </h4>
    <p>
     pow(x,n/2)*pow(x,n/2)*x
    </p>
    <h4>
     <a id="n__10">
     </a>
     n 为偶数
    </h4>
    <p>
     pow(x,n/2)*pow(x,n/2)
    </p>
    <pre><code class="prism language-c++">#define _CRT_SECURE_NO_WARNINGS 1
class Solution {
public:
    double myPow(double x, int n) {
        if (n &lt; 0)return 1.0 / pow(x, -(long long)n);//判断大于零还是小于零
        else return pow(x, n);
    };
    double pow(double x, int n)
    {
        if (n == 0)return 1.0;
        double tmp = pow(x, n / 2);       //快速幂:把n次方不断拆分
        return n % 2 == 0 ? tmp * tmp : tmp * tmp * x;
    };

};
</code></pre>
