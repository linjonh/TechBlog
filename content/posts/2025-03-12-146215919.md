    <h4>
     <strong>
      2. 表操作
     </strong>
    </h4>
    <h5>
     创建表
    </h5>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>CREATE TABLE table_name (
    column1 datatype constraints,
    column2 datatype constraints,
    ...
) COMMENT '表注释';</pre>
    <p>
     示例：
    </p>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>CREATE TABLE employees (
    id INT PRIMARY KEY AUTO_INCREMENT COMMENT '员工编号',
    name VARCHAR(50) NOT NULL COMMENT '姓名',
    age TINYINT UNSIGNED COMMENT '年龄',
    salary DECIMAL(10, 2) COMMENT '薪水'
) COMMENT '员工表';</pre>
    <h5>
     查看所有表
    </h5>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>SHOW TABLES;</pre>

    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>DESC table_name;</pre>
    <p>
     示例：
    </p>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>DESC employees;</pre>

    <h4>
     <strong>
      3. 数据操作
     </strong>
    </h4>
    <h5>
     插入数据
    </h5>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>INSERT INTO table_name (column1, column2, ...)
VALUES (value1, value2, ...);</pre>
    <p>
     示例：
    </p>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>INSERT INTO employees (name, age, salary)
VALUES ('张三', 25, 5000.00);</pre>
    <h5>
     查询数据
    </h5>
    <ul>
     <li>
      <p>
       查询所有数据：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM table_name;</pre>
      <p>
       示例：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM employees;</pre>
     </li>
     <li>
      <p>
       查询特定列：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT column1, column2 FROM table_name;</pre>
      <p>
       示例：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT name, salary FROM employees;</pre>
     </li>
     <li>
      <p>
       条件查询：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM table_name WHERE condition;</pre>
      <p>
       示例：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM employees WHERE age &gt; 30;</pre>
     </li>
     <li>
      <p>
       排序查询：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM table_name ORDER BY column_name ASC|DESC;</pre>
      <p>
       示例：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM employees ORDER BY salary DESC;</pre>
     </li>
     <li>
      <p>
       限制查询结果：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM table_name LIMIT number;</pre>
      <p>
       示例：
      </p>
      <p>
       sql
      </p>
      <p>
       复制
      </p>
      <pre>SELECT * FROM employees LIMIT 5;</pre>
     </li>
    </ul>

    </h4>
    <h5>
     主键约束
    </h5>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>CREATE TABLE table_name (
    id INT PRIMARY KEY,
    ...
);</pre>

     唯一约束
    </h5>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>CREATE TABLE table_name (
    email VARCHAR(100) UNIQUE,
    ...
);</pre>
    <h5>
     非空约束
    </h5>
    <p>
     sql
    </p>
    <p>
     复制
    </p>
    <pre>CREATE TABLE table_name (
    name VARCHAR(50) NOT NULL,
    ...
);</pre>

     外键约束
    </h5>
    <p>
     sql
    </p>
    <p>
     复制
