     这道题我们只要标记每个字符串出现在哪几行就行了，但是直接用 vector存的话就会产生重复，有的字符串可能在一行内出现了两次，就会打印两次，这时候我们可以套一个set来去重
    </p>
    <pre><code class="language-cpp">#include &lt;iostream&gt;
#include &lt;unordered_map&gt;
#include &lt;set&gt;
using namespace std;
unordered_map &lt;string,set&lt;int&gt;&gt; mp;
int n;
int main()
{
	cin &gt;&gt; n;
	for(int i = 1;i&lt;=n;i++)
	{
		int t;cin &gt;&gt; t;
		string s;
		for(int j = 1;j&lt;=t;j++)
		{
			cin &gt;&gt; s;
			mp[s].insert(i);
		}
	}
	int k;cin &gt;&gt; k;
	string t;
	while(k--)
	{
		cin &gt;&gt; t;
		for(auto e : mp[t])
		{
			cout &lt;&lt; e &lt;&lt; " ";
		 } 
		 cout &lt;&lt; endl;
	}
	
	
	
	
	
	
	return 0;
}</code></pre>
