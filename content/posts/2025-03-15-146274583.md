    queens:=make([]int,n)
    col:=make([]bool,n)
    diag1:=make([]bool,n*2-1)
    diag2:=make([]bool,n*2-1)
    ans:=[][]string{}
    var dfs func(int)
    dfs=func(r int){
        if r==n{
            board:=make([]string,n)
            for i,c:=range queens{
                board[i]=strings.Repeat(".",c)+"Q"+strings.Repeat(".",n-c-1)
            }
            ans=append(ans,board)

                queens[r]=c

            }
        }
    }
    dfs(0)
    return ans
}
