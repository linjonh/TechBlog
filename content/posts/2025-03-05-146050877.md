    painter.begin(this);

    {
        pen.setColor(colors[i]);
        pen.setWidth(widths[i]);
        painter.setPen(pen);
        painter.drawLine(lines[i]);
    }
    painter.end();
}

void Widget::mouseMoveEvent(QMouseEvent *event)
{
    end = event-&gt;pos();
    QLine line(start,end);
    lines &lt;&lt; line;
    if (!eraser)
    {
        colors &lt;&lt; color;
        widths &lt;&lt; width;
    }
    else
    {
        colors &lt;&lt; backgroundColor;
        widths &lt;&lt; width;
    }
    start = end;
