     2.1 基础路由配置
    </h4>
    <p>
     JSX
    </p>
    <blockquote>
     <pre><code>// App.js
import { 
  BrowserRouter,
  Routes,
  Route,
  Link,
  Outlet
} from 'react-router-dom';

// 页面组件
function Home() { return &lt;h2&gt;首页&lt;/h2&gt; }
function About() { return &lt;h2&gt;关于我们&lt;/h2&gt; }

function App() {
  return (
    &lt;BrowserRouter&gt;
      &lt;nav&gt;
        &lt;Link to="/"&gt;首页&lt;/Link&gt;
        &lt;Link to="/about"&gt;关于&lt;/Link&gt;
      &lt;/nav&gt;
      
      &lt;Routes&gt;
        &lt;Route path="/" element={&lt;Home /&gt;} /&gt;
        &lt;Route path="/about" element={&lt;About /&gt;} /&gt;
      &lt;/Routes&gt;
    &lt;/BrowserRouter&gt;
  );
}
</code></pre>
    </blockquote>

     2.4 完整示例代码
    </h4>
    <p>
     JSX
    </p>
    <blockquote>
     <pre><code>import React from 'react';
import {
  BrowserRouter,
  Routes,
  Route,
  Link,
  useParams,
  Navigate
} from 'react-router-dom';

// 页面组件
function Home() { return &lt;h2&gt;首页&lt;/h2&gt; }
function About() { return &lt;h2&gt;关于我们&lt;/h2&gt; }
function ProductList() {
  return (
    &lt;div&gt;
      &lt;h2&gt;产品列表&lt;/h2&gt;
      &lt;ul&gt;
        {[1,2,3].map(id =&gt; (
          &lt;li key={id}&gt;
            &lt;Link to={`/products/${id}`}&gt;产品{id}&lt;/Link&gt;
          &lt;/li&gt;
        ))}
      &lt;/ul&gt;
    &lt;/div&gt;
  );
}
function ProductDetail() {
  const { id } = useParams();
  return &lt;h2&gt;产品详情：{id}&lt;/h2&gt;;
}
function NotFound() { return &lt;h2&gt;404 页面不存在&lt;/h2&gt; }

function App() {
  return (
    &lt;BrowserRouter&gt;
      &lt;nav style={<!-- -->{ padding: 20, background: '#eee' }}&gt;
        &lt;Link to="/" style={<!-- -->{ marginRight: 20 }}&gt;首页&lt;/Link&gt;
        &lt;Link to="/about" style={<!-- -->{ marginRight: 20 }}&gt;关于&lt;/Link&gt;
        &lt;Link to="/products"&gt;产品中心&lt;/Link&gt;
      &lt;/nav&gt;

      &lt;Routes&gt;
        &lt;Route path="/" element={&lt;Home /&gt;} /&gt;
        &lt;Route path="/about" element={&lt;About /&gt;} /&gt;
        &lt;Route path="/products" element={&lt;ProductList /&gt;} /&gt;
        &lt;Route path="/products/:id" element={&lt;ProductDetail /&gt;} /&gt;
        &lt;Route path="/old-about" element={&lt;Navigate to="/about" replace /&gt;} /&gt;
        &lt;Route path="*" element={&lt;NotFound /&gt;} /&gt;
      &lt;/Routes&gt;
    &lt;/BrowserRouter&gt;
  );
}

export default App;
</code></pre>
    </blockquote>

    <h4 id="4.1%20%E6%A0%B8%E5%BF%83%E6%A6%82%E5%BF%B5" name="4.1%20%E6%A0%B8%E5%BF%83%E6%A6%82%E5%BF%B5">
     4.1 核心概念
    </h4>
    <ol>
     <li>
      路由配置必须包裹在
      <code>
       &lt;BrowserRouter&gt;
      </code>
      中
     </li>
     <li>
      <code>
       &lt;Routes&gt;
      </code>
      自动选择最佳匹配路由
     </li>
     <li>
      <code>
       element
      </code>
      属性接收React组件（v6新特性）
     </li>
     <li>
      动态参数使用
      <code>
       :paramName
      </code>
      语法
     </li>
     <li>
      索引路由使用
      <code>
       index
      </code>
      属性标记
     </li>
    </ol>
    <h4 id="4.2%20%E6%9C%80%E4%BD%B3%E5%AE%9E%E8%B7%B5" name="4.2%20%E6%9C%80%E4%BD%B3%E5%AE%9E%E8%B7%B5">
     4.2 最佳实践
    </h4>
    <ol>
     <li>
      使用
      <code>
       &lt;Link&gt;
      </code>
      代替
      <code>
       &lt;a&gt;
      </code>
      标签避免页面刷新
     </li>
     <li>
      嵌套路由配合
      <code>
       &lt;Outlet&gt;
      </code>
      实现布局复用
     </li>
     <li>
      <code>
       useNavigate
      </code>
      实现编程式导航
     </li>
     <li>
      <code>
       useLocation
      </code>
      获取路由状态信息
     </li>
     <li>
      <code>
       loader
      </code>
      和
      <code>
       action
      </code>
      处理数据加载（v6.4+）
     </li>
    </ol>
    <h4 id="4.3%20%E5%B8%B8%E8%A7%81%E9%97%AE%E9%A2%98" name="4.3%20%E5%B8%B8%E8%A7%81%E9%97%AE%E9%A2%98">
     4.3 常见问题
    </h4>
    <p>
     JAVASCRIPT
    </p>
    <blockquote>
     <pre><code>// 错误：直接使用&lt;a&gt;标签
&lt;a href="/about"&gt;关于&lt;/a&gt; ❌

// 正确：使用Link组件
&lt;Link to="/about"&gt;关于&lt;/Link&gt; ✅

// 路由匹配优先级问题
&lt;Route path="users/:id" /&gt;
&lt;Route path="users/new" /&gt; ❌ 顺序错误

// 正确顺序
&lt;Route path="users/new" /&gt;
&lt;Route path="users/:id" /&gt; ✅
</code></pre>
    </blockquote>
