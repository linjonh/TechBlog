     这道题跟全排列类似，但是又有一点不同，选出了这r个数后，不分每个数的排列顺序，所以不能直接使用全排列的思想，还有场宽这个词之前没听过，刚开始以为直接在数前面加两个空格，然后挂了，后面才发现不是这样，场宽是指数据项的宽度，可以用printf（”%kd“，n）；来输出场宽为k的数字n。

    <pre><code class="hljs">#include &lt;stdio.h&gt;

{
    //全部选择完了
    if(s==m+1)
    {
        for(int i=1;i&lt;=m;i++)
        {
            //场宽为3
            printf("%3d",a[i]);
        }
        printf("\n");

    }
    for(int i=a[s-1]+1;i&lt;=n;i++)

        //从小到大寻找第s个数的值
            a[s]=i;
        //寻找第s+1个数；
            hly(s+1);

                            num++;
                    }

                for(x=i+1;x&lt;=i+j;x++){

     寻找第一个成员站的位置，看能不能站下站完k名成员，特殊情况，k小于0时，站法零种，k=1时重复计算了，方法数要除以2。
    </p>
    <pre><code class="hljs">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
char a[205][205];
int main()
{
    int n,m,l,s=0,flag;
    scanf("%d %d %d",&amp;n,&amp;m,&amp;l);
    //初始化地图
    for(int i=1;i&lt;=n;i++)
        scanf(" %s",a[i]);
    //特例，没有队员，站不了
    if(l&lt;=0)
    {
        printf("0\n");
        return;
    }
    //寻找第一位队员站的位置
    for(int i=1;i&lt;=n;i++){
        for(int j=0;j&lt;m;j++){
            if(a[i][j]=='.'){
                flag=0;
                //往右寻找能不能站完
                for(int k=0;k&lt;=l-1;k++){
                    if(a[i][j+k]!='.')
                        flag=1;
                }
                if(flag==0)
                    s++;
                flag=0;
                //向下寻找能不能站完
                for(int k=0;k&lt;=l-1;k++){
                    if(a[i+k][j]!='.')
                        flag=1;
                }
                if(flag==0)
                    s++;
            }
        }
    }
    //特列，重复计算了，方法除以2；
    if(l==1)
        s/=2;
    printf("%d\n",s);
    return 0;
}
