    <h2>
     一、项目背景
    </h2>
    <p>
     在客户端/服务器架构开发中，开发者经常面临网络协议栈的调试难题。本文介绍如何基于Qt 6.6.3框架，从零构建跨平台的网络调试助手，支持TCP/UDP协议的双向测试。

    <h2>
     三、核心模块实现解析
    </h2>
    <h2>
     3.1 TCP服务器端实现
    </h2>
    <p>
     <strong>
      QTcpServer监听机制：
     </strong>
    </p>
    <pre><code>// 创建TCP服务器实例
m_tcpServer = new QTcpServer(this);
connect(m_tcpServer, &amp;QTcpServer::newConnection, [=](){
    while(m_tcpServer-&gt;hasPendingConnections()) {
        QTcpSocket* client = m_tcpServer-&gt;nextPendingConnection();
        m_clients.insert(client-&gt;peerAddress().toString(), client);
        emit newClientConnected(client-&gt;peerName());
    }
});

// 启动监听（支持IPv4/IPv6双栈）
if(!m_tcpServer-&gt;listen(QHostAddress::Any, port)) {
    qDebug() &lt;&lt; "Listen failed:" &lt;&lt; m_tcpServer-&gt;errorString();
}</code></pre>
    <p>
     <strong>
      关键技术点：
     </strong>
    </p>
    <ul>
     <li>
      使用QHostAddress::Any实现双栈监听
     </li>
     <li>
      连接管理采用QHash&lt;QString, QTcpSocket*&gt;存储客户端
     </li>
     <li>
      通过pendingConnection队列处理并发连接
     </li>
    </ul>
    <h2>
     3.2 TCP客户端实现
    </h2>
    <p>
     <strong>
      异步连接与数据收发：
     </strong>
    </p>
    <pre><code>// 建立连接
m_tcpSocket-&gt;connectToHost(ip, port);
connect(m_tcpSocket, &amp;QTcpSocket::connected, [=](){
    updateStatus("Connected to " + ip);
});

// 数据接收（处理分包和粘包）
connect(m_tcpSocket, &amp;QTcpSocket::readyRead, [=](){
    QByteArray data = m_tcpSocket-&gt;readAll();
    processNetworkData(data);
});

// 发送二进制数据（支持Hex模式）
void TcpClient::sendData(const QByteArray &amp;data)
{
    if(m_tcpSocket-&gt;state() == QAbstractSocket::ConnectedState) {
        qint64 bytesWritten = m_tcpSocket-&gt;write(data);
        if(bytesWritten == -1) {
            handleError(m_tcpSocket-&gt;error());
        }
    }
}
</code></pre>
    <h2>
     3.3 UDP通信实现
    </h2>
    <p>
     <strong>
      QUdpSocket绑定与广播：
     </strong>
    </p>
    <pre><code>// 服务器端绑定
m_udpSocket-&gt;bind(port, QUdpSocket::ShareAddress);

// 客户端发送数据报
qint64 sentBytes = m_udpSocket-&gt;writeDatagram(data, QHostAddress(ip), port);

// 接收处理（支持组播）
void UdpServer::readPendingDatagrams()
{
    while(m_udpSocket-&gt;hasPendingDatagrams()) {
        QByteArray datagram;
        datagram.resize(m_udpSocket-&gt;pendingDatagramSize());
        QHostAddress sender;
        quint16 senderPort;
        m_udpSocket-&gt;readDatagram(datagram.data(), datagram.size(), 
                                 &amp;sender, &amp;senderPort);
        processDatagram(sender.toString(), senderPort, datagram);
    }
}</code></pre>

     四、关键技术实现细节

    <h2>
     4.1 日志系统设计
    </h2>
    <pre><code>class LogHandler : public QObject {
    Q_OBJECT
public:
    explicit LogHandler(QObject *parent = nullptr)
        : QObject(parent), m_logFile("network_debug.log") 
    {
        if(!m_logFile.open(QIODevice::WriteOnly | QIODevice::Append)) {
            qWarning() &lt;&lt; "Cannot open log file";
        }
        m_stream.setDevice(&amp;m_logFile);
    }

    void writeLog(const QString &amp;msg) {
        QString timestamp = QDateTime::currentDateTime()
                           .toString("yyyy-MM-dd hh:mm:ss.zzz");
        m_stream &lt;&lt; timestamp &lt;&lt; " - " &lt;&lt; msg &lt;&lt; Qt::endl;
        m_stream.flush();
    }

private:
    QFile m_logFile;
    QTextStream m_stream;
};</code></pre>
    <h2>
     4.2 IP地址验证模块
    </h2>
    <p>
     采用正则表达式实现严格的IP/域名验证：
    </p>
    <pre><code>bool validateAddress(const QString &amp;input) {
    // IPv4正则表达式
    const QString ipv4Pattern = 
        R"(^((25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)\.){3}(25[0-5]|2[0-4][0-9]|[01]?[0-9][0-9]?)$)";
    
    // IPv6正则表达式（简化版）
    const QString ipv6Pattern = 
        R"(^([0-9a-fA-F]{1,4}:){7}[0-9a-fA-F]{1,4}$)";
    
    // 域名验证
    const QString domainPattern = 
        R"(^(?:(?!-)[A-Za-z0-9-]{1,63}(?&lt;!-)\.)+[A-Za-z]{2,6}$)";

    QRegularExpressionValidator validator(
        QRegularExpression(QString("(%1|%2|%3)").arg(ipv4Pattern, ipv6Pattern, domainPattern))
    );
    
    int pos = 0;
    return validator.validate(input, pos) == QValidator::Acceptable;
}</code></pre>
    <h2>
     4.3 自动化测试引擎
    </h2>
    <p>
     基于QTimer的定时发送框架：
    </p>
    <pre><code>class AutoTestEngine : public QObject {
    Q_OBJECT
public:
    explicit AutoTestEngine(QObject *parent = nullptr)
        : QObject(parent), m_timer(new QTimer(this))
    {
        connect(m_timer, &amp;QTimer::timeout, this, &amp;AutoTestEngine::onTimeout);
    }

    void startTest(int intervalMs, const QByteArray &amp;testData) {
        m_testData = testData;
        m_timer-&gt;start(intervalMs);
    }

private slots:
    void onTimeout() {
        if(!m_testData.isEmpty()) {
            emit dataReady(m_testData);
        }
    }

signals:
    void dataReady(const QByteArray &amp;data);

private:
    QTimer *m_timer;
    QByteArray m_testData;
};</code></pre>
