    <h3>
     一、userEffect是什么
    </h3>
    <p>
     在React组件中，副作用操作是指那些会被影响的其他的组件，而且不能在渲染过程中完成相关操作。其中，常见的副作用就是从API接口中获取数据。而
     <code>
      useEffect
     </code>
     提供了一种在函数组件中处理副作用的方式，使得函数组件也能处理这些复杂的逻辑。
    </p>
    <h3>
     二、怎么使用
    </h3>
    <p>
     <code>
      useEffect
     </code>
     函数接收两个参数：
    </p>
    <ul>
     <li>
      <strong>
       第一个参数
      </strong>
      ：是一个回调函数，该回调函数包含了副作用操作的代码。
     </li>
     <li>
      <strong>
       第二个参数
      </strong>
      ：是一个可选的依赖项数组，用于控制
      <code>
       useEffect
      </code>
      的执行时机。
     </li>
     <li>
      回调函数、依赖项
     </li>
     <li>
      <pre><code class="hljs">import React, { useEffect } from 'react';

function Example() {
  useEffect(() =&gt; {
    // 副作用操作代码
    console.log('副作用操作执行');

  }, []); // 依赖项数组

  return &lt;div&gt;示例组件&lt;/div&gt;;
}

export default Example;</code></pre>
      <p>
      </p>
     </li>
    </ul>
    <h3>
     三、不同的使用场景
    </h3>
    <h5>
     1、.仅在组件挂载时执行一次（依赖项数组为空）
    </h5>
    <pre><code class="hljs">import React, { useEffect } from 'react';

function ComponentDidMountExample() {
  useEffect(() =&gt; {
    console.log('组件挂载完成');

    // 清理函数，在组件卸载时执行
    return () =&gt; {
      console.log('组件卸载');
    };
  }, []);

  return &lt;div&gt;仅在挂载时执行一次的示例&lt;/div&gt;;
}

export default ComponentDidMountExample;</code></pre>
    <h5>
     2、在组件挂载和更新时都执行（不传递依赖项数组）
    </h5>
    <pre><code class="hljs">import React, { useState, useEffect } from 'react';

function EveryRenderExample() {
  const [count, setCount] = useState(0);

  useEffect(() =&gt; {
    console.log('组件挂载或更新');
  });

  return (
    &lt;div&gt;
      &lt;p&gt;计数: {count}&lt;/p&gt;
      &lt;button onClick={() =&gt; setCount(count + 1)}&gt;增加&lt;/button&gt;
    &lt;/div&gt;
  );
}

export default EveryRenderExample;</code></pre>
    <h5>
     3、仅在特定依赖项变化时执行
    </h5>
    <pre><code class="hljs">import React, { useState, useEffect } from 'react';

function DependencyChangeExample() {
  const [count, setCount] = useState(0);
  const [name, setName] = useState('');

  useEffect(() =&gt; {
    console.log('count 或 name 发生变化');
  }, [count, name]);

  return (
    &lt;div&gt;
      &lt;p&gt;计数: {count}&lt;/p&gt;
      &lt;button onClick={() =&gt; setCount(count + 1)}&gt;增加&lt;/button&gt;
      &lt;input
        type="text"
        value={name}
        onChange={(e) =&gt; setName(e.target.value)}
      /&gt;
    &lt;/div&gt;
  );
}

export default DependencyChangeExample;</code></pre>
    <h3>
     四、使用的好处
    </h3>
    <p>
     <strong>
      代码复用
     </strong>
     ：
     <code>
      useEffect
     </code>
     可以将副作用逻辑封装在一个函数中，提高代码的复用性。
    </p>
    <p>
     <strong>
      逻辑分离
     </strong>
     ：将副作用操作与组件的渲染逻辑分离，使代码更加清晰和易于维护。
    </p>
    <p>
     <strong>
      函数式编程
     </strong>
     ：使用
     <code>
      useEffect
     </code>
     可以在函数组件中处理副作用，避免了类组件的复杂性，使代码更符合函数式编程的思想。
    </p>
