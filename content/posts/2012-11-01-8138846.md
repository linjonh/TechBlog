    <pre><code class="language-java">import java.lang.reflect.ParameterizedType;

public class Test&lt;T&gt; {
	public Class&lt;T&gt; getTClass() {
		return (Class&lt;T&gt;) ((ParameterizedType) getClass()
				.getGenericSuperclass()).getActualTypeArguments()[0];
	}

	public void test() {
		System.out.println(getTClass());
	}

	public static void main(String s[]) {
		(new Test&lt;String&gt;() {
		}).test();
	}

}
