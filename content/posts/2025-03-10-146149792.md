    <h3>
     <a id="_PostgreSQL_16_0">
     </a>
     安装 PostgreSQL 16
    </h3>
    <ol>
     <li>
      <code>
       apt install postgresql-16
      </code>
     </li>
     <li>
      <code>
       passwd postgres
      </code>
      ，修改 postgres 用户密码
     </li>
     <li>
      <code>
       su postgres
      </code>
     </li>
     <li>
      <code>
       psql -U postgres
      </code>
      , 以 postgres 的身份登录数据库
     </li>
     <li>
      <code>
       alter user postgres with password 'abc123';
      </code>
     </li>
     <li>
      \q 退出
     </li>
     <li>
      /etc/postgresql/16/main/postgresql.conf 可修改
      <code>
       #listen_addresses = '*'
      </code>
      ，可修改端口 25432
     </li>
     <li>
      /etc/postgresql/16/main/pg_hba.conf 修改连接
      <pre><code># IPv4 local connections:
host    all             all             127.0.0.1/32            trust
host    all             all             0.0.0.0/0               scram-sha-256
</code></pre>
     </li>
     <li>
      <code>
       systemctl restart postgresql
      </code>
      重启
     </li>
     <li>
      <code>
       su root
      </code>
      ,
      <code>
       lsof -i:25432
      </code>
      可以查看是否运行，
      <code>
       systemctl disable postgresql
      </code>
      可禁止自启动
     </li>
     <li>
      <code>
       apt install ufw
      </code>
      ,
      <code>
       ufw enable
      </code>
      ,
      <code>
       ufw allow 25432/tcp
      </code>
     </li>
    </ol>
    <h3>
     <a id="_PostGis_3_18">
     </a>
     安装 PostGis 3
    </h3>
    <ol>
     <li>
      <code>
       apt install postgresql-16-postgis-3
      </code>
     </li>
     <li>
      <code>
       su postgres
      </code>
     </li>
     <li>
      <code>
       psql -U postgres
      </code>
     </li>
     <li>
      <code>
       create database "test_data";
      </code>
     </li>
     <li>
      \c test_data 连接
     </li>
     <li>
      <code>
       create extension postgis;
      </code>
     </li>
     <li>
      <code>
       select PostGIS_version();
      </code>
      验证安装
     </li>
     <li>
      \c postgres
     </li>
     <li>
      <code>
       select * from pg_database;
      </code>
     </li>
     <li>
      <code>
       drop database "test_data";
      </code>
     </li>
    </ol>
