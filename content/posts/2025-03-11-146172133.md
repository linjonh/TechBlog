/**
 * Snowflake 雪花算法生成器
 * 生成的 64 位 ID 结构：
 *  1 位    保留位（始终为0，防止负数）
 *  41 位   时间戳（毫秒级，当前时间减去自定义纪元）
 *  5 位    数据中心ID
 *  5 位    机器ID
 *  12 位   序列号（同一毫秒内生成多个 ID）
 */
class Snowflake
{
    // 自定义纪元（起始时间戳），可根据实际情况调整
    private $twepoch = 1288834974657;

    // 各部分占用的位数
    private $workerIdBits     = 5;   // 机器ID位数
    private $datacenterIdBits = 5;   // 数据中心ID位数
    private $sequenceBits     = 12;  // 序列号位数

    // 各部分的最大数值（通过位运算计算）
    private $maxWorkerId;
    private $maxDatacenterId;
    private $sequenceMask;

    // 各部分左移位数
    private $workerIdShift;
    private $datacenterIdShift;
    private $timestampLeftShift;

    // 工作机器ID和数据中心ID（取值范围 0 ~ 最大值）
    private $workerId;
    private $datacenterId;

    // 序列号，记录同一毫秒内生成的ID序号
    private $sequence = 0;

    // 上一次生成ID的时间戳，初始化为 -1
    private $lastTimestamp = -1;

    /**
     * 构造函数
     *
     * @param int $workerId      机器ID（0 ~ 31）
     * @param int $datacenterId  数据中心ID（0 ~ 31）
     */
    public function __construct($workerId, $datacenterId)
    {
        // 计算各部分最大值：如 -1 &lt;&lt; 5 得到 0b11111 (31)
        $this-&gt;maxWorkerId     = -1 ^ (-1 &lt;&lt; $this-&gt;workerIdBits);
        $this-&gt;maxDatacenterId = -1 ^ (-1 &lt;&lt; $this-&gt;datacenterIdBits);
        // 序列号最大值（4095：12位全1）
        $this-&gt;sequenceMask    = -1 ^ (-1 &lt;&lt; $this-&gt;sequenceBits);

        // 检查传入的机器ID和数据中心ID是否合法
        if ($workerId &gt; $this-&gt;maxWorkerId || $workerId &lt; 0) {
            throw new Exception("机器ID必须在 0 到 " . $this-&gt;maxWorkerId . " 之间");
        }
        if ($datacenterId &gt; $this-&gt;maxDatacenterId || $datacenterId &lt; 0) {
            throw new Exception("数据中心ID必须在 0 到 " . $this-&gt;maxDatacenterId . " 之间");
        }
        $this-&gt;workerId     = $workerId;
        $this-&gt;datacenterId = $datacenterId;

        // 计算各部分需要左移的位数
        // 序列号占用 12 位，因此机器ID需要左移 12 位
        $this-&gt;workerIdShift     = $this-&gt;sequenceBits;
        // 数据中心ID需要左移 12 + 5 = 17 位
        $this-&gt;datacenterIdShift = $this-&gt;sequenceBits + $this-&gt;workerIdBits;
        // 时间戳需要左移 12 + 5 + 5 = 22 位
        $this-&gt;timestampLeftShift = $this-&gt;sequenceBits + $this-&gt;workerIdBits + $this-&gt;datacenterIdBits;
    }

    /**
     * 生成下一个唯一ID
     *
     * @return int 64位的唯一ID
     */
    public function nextId()
    {
        // 获取当前毫秒级时间戳
        $timestamp = $this-&gt;timeGen();

        // 如果当前时间小于上一次生成ID的时间戳，则说明系统时钟回拨，无法生成ID
        if ($timestamp &lt; $this-&gt;lastTimestamp) {
            throw new Exception("时钟回拨异常，拒绝生成ID. 差值: " . ($this-&gt;lastTimestamp - $timestamp) . " 毫秒");
        }

        // 如果当前时间与上次生成ID时间相同，表示同一毫秒内生成多个ID
        if ($timestamp == $this-&gt;lastTimestamp) {
            // 序列号自增，并通过位运算确保在 12 位内（最大值为 4095）
            $this-&gt;sequence = ($this-&gt;sequence + 1) &amp; $this-&gt;sequenceMask;
            // 如果序列号达到最大值，则等待下一毫秒
            if ($this-&gt;sequence == 0) {
                $timestamp = $this-&gt;tilNextMillis($this-&gt;lastTimestamp);
            }
        } else {
            // 不同毫秒内，序列号重置为 0
            $this-&gt;sequence = 0;
        }

        // 更新上一次生成ID的时间戳
        $this-&gt;lastTimestamp = $timestamp;

        /**
         * 拼接各部分生成最终ID：
         * 1. (当前时间戳 - 自定义纪元) 左移 TIMESTAMP_LEFT_SHIFT 位
         * 2. 数据中心ID左移 DATACENTER_ID_SHIFT 位
         * 3. 机器ID左移 WORKER_ID_SHIFT 位
         * 4. 序列号（低12位）
         * 最终通过按位或（|）运算组合成一个 64 位的整数
         */
        $id = (($timestamp - $this-&gt;twepoch) &lt;&lt; $this-&gt;timestampLeftShift)
            | ($this-&gt;datacenterId &lt;&lt; $this-&gt;datacenterIdShift)
            | ($this-&gt;workerId &lt;&lt; $this-&gt;workerIdShift)
            | $this-&gt;sequence;

        return $id;
    }

    /**
     * 等待直到下一毫秒
     *
     * @param int $lastTimestamp 上次生成ID的时间戳
     * @return int 下一毫秒的时间戳
     */
    private function tilNextMillis($lastTimestamp)
    {
        $timestamp = $this-&gt;timeGen();
        // 循环等待直到时间戳发生变化（进入下一毫秒）
        while ($timestamp &lt;= $lastTimestamp) {
            $timestamp = $this-&gt;timeGen();
        }
        return $timestamp;
    }

    /**
     * 获取当前时间戳（毫秒级）
     *
     * @return int 当前毫秒级时间戳
     */
    private function timeGen()
    {
        // microtime(true) 返回秒为单位的小数，将其乘以 1000 转换为毫秒
        return (int) (microtime(true) * 1000);
    }
}

/**
 * 雪花算法分布式ID生成器
 * 结构：0 | 时间戳(41位) | 数据中心ID(5位) | 机器ID(5位) | 序列号(12位)
 */
class Snowflake
{
    // 起始时间戳（2024-01-01 00:00:00 UTC）
    const EPOCH = 1704067200000;
    
    // 位数分配
    const SEQUENCE_BITS = 12;   // 序列号位数
    const MACHINE_BITS = 5;     // 机器标识位数
    const DATACENTER_BITS = 5;  // 数据中心位数

    // 最大值计算（位运算优化）
    const MAX_SEQUENCE = -1 ^ (-1 &lt;&lt; self::SEQUENCE_BITS);
    const MAX_MACHINE = -1 ^ (-1 &lt;&lt; self::MACHINE_BITS);
    const MAX_DATACENTER = -1 ^ (-1 &lt;&lt; self::DATACENTER_BITS);

    // 移位偏移量计算
    const TIMESTAMP_SHIFT = self::SEQUENCE_BITS + self::MACHINE_BITS + self::DATACENTER_BITS;
    const DATACENTER_SHIFT = self::SEQUENCE_BITS + self::MACHINE_BITS;
    const MACHINE_SHIFT = self::SEQUENCE_BITS;

    private $machineId;      // 机器标识ID
    private $datacenterId;   // 数据中心ID
    private $sequence = 0;   // 序列号
    private $lastTimestamp = -1; // 上次生成时间戳

    /**
     * 构造函数
     * @param int $machineId    机器ID (0~31)
     * @param int $datacenterId 数据中心ID (0~31)
     */
    public function __construct($machineId, $datacenterId)
    {
        // 参数校验
        if ($machineId &gt; self::MAX_MACHINE || $machineId &lt; 0) {
            throw new InvalidArgumentException("机器ID值超出范围（0～".self::MAX_MACHINE."）");
        }
        if ($datacenterId &gt; self::MAX_DATACENTER || $datacenterId &lt; 0) {
            throw new InvalidArgumentException("数据中心ID值超出范围（0～".self::MAX_DATACENTER."）");
        }
        
        $this-&gt;machineId = $machineId;
        $this-&gt;datacenterId = $datacenterId;
    }

    /**
     * 生成下一个分布式ID
     * @return int 64位整型ID（PHP中实际返回字符串类型以保证精度）
     */
    public function generateId()
    {
        $timestamp = $this-&gt;getCurrentTimestamp();

        // 时钟回拨检查
        if ($timestamp &lt; $this-&gt;lastTimestamp) {
            $diff = $this-&gt;lastTimestamp - $timestamp;
            throw new RuntimeException("系统时钟回拨 {$diff} 毫秒");
        }

        // 同一毫秒内生成
        if ($this-&gt;lastTimestamp == $timestamp) {
            $this-&gt;sequence = ($this-&gt;sequence + 1) &amp; self::MAX_SEQUENCE;
            
            // 序列号耗尽，等待下一毫秒
            if ($this-&gt;sequence == 0) {
                $timestamp = $this-&gt;waitNextMillis($this-&gt;lastTimestamp);
            }
        } else {
            // 新时间戳重置序列号
            $this-&gt;sequence = 0;
        }

        $this-&gt;lastTimestamp = $timestamp;

        // 组合各部分数据
        $id = (($timestamp) &lt;&lt; self::TIMESTAMP_SHIFT)
            | ($this-&gt;datacenterId &lt;&lt; self::DATACENTER_SHIFT)
            | ($this-&gt;machineId &lt;&lt; self::MACHINE_SHIFT)
            | $this-&gt;sequence;

        // 返回字符串类型以避免32位系统精度丢失
        return (string)$id;
    }

    /**
     * 阻塞到下一毫秒
     * @param int $lastTimestamp 上次生成时间戳
     * @return int 当前时间戳
     */
    private function waitNextMillis($lastTimestamp)
    {
        $timestamp = $this-&gt;getCurrentTimestamp();
        while ($timestamp &lt;= $lastTimestamp) {
            usleep(1000); // 休眠1毫秒
            $timestamp = $this-&gt;getCurrentTimestamp();
        }
        return $timestamp;
    }

    /**
     * 获取当前时间戳（毫秒）
     * @return int 毫秒时间戳
     */
    private function getCurrentTimestamp()
    {
        return round(microtime(true) * 1000) - self::EPOCH;
    }
}

    const DATA_CENTER_ID_SHIFT = self::MACHINE_ID_SHIFT + self::MACHINE_ID_BITS; // 数据中心ID的偏移量

    private function timeGen() {

/**
 * 雪花算法（Snowflake）PHP实现
 * 
 * 雪花算法是由Twitter开发的分布式ID生成算法，能够在分布式系统中产生唯一的ID
 * 该算法生成的ID结构为64位长整型，由以下部分组成：
 * 1. 1位符号位，固定为0（表示正数）

        while ($currentTimestamp &lt;= $lastTimestamp) {
            $currentTimestamp = $this-&gt;timestamp();
        }
