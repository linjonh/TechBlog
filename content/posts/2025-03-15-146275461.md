     一、组件通信基础（通俗版）
    </h3>
    <h4 id="1.%20%E4%BB%80%E4%B9%88%E6%98%AF%E7%BB%84%E4%BB%B6%E9%80%9A%E4%BF%A1%EF%BC%9F" name="1.%20%E4%BB%80%E4%B9%88%E6%98%AF%E7%BB%84%E4%BB%B6%E9%80%9A%E4%BF%A1%EF%BC%9F">
     1. 什么是组件通信？
    </h4>
    <p>
     就像人与人之间需要说话交流，组件之间也需要「沟通」来传递数据。当父组件（上级）需要给子组件（下级）传递信息时，最常用的方法就是
     <strong>
      Props
     </strong>
     传值。
    </p>
    <h4 id="2.%20%E7%BB%8F%E5%85%B8%E6%AF%94%E5%96%BB%E7%90%86%E8%A7%A3" name="2.%20%E7%BB%8F%E5%85%B8%E6%AF%94%E5%96%BB%E7%90%86%E8%A7%A3" style="background-color:transparent">
     2. 经典比喻理解
    </h4>
    <p>
     想象组件是一个智能水杯：
    </p>
    <ul>
     <li>
      <strong>
       父组件
      </strong>
      = 手机APP（控制中心）
     </li>
     <li>
      <strong>
       子组件
      </strong>
      = 水杯本体
     </li>
     <li>
      <strong>
       Props
      </strong>
      = 蓝牙传输的温度设置指令
     </li>
    </ul>
    <hr/>
    <h3 id="%E4%BA%8C%E3%80%81Props%20%E6%A0%B8%E5%BF%83%E6%9C%BA%E5%88%B6%E8%AF%A6%E8%A7%A3" name="%E4%BA%8C%E3%80%81Props%20%E6%A0%B8%E5%BF%83%E6%9C%BA%E5%88%B6%E8%AF%A6%E8%A7%A3">
     二、Props 核心机制详解
    </h3>
    <h4 id="1.%20%E5%8D%95%E5%90%91%E6%95%B0%E6%8D%AE%E6%B5%81%E5%8E%9F%E5%88%99" name="1.%20%E5%8D%95%E5%90%91%E6%95%B0%E6%8D%AE%E6%B5%81%E5%8E%9F%E5%88%99">
     1. 单向数据流原则
    </h4>
    <p>
     数据只能从父组件
     <strong>
      向下流动
     </strong>
     ，子组件不能直接修改接收到的数据。就像公司里领导给下属布置任务，下属不能直接修改任务书，但可以反馈建议。
    </p>
    <h5 id="%E9%94%99%E8%AF%AF%E7%A4%BA%E8%8C%83%EF%BC%9A" name="%E9%94%99%E8%AF%AF%E7%A4%BA%E8%8C%83%EF%BC%9A">
     错误示范：
    </h5>
    <p>
     JAVASCRIPT
    </p>
    <blockquote>
     <pre><code class="hljs">// 子组件中直接修改 props（相当于篡改任务书）
function Child({ count }) {
  const handleClick = () =&gt; {
    count++  // ❌ 大忌！会引发警告
  }
}
</code></pre>
    </blockquote>
    <h4 id="2.%20%E7%B1%BB%E5%9E%8B%E9%AA%8C%E8%AF%81%E6%9C%BA%E5%88%B6" name="2.%20%E7%B1%BB%E5%9E%8B%E9%AA%8C%E8%AF%81%E6%9C%BA%E5%88%B6">
     2. 类型验证机制
    </h4>

    <h4 id="%E6%A1%88%E4%BE%8B%E5%9C%BA%E6%99%AF%EF%BC%9A%E6%B8%A9%E5%BA%A6%E6%8E%A7%E5%88%B6%E5%99%A8" name="%E6%A1%88%E4%BE%8B%E5%9C%BA%E6%99%AF%EF%BC%9A%E6%B8%A9%E5%BA%A6%E6%8E%A7%E5%88%B6%E5%99%A8">
     案例场景：温度控制器

    <ul>
     <li>

    <h4 id="1.%20Vue%203%20%E5%AE%9E%E7%8E%B0" name="1.%20Vue%203%20%E5%AE%9E%E7%8E%B0">
     1. Vue 3 实现
    </h4>
    <p>
     VUE
    </p>
    <blockquote>
     <pre><code class="hljs">&lt;!-- 父组件 Parent.vue --&gt;
&lt;template&gt;
  &lt;div class="parent"&gt;
    &lt;h2&gt;当前温度：{<!-- -->{ temp }}℃&lt;/h2&gt;
    &lt;!-- 传递温度数据和调节方法 --&gt;
    &lt;Thermometer 
      :temperature="temp" 
      @adjust="handleAdjust"
    /&gt;
  &lt;/div&gt;
&lt;/template&gt;

&lt;script setup&gt;
import { ref } from 'vue'
import Thermometer from './Thermometer.vue'

const temp = ref(25)  // 响应式温度数据

// 处理温度变化事件
const handleAdjust = (newTemp) =&gt; {
  temp.value = newTemp
}
&lt;/script&gt;
</code></pre>
    </blockquote>
    <p>
     VUE
    </p>
    <blockquote>
     <pre><code class="hljs">&lt;!-- 子组件 Thermometer.vue --&gt;
&lt;template&gt;
  &lt;div 
    class="thermometer" 
    :style="{ backgroundColor: tempColor }"
  &gt;
    &lt;div class="mercury" :style="mercuryStyle"&gt;&lt;/div&gt;
    &lt;button @click="increment"&gt;➕&lt;/button&gt;
    &lt;button @click="decrement"&gt;➖&lt;/button&gt;
  &lt;/div&gt;
&lt;/template&gt;

&lt;script setup&gt;
// 定义接收的 Props
const props = defineProps({
  temperature: {
    type: Number,
    required: true,
    validator: value =&gt; value &gt;= -20 &amp;&amp; value &lt;= 50
  }
})

// 根据温度计算颜色
const tempColor = computed(() =&gt; {
  return props.temperature &gt; 30 ? 'red' : 'blue'
})

// 温度变化事件
const emit = defineEmits(['adjust'])

const increment = () =&gt; {
  emit('adjust', props.temperature + 1)
}

const decrement = () =&gt; {
  emit('adjust', props.temperature - 1)
}
&lt;/script&gt;
</code></pre>
    </blockquote>
    <h4 id="2.%20React%20%E5%AE%9E%E7%8E%B0" name="2.%20React%20%E5%AE%9E%E7%8E%B0">

import { useState } from 'react'
import Thermometer from './Thermometer'

function Parent() {

  const handleAdjust = (newTemp) =&gt; {

  }

  return (
    &lt;div className="parent"&gt;
      &lt;h2&gt;当前温度：{temp}℃&lt;/h2&gt;
      &lt;Thermometer 
        temperature={temp} 
        onAdjust={handleAdjust}
      /&gt;
    &lt;/div&gt;
  )
}
</code></pre>
    </blockquote>
    <p>
     JSX
    </p>
    <blockquote>
     <pre><code class="hljs">// 子组件 Thermometer.jsx
import PropTypes from 'prop-types'

function Thermometer({ temperature, onAdjust }) {
  // 计算温度颜色
  const tempColor = temperature &gt; 30 ? 'red' : 'blue'

  // 温度调节方法
  const handleIncrement = () =&gt; onAdjust(temperature + 1)
  const handleDecrement = () =&gt; onAdjust(temperature - 1)

  return (
    &lt;div className="thermometer" style={<!-- -->{ backgroundColor: tempColor }}&gt;
      &lt;div 
        className="mercury"
        style={<!-- -->{ height: `${(temperature + 20) * 2}%` }}
      &gt;&lt;/div&gt;
      &lt;button onClick={handleIncrement}&gt;➕&lt;/button&gt;
      &lt;button onClick={handleDecrement}&gt;➖&lt;/button&gt;
    &lt;/div&gt;
  )
}

// 类型验证
Thermometer.propTypes = {
  temperature: PropTypes.number.isRequired,
  onAdjust: PropTypes.func.isRequired
}
</code></pre>

     <pre><code class="hljs">// 子组件 thermometer.js
Component({
  properties: {
    temperature: {
      type: Number,
      value: 25,
      observer(newVal) {
        this.updateColor(newVal)
      }
    }
  },

  methods: {
    increment() {
      this.triggerEvent('adjust', { temp: this.data.temperature + 1 })
    },
    
    decrement() {
      this.triggerEvent('adjust', { temp: this.data.temperature - 1 })
    },
    
    updateColor(temp) {
      this.setData({
        tempColor: temp &gt; 30 ? 'red' : 'blue',
        mercuryHeight: (temp + 20) * 2
      })
    }
  }
})
</code></pre>

    <h4 id="Q1%3A%20%E4%B8%BA%E4%BB%80%E4%B9%88%E6%88%91%E7%9A%84%E5%AD%90%E7%BB%84%E4%BB%B6%E4%B8%8D%E6%9B%B4%E6%96%B0%EF%BC%9F" name="Q1%3A%20%E4%B8%BA%E4%BB%80%E4%B9%88%E6%88%91%E7%9A%84%E5%AD%90%E7%BB%84%E4%BB%B6%E4%B8%8D%E6%9B%B4%E6%96%B0%EF%BC%9F">

    <p>

     </strong>

    </p>
    <ol>

// 子组件触发事件修改

</code></pre>
