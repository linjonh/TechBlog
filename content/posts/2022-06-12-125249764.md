```javascript
onReachBottom: function () {
    var that = this
    var length = this.__data__.list.length
    var arr = this.__data__.list
    wx.request({
      // 请求地址
      url: '',//填写自己的请求地址
      //传页码和每页显示的数量
      data: {
        length: length
      },
      success: function (res) {
        //判断是否有数据，如果没有数据给吃提示
        if (res.data.data != '') {
          wx.showToast({
            title: '正在加载中',
            icon: 'loading',
            duration: 2000
          })
          //把值发送到前端进行渲染
          that.setData({
            list: arr.concat(res.data.data)
          })
        } else {
          wx.showToast({
            title: '没有更多数据！',
            icon: 'error',
            duration: 2000
          })
        }
      }
    })
  },
```
