const express = require('express');

// 加载MySQL模块
const mysql = require('mysql');

// 加载bodyParser模块
const bodyParser = require('body-parser');

// 加载MD5模块
const md5 = require('md5');

// 创建MySQL连接池
const pool = mysql.createPool({
  host: '127.0.0.1', //MySQL服务器地址
  port: 3306, //MySQL服务器端口号
  user: 'root', //数据库用户的用户名
  password: '', //数据库用户密码
  database: 'reg_log', //数据库名称
  connectionLimit: 20, //最大连接数
  charset: 'utf8' //数据库服务器的编码方式
});

// 创建服务器对象
const server = express();

server.use(bodyParser.urlencoded({
  extended: false
}));


// 加载CORS模块
const cors = require('cors');

// 使用CORS中间件
server.use(cors({
  origin: ['http://localhost:8080', 'http://127.0.0.1:8080']
}));

//用户注册接口
server.post('/register', (req, res) =&gt; {
  //console.log(md5('12345678'));
  // 获取用户名和密码信息
  let username = req.body.username;
  let password = req.body.password;
  //以username为条件进行查找操作，以保证用户名的唯一性
  let sql = 'SELECT COUNT(id) AS count FROM reg_log WHERE username=?';
  pool.query(sql, [username], (error, results) =&gt; {
    if (error) throw error;
    let count = results[0].count;
    if (count == 0) {
      // 将用户的相关信息插入到数据表
      sql = 'INSERT reg_log(username,password) VALUES(?,MD5(?))';
      pool.query(sql, [username, password], (error, results) =&gt; {
        if (error) throw error;
        res.send({
          message: 'ok',
          code: 200
        });
      })
    } else {
      res.send({
        message: 'user exists',
        code: 201
      });
    }
  });
});

// 用户登录接口
server.post('/login', (req, res) =&gt; {
  //获取用户名和密码信息
  let username = req.body.username;
  let password = req.body.password;
  // SQL语句
  let sql = 'SELECT id,username FROM reg_log WHERE username=? AND password=MD5(?)';
  pool.query(sql, [username, password], (error, results) =&gt; {
    if (error) throw error;
    if (results.length == 0) { //登录失败
      res.send({
        message: 'login failed',
        code: 201
      });
    } else { //登录成功
      res.send({
        message: 'ok',
        code: 200,
        result: results[0]
      });
    }
  });

});

// 指定服务器对象监听的端口号
server.listen(3000, () =&gt; {
  console.log('server is running...');
});</code></pre>
    <h4 style="background-color:transparent;">
     1.代码释义
    </h4>
    <p id="">
     这段代码是一个使用 Node.js 和 Express 框架构建的简单 Web 服务器，它集成了 MySQL 数据库进行用户注册和登录的功能。下面是对代码主要部分的详细解释：
    </p>
    <ol>
     <li>
      <strong>
       引入必要的模块
      </strong>
      ：
      <ul>
       <li>
        <code>
         express
        </code>
        ：用于创建Web服务器。
       </li>
       <li>
        <code>
         mysql
        </code>
        ：用于与 MySQL 数据库进行交互。
       </li>
       <li>
        <code>
         bodyParser
        </code>
        ：用于解析传入的请求体，特别是 POST 和 PUT 请求。
       </li>
       <li>
        <code>
         md5
        </code>
        ：用于对密码进行 MD5 加密，虽然 MD5 在现代安全标准下被认为是不安全的，但这里仅作为示例。
       </li>
       <li>
        <code>
         cors
        </code>
        ：用于处理跨源资源共享（CORS），允许来自不同源的 Web 页面请求服务器资源。
       </li>
      </ul>
     </li>
     <li>
      <strong>
       创建MySQL连接池
      </strong>
      ：
      <ul>
       <li>
        使用
        <code>
         mysql.createPool
        </code>
        方法创建一个连接池，这样可以更有效地管理数据库连接。连接池配置包括数据库服务器的地址、端口、用户名、密码、数据库名、最大连接数以及字符集等。
       </li>
      </ul>
     </li>
     <li>
      <strong>
       创建Express服务器
      </strong>
      ：
      <ul>
       <li>
        使用
        <code>
         express()
        </code>
        函数创建一个 Express 应用实例。
       </li>
      </ul>
     </li>
     <li>
      <strong>
       配置中间件
      </strong>
      ：
      <ul>
       <li>
        使用
        <code>
         bodyParser.urlencoded
        </code>
        中间件来解析表单数据。
       </li>
       <li>
        使用
        <code>
         cors
        </code>
        中间件来允许来自特定源的跨域请求。
       </li>
      </ul>
     </li>
     <li>
      <strong>
       定义路由和处理器
      </strong>
      ：
      <ul>
       <li>
        <strong>
         注册接口
        </strong>
        (
        <code>
         /register
        </code>
        )：
        <ul>
         <li>
          接收POST请求，从请求体中获取用户名和密码。
         </li>
         <li>
          首先检查用户名是否已存在（通过查询数据库）。
         </li>
         <li>
          如果用户名不存在，则将用户名和密码（密码经过MD5加密）插入数据库，并返回成功消息。
         </li>
         <li>
          如果用户名已存在，则返回用户已存在的消息。
         </li>
        </ul>
       </li>
       <li>
        <strong>
         登录接口
        </strong>
        (
        <code>
         /login
        </code>
        )：
        <ul>
         <li>
          同样接收POST请求，从请求体中获取用户名和密码。
         </li>
         <li>
          使用SQL查询检查用户名和密码（密码已加密）是否匹配。
         </li>
         <li>
          如果匹配，返回成功消息和用户信息；如果不匹配，返回登录失败的消息。
         </li>
        </ul>
       </li>
      </ul>
     </li>
     <li>
      <strong>
       启动服务器
      </strong>
      ：
      <ul>
       <li>
        使用
        <code>
         server.listen
        </code>
        方法启动服务器，并监听 3000 端口。当服务器成功启动时，控制台会打印 “server is running...”。
       </li>
      </ul>
     </li>
    </ol>

&lt;!-- 注册页面 --&gt;
  &lt;div class="reg"&gt;
    &lt;mt-header title="用户注册"&gt;
      &lt;router-link slot="left" to="/"&gt; 返回首页 &lt;/router-link&gt;

    &lt;/mt-header&gt;
    &lt;mt-field
      type="text"
      label="用户名"
      placeholder="请输入用户名"
      v-model="name"
      :state="nameState"
      @blur.native.capture="checkName"&gt;&lt;/mt-field&gt;
      &lt;!-- @blur失焦，输入框在输入完成、移到其他地方时进行验证 --&gt;
    &lt;mt-field
      type="password"
      label="密码"
      placeholder="请输入密码"
      v-model="pwd"
      :state="pwdState"
      @blur.native.capture="checkPwd"&gt;&lt;/mt-field&gt;
    &lt;mt-field

      label="确认密码"
      placeholder="再次输入密码"
      v-model="repwd"
      :state="repwdState"
      @blur.native.capture="checkrePwd"&gt;&lt;/mt-field&gt;
    &lt;mt-field
      label="邮箱"

      type="email"
      v-model="email"
      :state="emailState"
      @blur.native.capture="checkEmail"&gt;&lt;/mt-field&gt;
    &lt;mt-field
      label="手机号"
      placeholder="请输入手机号"
      type="tel"
      v-model="phone"
      :state="phoneState"
      @blur.native.capture="checkPhone"&gt;&lt;/mt-field&gt;
    &lt;mt-field
      label="生日"
      placeholder="请输入生日"
      type="date"
      v-model="birthday"&gt;&lt;/mt-field&gt;
    &lt;mt-button class="btn" type="primary" size="large" @click="checkForm"&gt;注册&lt;/mt-button&gt;
  &lt;/div&gt;

&lt;style scope&gt;
.reg .btn {
  background-color: #d86931;

&lt;script&gt;
export default {
  data() {
    return {
      name: "", //双向数据绑定 用户名输入框
      nameState: "", //处理用户名输入框的动态更新
      pwd: "",
      pwdState: "",
      repwd: "",
      repwdState: "",
      email: "",
      emailState: "",
      phone: "",
      phoneState: "",
      birthday: "",
    };
  },
  methods: {
    // 封装各个字段的验证方法：
    // 1.验证用户名
    checkName() {
      //事件处理函数：处理点击按钮验证表单
      let reg = /^[a-zA-Z0-9\u4E00-\u9FA5]{2,6}$/;
      if (reg.test(this.name)) {
        this.nameState = "success";
        return true;
      } else {
        this.nameState = "error";
        return false;
      }
    },
    // 2.验证密码
    checkPwd() {
      let reg = /^\d{6}$/;
      if (reg.test(this.pwd)) {
        this.pwdState = "success";
        return true;
      } else {
        this.pwdState = "error";
        return false;
      }
    },
    // 3.重复输入密码
    checkrePwd() {
      let reg = /^\d{6}$/;
      if (reg.test(this.repwd) &amp;&amp; this.pwd == this.repwd) {
        this.repwdState = "success";
        return true;
      } else {
        this.repwdState = "error";
        return false;
      }
    },
    // 4.验证邮箱格式
    checkEmail() {
      let reg = /^([a-zA-Z]|[0-9])(\w|\-)+@[a-zA-Z0-9]+\.([a-zA-Z]{2,4})$/;
      if (reg.test(this.email)) {
        this.emailState = "success";
        return true;
      } else {
        this.emailState = "error";
        return false;
      }
    },
    // 5.验证手机号
    checkPhone() {
      let reg = /^1[3|5|7|8][0-9]\d{8}$/;
      if (reg.test(this.phone)) {
        this.phoneState = "success";
        return true;
      } else {
        this.phoneState = "error";
        return false;
      }
    },

    checkForm() {
      // 点击注册按钮后调用此方法，验证用户名、密码、二次密码是否均正确，正确则发送axios请求
      if (this.checkName() &amp;&amp; this.checkPwd() &amp;&amp; this.checkrePwd()) {
        console.log(`验证成功，执行注册业务......`);
        // 发送注册(post)请求
        this.axios
          .post("/register", `username=${this.name}&amp;password=${this.pwd}`)
          .then((result) =&gt; {
            console.log(result);
            if (result.data.code == 200) {
              // 弹窗提示注册成功
              this.$toast({
                message: "注册成功",
                position: "bottom",
                duration: 3000,
              });
              // 注册成功后直接跳转到登录页
              this.$router.push("/login");
            } else if (result.data.code == 201) {
              this.$toast({
                message: "用户已存在，请重新注册",
                position: "bottom",
                duration: 3000,
              });
            }
          });
      }
    },
  },
};

&lt;!-- 登录页面 --&gt;

      let reg = /^[a-zA-Z0-9\u4E00-\u9FA5]{2,6}$/; //2-6位的数字、字母、汉字

        this.nameState = "success";
        return true;
      } else {
        this.nameState = "error";
        return false;
      }
    },

      if (reg.test(this.pwd)) {
        this.pwdState = "success";
        return true;
      } else {
        this.pwdState = "error";
        return false;

    },

        // 发送登录(post)请求
        this.axios
          .post("/login", `username=${this.name}&amp;password=${this.pwd}`)
          .then((result) =&gt; {
            console.log(result);
            if (result.data.code == 200) {
              // 弹窗提示登录成功

              });
              // 修改vuex中用户登录状态
              this.$store.commit("loginok", this.name);
              // 将islogin与username存入sessionStorage
              sessionStorage.setItem("islogin", true);
              sessionStorage.setItem("username", this.name);
              // 登录成功后直接跳转到首页
              this.$router.push("/club");
            } else {

                duration: 3000,
              });
            }
          });
      }
    },
  },
};
