    <h3>
     应用场景
    </h3>
    <p>
     假设我们有一个
     <a href="https://so.csdn.net/so/search?q=%E4%B8%8B%E6%8B%89%E6%A1%86&amp;spm=1001.2101.3001.7020" title="下拉框">
      下拉框
     </a>
     组件，当下拉框展开的时候，点击下拉框之外的元素可以自动关闭下拉框。
    </p>
    <h2>
     <a name="t1">
     </a>
     <a id="ClickOutside_2">
     </a>
     一 ClickOutside代码示例
    </h2>
    <p>
     在
     <a href="https://so.csdn.net/so/search?q=vue3&amp;spm=1001.2101.3001.7020" title="vue3">
      vue3
     </a>
     中使用ClickOutside
    </p>
    <pre><code class="language-javascript">// 导入自定义指令
import { ClickOutside as vClickOutside } from 'element-plus';

// 绑定指令触发对应事件
&lt;div id="d1" v-click-outside="closeOrder"&gt;
    暂无数据
&lt;/div&gt;

// 写相关逻辑事件
const closeOrder = () =&gt; {
    console.log('点击了d1区域之外的位置')
}
</code></pre>
    <p>
     id="d1"记得写，否则不生效
    </p>
    <p>
     <br/>
     <a href="https://so.csdn.net/so/search?q=vue2&amp;spm=1001.2101.3001.7020" title="vue2">
      <br/>
      vue2
     </a>
     中使用ClickOutside
    </p>
    <pre><code class="language-javascript">// 导入指令
import { ClickOutside } from 'element-plus';

// 映射该指令
directives:{ ClickOutside }

// 绑定指令触发对应事件
&lt;div id="d1" v-clickOutside="closeOrder"&gt;
    暂无数据
&lt;/div&gt;

// 写相关逻辑事件
closeOrder(){
    console.log('点击了d1区域之外的位置')
}
</code></pre>
    <p>
    </p>

    </p>

const isChooseItem = ref("")
