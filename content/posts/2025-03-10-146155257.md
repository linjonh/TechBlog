     本机macos

    <h3>
     <a id="lima_3">
     </a>
     安装lima
    </h3>
    <pre><code class="prism language-powershell">brew install lima
</code></pre>
    <h3>
     <a id="_8">
     </a>
     创建配置
    </h3>
    <pre><code class="prism language-powershell"><span class="token function">echo</span> <span class="token string">"\\ndynamic:\n  big-sur:\n    image: docker://docker:git\n  linux:\n    image: docker.io/limasoftware/ubuntu:20.04 \\n"</span> &gt; ~<span class="token operator">/</span><span class="token punctuation">.</span>lima/default<span class="token punctuation">.</span>yaml
</code></pre>
    <h3>
     <a id="default_13">
     </a>
     启动名叫default的虚拟机
    </h3>
    <pre><code class="prism language-powershell">limactl <span class="token function">start</span> default
</code></pre>
    <p>
     <img alt="在这里插入图片描述" src="https://i-blog.csdnimg.cn/blog_migrate/89749058c2f1553ccaade3a4100db573.png"/>
    </p>
    <h3>
     <a id="lima_19">
     </a>
     进入lima
    </h3>
    <pre><code class="prism language-powershell">lima
</code></pre>
    <p>
     <img alt="在这里插入图片描述" src="https://i-blog.csdnimg.cn/direct/52d5b41859a94e01bed41ffdb335aa4d.png"/>
    </p>
    <h3>
     <a id="docker_25">
     </a>
     安装&amp;使用docker
    </h3>
    <pre><code class="prism language-powershell">sudo snap install docker
sudo <span class="token operator">-</span>i 
docker <span class="token function">ps</span>
</code></pre>
    <p>
     <img alt="在这里插入图片描述" src="https://i-blog.csdnimg.cn/direct/14e76e2858bb4a0593bbd100f2fd5bdb.png"/>
    </p>
    <h3>
     <a id="_33">
     </a>
     更改代理
    </h3>
    <p>
     可能会遇到下载不了的问题
    </p>
    <pre><code class="prism language-powershell">mkdir <span class="token operator">/</span>etc/docker
touch <span class="token operator">/</span>etc/docker/daemon<span class="token punctuation">.</span>json
vim <span class="token operator">/</span>etc/docker/daemon<span class="token punctuation">.</span>json
</code></pre>
    <p>
     写入代理地址
    </p>
    <pre><code class="prism language-powershell"><span class="token punctuation">{<!-- --></span>
  <span class="token string">"registry-mirrors"</span>: <span class="token punctuation">[</span><span class="token string">"https://镜像加速地址"</span><span class="token punctuation">]</span>
<span class="token punctuation">}</span>
</code></pre>
    <p>
     重启docker
    </p>
    <pre><code class="prism language-powershell">sudo systemctl restart docker
</code></pre>
