     问题描述：微信小程序报错VM1075:1 tabBar.list 需至少包含 2 项
    </p>
    <p>
     解决方法：
    </p>
    <p>
     app.json文件夹中的
     <span style="color:#f33b45;">
      tabBar--&gt;list
     </span>
     中需要有两个或两个以上的内容
    </p>
    <p>
     "tabBar": {
     <!-- -->
    </p>
    <p>
     "color": "#999",
    </p>
    <p>
     "backgroundColor": "#fafafa",
    </p>
    <p>
     "selectedColor": "#333",
    </p>
    <p>
     "borderStyle": "white",
    </p>
    <p>
     "
     <span style="color:#f33b45;">
      list
     </span>
     ": [{
     <!-- -->
    </p>
    <p>
     "text": "1个",
    </p>
    <p>
     "pagePath": "你的path"
    </p>
    <p>
     },{
     <!-- -->
    </p>
    <p>
     "text": "2个",
    </p>
    <p>
     "pagePath": "你的path"
    </p>
    <p>
     }],
    </p>
    <p>
     "items": [{
     <!-- -->
    </p>
    <p>
     "name": "1个",
    </p>
    <p>
     "pagePath": "你的path"
    </p>
    <p>
     },{
     <!-- -->
    </p>
    <p>
     "name": "2个",
    </p>
    <p>
     "pagePath": "你的path"
    </p>
    <p>
     }],
    </p>
    <p>
     "position": "bottom"
    </p>
    <p>
     },
    </p>
