     一、动态链表 - list (了解)
    </h2>
    <blockquote>
     <div>
      <span style="color:#fe2c24">
       <strong>
        new 和 delete 是非常耗时的操作
       </strong>
      </span>
     </div>
     <div>
      <span style="color:#0d0016">
       <strong>
        在算法比赛中，一般不会使使用 new 和 delete 去模拟实现一个链表。

      </span>
     </div>

      <span style="color:#0d0016">

    <h3>
    </h3>
    <h3>
     1.2 push_front/push_back
    </h3>
    <blockquote>
     <p>
      1） push_front ： 头插
     </p>
     <p>
      2）push_back：尾插
     </p>
     <p>
      <span style="color:#fe2c24">
       <strong>
        时间复杂度：O（1）
       </strong>
      </span>
     </p>
    </blockquote>
    <pre><code class="hljs">#include &lt;iostream&gt;
#include &lt;cstdio&gt;
#include &lt;list&gt;
using namespace std;

void print(list&lt;int&gt;&amp; i)
{
	for(auto e:i)
	{
		cout &lt;&lt; e &lt;&lt; " " ;
	}
	cout &lt;&lt; endl;
}

int main()
{
	list&lt;int&gt; lt;//创建一个存储int 类型的链表 
	
	//尾插
	for(int i = 1;i&lt;=5 ;i++)
	{
		lt.push_back(i); 
		print(lt);
	 } 
	 //头插
	 for(int i = 1;i &lt;= 5 ; i++)
	 {
	 	lt.push_front(i);
		 print(lt); 
	  } 
	return 0;
}</code></pre>

    <h3>
     2.2 单向链表
    </h3>
    <p class="img-center">
     <img alt="" height="991" src="https://i-blog.csdnimg.cn/direct/2ba2babdae92423bab90ca62247b99b8.png" width="1176"/>
    </p>
    <pre><code class="hljs">#include &lt;iostream&gt;
#include &lt;cstdio&gt;
using namespace std;

const int N = 1e5 + 10;
const int M = 1e6 + 10;

//链表
int h,id,e[N],ne[N];
int mp[M];//mp[i]用来标记i 这个元素存储再什么位置 
int main()
{
	int q; 
	cin &gt;&gt; q;
	
	//初始化
	id++;
	e[id] = 1;
	mp[1] = id; 
	
	while(q--)
	{
		int op,x,y;
		cin &gt;&gt; op &gt;&gt; x;
		int p = mp[x];//x 存的位置 
		
		if(op == 1)//尾部插入 
		{
			cin &gt;&gt; y;
			id++;
			e[id] = y;
			ne[id] = ne[p];
			ne[p] = id;
			
			mp[y] = id;//标记 y 这个位置 
		}
		else if(op == 2)
		{
			cout &lt;&lt; e[ne[p]] &lt;&lt; endl;
		}
		else//删除 x 后面的元素 
		{
			ne[p] = ne[ne[p]];
		}
	}
	return 0;
}</code></pre>

    <h3>
     2.3 队列安排
    </h3>
    <p>
     <a href="https://www.luogu.com.cn/problem/P1160" rel="nofollow" title="P1160 队列安排 - 洛谷">
      P1160 队列安排 - 洛谷
     </a>
    </p>
    <p class="img-center">
     <img alt="" height="1238" src="https://i-blog.csdnimg.cn/direct/71fa1d369e134d03b6d18ef16a714fb4.png" width="1215"/>
    </p>
    <pre><code class="hljs">#include &lt;iostream&gt;
#include &lt;cstdio&gt;
using namespace std;

const int N = 1e5 + 10;
//双向链表 
int pre[N],ne[N],h;

int n,m;
bool st[N];//st[i] 表示 i 这个同学是否出队 
int main()
{
	cin &gt;&gt; n;
	//初始化
	 pre[1] = h;
	 ne[h] = 1;
	 
	 for(int i = 2;i&lt;=n ; i++)
	 {
	 	int k,p;
	 	cin &gt;&gt; k &gt;&gt; p;
	 	if(p == 0)
	 	{
	 		//i 放在 k 的左边
			 ne[i] = k;
			 pre[i] = pre[k];
			 ne[pre[k]]	= i;
			 pre[k] = i;
		}
		else//i 放在 k 的右边 
		{
			pre[i] = k;
			ne[i] = ne[k];
			pre[ne[k]] = i;
			ne[k] = i;
		}
	 }
	 
	 cin &gt;&gt; m;
	 while(m--)
	 {
	 	int x;
	 	cin &gt;&gt; x;
	 	if(st[x]) continue;
	 	
	 	ne[pre[x]] = ne[x];
	 	pre[ne[x]] = pre[x];
	 	st[x] = true;//表示X已经删除 
	 	
	 }
	 for(int i = ne[h];i ; i = ne[i])
	 {
	 	cout &lt;&lt; i &lt;&lt; " ";
	 }
	return 0;
} </code></pre>

      P1996 约瑟夫问题 - 洛谷
     </a>
    </p>
    <p class="img-center">
     <img alt="" height="1027" src="https://i-blog.csdnimg.cn/direct/8bf93d5c1b604710afc759fe17c47638.png" width="1163"/>
    </p>
    <pre><code class="hljs">#include &lt;iostream&gt;
#include &lt;cstdio&gt;
using namespace std;

const int N = 110;
int n,m,ne[N];

int main()
{
	cin &gt;&gt; n &gt;&gt; m;
	//创建循环链表
	for(int i = 1 ; i &lt; n ; i++)
	{
		ne[i] = i + 1;	
	} 
	ne[n] = 1;
	
	//模拟游戏过程
	int t = n;
	for(int i = 1;i&lt;=n;i++)//执行n次出圈操作 
	{
		for(int j = 1; j &lt; m ; j++)
		{
			t = ne[t];
		}
		cout &lt;&lt; ne[t] &lt;&lt; " ";
		ne[t] = ne[ne[t]];
	} 
}</code></pre>
