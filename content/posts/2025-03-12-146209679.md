     9. 冒泡排序法：
    </h4>
    <pre><code class="language-cpp">#include &lt;stdio.h&gt;
#define CRT_SECURE_NO_WARNINGS
int main() {
	int arr[10];
	int t = 0;
	for (int i = 0; i &lt; 10; i++) {
		printf("请输入第%d个数：", i + 1);
		scanf("%d", &amp;arr[i]);
	}
	for (int i = 0; i &lt; 10; i++) {
		/*printf("请输入第%d个数：", i + 1);
		scanf("%d", &amp;arr[i]);*/
		for (int j = 0; j &lt; 10 - 1 - i; j++) {    
			if (arr[j + 1] &gt; arr[j]) {    //从大到小，arr[j]&gt;arr[j+1]为从小到大
				t = arr[j];
				arr[j] = arr[j + 1];
				arr[1+ j] = t;
			}
		}
	}
	for (int i = 0; i &lt; 10; i++) {
		printf("%d\t",arr[i]);
	}
	
}</code></pre>
