     面试题33：二叉搜索树的后续遍历序列
    </h4>
    <h5>
     <a id="_true_false_2">
     </a>
     输入一个整数数组，判断该数组是不是某二叉搜索树的后序遍历结果。如果是则返回
     <code>
      true
     </code>
     ，否则返回
     <code>
      false
     </code>
     。假设输入的数组的任意两个数字都互不相同。
    </h5>
    <pre><code class="prism language-c++">class Solution {
public:
    bool verifyPostorder(vector&lt;int&gt;&amp; postorder) {
        if(postorder.empty())
            return true;
        return devide_and_rule(0,postorder.size()-1,postorder);	// 递归
    }
    bool devide_and_rule(int left,int right,vector&lt;int&gt;&amp; postorder){
        if(left&gt;=right)	// 只有一个节点
            return true;
        int root = postorder[right];	// 根是最后一个节点
        int i=right-1;
        while(i&gt;=0 &amp;&amp; postorder[i]&gt;root)	// 从右向左找左子树
            i--;
        int mid = i;
        while(i&gt;=0 &amp;&amp; postorder[i]&lt;root)	// 判断左子树中是否还存在比根大的值，若存在返回fause
            i--;
        if(i&gt;=0)			// 正常是-1
            return false;
        return(devide_and_rule(left,mid,postorder)&amp;&amp;devide_and_rule(mid+1,right-1,postorder));	// 递归判断左子树和右子树
    }
};


</code></pre>
