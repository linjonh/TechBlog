    <p>
     递归实现求二叉树的层数，求以node为根节点的子树的高度
    </p>
    <pre><code class="language-cpp">int level(Node* node)
    {
        if(node == nullptr)
        {
            return 0;
        }
        int left = level(node-&gt;left_);
        int right = level(node-&gt;right_);
        return left &gt; right ? left + 1 : right + 1;
    }</code></pre>
    <p>
     递归求二叉树节点的个数
    </p>
    <pre><code class="language-cpp">int number(Node *node)
    {
        if(node == nullptr)
        {
            return 0;
        }
        int left = number(node-&gt;left_);
        int right = number(node-&gt;right_);
        return left + right + 1;
    }</code></pre>
    <pre><code class="language-cpp">// 递归层数遍历
    void levelOrder()
    {
        int h = level();
        for(int i = 0; i &lt; h; i++)
        {
            levelOrder(root_, i);
        }
    }
    void levelOrder(Node *node, int i)
    {
        if(node == nullptr)
        {
            return;
        }
        if(i == 0)
        {
            cout &lt;&lt; node-&gt;data_ &lt;&lt; " ";
            return;
        }
        levelOrder(node-&gt;left_, i - 1);
        levelOrder(node-&gt;right_, i - 1);
    }</code></pre>
