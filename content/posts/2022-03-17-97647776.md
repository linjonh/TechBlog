    <p>
     程序中用到了图形库操作，把一张图片分割，用一个二维数组代表各个碎片，初始打乱数组就可以得到打乱的图片。
    </p>
    <pre class="has"><code class="language-cpp">void get_arr()
{
	int book[10] = { 0 };
	srand((unsigned int)time(NULL));                 //设置时间种子
	for(int i=0; i&lt;3; i++)                           //随机arr数组，打乱图片顺序
		for (int j = 0; j &lt; 3; j++)
		{
			if (i == 2 &amp;&amp; j == 2)
				break;
			while (1)
			{
				int s = rand() % 8;
				if (book[s] == 0)
				{
					book[s] = 1;
					arr[i][j] = s;
					break;
				}
			}
		}
	arr[2][2] = 9;
}</code></pre>

    <pre class="has"><code class="language-cpp">#include &lt;stdio.h&gt;
#include &lt;stdlib.h&gt;
#include &lt;graphics.h&gt;
#include &lt;windows.h&gt;
#include &lt;conio.h&gt;
#include&lt;time.h&gt; 
#include &lt;iostream&gt;
#include &lt;algorithm&gt;
#define N 600             //图片长和宽，为便于计算取600
#define M N/3             //分为9块，每块有
using namespace std;
IMAGE t, over;
int arr[5][5];

void get_arr()
{
	int book[10] = { 0 };
	srand((unsigned int)time(NULL));                 //设置时间种子
	for(int i=0; i&lt;3; i++)                           //随机arr数组，打乱图片顺序
		for (int j = 0; j &lt; 3; j++)
		{
			if (i == 2 &amp;&amp; j == 2)
				break;
			while (1)
			{
				int s = rand() % 8;
				if (book[s] == 0)
				{
					book[s] = 1;
					arr[i][j] = s;
					break;
				}
			}
		}
	arr[2][2] = 9;
}
void choosephoto()
{
	int choose;
	srand((unsigned int)time(NULL));                 //设置时间种子
	choose = rand() % 5;							 //选择图片
	if(choose==0)
		loadimage(&amp;t, "少司命.jpeg", N, N);                //加载图片
	if (choose == 1)
		loadimage(&amp;t, "柯南.jpg", N, N);
	if (choose == 2)
		loadimage(&amp;t, "马里奥.jpeg", N, N);
	if (choose == 3)
		loadimage(&amp;t, "火影.jpeg", N, N);
	if (choose == 4)
		loadimage(&amp;t, "数字.png", N, N);
}
void Game()                          //显示拼图
{
	initgraph(N, N);
	
	for (int i = 0; i &lt; 3; i++)
	{
		for (int j = 0; j &lt; 3; j++)
		{
			putimage(j * M, i * M, M, M, &amp;t, arr[i][j]%3*M, arr[i][j]/3*M);     //切割图片
		}
	}
}
int GameOver()
{
	int temp = 1;
	for (int i = 0; i &lt; 3; i++)
	{
		for (int j = 0; j &lt; 3; j++)
		{
			if (arr[i][j] != 3 * i + j)
			{
				temp = 0;
				break;
			}
		}
		if (temp == 0)
			break;
	}
	if (temp == 1)
	{
		loadimage(&amp;over, "游戏结束.png", 400, 400);
		putimage(100, 100, &amp;over);
		return 1;
	}
}
void Gamestart()
{
	char ch;
	int tx, ty;
	
	while (1)
	{
		if (_kbhit()) {                     //检测键盘输入
			for (int i = 0; i &lt; 3; i++)
			{
				for (int j = 0; j &lt; 3; j++)
					if (arr[i][j] == 9)
					{
						tx = i;
						ty = j;
						break;
					}
			}
			ch = _getch();
			if (ch == 72 || ch=='w')                          //按键为w或者上
			{
				if (tx &gt;= 1)
				{
					//swap(arr[tx][ty], arr[tx - 1][ty]);	
					int x = arr[tx][ty];
					arr[tx][ty] = arr[tx - 1][ty];
					arr[tx - 1][ty] = x;
				}
			} 
			else if (ch == 75 || ch=='a')                //按键为a或者左
			{
				if (ty &gt;= 1)
				{
					int x = arr[tx][ty];
					arr[tx][ty] = arr[tx][ty - 1];
					arr[tx][ty - 1] = x;
				}
			}
			else if (ch == 80 || ch=='s')               //按键为s或者下
			{
				if (tx &lt; 2)
				{
					int x = arr[tx][ty];
					arr[tx][ty] = arr[tx + 1][ty];
					arr[tx + 1][ty] = x;
				}
			}
			else if (ch == 77 || ch=='d')                //按键为d或者右
			{
				if (ty &lt; 2)
				{
					int x = arr[tx][ty];
					arr[tx][ty] = arr[tx][ty + 1];
					arr[tx][ty + 1] = x;
				}
			}

			Game();                                   //输出拼图界面
			if (GameOver())
				break;
		}
	}
	while (ch=getchar())
		if (ch == '\n')
			break;

}

int main()
{	
	get_arr();                  //获取初始数组
	choosephoto();             //选择图片
	Game();                    //输出起始拼图
	Gamestart();               //游戏开始
	system("pause");
	closegraph();              //关闭画布
	return 0;
}</code></pre>
