  <div class="article_content clearfix" id="article_content">
   <link href="../../assets/css/kdoc_html_views-1a98987dfd.css" rel="stylesheet"/>
   <link href="../../assets/css/ck_htmledit_views-704d5b9767.css" rel="stylesheet"/>
   <div class="htmledit_views" id="content_views">
    <p>
     <strong>
      一、关闭数据库（2个节点实例都要关闭）
     </strong>
    </p>
    <p>
     srvctl stop database -d ora19c
    </p>
    <p>
    </p>
    <p>
     <strong>
      二、多路径控制文件
     </strong>
    </p>
    <p>
     打开其中一个节点到nomount状态
    </p>
    <p>
     sqlplus / as sysdba
    </p>
    <p>
     startup nomount;
    </p>
    <p>
     [oracle@ora19c1:/home/oracle]$ rman target /
    </p>
    <p>
     RMAN&gt; restore controlfile to '+OCR' from '+DATA/ORA19C/CONTROLFILE/Current.257.1057568499';
    </p>
    <p>
     ---Current.257.1057568499这种格式必须用rman 导入。
    </p>
    <p>
    </p>
    <p>
     <strong>
      三、修改spfile
     </strong>
    </p>
    <p>
     create pfile='/home/oracle/pfile.ora' from spfile;
    </p>
    <p>
     alter system set control_files='+DATA/ORA19C/CONTROLFILE/Current.257.1057568499','+OCR/ORA19C/CONTROLFILE/current.262.1116745615' scope=spfile;
    </p>
    <p>
     <img alt="" height="363" src="https://i-blog.csdnimg.cn/direct/efc81d1f2c99443a917db21442e897a4.png" width="1358"/>
    </p>
    <p>
    </p>
    <p>
     <strong>
      四、重启数据库
     </strong>
    </p>
    <p>
     shutdown immediate;
    </p>
    <p>
     srvctl start database -d ora19c
    </p>
