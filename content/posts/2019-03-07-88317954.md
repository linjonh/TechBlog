     输入
    </p>
    <p>
    </p>
    <p>
     多组输入，每组输入一个整数k(0&lt;=k&lt;=3000)
    </p>
    <p>
     输出
    </p>
    <p>
    </p>
    <p>
     每组输出一个数：2^k
    </p>
    <p>
     输入样例 1
    </p>
    <pre>2</pre>
    <p>
     输出样例 1
    </p>
    <pre>4</pre>
    <pre class="has"><code class="language-java">import java.math.*;
import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		Scanner cin = new Scanner(System.in);
		while(cin.hasNext()) {
			int n = cin.nextInt();
			BigDecimal rn = new BigDecimal("2");
			rn = rn.pow(n);
			System.out.println(rn);
		}
	}
}
</code></pre>
