     子组件
    </p>
    <pre><code class="prism language-vue">&lt;template&gt;
  &lt;el-dialog title="接口实例详情" @open="onOpen" @close="onClose"&gt;
    // 主体内容
    &lt;div slot="footer"&gt;
      &lt;el-button @click="close"&gt;取消&lt;/el-button&gt;
      &lt;el-button type="primary" @click="handelConfirm"&gt;确定&lt;/el-button&gt;
    &lt;/div&gt;
  &lt;/el-dialog&gt;
&lt;/template&gt;
&lt;script&gt;
export default {
  props: {
    visible: Boolean
  },
  methods: {
    // 弹出框打开事件
    onOpen() {},
    onClose() {
      this.$refs['GxptServiceDialog'].resetFields()
    },
    close() {
      this.$emit('update:visible', false)
    },
    handelConfirm() {
      this.$refs['gialog'].validate(valid =&gt; {
        if (!valid) return
        this.close()
      })
    }
  }
}  
</code></pre>

<span class="token keyword">export</span> <span class="token keyword">default</span> <span class="token keyword">new</span> <span class="token class-name">Vue</span><span class="token punctuation">(</span><span class="token punctuation">)</span>

     子组件
    </p>
    <pre><code class="prism language-vue">&lt;template&gt;
  &lt;el-dialog title="接口实例详情" @open="onOpen" @close="onClose" :visible.sync="visible"&gt;
    // 主体内容
    &lt;div slot="footer"&gt;
      &lt;el-button @click="close"&gt;取消&lt;/el-button&gt;
      &lt;el-button type="primary" @click="handelConfirm"&gt;确定&lt;/el-button&gt;
    &lt;/div&gt;
  &lt;/el-dialog&gt;
&lt;/template&gt;
&lt;script&gt;
export default {
  data() {
    return {
      visible: false
    }
  },
  mounted() {
    bus.$on('isVisible', data =&gt; {
      this.visible = data
    })
  },
  methods: {
    // 弹出框打开事件
    onOpen() {},
    onClose() {
      this.$refs['dialog'].resetFields()
    },
    close() {
      this.$emit('update:visible', false)
    },
    handelConfirm() {
      this.$refs['gialog'].validate(valid =&gt; {
        if (!valid) return
        this.close()
      })
    }
  }
}  
</code></pre>
