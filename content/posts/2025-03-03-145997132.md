    <p>
     你可以使用
     <code>
      requests
     </code>
     和
     <code>
      BeautifulSoup
     </code>
     来爬取《唐诗三百首》和《宋词三百首》的数据。以下是一个基本的 Python 爬虫示例，它从
     <strong>
      <a class="link-info" href="https://www.gushiwen.cn/gushi/tangshi.aspx" rel="nofollow" title="中华诗词网">
       中华诗词网
      </a>
     </strong>
     或类似的网站获取数据并保存为 JSON 文件。
    </p>
    <p>
    </p>
    <pre><code class="language-python">import requests
from bs4 import BeautifulSoup
import json
import time

# 爬取的URL列表，这里以唐诗为例
# base_url = "https://www.gushiwen.cn/gushi/tangshi.aspx"
# 这里是爬取宋词的URL
base_url = "https://www.gushiwen.cn/gushi/songsan.aspx"


# 解析诗词页面
def get_poem_links(url):
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/119.0.0.0 Safari/537.36"
    }
    response = requests.get(url, headers=headers)
    response.encoding = 'utf-8'
    soup = BeautifulSoup(response.text, 'html.parser')

    # 获取每首诗的链接
    links = []
    for a_tag in soup.select(".typecont a"):
        links.append("https://www.gushiwen.cn" + a_tag['href'])

    return links


# 解析诗词内容
def get_poem_data(poem_url):
    response = requests.get(poem_url)
    response.encoding = 'utf-8'
    soup = BeautifulSoup(response.text, 'html.parser')

    title = soup.select_one("h1").text.strip()
    author_tag = soup.select_one(".source")
    author = author_tag.text.strip() if author_tag else "未知"

    content_div = soup.select_one(".contson")
    content = "\n".join([line.strip() for line in content_div.stripped_strings])

    return {"title": title, "author": author, "content": content}


# 主函数
def scrape_tang_poems():
    poem_links = get_poem_links(base_url)
    poems = []

    for index, link in enumerate(poem_links):
        try:
            poem = get_poem_data(link)
            poems.append(poem)
            print(f"爬取成功: {poem['title']}")
            time.sleep(1)  # 避免请求过快被封
        except Exception as e:
            print(f"爬取失败: {link}, 错误: {e}")

    # 保存为 JSON 文件
    with open("tang_poems_songci.json", "w", encoding="utf-8") as f:
        json.dump(poems, f, ensure_ascii=False, indent=4)

    print("所有诗词已保存！")


# 运行爬虫
scrape_tang_poems()
