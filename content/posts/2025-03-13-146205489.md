	cin &gt;&gt; n &gt;&gt; m;
	for(int i = 1; i &lt;= n; i++){
		int x; cin &gt;&gt; x;
		if(x &gt; 0){
			r[x]++;
		}
		else if(x &lt; 0){
			l[-x]++;
		}
		else cnt = 1;
	}
	for(int i = 1; i &lt; N; i++){
		l[i] += l[i - 1];
		r[i] += r[i - 1];
	}
	int ans = 0;
	for(int i = 0; i &lt;= m; i++){
		int ll = l[i];
		if(m - 2 * i &gt; 0){
			ll += r[m - 2 * i];
		}
		ans = max(ans, ll);
		int rr = r[i];
		if(m - 2 * i &gt; 0){
			rr += l[m - 2 * i];
		}
		ans = max(ans, rr);
	}
	cout &lt;&lt; ans + cnt &lt;&lt; endl;
	
	
	return 0;
} </code></pre>
