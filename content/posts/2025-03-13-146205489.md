洛谷P10904挖矿

    
    
    #include<bits/stdc++.h>
    using namespace std;
    int n, m;
    const int N = 2000010;
    int l[N], r[N];
    int cnt;
    int main(){
    	cin >> n >> m;
    	for(int i = 1; i <= n; i++){
    		int x; cin >> x;
    		if(x > 0){
    			r[x]++;
    		}
    		else if(x < 0){
    			l[-x]++;
    		}
    		else cnt = 1;
    	}
    	for(int i = 1; i < N; i++){
    		l[i] += l[i - 1];
    		r[i] += r[i - 1];
    	}
    	int ans = 0;
    	for(int i = 0; i <= m; i++){
    		int ll = l[i];
    		if(m - 2 * i > 0){
    			ll += r[m - 2 * i];
    		}
    		ans = max(ans, ll);
    		int rr = r[i];
    		if(m - 2 * i > 0){
    			rr += l[m - 2 * i];
    		}
    		ans = max(ans, rr);
    	}
    	cout << ans + cnt << endl;
    	
    	
    	return 0;
    }
