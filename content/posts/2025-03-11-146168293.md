##### `c/c++`

```cpp
#include <vector>
#include <algorithm>

class Solution
{
public:
    int threeSumClosest(std::vector<int> &nums, int target)
    {
        int sz = nums.size();
        if (sz == 3)
            return nums[0] + nums[1] + nums[2];

        std::sort(nums.begin(), nums.end());

        int min_sum = nums[0] + nums[1] + nums[2];
        int min_sum_df = std::abs(target - min_sum);

        int i = 0;

        while (i < sz - 2)
        {
            int l = i + 1, r = sz - 1;

            while (l < r)
            {
                int sum = nums[i] + nums[l] + nums[r];
                int df = std::abs(target - sum);

                if (df < min_sum_df)
                {
                    min_sum = sum;
                    min_sum_df = df;
                }

                if (sum > target)
                    --r;
                else if (sum < target)
                    ++l;
                else
                    return sum;
            }
            ++i;
        }

        return min_sum;
    }
};
```

```go
package main

import (
	"math"
	"sort"

func threeSumClosest(nums []int, target int) int {

	n := len(nums)

	if len(nums) == 3 {
		return nums[0] + nums[1] + nums[2]
	}

	sort.Ints(nums)

	min_sum := nums[0] + nums[1] + nums[2]
	min_sum_df := math.Abs(float64(target) - float64(min_sum))

	i := 0

	for i < n-2 {
		l, r := i+1, n-1

		for l < r {
			sum := nums[i] + nums[l] + nums[r]
			sum_df := math.Abs(float64(target) - float64(sum))

			if sum_df < min_sum_df {
				min_sum = sum
				min_sum_df = sum_df
			}

			if sum < target {
				l++
			} else if sum > target {
				r--
			} else {
				return sum
			}
		}

		i++
	}

	return min_sum
}
```
