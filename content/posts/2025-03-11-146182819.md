    <p id="u85583158">
     chromium用于渲染页面 node.js用于后台操作 native用于操作os
    </p>
    <p class="img-center">
     <img alt="" height="147" id="uf2c03135" src="https://i-blog.csdnimg.cn/img_convert/51e9efdd6cf4afdd21d3d816fd5ac060.png" width="696"/>
    </p>
    <p id="ufce0bbe5">
     这是进程通信流程 当需要操作os的的时候就需要用到进程通信 例如通过前端向磁盘写入文件
    </p>
    <p id="ucb7a8ff0">
    </p>
    <p class="img-center">
     <img alt="" height="1287" id="u33cb150c" src="https://i-blog.csdnimg.cn/img_convert/e3e9f505f10f7c271829cb8405a52fad.png" width="2028"/>
    </p>
    <p id="u657282aa">
     进入项目文件并且初始化项目
    </p>
    <pre id="o4LFR"><code>cd TingHack
npm init</code></pre>
    <p id="ub3dae6e3">
     必填项
    </p>
    <p id="u2254339c">
    </p>
    <p class="img-center">
     <img alt="" height="234" id="ue1b99d52" src="https://i-blog.csdnimg.cn/img_convert/7e5136c24142a04bdfe4eb1a4ba2d861.png" width="832"/>
    </p>
    <p id="u15306224">
     scripts需要修改
    </p>
    <p id="u271f699e">
    </p>
    <p class="img-center">
     <img alt="" height="83" id="ucaeae4e6" src="https://i-blog.csdnimg.cn/img_convert/161fae9994f4a1829214480447db2981.png" width="314"/>
    </p>
    <p id="u60e04dc0">
     最终代码 package.json
    </p>
    <pre id="JlPlW"><code>{
  "name": "tinghack",
  "version": "1.0.0",
  "main": "main.js",
  "scripts": {
    "start": "electron ."
  },
  "author": "Ting",
  "license": "ISC",
  "description": "TingHack"
}</code></pre>
    <p id="u0a4fd32d">
     先设置镜像源然后 下载相关工具 之后编译器有了代码提示
    </p>
    <pre id="OfXTJ"><code>npm config set electron_mirror https://npm.taobao.org/mirrors/electron/ 
npm install --save-dev electron</code></pre>
    <p id="ucf53f7f0">
     执行 以下代码就有ui了
    </p>
    <pre id="sCXRA"><code>npm start</code></pre>
    <p id="ucf8d1bf0">
    </p>
    <p class="img-center">
     <img alt="" height="486" id="u0a7fd9f1" src="https://i-blog.csdnimg.cn/img_convert/a29c8620d6276263955d69515e7a6ee6.png" width="1525"/>
    </p>
    <p id="u575f4940">
     配置自动重启 每次修改代码后自动执行npm start
    </p>
    <pre id="nlgOc"><code>npm i nodemon -D</code></pre>
    <p id="u591af4a4">
     修改package.json
    </p>
    <pre id="xsD1t"><code>  "scripts": {
    "start": "nodemon --exec electron ."
  },</code></pre>
    <p id="u916b39e1">
     新建nodemon.json 代码如下
    </p>
    <pre id="cQtlB"><code>{
  "ignore":[
    "node modules",
    "dist"
  ],
  "restartable":"r",
  "watch":["*.*"],
  "ext":"html,js,css"
}
</code></pre>
    <p id="ud955bf9f">
     主进程（main.js）如果需要与渲染进程进行通信需要一个中间的进程 即预加载进程 （preload.js）
    </p>
    <p id="u4d8e901d">
     三个进行执行顺序是 主进程 预加载进程 渲染渲染进程 ，其中预加载进行可以使用主进程从一些函数 方法
    </p>
    <p id="u0263e5d9">
     其中渲染进程向主进程通信
    </p>
    <pre id="D1SOO"><code>const {contextBridge, ipcRenderer} = require('electron')
......
ipcRenderer.send('file-save', data)


const {app, BrowserWindow,ipcMain} = require('electron')
......
ipcMain.on('file-save', writeFile)</code></pre>
    <p id="u3897bd40">
     主进程 渲染进程双向通信
    </p>
    <pre id="zPdIh"><code>const {contextBridge, ipcRenderer} = require('electron')
......
async readFile() {
    return await ipcRenderer.invoke('file-read')
}


const {app, BrowserWindow,ipcMain} = require('electron')
......
ipcMain.handle('file-read',readFile)
</code></pre>
    <p id="u7d1dfc13">
     打包项目
    </p>
    <pre id="B9Pwc"><code>npm install electron-builder -D</code></pre>
    <p id="uc2c638a0">
     可自定义的设置项如下
    </p>
    <pre id="kdLpD"><code>{
  "name": "video-tools",
  "version": "1.0.0",
  "main": "main.js",
  "scripts": {
    "start": "electron .",
    "build": "electron-builder"
  },
  "build": {
    "appId": "com.atguigu.video",
    "win": {
      "icon": "./logo.ico",
      "target": [
        {
          "target": "nsis",
          "arch": ["x64"]
        }
      ]
    },
    "nsis": {
      "oneClick": false,
      "perMachine": true,
      "allowToChangeInstallationDirectory": true
    }
  },
  "devDependencies": {
    "electron": "^30.0.0",
    "electron-builder": "^24.13.3"
  },
  "author": "宝码香车",
  "license": "ISC",
  "description": "A video processing program based on Electron"
}
</code></pre>
