    <h4>
     一、操作环境
    </h4>
    <p>
     操作系统:  Windows 10 专业版、IDE:DevEco Studio 3.1、SDK:HarmonyOS 3.1
    </p>
    <h4>
     二、Toast组件
    </h4>
    <p>
     <strong>
      Toast
     </strong>
     的应用场景也非常广泛，比如网络请求出错了可以弹一个
     <strong>
      Toast
     </strong>
     提示等。
     <code>
      @ohos.prompt
     </code>
     模块里提供显示一个
     <strong>
      Toast
     </strong>
     的 API 如下所示：
    </p>
    <pre><code>declare namespace prompt {
  // 显示一个Toast
  function showToast(options: ShowToastOptions):void;
}

interface ShowToastOptions { // Toast 配置参数
message: string; // Toast 显示文本
duration?: number; // Toast 显示时长
bottom?: string | number; // Toast 距离屏幕底部距离
}</code></pre>
<ul>
<li>
<strong>
options
</strong>
：设备
<strong>
Toast
</strong>
显示特性，
<code>
ShowToastOptions
</code>
参数说明如下：
<ul>
<li>
<strong>
message
</strong>
：提示文本，必填项。
</li>
<li>
<strong>
duration：Toast
</strong>
显示时间，单位毫秒，范围 [1500, 10000]，默认 1500。
</li>
<li>
<strong>
bottom
</strong>
：设置
<strong>
Toast
</strong>
的显示位置距离底部的间距。
</li>
</ul>
</li>
</ul>
<p>
</p>
<h4>
三、示例
</h4>
<p>
代码
</p>
<pre><code>import prompt from '@ohos.prompt';
import promptAction from '@ohos.promptAction';

@Entry @Component struct ToastTest {

build() {
Column({space: 10}) {

      Button("无参数Toast")
        .onClick(() =&gt; {
          promptAction.showToast({
            message: "默认Toast"
          })
        })

      Button("带参数Toast")
        .onClick(() =&gt; {
          promptAction.showToast({
            message: "bottom为300的位置", // 显示文本
            duration: 8000,              // 显示时长
            bottom: 300                  // 距离底部的距离
          })
        })
    }
    .width('100%')
    .height('100%')
    .padding(10)

}
}</code></pre>
