     解题思路：
    </h4>
    <p>
     这题我们可以看到a，b这两个数的值是非常大的（已经超过了long long）因此我们需要自己写一个加法的代码，我们可以用一个数组把数字的每一位存起来，在想小学数学那样一位一位的计算就可以了
    </p>
    <p class="img-center">
     <img alt="" height="421" src="https://i-blog.csdnimg.cn/direct/c7070bc3315e4fbc88ad0baf3cf7c371.png" width="515"/>
    </p>
    <h4 id="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A" name="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A">
     解题代码：
    </h4>
    <pre><code class="language-cpp">#include &lt;iostream&gt;

using namespace std;

const int N = 1010;

int a[N], b[N], c[N];
int la, lb, lc;

void add(int c[], int a[], int b[])
{
	for (int i = 0; i &lt; lc; i++)
	{
		c[i] += a[i] + b[i];
		c[i + 1] = c[i] / 10;
		c[i] %= 10;
	}
	if (c[lc]) lc++;
}


int main()
{
	string x, y; cin &gt;&gt; x &gt;&gt; y;
	// 将数据存入数组
	la = x.size();
	lb = y.size();
	lc = max(la, lb);
	for (int i = la - 1; i &gt;= 0; i--) a[la - 1 - i] = x[i] - '0';
	for (int i = lb - 1; i &gt;= 0; i--) b[lb - 1 - i] = y[i] - '0';
	add(c, a, b);// c = a + b

	for (int i = lc - 1; i &gt;= 0; i--) cout &lt;&lt; c[i];

	return 0;
}</code></pre>

     解题思路：
    </h4>
    <p>
     这题和上面的高精度加法类似，都是不能简单的用一个long long 的变量就能搞定的，因此还是需要我们模拟一下小学时候学的减法，列竖式来一个一个算
    </p>
    <p class="img-center">
     <img alt="" height="423" src="https://i-blog.csdnimg.cn/direct/df4c6d37d4c34a6eb9ece988014005df.png" width="438"/>

    <h4 id="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A" name="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A">
     解题代码：
    </h4>
    <pre><code class="language-cpp">
#include &lt;iostream&gt;

using namespace std;

const int N = 1e5 + 10;

int a[N], b[N], c[N];
int la, lb, lc;

bool bigger(string x, string y)
{
	if (x.size() != y.size())
	{
		return y.size() &gt; x.size();
	}
	return y &gt; x;
}

void sub(int c[], int a[], int b[])
{
	for (int i = 0; i &lt; lc; i++)
	{
		c[i] += a[i] - b[i];
		if (c[i] &lt; 0)
		{
			c[i + 1]--;
			c[i] += 10;
		}
	}
	while (1 != lc &amp;&amp; c[lc - 1] == 0) lc--;
}

int main()
{
	string x, y; cin &gt;&gt; x &gt;&gt; y;
	if (bigger(x, y))
	{
		swap(x, y);
		cout &lt;&lt; '-';
	}
	la = x.size(); lb = y.size(); lc = max(la, lb);
	for (int i = la - 1; i &gt;= 0; i--) a[la - 1 - i] = x[i] - '0';
	for (int i = lb - 1; i &gt;= 0; i--) b[lb - 1 - i] = y[i] - '0';

	sub(c, a, b); // c = a - b;
	for (int i = lc - 1; i &gt;= 0; i--) cout &lt;&lt; c[i];
	return 0;
}</code></pre>

    <h4 id="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A" name="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A">
     解题代码：
    </h4>
    <pre><code class="language-cpp">#include &lt;iostream&gt;

using namespace std;

const int N = 2010;

int a[N], b[N], c[N];
int la, lb, lc;

void mul(int c[], int a[], int b[])
{
	for (int i = 0; i &lt; la; i++)
	{
		for (int j = 0; j &lt; lb; j++)
		{
			c[i + j] += a[i] * b[j]; // 无进位乘法
		}
	}
	// 处理进位
	for (int i = 0; i &lt; lc; i++)
	{
		c[i + 1] += c[i] / 10;
		c[i] %= 10;
	}
	while (lc != 1 &amp;&amp; c[lc - 1] == 0) lc--;
}

int main()
{
	string x, y; cin &gt;&gt; x &gt;&gt; y;
	la = x.size(); lb = y.size(); lc = la + lb;

	for (int i = la - 1; i &gt;= 0; i--) a[la - 1 - i] = x[i] - '0';
	for (int i = lb - 1; i &gt;= 0; i--) b[lb - 1 - i] = y[i] - '0';

	mul(c, a, b);// c = a * b;

	for (int i = lc - 1; i &gt;= 0; i--) cout &lt;&lt; c[i];

	return 0;
}</code></pre>

    <h4 id="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A" name="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A">
     解题代码：
    </h4>
    <pre><code class="language-cpp">#include &lt;iostream&gt;

using namespace std;

typedef long long LL;

const int N = 5010;

int a[N], c[N];
LL b;
int la, lc;

void div(int c[], int a[], LL b)
{
	LL t = 0;
	for (int i = lc - 1; i &gt;= 0; i--)
	{
		t = t * 10 + a[i];
		c[i] = t / b;
		t = t % b;
	}
	while (lc != 1 &amp;&amp; c[lc - 1] == 0) lc--;
}

int main()
{
	string x; cin &gt;&gt; x &gt;&gt; b;
	la = x.size(); lc = la;
	for (int i = la - 1; i &gt;= 0; i--) a[la - 1 - i] = x[i] - '0';

	div(c, a, b); // c = a / b

	for (int i = lc - 1; i &gt;= 0; i--) cout &lt;&lt; c[i];

	return 0;
}</code></pre>

     解题思路：
    </h4>
    <p>
     因为题目给的数量不大，因此我们可以暴力枚举，把所有情况全部意义罗列出来，在判断是否符合题目要求，符合直接返回即可（因为这个题是我们需要我们找到最后一个符合要求的地毯，因此我们可以之后从后往前遍历来优化代码）
    </p>
    <h4 id="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A" name="%C2%A0%20%C2%A0%20%C2%A0%20%C2%A0%20%E8%A7%A3%E9%A2%98%E4%BB%A3%E7%A0%81%EF%BC%9A">
     解题代码：
    </h4>
    <pre><code class="language-cpp">
#include &lt;iostream&gt;

using namespace std;

const int N = 1e4 + 10;

int a[N], b[N], g[N], k[N];

int main()
{
	int n; cin &gt;&gt; n;
	for (int i = 1; i &lt;= n; i++)
	{
		cin &gt;&gt; a[i] &gt;&gt; b[i] &gt;&gt; g[i] &gt;&gt; k[i];
	}
	int x, y; cin &gt;&gt; x &gt;&gt; y;
	int flag = 0;
	for (int i = n; i &gt;= 0; i--)
	{
		if (a[i] &lt;= x &amp;&amp; b[i] &lt;= y &amp;&amp;
			a[i] + g[i] &gt;= x &amp;&amp; b[i] + k[i] &gt;= y)
		{
			cout &lt;&lt; i &lt;&lt; endl;
			flag++;
			break;
		}
	}
	if (!flag) cout &lt;&lt; -1 &lt;&lt; endl;

	return 0;
}</code></pre>

using namespace std;

typedef long long LL;

int days[13] = { 0,31,29,31,30,31,30,31,31,30,31,30,31 };

     解题思路：
    </h4>
    <p>
     这个题也是可以枚举的，我们可以根据第一排是否1有雷推出下一排有没有雷（是否有雷用0 1表示）那么
     <strong>
      下一排有没有雷就=上一排的地雷+当前排的地雷 - 当前排的数字
     </strong>
    </p>

     解题代码：
    </h4>
    <pre><code class="language-cpp">#include &lt;iostream&gt;

using namespace std;

const int N = 1e4 + 10;

int n;

int a[N], b[N];

int cheak1()
{
	a[1] = 0;
	for (int i = 2; i &lt;= n + 1; i++)
	{
		a[i] = b[i - 1] - a[i - 1] - a[i - 2];
		if (a[i] &gt; 1 || a[i] &lt; 0)
			return 0;
	}
	if (a[n + 1])return 0;
	return 1;
}

int cheak2()
{
	a[1] = 1;
	for (int i = 2; i &lt;= n + 1; i++)
	{
		a[i] = b[i - 1] - a[i - 1] - a[i - 2];
		if (a[i] &gt; 1 || a[i] &lt; 0)
			return 0;
	}
	if (a[n + 1])return 0;
	return 1;
}

int main()
{
	cin &gt;&gt; n;
	for (int i = 1; i &lt;= n; i++)
	{
		cin &gt;&gt; b[i];
	}
	int ret = 0;
	ret += cheak1();
	ret += cheak2();
	cout &lt;&lt; ret &lt;&lt; endl;

	return 0;
}</code></pre>
