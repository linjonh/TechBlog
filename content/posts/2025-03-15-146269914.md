    <h4>
     二，思路
    </h4>
    <ul>
     <li>
      基础二分查找只要找到边界值和中间的下标值就行了
     </li>
     <li>
      i=0 j=行*高-1(对应基础版的length-1)
     </li>
     <li>
      中间下标=[中间值/行][中间值%行]
     </li>
     <li>
      比如 俩行俩列 [1,1] = 3(从0开始) , [3 /2(行)] [3 %2(行)]
     </li>
    </ul>
    <h4>
     三，代码
    </h4>
    <pre><code class="language-java">public class Leetcode74 {
    public static void main(String[] args) {
        System.out.println(new Leetcode74().searchMatrix(new int[][]{<!-- -->{1,3,5,7},{10,11,16,20},{23,30,34,60}}, 3));
    }
    public boolean searchMatrix(int[][] matrix, int target) {
        int i=0;
        int j=matrix[0].length*matrix.length-1;
        int len=matrix[0].length;
        while(i&lt;=j){
            int m=(i+j)&gt;&gt;&gt;1;
            if(matrix[m/len][m%len]&gt;target){
                j=m-1;
            }else if(matrix[m/len][m%len]&lt;target){
                i=m+1;
            }else{
                return true;
            }
        }
        return false;
    }
}
