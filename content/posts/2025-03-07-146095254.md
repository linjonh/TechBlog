    <h4>
     更换&lt;n-switch&gt;的颜色
    </h4>
    <blockquote>
     <p>
      使用     :theme-overrides="themeOverrides"
     </p>
    </blockquote>
    <pre><code class="language-html">&lt;n-switch size="small" :theme-overrides="themeOverrides" 
          v-model:value="app.enable" @update:value="toggleEnable(app)"/&gt;</code></pre>
    <pre><code class="language-javascript">const themeOverrides = {
    railColorActive: '#42af38', // 选中时的轨道颜色
};</code></pre>
    <h4>
     全局更换主题色,默认原本是绿色,现在全局更换的到蓝色
    </h4>
    <h6>
     1.自定义主题的js
    </h6>
    <pre><code class="language-javascript">
import { defineComponent } from 'vue';

export default {
    common: {
        // 设置主色为蓝色
        primaryColor: '#2080f0',
        primaryColorHover: '#4098fc',
        primaryColorPressed: '#1060c9',
        primaryColorSuppl: '#4098fc',
        infoColor: '#2080f0',
        infoColorHover: '#4098fc',
        infoColorPressed: '#1060c9',
        infoColorSuppl: '#4098fc'
    }
};</code></pre>
    <h6>
     2:在app.vue中使用
    </h6>
    <p>
     以下是我的App.vue整个代码
    </p>
    <pre><code class="language-javascript">&lt;template&gt;
    &lt;n-config-provider style="height: 100%;" :locale="zhCN" :date-locale="dateZhCN" :theme-overrides="themeOverrides"&gt;
        &lt;n-loading-bar-provider&gt;
            &lt;n-dialog-provider&gt;
                &lt;n-notification-provider&gt;
                    &lt;n-message-provider&gt;
                        &lt;router-view&gt;&lt;/router-view&gt;
                        &lt;naive-provider-content/&gt;
                    &lt;/n-message-provider&gt;
                &lt;/n-notification-provider&gt;
            &lt;/n-dialog-provider&gt;
        &lt;/n-loading-bar-provider&gt;
    &lt;/n-config-provider&gt;
&lt;/template&gt;

&lt;script setup&gt;
import {
    useLoadingBar,
    useDialog,
    useMessage,
    useNotification,
    NDialogProvider,
    NConfigProvider,
    NLoadingBarProvider,
    NMessageProvider,
    NNotificationProvider
} from 'naive-ui'
import {zhCN, dateZhCN} from 'naive-ui'
import {defineComponent, h, onMounted} from "vue";
import themeOverrides  from "@/styles/theme.js";


// 挂载naive组件的方法至window, 以便在路由钩子函数和请求函数里面调用
function registerNaiveTools() {
    window.$loadingBar = useLoadingBar()
    window.$dialog = useDialog()
    window.$message = useMessage()
    window.$success = message('success')
    window.$warning = message('warning')
    window.$error = message('error')
    window.$info = message('info')
    window.$notification = useNotification()
}

function message(
    type
) {
    return (content, duration = 1000, options) =&gt; {
        return new Promise((resolve) =&gt; {
            window.$message?.[type](content, {
                duration,
                ...options,
                onLeave: () =&gt; {
                    if (options?.onLeave) {
                        options.onLeave()
                    }
                    resolve()
                }
            })
        })
    }
}


const NaiveProviderContent = defineComponent({
    name: 'NaiveProviderContent',
    setup() {
        // console.debug(111)
        registerNaiveTools()
    },
    render() {
        return h('div')
    }
})




&lt;/script&gt;
