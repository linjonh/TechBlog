# 小程序各类编码转换

#### 小程序Unicode编码(\u方式)

```js
	// 中文转换为Unicode编码
	var str = "我是张三";
	escape(str).replace(/\%u/g,'/u');

  /**
   * unicode解码
   * @param {} str 
   */
 	decodeUnicode:function(str) {
	    str = str.replace(/\\/g, "%");
	    return unescape(str);
 	}

```

var str = "年后的方法发生公司方法";
    str = encodeURIComponent(str);  //编码
