    <h4>
     二，思路
    </h4>
    <ul>
     <li>
      找到最左和最右俩个点，利用俩个二分查找
     </li>
     <li>
      俩种情况一种存在一种不存在
     </li>
     <li>
      存在：nums[m]==target右边-1  结果i
     </li>
     <li>
      存在：nums[m]==target左边+1  但是结果下标减去i-1
     </li>
     <li>
      上面俩种情况可以举例子 1 2 2 3 找2 自己推理一下就明白
     </li>
     <li>
      不存在：一种情况nums为空的情况 一种是数据没找到
     </li>
    </ul>
    <h4>
     三，代码
    </h4>
    <pre><code class="language-java">import java.util.Arrays;

public class Leetcode34 {
    public static void main(String[] args) {
        System.out.println(Arrays.toString(new Leetcode34().searchRange(new int[]{1}, 1)));
    }

    public int[] searchRange(int[] nums, int target) {
        int[] arr = new int[2];
        arr[0] = left(nums, target);
        if (arr[0] == -1)
            arr[1] = -1;
        else
            arr[1] = right(nums, target);
        return arr;
    }

    public int left(int[] nums, int target) {
        if (nums.length &gt; 0) {
            int i = 0;
            int j = nums.length - 1;
            while (i &lt;= j) {
                int m = (i + j) &gt;&gt;&gt; 1;
                if (nums[m] &lt; target) {//左
                    i = m + 1;
                } else {
                    j = m - 1;
                }
            }
            if (i &lt; nums.length &amp;&amp; nums[i] == target) return i;
        }
        return -1;
    }

    public int right(int[] nums, int target) {
        if (nums.length &gt; 0) {
            int i = 0;
            int j = nums.length - 1;
            while (i &lt;= j) {
                int m = (i + j) &gt;&gt;&gt; 1;
                if (nums[m] &lt;= target) {//左
                    i = m + 1;
                } else {
                    j = m - 1;
                }
            }
            if (i - 1 &lt; nums.length &amp;&amp; nums[i - 1] == target) return i - 1;
        }
        return -1;
    }
}
