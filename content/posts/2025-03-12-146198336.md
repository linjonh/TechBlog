    <h3>
     <a id="_3">
     </a>
     第一步
    </h3>
    <p>
     在Excel中按下Alt + F11，打开Visual Basic for Applications（VBA）编辑器。
     <br/>
     <img alt="在这里插入图片描述" src="https://i-blog.csdnimg.cn/direct/1ccf67e6c76e45a3ac1c843f0e926435.png"/>
    </p>
    <h3>
     <a id="_7">
     </a>
     第二步
    </h3>
    <p>
     在VBA编辑器中，点击插入菜单，选择模块。
     <br/>
     <img alt="在这里插入图片描述" src="https://i-blog.csdnimg.cn/direct/dcd05b811b69410eb7bee9d993845c2c.png"/>
    </p>
    <h3>
     <a id="_13">
     </a>
     第三步
    </h3>
    <p>
     在新建的模块中，输入以下VBA代码：
     <br/>
     <img alt="在这里插入图片描述" src="https://i-blog.csdnimg.cn/direct/4bf59069b62149febc8e65684a1e15fb.png"/>
    </p>
    <blockquote>
     <p>
      代码，复制粘贴进去
     </p>
    </blockquote>
    <pre><code class="prism language-vat">Option Explicit

Sub RemovePassword()
    Dim i As Integer, j As Integer, k As Integer
    Dim l As Integer, m As Integer, n As Integer
    Dim i1 As Integer, i2 As Integer, i3 As Integer
    Dim i4 As Integer, i5 As Integer, i6 As Integer
    
    For i = 65 To 66
        For j = 65 To 66
            For k = 65 To 66
                For l = 65 To 66
                    For m = 65 To 66
                        For i1 = 65 To 66
                            For i2 = 65 To 66
                                For i3 = 65 To 66
                                    For i4 = 65 To 66
                                        For i5 = 65 To 66
                                            For i6 = 65 To 66
                                                For n = 32 To 126
                                                    On Error Resume Next
                                                    ActiveSheet.Unprotect Chr(i) &amp; Chr(j) &amp; Chr(k) &amp; _
                                                                           Chr(l) &amp; Chr(m) &amp; Chr(i1) &amp; _
                                                                           Chr(i2) &amp; Chr(i3) &amp; Chr(i4) &amp; _
                                                                           Chr(i5) &amp; Chr(i6) &amp; Chr(n)
                                                    On Error GoTo 0
                                                    If ActiveSheet.ProtectContents = False Then
                                                        MsgBox "密码为：" &amp; Chr(i) &amp; Chr(j) &amp; _
                                                               Chr(k) &amp; Chr(l) &amp; Chr(m) &amp; Chr(i1) &amp; _
                                                               Chr(i2) &amp; Chr(i3) &amp; Chr(i4) &amp; Chr(i5) &amp; _
                                                               Chr(i6) &amp; Chr(n)
                                                        Exit Sub
                                                    End If
                                                Next n
                                            Next i6
                                        Next i5
                                    Next i4
                                Next i3
                            Next i2
                        Next i1
                    Next m
                Next l
            Next k
        Next j
    Next i
End Sub
</code></pre>
