    <p>
     拓扑图
     <br/>
     <img alt="在这里插入图片描述" src="https://i-blog.csdnimg.cn/direct/90d9ad92b08246cdb934e1a660090660.png">
      <br/>
      要求：
     </img>
    </p>
    <ol>
     <li>
      PC1划分在vlan10；PC2划分在vlan20；主机IP和网关如图所示，这里不做展示
     </li>
     <li>
      LSW1配置vlan10的根交换机，vlan20的备根交换机；LSW2配置vlan20的根交换机，vlan10的备根交换机
     </li>
     <li>
      LSW1配置vlan10的VRRP作为vlan10 的网关；LSW2配置vlan20的VRRP作为vlan20的网关
     </li>
    </ol>
    <p>
     LSW1配置vlan
    </p>
    <pre><code class="prism language-bash">vlan batch <span class="token number">10</span> <span class="token number">20</span>
<span class="token comment">#</span>
interface GigabitEthernet0/0/1
 port link-type trunk
 port trunk allow-pass vlan <span class="token number">10</span> <span class="token number">20</span>
<span class="token comment">#</span>
interface GigabitEthernet0/0/2
 port link-type trunk
 port trunk allow-pass vlan <span class="token number">10</span> <span class="token number">20</span>
</code></pre>
    <p>
     LSW2配置vlan
    </p>
    <pre><code class="prism language-bash">vlan batch <span class="token number">10</span> <span class="token number">20</span>
<span class="token comment">#</span>
interface GigabitEthernet0/0/1
 port link-type trunk
 port trunk allow-pass vlan <span class="token number">10</span> <span class="token number">20</span>
<span class="token comment">#</span>
interface GigabitEthernet0/0/2
 port link-type trunk
 port trunk allow-pass vlan <span class="token number">10</span> <span class="token number">20</span>
</code></pre>
    <p>
     LSW3配置vlan
    </p>
    <pre><code class="prism language-bash">vlan batch <span class="token number">10</span> <span class="token number">20</span>
<span class="token comment">#</span>
interface Ethernet0/0/1
 port link-type access
 port default vlan <span class="token number">10</span>
<span class="token comment">#</span>
interface Ethernet0/0/2
 port link-type access
 port default vlan <span class="token number">20</span>
<span class="token comment">#</span>
interface GigabitEthernet0/0/1
 port link-type trunk
 port trunk allow-pass vlan <span class="token number">10</span> <span class="token number">20</span>
<span class="token comment">#</span>
interface GigabitEthernet0/0/2
 port link-type trunk
 port trunk allow-pass vlan <span class="token number">10</span> <span class="token number">20</span>
</code></pre>
    <p>
     下面开始配置MSTP
     <br/>
     LSW1配置MSTP
    </p>
    <pre><code class="prism language-bash">stp mode mstp
<span class="token comment">#</span>
stp region-configuration
 region-name re-mstp
 revision-level <span class="token number">1</span>
 instance <span class="token number">1</span> vlan <span class="token number">10</span>
 instance <span class="token number">2</span> vlan <span class="token number">20</span>
 active region-configuration
<span class="token comment">#</span>
stp instance <span class="token number">1</span> root primary
stp instance <span class="token number">2</span> root secondary
</code></pre>
    <p>
     LSW2配置MSTP
    </p>
    <pre><code class="prism language-bash">stp mode mstp
<span class="token comment">#</span>
stp region-configuration
 region-name re-mstp
 revision-level <span class="token number">1</span>
 instance <span class="token number">1</span> vlan <span class="token number">10</span>
 instance <span class="token number">2</span> vlan <span class="token number">20</span>
 active region-configuration
<span class="token comment">#</span>
stp instance <span class="token number">1</span> root secondary
stp instance <span class="token number">2</span> root primary
</code></pre>
    <p>
     LSW3配置MSTP
    </p>
    <pre><code class="prism language-bash">stp mode mstp
<span class="token comment">#</span>
stp region-configuration
 region-name re-mstp
 revision-level <span class="token number">1</span>
 instance <span class="token number">1</span> vlan <span class="token number">10</span>
 instance <span class="token number">2</span> vlan <span class="token number">20</span>
 active region-configuration
</code></pre>
    <p>
     下面开始配置VRRP
     <br/>
     LSW1配置VRRP
    </p>
    <pre><code class="prism language-bash">interface Vlanif10
 <span class="token function">ip</span> address <span class="token number">192.168</span>.10.253 <span class="token number">255.255</span>.255.0
 vrrp vrid <span class="token number">10</span> virtual-ip <span class="token number">192.168</span>.10.1
 vrrp vrid <span class="token number">10</span> priority <span class="token number">120</span>
 vrrp vrid <span class="token number">10</span> preempt-mode timer delay <span class="token number">30</span>
<span class="token comment">#</span>
interface Vlanif20
 <span class="token function">ip</span> address <span class="token number">192.168</span>.20.253 <span class="token number">255.255</span>.255.0
 vrrp vrid <span class="token number">20</span> virtual-ip <span class="token number">192.168</span>.20.1
 vrrp vrid <span class="token number">20</span> priority <span class="token number">100</span>		<span class="token comment"># 设置优先级100（默认值是100，也可以不设置）</span>
</code></pre>
    <p>
     LSW2配置VRRP
    </p>
    <pre><code class="prism language-bash">interface Vlanif10
 <span class="token function">ip</span> address <span class="token number">192.168</span>.10.254 <span class="token number">255.255</span>.255.0
 vrrp vrid <span class="token number">10</span> virtual-ip <span class="token number">192.168</span>.10.1
 vrrp vrid <span class="token number">10</span> priority <span class="token number">100</span>		<span class="token comment"># 设置优先级100（默认值是100，也可以不设置）</span>
<span class="token comment">#</span>
interface Vlanif20
 <span class="token function">ip</span> address <span class="token number">192.168</span>.20.254 <span class="token number">255.255</span>.255.0
 vrrp vrid <span class="token number">20</span> virtual-ip <span class="token number">192.168</span>.20.1
 vrrp vrid <span class="token number">20</span> priority <span class="token number">120</span>
 vrrp vrid <span class="token number">20</span> preempt-mode timer delay <span class="token number">30</span>
</code></pre>
