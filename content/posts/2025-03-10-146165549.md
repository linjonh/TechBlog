#include &lt;vector&gt;
using namespace std;
int n;string d;
vector &lt;int&gt; t;
void solve()
{
	while(n--)
	{
		int pos = 0;
		for(int i = 0;i&lt;t.size();i++)
		{
			t[i]*=2;
			if(pos){
				t[i]+=1;
				pos = 0;
			}
			
			if(t[i]&gt;=10) pos = 1;
			t[i]%=10;
			
		}
		if(pos) t.push_back(1);
	}
	
	
	
	
}
int main()
{
	cin &gt;&gt; n &gt;&gt; d;
	int flag;
	for(int i = d.size()-1;i&gt;=0;i--)
	{
		if(d[i] != '.') t.push_back(d[i]-'0');
		else{
			flag = d.size()-i-1;
		}
	}
	
	solve();

	if(t[flag-1] &gt;=5)
	{
		t[flag]+=1;
	}
	for(int i = t.size()-1;i&gt;=flag;i--)
	{
		cout &lt;&lt; t[i];
	}
	
	return 0;
}</code></pre>
