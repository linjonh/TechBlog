     widget.h

    <pre><code class="language-cpp">#ifndef WIDGET_H
#define WIDGET_H

#include &lt;QWidget&gt;
#include &lt;QMediaPlayer&gt;
#include &lt;QMediaPlaylist&gt;
#include &lt;QVideoWidget&gt;
#include &lt;QFileDialog&gt;

namespace Ui {
class Widget;
}

class Widget : public QWidget
{
Q_OBJECT
protected:
void paintEvent(QPaintEvent \*event);

public:
explicit Widget(QWidget \*parent = 0);
~Widget();

private slots:
void on_pushButton_open_clicked(); //打开
void on_pushButton_pre_clicked(); //上一个
void on_pushButton_start_stop_clicked(); //暂停——播放
void on_pushButton_next_clicked(); //下一个
void on_pushButton_stop_clicked(); //结束
void on_horizontalSlider_pace_valueChanged(int value); //进度
void on_verticalSlider_volume_valueChanged(int value); //音量

private:
Ui::Widget *ui;
QMediaPlayer *myplayer = nullptr; //播放者
QMediaPlaylist *myplayerlist = nullptr; //播放列表
QVideoWidget *mywidget = nullptr; //在哪播放

    bool isstart = false;                       //初始播放状态为false

};

#endif // WIDGET_H
</code></pre>
<h4>
<strong>
·
</strong>
widget.cpp
</h4>
<pre><code class="language-cpp">#include "widget.h"
#include "ui_widget.h"

Widget::Widget(QWidget \*parent) :
QWidget(parent),
ui(new Ui::Widget)
{
ui-&gt;setupUi(this);

    //播放者
    myplayer = new QMediaPlayer;
    //播放列表
    myplayerlist = new QMediaPlaylist;
    //在哪播放
    mywidget = new QVideoWidget(ui-&gt;label);

    myplayer-&gt;setPlaylist(myplayerlist);
    myplayer-&gt;setVideoOutput(mywidget);
    mywidget-&gt;resize(ui-&gt;label-&gt;size());    //设置播放界面大小
    setWindowTitle("QT-视频播放器");          //设置界面名称

    myplayer-&gt;setVolume(50);                     //设置初始音量为50
    ui-&gt;verticalSlider_volume-&gt;setRange(0,100);//设置音量范围
    ui-&gt;verticalSlider_volume-&gt;setValue(50);   //设置滑动杆的初始位置为50
    //设置播放进度
    connect(myplayer,&amp;QMediaPlayer::positionChanged,ui-&gt;horizontalSlider_pace,&amp;QSlider::setValue);
    connect(myplayer,&amp;QMediaPlayer::durationChanged,ui-&gt;horizontalSlider_pace,[&amp;](qint64 temp){
                ui-&gt;horizontalSlider_pace-&gt;setRange(0,(int)temp);
            });

}

Widget::~Widget()
{
delete ui;
}

void Widget::paintEvent(QPaintEvent _event)
{
mywidget-&gt;resize(ui-&gt;label-&gt;size());//设置播放界面大小为标签大小
}
//打开
void Widget::on_pushButton_open_clicked()
{
//打开文件
QStringList mylist = QFileDialog::getOpenFileNames(this,"选择播放路径", //弹出"选择播放路径"窗口
"D:\\Qt\\220701QT\\VideoPlayer\\音视频", //播放路径
"allfiles(_._);;" //所以文件类型
"MP3(_.mp3);;" //音频
"MP4(\*.mp4);;"); //视频
myplayerlist-&gt;clear();
for(const auto &amp;k:mylist)
myplayerlist-&gt;addMedia(QUrl(k));
}
//上一个
void Widget::on_pushButton_pre_clicked()
{
myplayer-&gt;stop();
myplayerlist-&gt;previous();
myplayer-&gt;play();
}
//暂停——播放
void Widget::on_pushButton_start_stop_clicked()
{
if(isstart)
{
ui-&gt;pushButton_start_stop-&gt;setStyleSheet("QPushButton#pushButton_start_stop{border-image:url(:/播放.png)}");
isstart = false;
myplayer-&gt;pause();//暂停
}
else
{
ui-&gt;pushButton_start_stop-&gt;setStyleSheet("QPushButton#pushButton_start_stop{border-image:url(:/暂停.png)}");
isstart = true;
myplayer-&gt;play();
ui-&gt;pushButton_stop-&gt;setEnabled(true);
}
}
//下一个
void Widget::on_pushButton_next_clicked()
{
myplayer-&gt;stop();
myplayerlist-&gt;next();
myplayer-&gt;play();
}
//结束
void Widget::on_pushButton_stop_clicked()
{
myplayer-&gt;stop();
ui-&gt;pushButton_stop-&gt;setEnabled(false);
ui-&gt;pushButton_start_stop-&gt;setStyleSheet("QPushButton#pushButton_start_stop{border-image:url(:/播放.png)}");
}
//进度
void Widget::on_horizontalSlider_pace_valueChanged(int value)
{
myplayer-&gt;setPosition((qint64)value);
}
//音量
void Widget::on_verticalSlider_volume_valueChanged(int value)
{
myplayer-&gt;setVolume(value);
}
