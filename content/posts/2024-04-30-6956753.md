     </span>

    <p>
     <span style="font-size:18px">
      <span class="pun">
       &lt;?
      </span>

      </span>
      <span class="pun">

      </span>

     </span>

      </span>
      <span class="pln">
      </span>
      <span class="atn">
       xmlns:android
      </span>
      <span class="pun">
       =
      </span>
      <span class="atv">
       "http://schemas.android.com/apk/res/android"
      </span>
     </span>
     <span class="pln">

      <span style="font-size:18px">

      </span>

      <span class="pun">
       =
      </span>

       "@string/search_hint"
      </span>

     <span class="pln">

      </span>
