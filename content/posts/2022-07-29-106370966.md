				float differenceValue = tmp-AB;
				if(tmp&gt;= AB &amp; getGreatestCommonDivisor(i,j)==1) {
					if(minDifferenceValue == -1 || differenceValue &lt; minDifferenceValue) {
						retA = i;
						retB = j;
						minDifferenceValue = differenceValue;
					}
				}
			}
		}
		
		System.out.println(String.format("%d %d", retA,retB));
	}
	
	
	
	private static int getGreatestCommonDivisor(int n1, int n2) {
		int num1 = n1;
		int num2 = n2;
		if(num1&lt;num2){
            int temp=num1;
            num1=num2;
            num2=temp;
        }
 
        int remainder = num1%num2;
 
        while(remainder!=0){
            num1=num2;
            num2=remainder;
            remainder=num1%num2;
        }
        return num2;
	}
	
}
