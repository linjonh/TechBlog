    <h2 id="%E4%B8%80%E3%80%81%E9%A2%98%E7%9B%AE%E6%8F%8F%E8%BF%B0" name="%E4%B8%80%E3%80%81%E9%A2%98%E7%9B%AE%E6%8F%8F%E8%BF%B0">
     一、题目描述
    </h2>
    <p>
     You are given a string
     <code>
      s
     </code>
     containing lowercase letters and an integer
     <code>
      k
     </code>
     . You need to :
    </p>
    <ul>
     <li>
      First, change some characters of
      <code>
       s
      </code>
      to other lowercase English letters.
     </li>
     <li>
      Then divide
      <code>
       s
      </code>
      into
      <code>
       k
      </code>
      non-empty disjoint substrings such that each substring is a palindrome.
     </li>
    </ul>
    <p>
     Return
     <em>
      the minimal number of characters that you need to change to divide the string
     </em>
     .
    </p>
    <p>
    </p>
    <p>
     <strong>
      Example 1:
     </strong>
    </p>
    <pre><strong>Input:</strong> s = "abc", k = 2
<strong>Output:</strong> 1
<strong>Explanation:</strong> You can split the string into "ab" and "c", and change 1 character in "ab" to make it palindrome.
</pre>
    <p>
     <strong>
      Example 2:
     </strong>
    </p>
    <pre><strong>Input:</strong> s = "aabbc", k = 3
<strong>Output:</strong> 0
<strong>Explanation:</strong> You can split the string into "aa", "bb" and "c", all of them are palindrome.</pre>
    <p>
     <strong>
      Example 3:
     </strong>
    </p>
    <pre><strong>Input:</strong> s = "leetcode", k = 8
<strong>Output:</strong> 0

</pre>
    <p>
     <strong>
      Constraints:
     </strong>
    </p>
    <ul>
     <li>
      <code>
       1 &lt;= k &lt;= s.length &lt;= 100
      </code>
      .
     </li>
     <li>
      <code>
       s
      </code>
      only contains lowercase English letters.
     </li>
    </ul>
    <p>
    </p>
    <h2 id="%E4%BA%8C%E3%80%81%E8%A7%A3%E9%A2%98%E6%80%9D%E8%B7%AF" name="%E4%BA%8C%E3%80%81%E8%A7%A3%E9%A2%98%E6%80%9D%E8%B7%AF">
     二、解题思路
    </h2>
    <ul>
     <li>
      <p>
       时间复杂度：O(n^2*k)
      </p>
     </li>
     <li>
      <p>
       空间复杂度：O(n^2+n*k)
      </p>
     </li>
    </ul>
    <h3 id="%E3%80%90C%2B%2B%E3%80%91" name="%E3%80%90C%2B%2B%E3%80%91">
     【C++】
    </h3>
    <pre><code class="language-cpp">class Solution {
public:
    int palindromePartition(string s, int k) {
        vector&lt;vector&lt;int&gt;&gt; change(s.size(), vector&lt;int&gt;(s.size(), 0));
        for (int l = s.size() - 2; l &gt;= 0; --l) {
            for (int r = l + 1; r &lt; s.size(); ++r) {
                change[l][r] = change[l + 1][r - 1] + (s[l] == s[r] ? 0 : 1);
            }
        }
        vector&lt;vector&lt;int&gt;&gt; dp(k, vector&lt;int&gt;(s.size(), INT_MAX));
        dp[0] = move(change[0]);
        for (int i = 1; i &lt; k; ++i) {
            for (int r = i; r &lt;= s.size() - k + i; ++r) {
                for (int l = i; l &lt;= r; ++l) {
                    dp[i][r] = min(dp[i][r], dp[i - 1][l - 1] + change[l][r]);
                }
            }
        }
        return dp[k - 1][s.size() - 1];
    }
};</code></pre>
    <h3 id="%E3%80%90Java%E3%80%91" name="%E3%80%90Java%E3%80%91">
     【Java】
    </h3>
    <pre><code class="language-java">class Solution {
    public int palindromePartition(String s, int k) {
        int[][] change = new int[s.length()][s.length()];
        for (int l = s.length() - 2; l &gt;= 0; --l) {
            for (int r = l + 1; r &lt; s.length(); ++r) {
                change[l][r] = change[l + 1][r - 1] + (s.charAt(l) == s.charAt(r) ? 0 : 1);
            }
        }
        int[][] dp = new int[k][s.length()];
        for (int i = 0; i &lt; k; ++i) Arrays.fill(dp[i], Integer.MAX_VALUE);
        dp[0] = change[0];
        for (int i = 1; i &lt; k; ++i) {
            for (int r = i; r &lt;= s.length() - k + i; ++r) {
                for (int l = i; l &lt;= r; ++l) {
                    dp[i][r] = Math.min(dp[i][r], dp[i - 1][l - 1] + change[l][r]);
                }
            }
        }
        return dp[k - 1][s.length() - 1];
    }
}</code></pre>
