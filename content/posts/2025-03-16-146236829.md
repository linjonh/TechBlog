    </p>
    <p>

    <h2 style="background-color:transparent">
     <a href="https://leetcode.cn/problems/sort-an-array/" rel="nofollow" title="912. 排序数组">
      912. 排序数组
     </a>
    </h2>
    <p>
     <img alt="" height="1522" src="https://i-blog.csdnimg.cn/direct/9517cdc5eb4840468d3392a4854af713.png" width="1080"/>
    </p>
    <h3>
     解法
    </h3>
    <p>
     使用归并算法
    </p>
    <ol>
     <li>
      根据中间点划分区间, mid =(right + left ) / 2
     </li>
     <li>
      将左右区间排序
     </li>
     <li>
      合并两个有序数组
     </li>
     <li>
      处理没有遍历完的数组
     </li>
     <li>
      将排好序的数组tmp覆盖到原数组nums里
     </li>
    </ol>
    <p>
     优化: 将tmp数组new成一个全局变量,减少频繁的空间开销
    </p>
    <h3>
     代码
    </h3>
    <pre><code>class Solution {
    int[] tmp;
    public int[] sortArray(int[] nums) {
        int n = nums.length;
        tmp = new int[n];
        mergeSort(nums, 0, n - 1);
        return nums;
    }
    public void mergeSort(int[] nums, int left, int right){
        if(left &gt;= right) return;

        //1.找到中间点mid
        int mid = (left + right) / 2;
        //2.左右排好序
        mergeSort(nums, left, mid);
        mergeSort(nums, mid + 1, right);
        //3.合并有序数组 [left, mid] [mid + 1, right]
        int cur1 = left, cur2 = mid + 1, i = 0;
        while(cur1 &lt;= mid &amp;&amp; cur2 &lt;= right){
            tmp[i++] = nums[cur1] &lt;= nums[cur2] ? nums[cur1++] : nums[cur2++];
        }
        //处理没有遍历完的数组
        while(cur1 &lt;= mid) tmp[i++] = nums[cur1++];
        while(cur2 &lt;= right) tmp[i++] = nums[cur2++];
        //4.tmp覆盖nums
        for(int j = left; j &lt;= right; j++){
            nums[j] = tmp[j - left];
        }
    }
}</code></pre>
    <h2 style="background-color:transparent">
     <a href="https://leetcode.cn/problems/shu-zu-zhong-de-ni-xu-dui-lcof/description/" rel="nofollow" title="LCR 170. 交易逆序对的总数">
      LCR 170. 交易逆序对的总数
     </a>
     <strong>
      （数组中的逆序对）
     </strong>
    </h2>
    <p>
     <img alt="" height="1262" src="https://i-blog.csdnimg.cn/direct/8380c00db2424eb7912614513f4e776b.png" width="1192"/>
    </p>
    <p>
     <strong>
      解法
     </strong>
    </p>
    <p>
     解法一: 暴力枚举: 两层for循环
    </p>
    <p>
     解法二: 分治,N*logN
    </p>
    <p>
     将数组通过mid分为两部分,
    </p>
    <ol>
     <li>
      先找左半部分的逆序对, 再左排序
     </li>
     <li>
      再找右半部分的逆序对, 再右排序
     </li>
     <li>
      最后找一左一右组合的逆序对(此时
      <strong>
       左数组和右数组分别都是有序的
      </strong>
      ),再合并排序
     </li>
    </ol>
    <p>
     策略一: 找出该数(nums[cur1] )之前,有多少个(mid - cur2 + 1)数比我(nums[cur2] )大.
    </p>
    <p>
    </p>
    <ul>
     <li>
      当nums[cur1] &lt;= nums[cur2] 时 , cur1++;
     </li>
     <li>
      当nums[cur1] &gt; nums[cur2]时 ,ret += mid - cur1 + 1, cur2++
     </li>
    </ul>
    <p>
     <img alt="" height="474" src="https://i-blog.csdnimg.cn/direct/9f51c36c34ef4731a7cad78b51ff6dad.png" width="1404"/>
    </p>
    <p>
     策略二: 找出该数(nums[cur1] )之后,有多少个(right - cur2 + 1)数比我(nums[cur2] )小.
    </p>
    <ul>
     <li>
      当nums[cur1] &lt;= nums[cur2] 时 , cur2++;
     </li>
     <li>
      当nums[cur1] &gt; nums[cur2]时 ,ret += right - cur2 + 1, cur1++
     </li>
    </ul>
    <p>
     <img alt="" height="403" src="https://i-blog.csdnimg.cn/direct/ff589d2c87e24d898fbad09c3e0111b2.png" width="1195"/>
    </p>
    <ul>
     <li>
      <strong>
       时间复杂度
      </strong>
      ：O(nlogn)，这是归并排序的时间复杂度，其中 n 是数组的长度。
     </li>
     <li>
      <strong>
       空间复杂度
      </strong>
      ：O(n)，主要是临时数组
      <code>
       tmp
      </code>
      的空间开销。
     </li>
    </ul>
    <h3>
     代码
    </h3>
    <pre><code>class Solution {
    int[] tmp;
    public int reversePairs(int[] nums) {
        int n = nums.length;
        tmp = new int[n];
        return merge(nums, 0, n - 1);
    }
    public int merge(int[] nums, int left, int right){
        if(left &gt;= right) return 0;
        //找出中间点mid
        int mid = (left + right) / 2;
        int ret = 0;
        //左右排好序,返回结果
        ret += merge(nums, left, mid);
        ret += merge(nums, mid + 1, right);

        //合并有序数组,计算一左一右逆序对
        int cur1 = left, cur2 = mid + 1, i = 0;
        while(cur1 &lt;= mid &amp;&amp; cur2 &lt;= right){
            if(nums[cur1] &lt;= nums[cur2]){
                tmp[i++] = nums[cur1++];
            }else{
                ret += mid - cur1 + 1;
                tmp[i++] = nums[cur2++];
            }
            
        }
        //处理剩下的数组
        while(cur1 &lt;= mid) tmp[i++] = nums[cur1++];
        while(cur2 &lt;= right) tmp[i++] = nums[cur2++];

        for(int j = left; j &lt;= right; j++){
            nums[j] = tmp[j - left];
        }
        return ret;

    }
}</code></pre>
    <h2 style="background-color:transparent">
     <a href="https://leetcode.cn/problems/count-of-smaller-numbers-after-self/description/" rel="nofollow" title="315. 计算右侧小于当前元素的个数">
      315. 计算右侧小于当前元素的个数
     </a>
    </h2>
    <p>
     <img alt="" height="1931" src="https://i-blog.csdnimg.cn/direct/52585ca0ddbb431785d89ef93b697021.png" width="1378"/>
    </p>
    <h3>
     解法
    </h3>
    <p>
     解法: 归并排序
    </p>
    <p>
     和上一题类似,但是这里只能通过策略二: 降序解决
    </p>
    <p>
    </p>
    <p>
     <img alt="" height="839" src="https://i-blog.csdnimg.cn/direct/b2c2347ae38f475db08a7f6913ab4874.png" width="1798"/>
    </p>
    <p>
     注意: 最终结果是加在原始数组的下标中
    </p>
    <h3>
     代码
    </h3>
    <pre><code>class Solution {
    int[] ret;
    int[] index; //标记 nums中当前元素的原始下标
    int[] tmpIndex;
    int[] tmpNums;

    public List&lt;Integer&gt; countSmaller(int[] nums) {
        int n = nums.length;
        ret = new int[n];
        index = new int[n];
        tmpIndex = new int[n];
        tmpNums = new int[n];
        //初始化index数组
        for(int i = 0; i &lt; n; i++){
            index[i] = i;
        }
        merge(nums, 0, n - 1);
        List&lt;Integer&gt; l = new ArrayList&lt;&gt;();
        for(int x : ret){
            l.add(x);
        }
        return l;
    }
    public void merge(int[] nums, int left, int right){
        if(left &gt;= right) return;
        int mid = (left + right) / 2;
        merge(nums, left, mid);
        merge(nums, mid + 1, right);
        //[left, mid] [mid + 1, right] 处理一左一右的情况
        int cur1 = left, cur2 = mid + 1, i = 0;
        while(cur1 &lt;= mid &amp;&amp; cur2 &lt;= right){//降序排列
            if(nums[cur1] &lt;= nums[cur2]){
                tmpNums[i] = nums[cur2];
                tmpIndex[i++] = index[cur2++];//下标数组同步修改
            }else{
                ret[index[cur1]] += right - cur2 + 1;//结果在原数组下标对应值位置上
                tmpNums[i] = nums[cur1];
                tmpIndex[i++] = index[cur1++];
            }
        }
        while(cur1 &lt;= mid) {
            tmpNums[i] = nums[cur1];
            tmpIndex[i++] = index[cur1++];

        }
        while(cur2 &lt;= right){
            tmpNums[i] = nums[cur2];
            tmpIndex[i++] = index[cur2++];
        } 

        for(int j = left; j &lt;= right; j++){
            nums[j] = tmpNums[j - left];
            index[j] = tmpIndex[j - left];
        }
    }
}</code></pre>

    <h3>
     代码
    </h3>
    <p>
     策略一: 降序,谁大谁在前面
    </p>
    <pre><code>class Solution {
    int[] tmp; 
    public int reversePairs(int[] nums) {
     int n = nums.length;
        tmp = new int[n];
        return merge(nums, 0, n - 1);
    }
    public int merge(int[] nums, int left, int right){
        if(left &gt;= right) return 0;
        int mid = (left + right) / 2, ret = 0;

        ret += merge(nums, left, mid);
        ret += merge(nums, mid + 1, right);
        //[left, mid] [mid + 1, right]
        //先统计,后归并
        int x = left, y = mid + 1;
        while(x &lt;= mid &amp;&amp; y &lt;= right){//降序
            if(nums[x] / 2.0 &gt; nums[y] ){
                ret += right - y + 1;
                x++;
            }else{                
                y++;
            }
        }
        
        int cur1 = left, cur2 = mid + 1, i = 0;
        while(cur1 &lt;= mid &amp;&amp; cur2 &lt;= right){
            if(nums[cur1] &lt;= nums[cur2]){
                tmp[i++] = nums[cur2++];
            }else{                
                tmp[i++] = nums[cur1++];
            }
        }
        while(cur1 &lt;= mid) tmp[i++] = nums[cur1++];
        while(cur2 &lt;= right) tmp[i++] = nums[cur2++];

        for(int j = left; j &lt;= right; j++){
            nums[j] = tmp[j - left];
        }
        return ret;
    }
}</code></pre>
