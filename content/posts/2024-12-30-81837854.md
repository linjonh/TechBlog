# 学了几个小时后python后，我写的第一个python小游戏

不多说，直接上代码！

print('----------第一个pyhon------------')
  
print("1.lol  2.cf  3  dnf 4.吃鸡 5.冒险岛 6.剑侠")
  
temp = input("猜猜现在那个游戏最火？")
  
guess = int(temp)
  
while guess != 3:
  
temp = input("猜错了，请重新输入吧")
  
guess = int(temp)
  
if guess == 3:
  
print("厉害了，这都被你发现了！")
  
print("不过，没有奖励！哈哈")
  
else:
  
if guess > 3:
  
print("大了 大了")
  
else:
  
print("小了 小了")
  
print("游戏结束！不玩啦")
