    <pre><code class="language-cs">#include&lt;reg52.h&gt;


char duan[]={0xc0,0xf9,0xa4,0xb0,0x99,0x92,0x82,0xf8,0x80,0x90};
int num=0;

void init()
{
	TMOD=0x60;    //用计数器0的值： 0000 0110
	TH1=256-3;   // 用计数器0的值： TH0
	TL1=256-3;   // 用计数器0的值： TL0
	ET1=1;       // 用计数器0的值： ET0
	TR1=1;       // 用计数器0的值： TR0
	EA=1;
}

void display()
{
	P2=duan[num];
}

int main()
{
	init();
	while(1)
	{
		display();
	}
}

void zhongduan()interrupt 3
{
	num++;
	if(num==10)
	{
		num=0;
	}
}</code></pre>

     为此写了一个小程序来验证学习到的知识是否正确。主要功能就是控制P0的第0口的小灯间隔1秒亮灭，来看程序，这是使用16位计时器0的例子：
    </p>
    <pre><code class="language-cs">#include&lt;reg52.h&gt;

sbit LED = P0^0;
unsigned int count = 0;
void InitTime()
{
	TMOD = 0x01;
	TH0 = 0xB4;
	TL0 = 0x00;
	TR0 = 1;
}

int main()
{
	InitTime();
	
	while(1)
	{
		if(1 == TF0)
		{
			TF0 = 0; // 将硬件置1的溢出标志位置0，重新计时
			TH0 = 0xB4;
			TL0 = 0x00;
			count++;
			if(40 == count)
			{
				count = 0;
				LED = ~LED;
			}
		}
		
		
	}
	return 0;
}</code></pre>
