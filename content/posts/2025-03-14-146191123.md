    <pre><code class="language-cpp">void HAL_TIM_PeriodElapsedCallback(TIM_HandleTypeDef *htim)   //中断回调函数
{
	if(htim == (&amp;htim3))//定时器3中断
	{
		Data_Slow++;//阿里云数据上传减速
		if(Data_Slow == 2000)  Data_Slow = 0;

		Working_Pro();
	
	}
}

void Working_Pro()
{
	Key_Slow++;
	if(Key_Slow == 10)  Key_Slow = 0;//按键消抖
	
	if(Beep_flag == 1)//蜂鸣器开  着火了
	{
		Beep_Slow ++;
		Flame_Slow ++;
		
		if(Flame_Slow == 500)//火焰状态显示
		{
			Flame_Slow = 0;//复位
			Flame_flag ^= 1;//取反
		}
		
		if(Beep_Slow &lt; 200 &amp;&amp; Beep_data == 0)
		{
			Beep_data = 1;//开
		}
		else if(Beep_Slow &gt;= 200 &amp;&amp; Beep_Slow &lt; 400 &amp;&amp; Beep_data == 1)
		{
			Beep_data = 0;
		}
		else if(Beep_Slow &gt;= 400 &amp;&amp; Beep_Slow &lt; 600 &amp;&amp; Beep_data == 0)
		{
			Beep_data = 1;
		}
		else if(Beep_Slow &gt;= 600 &amp;&amp; Beep_Slow &lt; 1200 &amp;&amp; Beep_data == 1)
		{
			Beep_data = 0;
		}
		else if(Beep_Slow == 1200 &amp;&amp; Beep_data == 0)
		{
			Beep_Slow = 0;//复位
		}

	}
	else if(Beep_flag == 0)//蜂鸣器关
	{
		Beep_Slow = 0;
		Beep_data = 0;//关
	}
}</code></pre>
